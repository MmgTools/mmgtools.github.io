51 static int MMG5_adpspl(MMG5_pMesh mesh,MMG5_pSol met, int* warn) {
52  MMG5_pTetra pt;
53  MMG5_pxTetra pxt;
54  MMG5_pPoint p0,p1;
55  double len,lmax,o[3];
56  int k,ip,ip1,ip2,list[MMG3D_LMAX+2],ilist,src;
57  int ns,ier;
58  int8_t imax,j,i,i1,i2,ifa0,ifa1;
59  int8_t chkRidTet;
60  static int8_t mmgWarn = 0;
61 
62  *warn=0;
63  ns = 0;
64 
65  if ( met->size==6 ) chkRidTet=1;
66  else chkRidTet=0;
67 
68  for (k=1; k<=mesh->ne; k++) {
69  pt = &mesh->tetra[k];
70  if ( !MG_EOK(pt) || (pt->tag & MG_REQ) ) continue;
71  pxt = pt->xt ? &mesh->xtetra[pt->xt] : 0;
72 
73  /* find longest edge */
74  imax = -1; lmax = 0.0;
75  for (i=0; i<6; i++) {
76  if ( pt->xt && (pxt->tag[i] & MG_REQ) ) continue;
77  len = MMG5_lenedg(mesh,met,i,pt);
78 
79  if ( len > lmax ) {
80  lmax = len;
81  imax = i;
82  }
83  }
84  if ( imax==-1 ) {
85  if ( !mmgWarn ) {
86  fprintf(stderr,
87  "\n ## Warning: %s: at least 1 tetra with 4 required"
88  " or null edges.\n",__func__);
89  mmgWarn = 1;
90  }
91  continue;
92  }
93  if ( lmax < MMG3D_LOPTL ) continue;
94 
95  /* proceed edges according to lengths */
96  ifa0 = MMG5_ifar[imax][0];
97  ifa1 = MMG5_ifar[imax][1];
98  i = (pt->xt && (pxt->ftag[ifa1] & MG_BDY)) ? ifa1 : ifa0;
99  j = MMG5_iarfinv[i][imax];
100  i1 = MMG5_idir[i][MMG5_inxt2[j]];
101  i2 = MMG5_idir[i][MMG5_iprv2[j]];
102  ip1 = pt->v[i1];
103  ip2 = pt->v[i2];
104  p0 = &mesh->point[ip1];
105  p1 = &mesh->point[ip2];
106 
107  /* Case of a boundary face */
108  if ( pt->xt && (pxt->ftag[i] & MG_BDY) ) {
109  if ( !(MG_GET(pxt->ori,i)) ) continue;
110 
111  ier = MMG3D_splsurfedge( mesh,met,k,pt,pxt,imax,2,chkRidTet,warn );
112 
113  if ( ier==-1 ) { return -1; }
114  else if ( !ier ) { continue; }
115  else if ( ier==2 ) { break; }
116 
117  ++ns;
118  }
119  else {
120  /* Case of an internal face */
121  if ( (p0->tag & MG_BDY) && (p1->tag & MG_BDY) ) continue;
122  ilist = MMG5_coquil(mesh,k,imax,list);
123  if ( !ilist ) continue;
124  else if ( ilist<0 ) return -1;
125  o[0] = 0.5*(p0->c[0] + p1->c[0]);
126  o[1] = 0.5*(p0->c[1] + p1->c[1]);
127  o[2] = 0.5*(p0->c[2] + p1->c[2]);

131 #else
132  src = 1;
133 #endif
134  ip = MMG3D_newPt(mesh,o,MG_NOTAG,src);
135 
136  if ( !ip ) {
137  /* reallocation of point table */
138  MMG3D_POINT_REALLOC(mesh,met,ip,mesh->gap,
139  *warn=1;
140  break
141  ,o,MG_NOTAG,src);
142  }
143  if ( met->m ) {
144  ier = MMG5_intmet(mesh,met,k,imax,ip,0.5);
145  if ( !ier ) {
146  MMG3D_delPt(mesh,ip);
147  return -1;
148  }
149  else if (ier < 0 ) {
150  MMG3D_delPt(mesh,ip);
151  continue;
152  }
153  }

155  if ( ier == 1 )

157 
158  if ( ier < 0 ) {

160  return -1;
161  }
162  else if ( ier == 0 || ier == 2 ) {
163  MMG3D_delPt(mesh,ip);
164  }
165  else {
166  ns++;
167  }
168  }
169  }
170 
171  return ns;
172 }

184 static int MMG5_adpcol(MMG5_pMesh mesh,MMG5_pSol met) {
185  MMG5_pTetra pt;
186  MMG5_pxTetra pxt;
187  MMG5_pPoint p0,p1;
188  double len,lmin;
189  int k,ip,iq,list[MMG3D_LMAX+2],ilist,lists[MMG3D_LMAX+2],ilists,nc;
190  int ier;
191  int16_t tag;
192  int8_t imin,j,i,i1,i2,ifa0,ifa1;
193  static int8_t mmgWarn = 0;
194 
195  nc = 0;
196  for (k=1; k<=mesh->ne; k++) {
197  pt = &mesh->tetra[k];
198  if ( !MG_EOK(pt) || (pt->tag & MG_REQ) ) continue;
199  pxt = pt->xt ? &mesh->xtetra[pt->xt] : 0;
200 
201  /* find shortest edge */
202  imin = -1; lmin = DBL_MAX;
203  for (i=0; i<6; i++) {
204  if ( pt->xt && (pxt->tag[i] & MG_REQ) ) continue;
205  len = MMG5_lenedg(mesh,met,i,pt);
206 
207  if ( len < lmin ) {
208  lmin = len;
209  imin = i;
210  }
211  }
212  if ( imin==-1 ) {
213  if ( !mmgWarn ) {
214  fprintf(stderr,
215  "\n ## Warning: %s: at least 1 tetra with 4 required"
216  " or null edges.\n",__func__);
217  mmgWarn = 1;
218  }
219  continue;
220  }
221  if ( lmin > MMG3D_LOPTS ) continue;
222 
223  // Case of an internal tetra with 4 ridges vertices.
224  if ( lmin == 0 ) continue;
225 
226  ifa0 = MMG5_ifar[imin][0];
227  ifa1 = MMG5_ifar[imin][1];
228  i = (pt->xt && (pxt->ftag[ifa1] & MG_BDY)) ? ifa1 : ifa0;
229  j = MMG5_iarfinv[i][imin];
230  i1 = MMG5_idir[i][MMG5_inxt2[j]];
231  i2 = MMG5_idir[i][MMG5_iprv2[j]];
232  ip = pt->v[i1];
233  iq = pt->v[i2];
234  p0 = &mesh->point[ip];
235  p1 = &mesh->point[iq];
236  if ( (p0->tag > p1->tag) || (p0->tag & MG_REQ) ) continue;
237 
238 
239  /* Case of a boundary face */
240  ilist = 0;
241  if ( pt->xt && (pxt->ftag[i] & MG_BDY) ) {
242  tag = pxt->tag[MMG5_iarf[i][j]];
243  if ( tag & MG_REQ ) continue;
244  tag |= MG_BDY;
245  if ( p0->tag > tag ) continue;
246  if ( ( tag & MG_NOM ) && (mesh->adja[4*(k-1)+1+i]) ) continue;
247 
248  if (MMG5_boulesurfvolp(mesh,k,i1,i,
249  list,&ilist,lists,&ilists,(p0->tag & MG_NOM)) < 0 )
250  return -1;
251 
252  ilist = MMG5_chkcol_bdy(mesh,met,k,i,j,list,ilist,lists,ilists,0,0,2,0,0);
253  }
254  /* Case of an internal face */
255  else {
256  if ( p0->tag & MG_BDY ) continue;
257  ilist = MMG5_boulevolp(mesh,k,i1,list);
258  ilist = MMG5_chkcol_int(mesh,met,k,i,j,list,ilist,2);
259  }
260  if ( ilist > 0 ) {
261  ier = MMG5_colver(mesh,met,list,ilist,i2,2);
262  if ( ier < 0 ) return -1;
263  else if ( ier ) {
264  MMG3D_delPt(mesh,ier);
265  nc++;
266  }
267  }
268  else if (ilist < 0 ) return -1;
269  }
270 
271  return nc;
272 }
