39 int MMGS_newPt(MMG5_pMesh mesh,double c[3],double n[3]) {
40  MMG5_pPoint ppt;
41  int curpt;
42 
43  if ( !mesh->npnil ) return 0;
44 
45  curpt = mesh->npnil;
46  if ( mesh->npnil > mesh->np ) mesh->np = mesh->npnil;
47  ppt = &mesh->point[curpt];
48  memcpy(ppt->c,c,3*sizeof(double));
49  if ( n )
50  memcpy(ppt->n,n,3*sizeof(double));
51  ppt->tag &= ~MG_NUL;
52  mesh->npnil = ppt->tmp;
53  ppt->tmp = 0;
54 
55  return curpt;
56 }

127 static inline
128 int MMGS_memOption_memSet(MMG5_pMesh mesh) {
129  size_t usedMem,avMem,reservedMem,npadd;
130  int bytes;
131 
132  MMG5_memOption_memSet(mesh);
133 
134  /* init allocation need MMG5_MEMMIN B */
135  reservedMem = MMG5_MEMMIN;
136 
137  /* Compute the needed initial memory */
138  usedMem = reservedMem + (mesh->np+1)*sizeof(MMG5_Point)
139  + (mesh->nt+1)*sizeof(MMG5_Tria) + (3*mesh->nt+1)*sizeof(int)
140  + (mesh->np+1)*sizeof(double);
141 
142  if ( usedMem > mesh->memMax ) {
143  fprintf(stderr,"\n ## Error: %s: %zu MB of memory ",__func__,mesh->memMax/MMG5_MILLION);
144  fprintf(stderr,"is not enough to load mesh. You need to ask %zu MB minimum\n",
145  usedMem/MMG5_MILLION+1);
146  return 0;
147  }
148 
149  /* point+xpoint+tria+adja+aniso sol */
150  bytes = sizeof(MMG5_Point) + sizeof(MMG5_xPoint) +
151  2*sizeof(MMG5_Tria) + 3*sizeof(int) + 6*sizeof(double);
152 
153  avMem = mesh->memMax-usedMem;
154 
155  /* If npadd is exactly the maximum memory available, we will use all the
156  * memory and the analysis step will fail. As arrays may be reallocated, we
157  * can have smaller values for xpmax and ntmax (npadd/2). */
158  npadd = avMem/(2*bytes);
159  mesh->npmax = MG_MIN(mesh->npmax,mesh->np+npadd);
160  mesh->ntmax = MG_MIN(mesh->ntmax,2*npadd+mesh->nt);
161 
162  if ( abs(mesh->info.imprim) > 4 || mesh->info.ddebug ) {
163  fprintf(stdout," MAXIMUM MEMORY AUTHORIZED (MB) %zu\n",
164  mesh->memMax/MMG5_MILLION);
165  }
166 
167  if ( abs(mesh->info.imprim) > 5 || mesh->info.ddebug ) {
168  fprintf(stdout," MMG2D_NPMAX %d\n",mesh->npmax);
169  fprintf(stdout," MMG2D_NTMAX %d\n",mesh->ntmax);
170  }
171 
172  return 1;
173 }

202  int k;
203

205  fprintf(stderr," Exit program.\n");

212 
213 
214  mesh->namax = mesh->na;
215  if ( mesh->na ) {
216  MMG5_ADD_MEM(mesh,(mesh->na+1)*sizeof(MMG5_Edge),"initial edges",return 0);
217  MMG5_SAFE_CALLOC(mesh->edge,(mesh->na+1),MMG5_Edge,return 0);
218  }
219 
220  /* keep track of empty links */
221  mesh->npnil = mesh->np + 1;
222  mesh->nenil = mesh->nt + 1;
223 
224  for (k=mesh->npnil; k<mesh->npmax-1; k++)
225  mesh->point[k].tmp = k+1;
226 
227  for (k=mesh->nenil; k<mesh->ntmax-1; k++)
228  mesh->tria[k].v[2] = k+1;
229 
230  return 1;
231 }
