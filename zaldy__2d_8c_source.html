38 int MMG2D_newPt(MMG5_pMesh mesh,double c[2],int16_t tag) {
39  MMG5_pPoint ppt;
40  int curpt;
41 
42  if ( !mesh->npnil ) return 0;
43 
44  curpt = mesh->npnil;
45  if ( mesh->npnil > mesh->np ) mesh->np = mesh->npnil;
46  ppt = &mesh->point[curpt];
47  memcpy(ppt->c,c,2*sizeof(double));
48  ppt->tag &= ~MG_NUL;
49  mesh->npnil = ppt->tmp;
50  ppt->tmp = 0;
51  ppt->tag = tag;
52 
53  return curpt;
54 }

156 static inline
157 int MMG2D_memOption_memSet(MMG5_pMesh mesh) {
158  size_t usedMem,avMem,reservedMem,npadd;
159  int ctri,bytes;
160 
161  MMG5_memOption_memSet(mesh);
162 
163  /* init allocation need MMG5_MEMMIN B */
164  reservedMem = MMG5_MEMMIN + mesh->nquad*sizeof(MMG5_Quad);
165 
166  /* Compute the needed initial memory */
167  usedMem = reservedMem + (mesh->np+1)*sizeof(MMG5_Point)
168  + (mesh->nt+1)*sizeof(MMG5_Tria) + (3*mesh->nt+1)*sizeof(int)
169  + (mesh->na+1)*sizeof(MMG5_Edge) + (mesh->np+1)*sizeof(double);
170 
171  if ( usedMem > mesh->memMax ) {
172  fprintf(stderr,"\n ## Error: %s: %zu MB of memory ",__func__,
173  mesh->memMax/MMG5_MILLION);
174  fprintf(stderr,"is not enough to load mesh. You need to ask %zu MB minimum\n",
175  usedMem/MMG5_MILLION+1);
176  return 0;
177  }
178 
179  ctri = 2;
180 
181  /* Euler-poincare: ne = 6*np; nt = 2*np; na = np/5 *
182  * point+tria+edges+adjt+ aniso sol */
183  bytes = sizeof(MMG5_Point) +
184  2*sizeof(MMG5_Tria) + 3*2*sizeof(int)
185  + 0.2*sizeof(MMG5_Edge) + 3*sizeof(double);
186 
187  avMem = mesh->memMax-usedMem;
188 
189  /* If npadd is exactly the maximum memory available, we will use all the
190  * memory and the analysis step will fail. As arrays may be reallocated, we
191  * can have smaller values for npmax and ntmax (npadd/2). */
192  npadd = avMem/(2*bytes);
193  mesh->npmax = MG_MIN(mesh->npmax,mesh->np+npadd);
194  mesh->ntmax = MG_MIN(mesh->ntmax,ctri*npadd+mesh->nt);
195  mesh->namax = MG_MIN(mesh->namax,ctri*npadd+mesh->na);
196 
197  if ( abs(mesh->info.imprim) > 4 || mesh->info.ddebug ) {
198  fprintf(stdout," MAXIMUM MEMORY AUTHORIZED (MB) %zu\n",
199  mesh->memMax/MMG5_MILLION);
200  }
201 
202  if ( abs(mesh->info.imprim) > 5 || mesh->info.ddebug ) {
203  fprintf(stdout," MMG2D_NPMAX %d\n",mesh->npmax);
204  fprintf(stdout," MMG2D_NTMAX %d\n",mesh->ntmax);
205  }
206 
207  return 1;
208 }

237 int MMG2D_setMeshSize_alloc( MMG5_pMesh mesh ) {
238  int k;
239 
240  MMG5_ADD_MEM(mesh,(mesh->npmax+1)*sizeof(MMG5_Point),"initial vertices",
241  printf(" Exit program.\n");
242  return 0);
243  MMG5_SAFE_CALLOC(mesh->point,mesh->npmax+1,MMG5_Point,return 0);
244 
245  MMG5_ADD_MEM(mesh,(mesh->ntmax+1)*sizeof(MMG5_Tria),"initial triangles",return 0);
246  MMG5_SAFE_CALLOC(mesh->tria,mesh->ntmax+1,MMG5_Tria,return 0);
247  memset(&mesh->tria[0],0,sizeof(MMG5_Tria));
248 
249  if ( mesh->nquad ) {
250  MMG5_ADD_MEM(mesh,(mesh->nquad+1)*sizeof(MMG5_Quad),"initial quadrilaterals",return 0);
251  MMG5_SAFE_CALLOC(mesh->quadra,(mesh->nquad+1),MMG5_Quad,return 0);
252  }
253 
254  mesh->namax = mesh->na;
255  if ( mesh->na ) {
256  MMG5_ADD_MEM(mesh,(mesh->namax+1)*sizeof(MMG5_Edge),"initial edges",return 0);
257  MMG5_SAFE_CALLOC(mesh->edge,(mesh->namax+1),MMG5_Edge,return 0);
258  }
259 
260  /* keep track of empty links */
261  mesh->npnil = mesh->np + 1;
262  mesh->nenil = mesh->nt + 1;
263  mesh->nanil = 0;
264 
265  for (k=mesh->npnil; k<mesh->npmax-1; k++) {
266  /* Set tangent field of point to 0 */
267  mesh->point[k].n[0] = 0;
268  mesh->point[k].n[1] = 0;
269  mesh->point[k].n[2] = 0;
270  /* link */
271  mesh->point[k].tmp = k+1;
272  }
273 
274  for (k=mesh->nenil; k<mesh->ntmax-1; k++)
275  mesh->tria[k].v[2] = k+1;
276 
277  return 1;
278 }
