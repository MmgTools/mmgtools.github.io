52 double caleltsig_ani(MMG5_pMesh mesh,MMG5_pSol met,int iel) {
53  MMG5_pTria pt;
54  MMG5_pPoint pa,pb,pc;
55  double ps1,ps2,abx,aby,abz,acx,acy,acz,dd,rap,anisurf;
56  double n[3],pv[3],l[3],*ncomp,*a,*b,*c;
57  int ia,ib,ic;
58 
59  pt = &mesh->tria[iel];
60  ia = pt->v[0];
61  ib = pt->v[1];
62  ic = pt->v[2];
63 
64  pa = &mesh->point[ia];
65  pb = &mesh->point[ib];
66  pc = &mesh->point[ic];
67 
68  a = &pa->c[0];
69  b = &pb->c[0];
70  c = &pc->c[0];
71 
72  /* area */
73  abx = b[0] - a[0];
74  aby = b[1] - a[1];
75  abz = b[2] - a[2];
76  acx = c[0] - a[0];
77  acy = c[1] - a[1];
78  acz = c[2] - a[2];
79 
80  pv[0] = aby*acz - abz*acy;
81  pv[1] = abz*acx - abx*acz;
82  pv[2] = abx*acy - aby*acx;
83 
84  dd = pv[0]*pv[0] + pv[1]*pv[1] + pv[2]*pv[2];
85  if ( dd < MMG5_EPSD2 ) return 0.0;
86  dd = 1.0 / sqrt(dd);
87 
88  // If one of the triangle vertex is not REF or GEO, it contains the normal at
89  // the C1 surface.
90  if ( !MG_EDG(pa->tag) ) {
91  memcpy(n,&pa->n[0],3*sizeof(double));
92  ps1 = n[0]*pv[0]+n[1]*pv[1]+n[2]*pv[2];
93  ps1 *= dd;
94  }
95  else if ( !MG_EDG(pb->tag) ) {
96  memcpy(n,&pb->n[0],3*sizeof(double));
97  ps1 = n[0]*pv[0]+n[1]*pv[1]+n[2]*pv[2];
98  ps1 *= dd;
99  }
100  else if ( !MG_EDG(pc->tag) ) {
101  memcpy(n,&pc->n[0],3*sizeof(double));
102  ps1 = n[0]*pv[0]+n[1]*pv[1]+n[2]*pv[2];
103  ps1 *= dd;
104  }
105  else {
106  // We must find the normal at the surface elsewhere. Arbitrary, we take it
107  // at point pa.
108  memcpy(n,&mesh->xpoint[pa->xp].n1[0],3*sizeof(double));
109  ps1 = n[0]*pv[0]+n[1]*pv[1]+n[2]*pv[2];
110  ps1 *= dd;
111 
112  if ( (pa->tag & MG_GEO) ) {
113  ncomp = &mesh->xpoint[pa->xp].n2[0];
114  ps2 = ncomp[0]*pv[0]+ncomp[1]*pv[1]+ncomp[2]*pv[2];
115  ps2 *= dd;
116  if ( fabs(1.0-fabs(ps1)) > fabs(1.0-fabs(ps2)) ) {
117  memcpy(n,ncomp,3*sizeof(double));
118  ps1 = ps2;
119  }
120  }
121  }
122 
123  /* if orientation is reversed with regards to orientation of vertices */
124  if ( ps1 < 0.0 ) return -1.0;
125 
126  anisurf = MMG5_surftri_ani(mesh,met,pt);
127  if ( anisurf == 0.0 ) return -1.0;
128 
129  l[0] = MMG5_lenSurfEdg_ani(mesh,met,ib,ic,( pt->tag[0] & MG_GEO ));
130  l[1] = MMG5_lenSurfEdg_ani(mesh,met,ia,ic,( pt->tag[1] & MG_GEO ));
131  l[2] = MMG5_lenSurfEdg_ani(mesh,met,ia,ib,( pt->tag[2] & MG_GEO ));
132 
133  rap = l[0]*l[0] + l[1]*l[1] + l[2]*l[2];
134  if ( rap < MMG5_EPSD2 ) return 0.0;
135  return anisurf / rap;
136 }

148  ib = pt->v[1];
149  ic = pt->v[2];
150

152  pb = &mesh->point[ib];
153  pc = &mesh->point[ic];
154 
155  a = &pa->c[0];
156  b = &pb->c[0];
157  c = &pc->c[0];
158 
159  /* area */
160  abx = b[0] - a[0];
161  aby = b[1] - a[1];
162  abz = b[2] - a[2];
163  acx = c[0] - a[0];
164  acy = c[1] - a[1];
165  acz = c[2] - a[2];
166  bcx = c[0] - b[0];
167  bcy = c[1] - b[1];
168  bcz = c[2] - b[2];
169 
170  pv[0] = aby*acz - abz*acy;
171  pv[1] = abz*acx - abx*acz;
172  pv[2] = abx*acy - aby*acx;
173

175  sqcal = sqrt(cal);
176 
177  if ( sqcal < MMG5_EPSD2 ) return 0.0;
178  invsqcal = 1.0 / sqcal;
179

181  memcpy(n,&pa->n[0],3*sizeof(double));
182  ps1 = n[0]*pv[0]+n[1]*pv[1]+n[2]*pv[2];
183  ps1 *= invsqcal;
184  }

186  memcpy(n,&pb->n[0],3*sizeof(double));
187  ps1 = n[0]*pv[0]+n[1]*pv[1]+n[2]*pv[2];
188  ps1 *= invsqcal;
189  }
190  else if ( !MG_EDG(pc->tag) ) {
191  memcpy(n,&pc->n[0],3*sizeof(double));
192  ps1 = n[0]*pv[0]+n[1]*pv[1]+n[2]*pv[2];
193  ps1 *= invsqcal;
194  }
195  else {
196  memcpy(n,&mesh->xpoint[pa->xp].n1[0],3*sizeof(double));
197  ps1 = n[0]*pv[0]+n[1]*pv[1]+n[2]*pv[2];
198  ps1 *= invsqcal;
199 
200  if ( (pa->tag & MG_GEO) ) {
201  ncomp = &mesh->xpoint[pa->xp].n2[0];
202  ps2 = ncomp[0]*pv[0]+ncomp[1]*pv[1]+ncomp[2]*pv[2];
203  ps2 *= invsqcal;
204  if ( fabs(1.0-fabs(ps1)) > fabs(1.0-fabs(ps2)) ) {
205  memcpy(n,ncomp,3*sizeof(double));
206  ps1 = ps2;
207  }
208  }
209  }
210 
211  /* if orientation is reversed with regards to orientation of vertex */
212  if ( ps1 < 0.0 ) return -1.0;
213  if ( cal > MMG5_EPSD2 ) {
214  /* qual = 2.*surf / length */
215  rap = abx*abx + aby*aby + abz*abz;
216  rap += acx*acx + acy*acy + acz*acz;
217  rap += bcx*bcx + bcy*bcy + bcz*bcz;
218  if ( rap > MMG5_EPSD2 )
219  return sqrt(cal) / rap;
220  else
221  return 0.0;
222  }
223  else
224  return 0.0;
225 }

235  o[2] = dd * (p0->c[2] + p1->c[2] + p2->c[2]);
236

283  MMG5_Hash hash;
284  double len,avlen,lmin,lmax;
285  int k,np,nq,amin,bmin,amax,bmax,ned,hl[9],nullEdge;
286  int8_t ia,i0,i1,i;
287  static double bd[9]= {0.0, 0.3, 0.6, 0.7071, 0.9, 1.3, 1.4142, 2.0, 5.0};
288 
289  memset(hl,0,9*sizeof(int));
290  ned = 0;
291  avlen = 0.0;
292  lmax = 0.0;
293  lmin = 1.e30;
294  amin = amax = bmin = bmax = 0;
295  nullEdge = 0;
296 
297  /* Hash all edges in the mesh */
298  if ( !MMG5_hashNew(mesh,&hash,mesh->np,7*mesh->np) ) return 0;
299 
300  for(k=1; k<=mesh->nt; k++) {
301  pt = &mesh->tria[k];
302  if ( !MG_EOK(pt) ) continue;
303 
304  for(ia=0; ia<3; ia++) {
305  i0 = MMG5_iprv2[ia];
306  i1 = MMG5_inxt2[ia];
307  np = pt->v[i0];
308  nq = pt->v[i1];
309 
310  if(!MMG5_hashEdge(mesh,&hash,np,nq,0)){
311  fprintf(stderr," ## Error: %s: function MMG5_hashEdge return 0\n",
312  __func__);
313  return 0;
314  }
315  }
316  }
317 
318  /* Pop edges from hash table, and analyze their length */
319  for(k=1; k<=mesh->nt; k++) {
320  pt = &mesh->tria[k];
321  if ( !MG_EOK(pt) ) continue;
322 
323  for(ia=0; ia<3; ia++) {
324  i0 = MMG5_iprv2[ia];
325  i1 = MMG5_inxt2[ia];
326  np = pt->v[i0];
327  nq = pt->v[i1];
328 
329  /* Remove edge from hash */
330  MMG5_hashGet(&hash,np,nq);
331 
332  if ( (!metRidTyp) && met->m && met->size>1 ) {
333  len = MMG5_lenSurfEdg33_ani(mesh,met,np,nq,(pt->tag[ia] & MG_GEO));
334  }
335  else
336  len = MMG5_lenSurfEdg(mesh,met,np,nq,(pt->tag[ia] & MG_GEO));
337 
338  if ( !len ) {
339  ++nullEdge;
340  }
341  else {
342  ned ++;
343  avlen += len;
344 
345  if( len < lmin ) {
346  lmin = len;
347  amin = np;
348  bmin = nq;
349  }
350 
351  if ( len > lmax ) {
352  lmax = len;
353  amax = np;
354  bmax = nq;
355  }
356 
357  /* Locate size of edge among given table */
358  for(i=0; i<8; i++) {
359  if ( bd[i] <= len && len < bd[i+1] ) {
360  hl[i]++;
361  break;
362  }
363  }
364  if( i == 8 ) hl[8]++;
365  }
366  }
367  }
368 
369  /* Display histogram */
370  MMG5_displayLengthHisto(mesh, ned, &avlen, amin, bmin, lmin,
371  amax, bmax, lmax, nullEdge, &bd[0], &hl[0],0);
372 
373  MMG5_DEL_MEM(mesh,hash.item);
374  return 1;
375 }

510 int8_t typelt(MMG5_pPoint p[3],int8_t *ia) {
511  double h1,h2,h3,hmi,hma,ux,uy,uz,vx,vy,vz,wx,wy,wz,dd;
512 
513  ux = p[1]->c[0] - p[0]->c[0];
514  uy = p[1]->c[1] - p[0]->c[1];
515  uz = p[1]->c[2] - p[0]->c[2];
516  h1 = ux*ux + uy*uy + uz*uz;
517 
518  vx = p[2]->c[0] - p[0]->c[0];
519  vy = p[2]->c[1] - p[0]->c[1];
520  vz = p[2]->c[2] - p[0]->c[2];
521  h2 = vx*vx + vy*vy + vz*vz;
522  hmi = h1;
523  hma = h2;
524  *ia = 2;
525  if ( h1 > h2 ) {
526  hmi = h2;
527  hma = h1;
528  *ia = 1;
529  }
530  wx = p[2]->c[0] - p[1]->c[0];
531  wy = p[2]->c[1] - p[1]->c[1];
532  wz = p[2]->c[2] - p[1]->c[2];
533  h3 = wx*wx + wy*wy + wz*wz;
534  if ( h3 < hmi ) {
535  hmi = h3;
536  *ia = 0;
537  }
538  else if ( h3 > hma )
539  hma = h3;
540 
541  /* needle */
542  if ( hmi < 0.01 * hma ) return 1;
543 
544  /* check obtuse angle */
545  dd = (ux*vx + uy*vy + uz*vz) / sqrt(h1*h2);
546  if ( dd < COS145 ) {
547  *ia = 0;
548  return 2;
549  }
550  dd = (vx*wx + vy*wy + vz*wz) / sqrt(h2*h3);
551  if ( dd < COS145 ) {
552  *ia = 2;
553  return 2;
554  }
555  dd = -(ux*wx + uy*wy + uz*wz) / sqrt(h1*h3);
556  if ( dd < COS145 ) {
557  *ia = 1;
558  return 2;
559  }
560 
561  return 0;
562 }
