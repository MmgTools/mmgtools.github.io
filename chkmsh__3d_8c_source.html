91 int MMG3D_chk_shellEdgeTag_oneDir(MMG5_pMesh mesh,int start, int na, int nb,
92  int16_t tag,int ref, int piv,int adj) {
93  MMG5_pTetra pt;
94  MMG5_pxTetra pxt;
95  int *adja;
96  int8_t i;
97 
98  while ( adj && (adj != start) ) {
99  pt = &mesh->tetra[adj];
100 
101  /* identification of edge number in tetra adj */
102  if ( !MMG3D_findEdge(mesh,pt,adj,na,nb,1,NULL,&i) ) return -1;
103 
104  /* update edge ref and tag */
105  if ( pt->xt ) {
106  pxt = &mesh->xtetra[pt->xt];
107  /* Test only BDY edges */
108  if ( pxt->tag[i] & MG_BDY ) {
109  assert (pxt->tag[i] == tag && "non consistent tags");
110  assert (pxt->edg[i] == ref && "non consistent refs");
111  }
112  }
113 
114  /* set new triangle for travel */
115  adja = &mesh->adja[4*(adj-1)+1];
116  if ( pt->v[ MMG5_ifar[i][0] ] == piv ) {
117  adj = adja[ MMG5_ifar[i][0] ] / 4;
118  piv = pt->v[ MMG5_ifar[i][1] ];
119  }
120  else {
121  adj = adja[ MMG5_ifar[i][1] ] /4;
122  piv = pt->v[ MMG5_ifar[i][0] ];
123  }
124  }
125 
126  return adj;
127 }

141 int MMG3D_chk_shellEdgeTag(MMG5_pMesh mesh,int start, int8_t ia,int16_t tag,int ref) {
142  MMG5_pTetra pt;
143  MMG5_pxTetra pxt;
144  int piv,na,nb,adj,*adja;
145 
146  pt = &mesh->tetra[start];
147 
148  assert( start >= 1 && MG_EOK(pt) && "invalid tetra" );
149  assert ( tag & MG_BDY && "Unexpected non boundary tag");
150 
151  pxt = NULL;
152  na = pt->v[MMG5_iare[ia][0]];
153  nb = pt->v[MMG5_iare[ia][1]];
154 
155  if ( pt->xt ) {
156  pxt = &mesh->xtetra[pt->xt];
157  /* Test only BDY edges */
158  if ( pxt->tag[ia] & MG_BDY ) {
159  assert (pxt->tag[ia] == tag && "non consistent tags"); ;
160  assert (pxt->edg[ia] == ref && "non consistent refs"); ;
161  }
162  }
163 
164  /* Travel in one direction */
165  adja = &mesh->adja[4*(start-1)+1];
166  adj = adja[MMG5_ifar[ia][0]] / 4;
167  piv = pt->v[MMG5_ifar[ia][1]];
168 
169  adj = MMG3D_chk_shellEdgeTag_oneDir(mesh,start,na,nb,tag,ref,piv,adj);
170 
171  /* If all shell has been travelled, stop, else, travel it the other sense */
172  if ( adj > 0 ) {
173  assert ( adj == start );
174  return 1;
175  }
176  else if ( adj < 0 ) return 0;
177 
178  assert(!adj);
179 
180  pt = &mesh->tetra[start];
181  adja = &mesh->adja[4*(start-1)+1];
182  adj = adja[MMG5_ifar[ia][1]] / 4;
183  piv = pt->v[MMG5_ifar[ia][0]];
184 
185  adj = MMG3D_chk_shellEdgeTag_oneDir(mesh,start,na,nb,tag,ref,piv,adj);
186 
187  if ( adj < 0 ) return 0;
188 
189  return 1;

200 void MMG3D_chkmeshedgestags(MMG5_pMesh mesh) {
201  MMG5_pTetra pt;
202  MMG5_pxTetra pxt;
203  MMG5_Hash hash;
204  int k,nt,i,ip1,ip2,tag;
205 
206  /* Rough eval of the number of boundary triangles */
207  nt = 0;
208  for (k=1; k<=mesh->ne; k++) {
209  pt = &mesh->tetra[k];
210  if ( !MG_EOK(pt) ) continue;
211  if ( !pt->xt ) continue;
212 
213  pxt = &mesh->xtetra[pt->xt];
214  for (i=0; i<4; i++) {
215  if ( pxt->ftag[i] & MG_BDY ) {
216  ++nt;
217  }
218  }
219  }
220  nt = nt/2 + 1;
221 
222  /* Travel mesh edges and hash boundary ones */
223  MMG5_hashNew(mesh,&hash,nt,3*nt);
224 
225  for (k=1; k<=mesh->ne; k++) {
226  pt = &mesh->tetra[k];
227  if ( !MG_EOK(pt) ) continue;
228  if ( !pt->xt ) continue;
229 
230  pxt = &mesh->xtetra[pt->xt];
231  for (i=0; i<6; i++) {
232  if ( pxt->tag[i] & MG_BDY ) {
233  ip1 = pt->v[MMG5_iare[i][0]];
234  ip2 = pt->v[MMG5_iare[i][1]];
235  tag = MMG5_hashEdgeTag ( mesh,&hash,ip1,ip2,pxt->tag[i]);
236  if ( tag != pxt->tag[i] ) {
237  fprintf(stderr,"Error: %s: %d: Non consistency at tet %d (%d), edge %d:%d--%d\n ",
238  __func__,__LINE__,k,MMG3D_indElt(mesh,k),i,ip1,ip2);
239  assert( tag == pxt->tag[i] && "edge tag error" );
240  }
241  }
242  }
243  }
244  MMG5_DEL_MEM(mesh,hash.item);
245 }
