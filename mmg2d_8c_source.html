80 static inline
81 int MMG2D_countLocalParamAtEdg( MMG5_pMesh mesh,MMG5_iNode **bdyRefs) {
82  int npar,k,ier;
83 
85  (*bdyRefs) = NULL;
86 
87  k = mesh->na? mesh->edge[1].ref : 0;
88 
89  /* Try to alloc the first node */
90  ier = MMG5_Add_inode( mesh, bdyRefs, k );
91  if ( ier < 0 ) {
92  fprintf(stderr,"\n ## Error: %s: unable to allocate the first boundary"
93  " reference node.\n",__func__);
94  return 0;
95  }
96  else {
97  assert(ier);
98  npar = 1;
99  }
100 
101  for ( k=1; k<=mesh->na; ++k ) {
102  ier = MMG5_Add_inode( mesh, bdyRefs, mesh->edge[k].ref );
103 
104  if ( ier < 0 ) {
105  fprintf(stderr,"\n ## Warning: %s: unable to list the edge references.\n"
106  " Uncomplete parameters file.\n",__func__);
107  break;
108  }
109  else if ( ier ) ++npar;
110  }
111 
112  return npar;
113 }

149 static inline
150 int MMG2D_writeLocalParam( MMG5_pMesh mesh ) {
151  MMG5_iNode *edgRefs,*triRefs;
152  int nparEdg,nparTri;
153  char *ptr,data[MMG5_FILESTR_LGTH];
154  FILE *out;
155 
157  strcpy(data,mesh->namein);
158 
159  ptr = MMG5_Get_filenameExt(data);
160 
161  if ( ptr ) *ptr = '\0';
162  strcat(data,".mmg2d");
163 
164  if ( !(out = fopen(data,"wb")) ) {
165  fprintf(stderr,"\n ** UNABLE TO OPEN %s.\n",data);
166  return 0;
167  }
168 
169  fprintf(stdout,"\n %%%% %s OPENED\n",data);
170 
171  nparEdg = MMG2D_countLocalParamAtEdg( mesh, &edgRefs);
172  if ( !nparEdg ) {
173  fclose(out);
174  return 0;
175  }
176 
177  nparTri = MMG5_countLocalParamAtTri( mesh, &triRefs);
178  if ( !nparTri ) {
179  fclose(out);
180  return 0;
181  }
182 
183  fprintf(out,"parameters\n %d\n",nparTri+nparEdg);
184 
186  if (! MMG2D_writeLocalParamAtEdg(mesh,edgRefs,out) ) {
187  fclose(out);
188  return 0;
189  }
190 
192  if (! MMG5_writeLocalParamAtTri(mesh,triRefs,out) ) {
193  fclose(out);
194  return 0;
195  }
196 
197  fclose(out);
198  fprintf(stdout," -- WRITING COMPLETED\n");
199 
200  return 1;
201 }

217 static inline
218 int MMG2D_defaultOption(MMG5_pMesh mesh,MMG5_pSol met,MMG5_pSol sol) {
219  mytime ctim[TIMEMAX];
220  double hsiz;
221  char stim[32];
222 
223  signal(SIGABRT,MMG2D_excfun);
224  signal(SIGFPE,MMG2D_excfun);
225  signal(SIGILL,MMG2D_excfun);
226  signal(SIGSEGV,MMG2D_excfun);
227  signal(SIGTERM,MMG2D_excfun);
228  signal(SIGINT,MMG2D_excfun);
229 
230  tminit(ctim,TIMEMAX);
231  chrono(ON,&(ctim[0]));
232 
233  if ( mesh->info.npar ) {
234  fprintf(stderr,"\n ## Error: %s: "
235  "unable to save of a local parameter file with"
236  " the default parameters values because local parameters"
237  " are provided.\n",__func__);
238  _LIBMMG5_RETURN(mesh,met,sol,MMG5_LOWFAILURE);
239  }
240 
241 
242  if ( mesh->info.imprim > 0 ) fprintf(stdout,"\n -- INPUT DATA\n");
243  /* load data */
244  chrono(ON,&(ctim[1]));
245 
246  if ( met && met->np && (met->np != mesh->np) ) {
247  fprintf(stderr,"\n ## WARNING: WRONG SOLUTION NUMBER. IGNORED\n");
248  MMG5_DEL_MEM(mesh,met->m);
249  met->np = 0;
250  }
251 
252  if ( sol && sol->np && (sol->np != mesh->np) ) {
253  fprintf(stderr,"\n ## WARNING: WRONG SOLUTION NUMBER. IGNORED\n");
254  MMG5_DEL_MEM(mesh,sol->m);
255  sol->np = 0;
256  }
257 
258  chrono(OFF,&(ctim[1]));
259  printim(ctim[1].gdif,stim);
260  if ( mesh->info.imprim > 0 )
261  fprintf(stdout," -- INPUT DATA COMPLETED. %s\n",stim);
262 
263  /* analysis */
264  chrono(ON,&(ctim[2]));
265  MMG2D_setfunc(mesh,met);
266  MMG2D_Set_commonFunc();
267 
268  if ( mesh->info.imprim > 0 ) {
269  fprintf(stdout,"\n %s\n MODULE MMG2D: IMB-LJLL : %s (%s)\n %s\n",
270  MG_STR,MMG_VERSION_RELEASE,MMG_RELEASE_DATE,MG_STR);
271  fprintf(stdout,"\n -- DEFAULT PARAMETERS COMPUTATION\n");
272  }
273 
274  /* scaling mesh and hmin/hmax computation*/
275  if ( !MMG2D_scaleMesh(mesh,met,sol) ) _LIBMMG5_RETURN(mesh,met,sol,MMG5_STRONGFAILURE);
276 
277  /* specific meshing + update hmin/hmax */
278  if ( mesh->info.optim ) {
279  if ( !MMG2D_doSol(mesh,met) ) {
280  if ( !MMG5_unscaleMesh(mesh,met,sol) )
281  _LIBMMG5_RETURN(mesh,met,sol,MMG5_STRONGFAILURE);
282  _LIBMMG5_RETURN(mesh,met,sol,MMG5_LOWFAILURE);
283  }
284  MMG2D_solTruncatureForOptim(mesh,met);
285  }
286  if ( mesh->info.hsiz > 0. ) {
287  if ( !MMG5_Compute_constantSize(mesh,met,&hsiz) ) {
288  if ( !MMG5_unscaleMesh(mesh,met,sol) ) _LIBMMG5_RETURN(mesh,met,sol,MMG5_STRONGFAILURE);
289  _LIBMMG5_RETURN(mesh,met,sol,MMG5_STRONGFAILURE);
290  }
291  }
292 
293  /* unscaling mesh */
294  if ( !MMG5_unscaleMesh(mesh,met,sol) ) _LIBMMG5_RETURN(mesh,met,sol,MMG5_STRONGFAILURE);
295 
296  /* Save the local parameters file */
297  mesh->mark = 0;
298  if ( !MMG2D_writeLocalParam(mesh) ) {
299  fprintf(stderr,"\n ## Error: %s: Unable to save the local parameters file.\n"
300  " Exit program.\n",__func__);
301  _LIBMMG5_RETURN(mesh,met,sol,MMG5_LOWFAILURE);
302  }
303 
304  _LIBMMG5_RETURN(mesh,met,sol,MMG5_SUCCESS);
305 }

315  for ( i=1; i< argc; ++i ) {
316  if ( !strcmp(argv[i],"-val") ) {
317  MMG2D_defaultValues(mesh);
318  return 0;
319  }
320  }
321 
322  /* Second step: read all other arguments. */
323  i = 1;
324  while ( i < argc ) {
325  if ( *argv[i] == '-' ) {
326  switch(argv[i][1]) {
327  case '?':
328  MMG2D_usage(argv[0]);
329  return 0;
330  case 'a':
331  if ( !strcmp(argv[i],"-ar") && ++i < argc ) {
332  if ( !MMG2D_Set_dparameter(mesh,met,MMG2D_DPARAM_angleDetection,
333  atof(argv[i])) )
334  return 0;
335  }
336  break;

338  if ( !MMG2D_Set_solSize(mesh,met,MMG5_Vertex,0,MMG5_Tensor) )
339  return 0;
340  break;
341  case 'd':
342  if ( !strcmp(argv[i],"-default") ) {
343  mesh->mark=1;

346  return 0;
347  }
348  break;
349  case 'h':
