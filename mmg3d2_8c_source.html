57  double vfrac,lam,area,o1[3],o2[3],o3[3];
58  int8_t i1,i2,i3;

60  i1 = MMG5_idir[i0][0];
61  i2 = MMG5_idir[i0][1];
62  i3 = MMG5_idir[i0][2];
63 
64  lam = v[i0] / (v[i0]-v[i1]);
65  o1[0] = ppt[i0]->c[0] + lam*(ppt[i1]->c[0]-ppt[i0]->c[0]);
66  o1[1] = ppt[i0]->c[1] + lam*(ppt[i1]->c[1]-ppt[i0]->c[1]);
67  o1[2] = ppt[i0]->c[2] + lam*(ppt[i1]->c[2]-ppt[i0]->c[2]);
68

70  o2[0] = ppt[i0]->c[0] + lam*(ppt[i2]->c[0]-ppt[i0]->c[0]);
71  o2[1] = ppt[i0]->c[1] + lam*(ppt[i2]->c[1]-ppt[i0]->c[1]);
72  o2[2] = ppt[i0]->c[2] + lam*(ppt[i2]->c[2]-ppt[i0]->c[2]);
73 
74  lam = v[i0] / (v[i0]-v[i3]);
75  o3[0] = ppt[i0]->c[0] + lam*(ppt[i3]->c[0]-ppt[i0]->c[0]);
76  o3[1] = ppt[i0]->c[1] + lam*(ppt[i3]->c[1]-ppt[i0]->c[1]);
77  o3[2] = ppt[i0]->c[2] + lam*(ppt[i3]->c[2]-ppt[i0]->c[2]);

105 double MMG3D_vfrac(MMG5_pMesh mesh,MMG5_pSol sol,int k,int pm) {
106  MMG5_pTetra pt;
107  MMG5_pPoint ppt[4];
108  double v[4],vfm,vfp,lam,eps,o[18];
109  int ip[4],nplus,nminus,nzero;
110  int8_t flag,cfg,ia;
111  int8_t i,i0,i1,imin1,imin2,iplus1,iplus2,iz;
112  uint8_t tau[4];
113  const uint8_t *taued;
114 
115  eps = MMG5_EPS*MMG5_EPS;
116  pt = &mesh->tetra[k];
117 
118  ip[0] = pt->v[0];
119  ip[1] = pt->v[1];
120  ip[2] = pt->v[2];
121  ip[3] = pt->v[3];
122 
123  ppt[0] = &mesh->point[ip[0]];
124  ppt[1] = &mesh->point[ip[1]];
125  ppt[2] = &mesh->point[ip[2]];
126  ppt[3] = &mesh->point[ip[3]];
127 
128  v[0] = sol->m[ip[0]] - mesh->info.ls;
129  v[1] = sol->m[ip[1]] - mesh->info.ls;
130  v[2] = sol->m[ip[2]] - mesh->info.ls;
131  v[3] = sol->m[ip[3]] - mesh->info.ls;
132 
133  /* Identify number of zero, positive and negative vertices, and corresponding
134  * indices */
135  nplus = nminus = nzero = 0;
136  imin1 = imin2 = iplus1 = iplus2 = iz = -1;
137 
138  for (i=0; i<4; i++) {
139  if ( fabs(v[i]) < eps ) {
140  nzero++;
141  if ( iz < 0 ) iz = i;
142  }
143  else if ( v[i] >= eps ) {
144  nplus++;
145  if ( iplus1 < 0 ) iplus1 = i;
146  else if ( iplus2 < 0 ) iplus2 = i;
147  }
148  else {
149  nminus++;
150  if ( imin1 < 0 ) imin1 = i;
151  else if ( imin2 < 0 ) imin2 = i;
152  }
153  }
154 
155  /* Degenerate case */
156  if ( nzero == 4 ) return 0.0;
157 
158  /* Whole tetra is positive */
159  if ( nminus == 0 ) {
160  vfp = MMG5_det4pt(ppt[0]->c,ppt[1]->c,ppt[2]->c,ppt[3]->c);
161  vfp = fabs(vfp);
162  if ( pm == 1 ) return vfp;
163  else return 0.0;
164  }
165 
166  /* Whole tetra is negative */
167  if ( nplus == 0 ) {
168  vfm = MMG5_det4pt(ppt[0]->c,ppt[1]->c,ppt[2]->c,ppt[3]->c);
169  vfm = fabs(vfm);
170  if ( pm == -1 ) return vfm;
171  else return 0.0;
172  }
173 
174  /* Exactly one vertex is negative */
175  if ( nminus == 1 ) {
176  return MMG3D_vfrac_1vertex(ppt,imin1,v,pm!=-1);
177  }
178 
179  /* Exactly one vertex is positive */
180  if ( nplus == 1 ) {
181  return MMG3D_vfrac_1vertex(ppt,iplus1,v,pm!=1);
182  }
183 
184  flag = 0;
185  for ( ia=0; ia<18; ++ia ) {
186  o[ia] = 0.0;
187  }
188 
189  /* We have exactly 2 negative vertices and 2 positive ones */
190  assert ( nplus==2 && nminus==2 );
191 
192  /* Config detection */
193  for ( ia=0; ia<6; ++ia ) {
194  i0 = MMG5_iare[ia][0];
195  i1 = MMG5_iare[ia][1];
196 
197  if ( fabs(v[i0]) < MMG5_EPSD2 || fabs(v[i1]) < MMG5_EPSD2 ) continue;
198  else if ( MG_SMSGN(v[i0],v[i1]) ) continue;
199 
200  MG_SET(flag,ia);
201 
202  /* Computation of the intersection between edges and isovalue */
203  lam = v[i0] / (v[i0]-v[i1]);
204  o[3*ia ] = ppt[i0]->c[0] + lam*(ppt[i1]->c[0]-ppt[i0]->c[0]);
205  o[3*ia+1] = ppt[i0]->c[1] + lam*(ppt[i1]->c[1]-ppt[i0]->c[1]);
206  o[3*ia+2] = ppt[i0]->c[2] + lam*(ppt[i1]->c[2]-ppt[i0]->c[2]);
207  }
208 
209  assert ( flag==30 || flag==45 || flag==51 );
210 
211  /* Set permutation of vertices : reference configuration 30 */
212  tau[0] = 0 ; tau[1] = 1 ; tau[2] = 2 ; tau[3] = 3;
213  taued = &MMG5_permedge[0][0];
214 
215  switch(flag){
216  case 45:
217  tau[0] = 1 ; tau[1] = 3 ; tau[2] = 2 ; tau[3] = 0;
218  taued = &MMG5_permedge[5][0];
219  break;
220 
221  case 51:
222  tau[0] = 1 ; tau[1] = 2 ; tau[2] = 0 ; tau[3] = 3;
223  taued = &MMG5_permedge[4][0];
224  break;
225  }
226 
227  if ( pm < 0 ) {
228  if ( v[tau[0]] < 0.0 ) {
229  /* compute the area that contains tau[0] and tau[1] */
230  cfg = 0;
231  }
232  else {
233  /* compute the area that contains tau[2] and tau[3] */
234  cfg = 2;
235  }
236  }
237  else {
238  assert ( pm > 0 );
239  if ( v[tau[0]] < 0.0 ) {
240  /* compute the area that contains tau[0] and tau[1] */
241  cfg = 2;
242  }
243  else {
244  cfg = 0;
245  }
246  }
247  assert ( cfg == 0 || cfg == 2 );
248 
249  /* Computation of the area depending on the detected config */
250  if ( cfg == 0 ) {
251  vfp = fabs ( MMG5_det4pt(ppt[tau[0]]->c,ppt[tau[1]]->c,&o[3*taued[3]],&o[3*taued[4]]) );
252  vfp += fabs ( MMG5_det4pt(ppt[tau[0]]->c,&o[3*taued[4]],&o[3*taued[3]],&o[3*taued[2]]) );
253  vfp += fabs ( MMG5_det4pt(ppt[tau[0]]->c,&o[3*taued[3]],&o[3*taued[1]],&o[3*taued[2]]) );
254 #ifdef NDEBUG
255  return vfp;
256 #endif
257  }
258  else if ( cfg == 2 ) {
259  vfm = fabs ( MMG5_det4pt(&o[3*taued[2]],&o[3*taued[4]],ppt[tau[2]]->c,ppt[tau[3]]->c) );
260  vfm += fabs ( MMG5_det4pt(&o[3*taued[2]],&o[3*taued[3]],ppt[tau[2]]->c,&o[3*taued[4]]) );
261  vfm += fabs ( MMG5_det4pt(&o[3*taued[1]],&o[3*taued[3]],ppt[tau[2]]->c,&o[3*taued[2]]) );
262 #ifdef NDEBUG
263  return vfm;
264 #endif
265  }
266 
267 #ifndef NDEBUG
268 
269  /* Compute the complementary area */
270  if ( cfg == 0 ) {
271  vfm = fabs ( MMG5_det4pt(&o[3*taued[2]],&o[3*taued[4]],ppt[tau[2]]->c,ppt[tau[3]]->c) );
272  vfm += fabs ( MMG5_det4pt(&o[3*taued[2]],&o[3*taued[3]],ppt[tau[2]]->c,&o[3*taued[4]]) );
273  vfm += fabs ( MMG5_det4pt(&o[3*taued[1]],&o[3*taued[3]],ppt[tau[2]]->c,&o[3*taued[2]]) );
274  }
275  else if ( cfg == 2 ) {
276  vfp = fabs ( MMG5_det4pt(ppt[tau[0]]->c,ppt[tau[1]]->c,&o[3*taued[3]],&o[3*taued[4]]) );
277  vfp += fabs ( MMG5_det4pt(ppt[tau[0]]->c,&o[3*taued[4]],&o[3*taued[3]],&o[3*taued[2]]) );
278  vfp += fabs ( MMG5_det4pt(ppt[tau[0]]->c,&o[3*taued[3]],&o[3*taued[1]],&o[3*taued[2]]) );
279  }
280 
281  /* vfp and vfm have been computed: check that the sum of both area is the
282  * area of the whole tetra */
283  double vf;
284  vf = fabs ( MMG5_det4pt(ppt[0]->c,ppt[1]->c,ppt[2]->c,ppt[3]->c) );
285 
286  assert ( fabs(vf-(vfp+vfm)) < MMG5_EPSOK );
287 
288  if ( cfg==0 ) {
289  return vfp;
290  }
291  else {
292  return vfm;
293  }
294 #endif
295 
296  /* Should not pass here */
297  return 0.0;
298 }

358 static inline int
359 MMG5_invsl(double A[3][3],double b[3],double r[3]) {
360  double detA;
361 
362  detA = A[0][0]*(A[1][1]*A[2][2] - A[2][1]*A[1][2]) \
363  - A[0][1]*(A[1][0]*A[2][2] - A[2][0]*A[1][2]) \
364  + A[0][2]*(A[1][0]*A[2][1] - A[2][0]*A[1][1]);
365  if ( detA < MMG5_EPSD ) return 0;
366  detA = 1.0 / detA;
367 
368  r[0] = b[0]*(A[1][1]*A[2][2] - A[2][1]*A[1][2]) \
369  - A[0][1]*(b[1]*A[2][2] - b[2]*A[1][2]) \
370  + A[0][2]*(b[1]*A[2][1] - b[2]*A[1][1]);
371 
372  r[1] = A[0][0]*(b[1]*A[2][2] - b[2]*A[1][2]) \
373  - b[0]*(A[1][0]*A[2][2] - A[2][0]*A[1][2]) \
374  + A[0][2]*(A[1][0]*b[2] - A[2][0]*b[1]);
375 
376  r[2] = A[0][0]*(A[1][1]*b[2] - A[2][1]*b[1]) \
377  - A[0][1]*(A[1][0]*b[2] - A[2][0]*b[1]) \
378  + b[0]*(A[1][0]*A[2][1] - A[2][0]*A[1][1]);
379 
380  r[0] *= detA;
381  r[1] *= detA;
382  r[2] *= detA;
383 
384  return 1;
385 }

403 MMG5_ismaniball(MMG5_pMesh mesh,MMG5_pSol sol,int k,int indp) {

408  static int8_t mmgWarn0 = 0;
409 
410  pt = &mesh->tetra[k];
411  np = pt->v[indp];
412  if ( fabs(sol->m[np]-mesh->info.ls) > MMG5_EPSD2 ) return 1;
413 
414  memset(bdy,0,(MMG3D_LMAX+1)*sizeof(int));
415  memset(list,0,(MMG3D_LMAX+1)*sizeof(int));
416 
417  /* Sign of a starting point in ball of np */
418  for (j=0; j<3; j++) {
419  ip = MMG5_idir[indp][j];
420  if ( sol->m[pt->v[ip]]-mesh->info.ls != 0.0 ) break;
421  }
422  if ( j == 3 ) {
423  if ( !mmgWarn0 ) {
424  mmgWarn0 = 1;
425  fprintf(stderr,"\n ## Warning: %s: at least 1 tetra with 4 null"
426  " values.\n",__func__);
427  }
428  return 0;
429  }
430 
431  v = sol->m[pt->v[ip]]-mesh->info.ls;
432  base = ++mesh->base;
433  pt->flag = base;
434  ilist = 0;
435  list[ilist] = 4*k+indp;
436  ilist++;
437 
438  /* travel list and pile up, by adjacency, faces of ball of np while they have at least
439  a vertex with same sign as v */
440  res = cur = 0;
441  while ( cur < ilist ) {
442  iel = list[cur] / 4;
443  i = list[cur] % 4;
444  pt = &mesh->tetra[iel];
445  adja = &mesh->adja[4*(iel-1)+1];
446 
447  /* Store a face for starting back enumeration with the opposite sign */
448  if ( !res ) {
449  for (j=0; j<3; j++) {
450  i1 = MMG5_idir[i][j];
451  v1 = sol->m[pt->v[i1]]-mesh->info.ls;
452  if ( ( v1 != 0.0 ) && !MG_SMSGN(v,v1) ) {
453  res = 4*iel + i;
454  break;
455  }
456  }
457  }
458 
459  /* Pile up faces sharing a vertex with same sign as v */
460  for (j=0; j<3; j++) {
461  i1 = MMG5_idir[i][MMG5_inxt2[j]];
462  i2 = MMG5_idir[i][MMG5_iprv2[j]];

470

473 
474  if ( pt1->flag == base ) continue;
475  for (ip=0; ip<4; ip++) {
476  if ( pt1->v[ip] == np ) break;
477  }

481  ilist++;

486  }

516  else
517  nopp++;

521  else if ( MG_SMSGN(v,v2) )

523  else

542  for (j=0; j<4; j++)
543  if ( pt1->v[j] == np ) break;
544  res = 4*jel+j;
545  }
546  }
547

549  bdy[ibdy] = list[l];
550  ibdy++;
551  }
552  }

555  if ( !res )
556  return 0;
557

560  pt = &mesh->tetra[iel];

562  pt->flag = base;
563 
564  memset(list,0,(MMG3D_LMAX+1)*sizeof(int));

567  ilist++;

570  i = list[cur] % 4;
571  pt = &mesh->tetra[iel];
572  adja = &mesh->adja[4*(iel-1)+1];
573 
574  /* Pile up faces sharing a vertex with opposite sign to v */
575  for (j=0; j<3; j++) {
576  i1 = MMG5_idir[i][MMG5_inxt2[j]];
577  i2 = MMG5_idir[i][MMG5_iprv2[j]];
578  v1 = sol->m[pt->v[i1]]-mesh->info.ls;
579  v2 = sol->m[pt->v[i2]]-mesh->info.ls;
580 
581  if ( v1 == 0.0 && v2 == 0.0 ) {
582  jel = adja[MMG5_idir[i][j]];
583  if( !jel ) continue;
584  jel /=4 ;
585  pt1 = &mesh->tetra[jel];
586  pt1->flag = base;
587  }
588 
589  else if ( ( ( v1 != 0.0 ) && (!MG_SMSGN(v,v1)) ) || ( ( v2 != 0.0 ) && (!MG_SMSGN(v,v2)) ) ) {
590  jel = adja[MMG5_idir[i][j]];
591  if( !jel ) continue;
592  jel /= 4;
593  pt1 = &mesh->tetra[jel];
594 
595  j0 = MMG5_idir[i][0];
596  j1 = MMG5_idir[i][1];
597  j2 = MMG5_idir[i][2];
598

600  v1 = sol->m[pt1->v[j1]]-mesh->info.ls;
601  v2 = sol->m[pt1->v[j2]]-mesh->info.ls;
602 
603  nzeros = nsame = nopp = 0;
604 
605  if ( v0 == 0.0 )
606  nzeros++;
607  else if ( MG_SMSGN(v,v0) )
608  nsame++;
609  else
610  nopp++;
611 
612  if ( v1 == 0.0 )
613  nzeros++;
