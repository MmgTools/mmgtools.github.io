51 void MMG5_swapTet(MMG5_pTetra tetras/*, int* adja*/, int* perm, int ind1, int ind2) {
52  MMG5_Tetra pttmp;
53  int tmp;
54 
55  /* Commentated part: swap for adja table if we don't free it in renumbering *
56  * function (faster but need of 4*mesh->nemax*sizeof(int) extra bytes ) */
57  /* int adjatmp,kadj,ifadj,j; */
58 
59  /* /\* 1-- swap the adja table *\/ */
60  /* /\* First: replace ind2 by ind1 in adjacents tetras of ind2 *\/ */
61  /* for ( j=1; j<5; j++ ) { */
62  /* if ( adja[4*(ind2-1)+j]/4 ) { */
63  /* kadj = adja[4*(ind2-1)+j]/4; */
64  /* ifadj = adja[4*(ind2-1)+j]%4; */
65  /* adja[4*(kadj-1)+1+ifadj] = 4*ind1 + adja[4*(kadj-1)+1+ifadj]%4; */
66  /* } */
67  /* } */
68 
69  /* /\* Second: replace ind1 by ind2 in adjacents tetras of ind1*\/ */
70  /* for ( j=1; j<5; j++ ) { */
71  /* if ( adja[4*(ind1-1)+j]/4 ) { */
72  /* kadj = adja[4*(ind1-1)+j]/4; */
73  /* ifadj = adja[4*(ind1-1)+j]%4; */
74  /* if ( kadj == ind1 ) */
75  /* adja[4*(ind2-1)+1+ifadj] = 4*ind2 + adja[4*(ind2-1)+1+ifadj]%4; */
76  /* else */
77  /* adja[4*(kadj-1)+1+ifadj] = 4*ind2 + adja[4*(kadj-1)+1+ifadj]%4; */
78  /* } */
79  /* } */
80 
81  /* /\* Third: swap adjacents for ind1 and ind2 *\/ */
82  /* for ( j=1; j<5; j++ ) { */
83  /* adjatmp = adja[4*(ind2-1)+j]; */
84  /* adja[4*(ind2-1)+j] = adja[4*(ind1-1)+j]; */
85  /* adja[4*(ind1-1)+j] = adjatmp; */
86  /* } */
87 
88  /* 2-- swap the tetrahedras */
89  memcpy(&pttmp ,&tetras[ind2],sizeof(MMG5_Tetra));
90  memcpy(&tetras[ind2],&tetras[ind1],sizeof(MMG5_Tetra));
91  memcpy(&tetras[ind1],&pttmp ,sizeof(MMG5_Tetra));
92 
93  /* 3-- swap the permutation table */
94  tmp = perm[ind2];
95  perm[ind2] = perm[ind1];
96  perm[ind1] = tmp;
97 }
