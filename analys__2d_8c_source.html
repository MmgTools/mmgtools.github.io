267 int MMG2D_singul(MMG5_pMesh mesh, int ref ) {
268  MMG5_pTria pt;
269  MMG5_pPoint ppt,p1,p2;
270  double ux,uy,uz,vx,vy,vz,dd;
271  int list[MMG2D_LONMAX+2],listref[MMG2D_LONMAX+2],k,ns,ng,nr,nm,nre,nc;
272  int8_t i;
273 
274  nre = nc = nm = 0;
275 
276  if ( ref == MMG5_UNSET ) {
277  /* reset the ppt->s tag */
278  for (k=1; k<=mesh->np; ++k) {
279  mesh->point[k].s = 0;
280  }
281  }
282  else {
284  /* First: mark all the points */
285  for ( k=1; k<=mesh->np; ++k ) {
286  mesh->point[k].s = 1;
287  }
288  /* Second: if the point belongs to a boundary edge of reference ref, remove
289  * the mark */
290  for ( k=1; k<=mesh->nt; ++k ) {
291  pt = &mesh->tria[k];
292 
293  for (i=0; i<3; i++) {
294  if ( !MG_EDG(pt->tag[i]) ) continue;
295  if ( pt->edg[i] != ref ) continue;
296 
297  mesh->point[pt->v[MMG5_iprv2[i]]].s = 0;
298  mesh->point[pt->v[MMG5_inxt2[i]]].s = 0;
299  }
300  }
301  }
302 
304  for (k=1; k<=mesh->nt; k++) {
305  pt = &mesh->tria[k];
306  if ( ! MG_EOK(pt) ) continue;
307 
308  for (i=0; i<3; i++) {
309  ppt = &mesh->point[pt->v[i]];
310 
311  if ( ppt->s ) continue;
312  ppt->s = 1;
313 
314  if ( !MG_VOK(ppt) || MG_CRN & ppt->tag ) continue;
315 
316  if ( !MG_EDG(ppt->tag) ) continue;
317 
318  ns = MMG5_bouler(mesh,mesh->adja,k,i,list,listref,&ng,&nr,MMG2D_LONMAX);
319 
320  if ( !ns ) continue;
321  if ( (ng+nr) > 2 ) {
322  /* Previous classification may be subject to discussion, and may depend on the user's need */
323  ppt->tag |= MG_NOM;
324  nm++;
325  /* Two ridge curves and one ref curve: non manifold situation */
326  /*if ( ng == 2 && nr == 1 ) {
327  ppt->tag |= MG_NOM;
328  nm++;
329  }
330  else {
331  ppt->tag |= MG_CRN + MG_REQ;
332  nre++;
333  nc++;
334  }*/
335  }
336  /* One ridge curve and one ref curve meeting at a point */
337  else if ( (ng == 1) && (nr == 1) ) {
338  ppt->tag |= MG_REQ;
339  nre++;
340  }
341  /* Evanescent ridge */
342  else if ( ng == 1 && !nr ){
343  ppt->tag |= MG_CRN + MG_REQ;
344  nre++;
345  nc++;
346  }
347  /* Evanescent ref curve */
348  else if ( nr == 1 && !ng ){
349  ppt->tag |= MG_CRN + MG_REQ;
350  nre++;
351  nc++;
352  }
353  /* Check ridge angle */
354  else {
355  assert ( ng == 2 || nr == 2 );
356  p1 = &mesh->point[list[1]];
357  p2 = &mesh->point[list[2]];
358  ux = p1->c[0] - ppt->c[0];
359  uy = p1->c[1] - ppt->c[1];
360  uz = p1->c[2] - ppt->c[2];
361  vx = p2->c[0] - ppt->c[0];
362  vy = p2->c[1] - ppt->c[1];
363  vz = p2->c[2] - ppt->c[2];
364  dd = (ux*ux + uy*uy + uz*uz) * (vx*vx + vy*vy + vz*vz);
365 
366  /* If both edges carry different refs, tag vertex as singular */
367  if ( listref[1] != listref[2] ) {
368  ppt->tag |= MG_REQ;
369  nc++;
370  }
371 
372  /* Check angle */
373  if ( fabs(dd) > MMG5_EPSD ) {
374  dd = (ux*vx + uy*vy + uz*vz) / sqrt(dd);
375  if ( dd > -mesh->info.dhd ) {
376  ppt->tag |= MG_CRN;
377  nc++;
378  }
379  }
380  }
381  }
382  }
383 
384  if ( abs(mesh->info.imprim) > 3 && nc+nre+nm > 0 )
385  fprintf(stdout," %d corners, %d singular points and %d non manifold points detected\n",nc,nre,nm);
386 
387  return 1;
388 }

505 int MMG2D_regnor(MMG5_pMesh mesh) {
506  MMG5_pTria pt;
507  MMG5_pPoint ppt,p1,p2;
508  double *tmp,dd,ps,lm1,lm2,nx,ny,ux,uy,nxt,nyt,res,res0,n[2];
509  int k,iel,ip1,ip2,nn,it,maxit;
510  int8_t i,ier;
511 
512  it = 0;
513  maxit = 10;
514  res0 = 0.0;
515  nn = 0;
516  lm1 = 0.4;
517  lm2 = 0.399;
518 
519  /* Temporary table for normal vectors */
520  MMG5_SAFE_CALLOC(tmp,2*mesh->np+1,double,return 0);
521 
522  /* Allocate a seed to each point */
523  for (k=1; k<=mesh->nt; k++) {
524  pt = &mesh->tria[k];
525  if ( !MG_EOK(pt) ) continue;
526 
527  for (i=0; i<3; i++) {
528  ppt = &mesh->point[pt->v[i]];
529  ppt->s = k;
530  }
531  }
532 
533  do {
534  /* Step 1: Laplacian */
535  for (k=1; k<=mesh->np; k++) {
536  ppt = &mesh->point[k];
537  if ( !MG_VOK(ppt) ) continue;
538  if ( MG_SIN(ppt->tag) || ppt->tag & MG_NOM ) continue;
539  if ( !MG_EDG(ppt->tag) ) continue;
540 
541  nx = ny = 0;
542  iel = ppt->s;
543  pt = &mesh->tria[iel];
544  i = 0;
545  if ( pt->v[1] == k ) i = 1;
546  if ( pt->v[2] == k ) i = 2;

549

552  MMG5_SAFE_FREE(tmp);
553  return 0;
554  }
555 
556  p1 = &mesh->point[ip1];
557  p2 = &mesh->point[ip2];
558 
559  /* If p1 is singular, update with the (adequately oriented) normal to the edge ppt-p1 */
560  if ( MG_SIN(p1->tag) || p1->tag & MG_NOM ) {
561  ux = p1->c[0] - ppt->c[0];
562  uy = p1->c[1] - ppt->c[1];
563  dd = ux*ux + uy*uy;
564  if ( dd > MMG5_EPSD ) {
565  dd = 1.0 / sqrt(dd);
566  ux *= dd;
567  uy *= dd;
568  }
569  nxt = -uy;
570  nyt = ux;
571 
572  ps = nxt*ppt->n[0] + nyt*ppt->n[1];
573  if ( ps < 0.0 ) {
574  nx += -nxt;
575  ny += -nyt;
576  }
577  else {
578  nx += nxt;
579  ny += nyt;
580  }
581  }
582  else {
583  nx += p1->n[0];
584  ny += p1->n[1];
585  }
586 
587  /* If p2 is singular, update with the (adequately oriented) normal to the edge ppt-p1 */
588  if ( MG_SIN(p2->tag) || p2->tag & MG_NOM ) {
589  ux = p2->c[0] - ppt->c[0];
590  uy = p2->c[1] - ppt->c[1];
591  dd = ux*ux + uy*uy;
592  if ( dd > MMG5_EPSD ) {
593  dd = 1.0 / sqrt(dd);
594  ux *= dd;
595  uy *= dd;
596  }
597  nxt = -uy;
598  nyt = ux;
599 
600  ps = nxt*ppt->n[0] + nyt*ppt->n[1];
601  if ( ps < 0.0 ) {
602  nx += -nxt;
603  ny += -nyt;
604  }
605  else {
606  nx += nxt;
607  ny += nyt;
608  }
609  }
610  else {
611  nx += p2->n[0];
612  ny += p2->n[1];
613  }
614 
615  /* Average normal normalization */
616  dd = nx*nx + ny*ny;
617  if ( dd > MMG5_EPSD2 ) {
618  dd = 1.0 / sqrt(dd);
619  nx *= dd;
620  ny *= dd;
621  }
622

624  tmp[2*(k-1)+1] = ppt->n[0] + lm1 * (nx - ppt->n[0]);
625  tmp[2*(k-1)+2] = ppt->n[1] + lm1 * (ny - ppt->n[1]);
626  }
627 
628  /* Antilaplacian operation */
629  res = 0.0;
630  for (k=1; k<=mesh->np; k++) {
631 
632  ppt = &mesh->point[k];
633  if ( !MG_VOK(ppt) ) continue;
634  if ( MG_SIN(ppt->tag) || ppt->tag & MG_NOM ) continue;
635  if ( !MG_EDG(ppt->tag) ) continue;
636 
637  nx = ny = 0;
638  iel = ppt->s;
639  pt = &mesh->tria[iel];
640  i = 0;
641  if ( pt->v[1] == k ) i = 1;
642  if ( pt->v[2] == k ) i = 2;
643 
644  ier = MMG2D_bouleendp(mesh,iel,i,&ip1,&ip2);
645  if ( !ier ) {
646  fprintf(stderr,"\n ## Error: %s: Abort.\n",__func__);
647  MMG5_SAFE_FREE(tmp);
648  return 0;
649  }
650 
651  p1 = &mesh->point[ip1];
652  p2 = &mesh->point[ip2];
653 
654  /* If p1 is singular, update with the (adequately oriented) normal to the edge ppt-p1 */
655  if ( MG_SIN(p1->tag) || p1->tag & MG_NOM ) {
656  ux = p1->c[0] - ppt->c[0];
657  uy = p1->c[1] - ppt->c[1];
658  dd = ux*ux + uy*uy;
659  if ( dd > MMG5_EPSD ) {

671  }

673  nx += nxt;
674  ny += nyt;
675  }
676  }

680  }
681 
682  /* If p2 is singular, update with the (adequately oriented) normal to the edge ppt-p1 */
683  if ( MG_SIN(p2->tag) || p2->tag & MG_NOM ) {
684  ux = p2->c[0] - ppt->c[0];
685  uy = p2->c[1] - ppt->c[1];
686  dd = ux*ux + uy*uy;
687  if ( dd > MMG5_EPSD ) {
688  dd = 1.0 / sqrt(dd);
689  ux *= dd;
690  uy *= dd;
691  }
692  nxt = -uy;
693  nyt = ux;
694 
695  ps = nxt*ppt->n[0] + nyt*ppt->n[1];
696  if ( ps < 0.0 ) {
697  nx += -nxt;
698  ny += -nyt;
699  }
700  else {
701  nx += nxt;
702  ny += nyt;
703  }
704  }
705  else {
706  nx += tmp[2*(ip2-1)+1];
707  ny += tmp[2*(ip2-1)+2];
708  }
709 
710  /* Average normal normalization */
711  dd = nx*nx + ny*ny;
712  if ( dd > MMG5_EPSD2 ) {
713  dd = 1.0 / sqrt(dd);
714  nx *= dd;
715  ny *= dd;
716  }
717 
718  /* Anti Laplacian operation */
719  n[0] = tmp[2*(k-1)+1] - lm2 * (nx - tmp[2*(k-1)+1]);
720  n[1] = tmp[2*(k-1)+2] - lm2 * (ny - tmp[2*(k-1)+2]);
721  res += (n[0]-ppt->n[0])*(n[0]-ppt->n[0]) + (n[1]-ppt->n[1])*(n[1]-ppt->n[1]);
722  ppt->n[0] = n[0];
723  ppt->n[1] = n[1];
724  nn++;
725  }
726 
727  /* Normalization */
728  for (k=1; k<=mesh->np; k++) {
729  ppt = &mesh->point[k];
730  if ( !MG_VOK(ppt) ) continue;
731  if ( MG_SIN(ppt->tag) || ppt->tag & MG_NOM ) continue;
732  if ( !MG_EDG(ppt->tag) ) continue;
733 
734  dd = ppt->n[0]*ppt->n[0] + ppt->n[1]*ppt->n[1];
735  if ( dd > MMG5_EPSD ) {
736  dd = 1.0 / sqrt(dd);
737  ppt->n[0] *= dd;
738  ppt->n[1] *= dd;
739  }
740  }
741 
742  if ( it == 0 ) res0 = res;
743  if ( res0 > MMG5_EPSD ) res = res / res0;
744 
745  if ( mesh->info.imprim < -1 || mesh->info.ddebug ) {
746  fprintf(stdout," iter %5d res %.3E",it,res);
747  fflush(stdout);
748  }
749  }
750  while ( ++it < maxit && res > MMG5_EPS );
751 
752 
753  if ( abs(mesh->info.imprim) > 4 )
754  fprintf(stdout," %d normals regularized: %.3e\n",nn,res);
755 
756  MMG5_SAFE_FREE(tmp);
757  return 1;
758 }
