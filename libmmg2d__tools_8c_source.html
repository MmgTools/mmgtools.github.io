84 int MMG2D_parsop(MMG5_pMesh mesh,MMG5_pSol met) {
85  int ret,ref,i,j,npar,rin,rex,split;
86  float fp1,fp2,fp3;
87  char *ptr,data[256];
88  FILE *in;
89  MMG5_pMat pm;
90  fpos_t position;
91 
92  /* Check for parameter file */
93  strcpy(data,mesh->namein);
94  ptr = strstr(data,".mesh");
95  if ( ptr ) *ptr = '\0';
96  strcat(data,".mmg2d");
97  in = fopen(data,"rb");
98 
99  if ( !in ) {
100  sprintf(data,"%s","DEFAULT.mmg2d");
101  in = fopen(data,"rb");
102  if ( !in )
103  return 1;
104  }
105  if ( mesh->info.imprim >= 0 )
106  fprintf(stdout,"\n %%%% %s OPENED\n",data);
107 
108  /* Read parameters */
109  while ( !feof(in) ) {
110  ret = fscanf(in,"%255s",data);
111  if ( !ret || feof(in) ) break;
112  for (i=0; i<strlen(data); i++) data[i] = tolower(data[i]);
113 
114  /* Read user-defined references for the LS mode */
115  if ( !strcmp(data,"lsreferences") ) {
116  ret = fscanf(in,"%d",&npar);
117  if ( !ret ) {
118  fprintf(stderr," %%%% Wrong format for lsreferences: %d\n",npar);
119  return (0);
120  }
121 
122  if ( !MMG2D_Set_iparameter(mesh,met,MMG2D_IPARAM_numberOfMat,npar) ) {
123  return 0;
124  }
125  for (i=0; i<mesh->info.nmat; i++) {
126  MMG_FSCANF(in,"%d",&ref);
127  fgetpos(in,&position);
128  MMG_FSCANF(in,"%255s",data);
129  split = MMG5_MMAT_NoSplit;
130  rin = rex = ref;
131  if ( strcmp(data,"nosplit") ) {
132  fsetpos(in,&position);
133  split = MMG5_MMAT_Split;
134  MMG_FSCANF(in,"%d",&rin);
135  MMG_FSCANF(in,"%d",&rex);
136  }
137  if ( !MMG2D_Set_multiMat(mesh,met,ref,split,rin,rex) ) {
138  return 0;
139  }
140  }
141  }
142  /* Read user-defined local parameters and store them in the structure info->par */
143  else if ( !strcmp(data,"parameters") ) {
144  ret = fscanf(in,"%d",&npar);
145 
146  if ( !ret ) {
147  fprintf(stderr," %%%% Wrong format for parameters: %d\n",npar);
148  return 0;
149  }
150  else if ( npar > MMG2D_LPARMAX ) {
151  fprintf(stderr," %%%% Too many local parameters %d. Abort\n",npar);
152  return 0;
153  }
154 
155  /* Allocate memory and fill the info->par table (adding one, corresponding to the command line data) */
156  if ( npar ) {
157  if ( !MMG2D_Set_iparameter(mesh,met,MMG2D_IPARAM_numberOfLocalParam,npar) )
158  return 0;
159 
160  for (i=0; i<mesh->info.npar; i++) {
161  ret = fscanf(in,"%d %255s",&ref,data);
162  if ( ret ) ret = fscanf(in,"%f %f %f",&fp1,&fp2,&fp3);
163 
164  if ( !ret ) {
165  fprintf(stderr," %%%% Wrong format: %s\n",data);
166  return (0);
167  }
168 
169  for (j=0; j<strlen(data); j++) data[j] = tolower(data[j]);
170  if ( !strcmp(data,"triangles") || !strcmp(data,"triangle") ) {
171  if ( !MMG2D_Set_localParameter(mesh,met,MMG5_Triangle,ref,fp1,fp2,fp3) ) {
172  return 0;
173  }
174  }
175  else if ( !strcmp(data,"edges") || !strcmp(data,"edge") ) {
176  if ( !MMG2D_Set_localParameter(mesh,met,MMG5_Edg,ref,fp1,fp2,fp3) ) {
177  return 0;
178  }
179  }
180  else {
181  fprintf(stderr," %%%% Wrong format: %s\n",data);
182  return 0;
183  }
184  }
185  }
186  }
187  else {
188  fprintf(stderr," %%%% Wrong format: %s\n",data);
189  return 0;
190  }
191  }
192 
193  fclose(in);
194  return 1;
195 }

206 int MMG2D_Get_numberOfNonBdyEdges(MMG5_pMesh mesh, int* nb_edges) {
207  MMG5_pTria pt,pt1;
208  MMG5_pEdge ped;
209  int *adja,k,i,j,i1,i2,iel;
210 
211  *nb_edges = 0;
212  if ( mesh->tria ) {
213  /* Create the triangle adjacency if needed */
214  if ( !mesh->adja ) {
215  if ( !MMG2D_hashTria( mesh ) ) {
216  fprintf(stderr,"\n ## Error: %s: unable to create "
217  "adjacency table.\n",__func__);
218  return 0;
219  }
220  }
221 
222  /* Count the number of non boundary edges */
223  for ( k=1; k<=mesh->nt; k++ ) {
224  pt = &mesh->tria[k];
225  if ( !MG_EOK(pt) ) continue;
226 
227  adja = &mesh->adja[3*(k-1)+1];
228 
229  for ( i=0; i<3; i++ ) {
230  iel = adja[i] / 3;
231  assert ( iel != k );
232 
233  pt1 = &mesh->tria[iel];
234 
235  if ( (!iel) || (pt->ref != pt1->ref) ||
236  ((pt->ref==pt1->ref) && MG_SIN(pt->tag[i])) ||
237  (mesh->info.opnbdy && pt->tag[i]) ) {
238  /* Do not treat boundary edges */
239  continue;
240  }
241  if ( k < iel ) {
242  /* Treat edge from the triangle with lowest index */
243  ++(*nb_edges);
244  }
245  }
246  }
247 
248  /* Append the non boundary edges to the boundary edges array */
249  if ( mesh->namax ) {
250  MMG5_ADD_MEM(mesh,(*nb_edges)*sizeof(MMG5_Edge),"non boundary edges",
251  printf(" Exit program.\n");
252  return 0);
253  MMG5_SAFE_RECALLOC(mesh->edge,(mesh->namax+1),(mesh->namax+(*nb_edges)+1),
254  MMG5_Edge,"non bdy edges arrray",return 0);
255  }
256  else {
257  MMG5_ADD_MEM(mesh,((*nb_edges)+1)*sizeof(MMG5_Edge),"non boundary edges",
258  printf(" Exit program.\n");
259  return 0);
260  MMG5_SAFE_RECALLOC(mesh->edge,0,((*nb_edges)+1),
261  MMG5_Edge,"non bdy edges arrray",return 0);
262  }
263 
264  j = mesh->namax+1;
265  for ( k=1; k<=mesh->nt; k++ ) {
266  pt = &mesh->tria[k];
267  if ( !MG_EOK(pt) ) continue;
268 
269  adja = &mesh->adja[3*(k-1)+1];
270 
271  for ( i=0; i<3; i++ ) {
272  i1 = MMG5_inxt2[i];
273  i2 = MMG5_iprv2[i];
274  iel = adja[i] / 3;
275  assert ( iel != k );
276 
277  pt1 = &mesh->tria[iel];
278 
279  if ( (!iel) || (pt->ref != pt1->ref) ||
280  ((pt->ref==pt1->ref) && MG_SIN(pt->tag[i])) ||
281  (mesh->info.opnbdy && pt->tag[i]) ) {
282  /* Do not treat boundary edges */
283  continue;
284  }
285  if ( k < iel ) {
286  /* Treat edge from the triangle with lowest index */
287  ped = &mesh->edge[j++];
288  assert ( ped );
289  ped->a = pt->v[i1];
290  ped->b = pt->v[i2];
291  ped->ref = pt->edg[i];
292  }
293  }
294  }
295  }
296  return 1;
297 }

369 int MMG2D_Get_adjaVerticesFast(MMG5_pMesh mesh, int ip,int start, int lispoi[MMG2D_LMAX])
370 {
371  MMG5_pTria pt;
372  int k,prevk,nbpoi,iploc,i,i1,i2,*adja;
373 
374  pt = &mesh->tria[start];
375 
376  for ( iploc=0; iploc<3; ++iploc ) {
377  if ( pt->v[iploc] == ip ) break;
378  }
379 
380  assert(iploc!=3);
381 
382  k = start;
383  i = iploc;
384  nbpoi = 0;
385  do {
386  if ( nbpoi == MMG2D_LMAX ) {
387  fprintf(stderr,"\n ## Warning: %s: unable to compute adjacent"
388  " vertices of the vertex %d:\nthe ball of point contain too many"
389  " elements.\n",__func__,ip);
390  return 0;
391  }
392  i1 = MMG5_inxt2[i];
393  lispoi[nbpoi] = mesh->tria[k].v[i1];
394  ++nbpoi;
395 
396  adja = &mesh->adja[3*(k-1)+1];
397  prevk = k;
398  k = adja[i1] / 3;
399  i = adja[i1] % 3;
400  i = MMG5_inxt2[i];
401  }
402  while ( k && k != start );
403 
404  if ( k > 0 ) return nbpoi;
405 
406  /* store the last point of the boundary triangle */
407  if ( nbpoi == MMG2D_LMAX ) {
408  fprintf(stderr,"\n ## Warning: %s: unable to compute adjacent vertices of the"
409  " vertex %d:\nthe ball of point contain too many elements.\n",
410  __func__,ip);
411  return 0;
412  }
413  i1 = MMG5_inxt2[i1];
414  lispoi[nbpoi] = mesh->tria[prevk].v[i1];
415  ++nbpoi;
416 
417  /* check if boundary hit */
418  k = start;
419  i = iploc;
420  do {
421  adja = &mesh->adja[3*(k-1)+1];
422  i2 = MMG5_iprv2[i];
423  k = adja[i2] / 3;
424  if ( k == 0 ) break;
425 
426  if ( nbpoi == MMG2D_LMAX ) {
427  fprintf(stderr,"\n ## Warning: %s: unable to compute adjacent vertices of the"
428  " vertex %d:\nthe ball of point contain too many elements.\n",
429  __func__,ip);
430  return 0;
431  }
432  i = adja[i2] % 3;
433  lispoi[nbpoi] = mesh->tria[k].v[i];
434  ++nbpoi;
435 
436  i = MMG5_iprv2[i];
437  }
438  while ( k );
439 
440  return nbpoi;
441 }
