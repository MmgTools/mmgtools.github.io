40 /* Calculate an estimate of the average (isotropic) length of edges in the mesh */
41 double MMG2D_estavglen(MMG5_pMesh mesh) {
42  MMG5_pTria pt;
43  MMG5_pPoint p1,p2;
44  int k,na;
45  double len,lent,dna;
46  int8_t i,i1,i2;
47 
48  na = 0;
49  lent = 0.0;
50 
51  for (k=1; k<=mesh->nt; k++) {
52  pt = &mesh->tria[k];
53  for (i=0; i<3; i++) {
54  i1 = MMG5_inxt2[i];
55  i2 = MMG5_iprv2[i];
56 
57  p1 = &mesh->point[pt->v[i1]];
58  p2 = &mesh->point[pt->v[i2]];
59 
60  len = (p2->c[0]-p1->c[0])*(p2->c[0]-p1->c[0]) + (p2->c[1]-p1->c[1])*(p2->c[1]-p1->c[1]);
61 
62  lent += sqrt(len);
63  na++;
64  }
65  }
66 
67  dna = (double)na;
68  dna = 1.0 / dna;
69  lent *= dna;
70 
71  return lent;
72 }

75 static
76 inline double MMG2D_caltri_iso_3pt(double *a,double *b,double *c) {
77  double abx,aby,acx,acy,bcx,bcy,area,h1,h2,h3,hm;
78 
79  abx = b[0] - a[0];
80  aby = b[1] - a[1];
81  acx = c[0] - a[0];
82  acy = c[1] - a[1];
83  bcx = c[0] - b[0];
84  bcy = c[1] - b[1];
85 
86  /* orientation */
87  area = abx*acy - aby*acx;
88  if ( area <= 0.0 ) return 0.0;
89 
90  /* edge lengths */
91  h1 = abx*abx + aby*aby;
92  h2 = acx*acx + acy*acy;
93  h3 = bcx*bcx + bcy*bcy;
94 
95  hm = h1 + h2 + h3;
96 
97  if ( hm > MMG2D_EPSD ) {
98  return area / hm;
99  }
100  else {
101  return 0.0;
102  }
103 }

235  c[i][j] = ppt->c[j];
236  }

242  v = &disp->m[2*np];
243  for (j=0; j<2; j++)
244  c[i][j] += tau*v[j];
245  }

248

256  ppt = &mesh->point[k];
