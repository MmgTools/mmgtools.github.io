35 /* Calculate the barycentric coordinates of point P(c[0],c[1]) in tria pt and the associated determinant */
36 /* l1 = barycentric coordinate with respect to p1, l2 = barycentric coordinate with respect to p2 */
37 int MMG2D_coorbary(MMG5_pMesh mesh,MMG5_pTria pt,double c[2],double* det,double* l1,double* l2) {
38  MMG5_pPoint p1,p2,p3;
39  double b2,b3;
40  static int8_t mmgWarn0=0;
41 
42  p1 = &mesh->point[pt->v[0]];
43  p2 = &mesh->point[pt->v[1]];
44  p3 = &mesh->point[pt->v[2]];
45 
46  /* Calculate det (p2-p1,p3-p1) */
47  *det = (p2->c[0]-p1->c[0])*(p3->c[1]-p1->c[1]) - (p2->c[1]-p1->c[1])*(p3->c[0]-p1->c[0]);
48  if ( *det < MMG5_EPSD ) {
49  if ( !mmgWarn0 ) {
50  mmgWarn0 = 1;
51  fprintf(stderr,"\n ## Warning: %s: at least 1 flat triangle. abort.\n",
52  __func__);
53  }
54  return 0;
55  }
56  *det = 1.0 / (*det);
57 
58  b2 = (c[0]-p1->c[0])*(p3->c[1]-p1->c[1]) - (c[1]-p1->c[1])*(p3->c[0]-p1->c[0]);
59  b3 = (p2->c[0]-p1->c[0])*(c[1]-p1->c[1]) -(p2->c[1]-p1->c[1])*(c[0]-p1->c[0]);
60  b2 *= (*det);
61  b3 *= (*det);
62  *l1 = 1.0 - (b2+b3);
63  *l2 = b2;
64 
65  return 1;
66 }

86 /* Check whether edge ppa-ppb crosses triangles pt (in the sense that two edges
87  of this triangle are crossed by (ppa-ppb), or only one, and ppa or ppb is a
88  vertex of pt; if at least one edge is crossed by ia-ib, return i+1, where i

90 int MMG2D_cutEdge(MMG5_pMesh mesh,MMG5_pTria pt,MMG5_pPoint ppa,MMG5_pPoint ppb) {
91  double la[3],lb[3],det;
92  int icompt,ireturn;
93  int8_t ier,i;
94 
95  ier = MMG2D_coorbary(mesh,pt,ppa->c,&det,&la[0],&la[1]);
96  if ( !ier ) return 0;
97  la[2] = 1.0-(la[0]+la[1]);
98 
99  ier = MMG2D_coorbary(mesh,pt,ppb->c,&det,&lb[0],&lb[1]);
100  if ( !ier ) return 0;
101  lb[2] = 1.0-(lb[0]+lb[1]);
102 
103  /* Check whether ppa or ppb is a vertex of pt */
104  for (i=0; i<3; i++) {
105  if ( fabs(la[i]-1.0) < 1.0e-12 ) {
106  if ( lb[i] < 0.0 ) return i+1;
107  else return 0;
108  }
109  if ( fabs(lb[i]-1.0) < 1.0e-12) {
110  if ( la[i] < 0.0 ) return i+1;
111  else return 0;
112  }
113  }
114 
115  icompt = 0;
116  for (i=0; i<3; i++) {
117  if ( ( la[i] >= 0.0 && lb[i] <= 0.0 ) || ( la[i] <= 0.0 && lb[i] >= 0.0 ) ) {
118  ireturn = i+1;
119  icompt++;
120  }
121  }
122 
123  if ( icompt > 1 ) return ireturn;
124  return 0;
125 }

129 int MMG2D_cutEdgeTriangle(MMG5_pMesh mesh,int k,int ia,int ib) {
130  MMG5_pTria pt;
131  MMG5_pPoint p1,p2,p3,ppa,ppb;
132  double a11,a21,a12,a22,area1,area2,area3,prod1,prod2,prod3;
133  int ibreak,iare;
134  int8_t i;
135 
136  ppa = &mesh->point[ia];
137  ppb = &mesh->point[ib];
138 
139  pt = &mesh->tria[k];
140  if ( !MG_EOK(pt) ) return 0;
141  ibreak = 0;
142 
143  if ( pt->v[0] == ib || pt->v[1] == ib || pt->v[2] == ib) ibreak = 1;
144 
145  p1 = &mesh->point[pt->v[0]];
146  p2 = &mesh->point[pt->v[1]];
147  p3 = &mesh->point[pt->v[2]];
148 
149  /* Calculation of the areas ia,ib,pi*/
150  a11 = ppb->c[0] - ppa->c[0];
151  a21 = ppb->c[1] - ppa->c[1];
152 
153  a12 = p1->c[0] - ppa->c[0];
154  a22 = p1->c[1] - ppa->c[1];
155  area1 = a11*a22 - a12*a21;
156 
157  a12 = p2->c[0] - ppa->c[0];
158  a22 = p2->c[1] - ppa->c[1];
159  area2 = a11*a22 - a12*a21;
160 
161  a12 = p3->c[0] - ppa->c[0];
162  a22 = p3->c[1] - ppa->c[1];
163  area3 = a11*a22 - a12*a21;
164 
165  prod1 = area1*area2;
166  prod2 = area3*area2;
167  prod3 = area3*area1;
168 
169  /* Both edges p2p3 and p1p3 corresponding to prod2 and prod3 are cut by edge ia,ib */
170  if ( prod1 > 0 && ((prod2 < 0 || prod3 < 0))) {
171  if ( (iare = MMG2D_cutEdge(mesh,pt,ppa,ppb)) ) {
172  return iare;
173  }
174  }
175 
176  /* Both edges corresponding to prod1 and prod3 are cut by edge ia,ib */
177  if ( prod2 > 0 && ((prod1 < 0 || prod3 < 0))) {
178  if ( (iare = MMG2D_cutEdge(mesh,pt,ppa,ppb)) ) {
179  return iare;
180  }
181  }
182 
183  /* Both edges corresponding to prod1 and prod2 are cut by edge ia,ib */
184  if ( prod3 > 0 && ((prod2 < 0 || prod1 < 0))) {
185  if ( (iare = MMG2D_cutEdge(mesh,pt,ppa,ppb)) ) {
186  return iare;
187  }
188  }
189 
190  /* Case where one vertex of pt is ia */
191  for(i=0; i<3; i++){
192  if ( pt->v[i] == ia || ibreak ) {
193  /* One vertex is ia, and the opposite edge is frankly crossed */
194  if ( (prod1 < 0) || (prod2 < 0) || (prod3 < 0) ) {
195  if ( (iare = MMG2D_cutEdge(mesh,pt,ppa,ppb)) ) {
196  return iare;
197  }
198  }
199  else {
200  /*check if ia-ib edge de pt*/
201  if ( ibreak && ( pt->v[(i+1)%3] == ia || pt->v[(i+2)%3] == ia ) ) {
202  return -3;
203  }
204  else if ( pt->v[i] == ia && ( pt->v[(i+1)%3] == ib || pt->v[(i+2)%3] == ib ) ) {
205  return -3;
206  }
207  }
208  }
209  }
210 
211  return 0;
212 }

215 int MMG2D_findTria(MMG5_pMesh mesh,int ip) {
216  MMG5_pTria pt,pt1;
217  int iel,base,iadr,*adja,iter,ier;
218  int mvDir[3],jel,i;
219  double l1,l2,l3,det,eps;
220  static int8_t mmgWarn0 = 0;
221 
222  ++mesh->base;
223  base = ++mesh->base;
224  iter = 0;
225  iel = 1;
226  do {
227  mvDir[0] = mvDir[1] = mvDir[2] = 0;
228  iter++;
229  pt = &mesh->tria[iel];
230  if ( !MG_EOK(pt) ) {
231  iel++;
232  if ( iel > mesh->nt ) return 0;
233  continue;
234  }
235 
236  if ( pt->base == base) {
237  if ( !mmgWarn0 ) {
238  mmgWarn0 = 1;
239  fprintf(stderr,"\n ## Warning: %s: numerical problem, please make"
240  " a bug report.\n",__func__);
241  }
242  return iel;
243  }
244  /* Check whether ip is one of the vertices of pt */
245  if ( (pt->v[0] == ip) || (pt->v[1] == ip) || (pt->v[2] == ip) ) return iel;
246 
247  pt->base = base;
248  iadr = 3*(iel-1)+1;
249  adja = &mesh->adja[iadr];
250 
251  /*compute the barycentric coordinates*/
252  ier = MMG2D_coorbary(mesh,pt,mesh->point[ip].c,&det,&l1,&l2);
253  if ( !ier ) return 0;
254 
255  l3 = 1-l1-l2;
256  /*warning -1e12 is too strict*/
257  eps = (-1e-9)*MMG2D_quickcal(mesh,pt);
258 
259  /*find in which direction we can move*/
260  if (l1<eps)
261  mvDir[0] = 1;
262  if (l2<eps)
263  mvDir[1] = 1;
264  if (l3<eps)
265  mvDir[2] = 1;
266 
267  /*if all the barycentric coordinates are positive, we find the triangle*/
268  if(!mvDir[0] && !mvDir[1] && !mvDir[2]) {
269  break;
270  }
271 
272  /*first, try to move in an optimal direction*/
273  iel = 0;
274  for(i=0;i<3; i++) {
275  if (!mvDir[i]) continue;
276  jel = adja[i]/3;
277 
278  if (!jel) continue;
279  pt1 = &mesh->tria[jel];
280 
281  if(pt1->base == mesh->base) continue;
282  iel = jel;
283  break;
284  }
285 
286  /*second, the optimal move is not possible, so try to move in one direction*/
287  if(i==3) {
288  for(i=0;i<3; i++) {
289  if (mvDir[i]) continue;
290  jel = adja[i]/3;
291  if (!jel) continue;
292  pt1 = &mesh->tria[jel];
293  if(pt1->base == mesh->base) continue;
294  iel = jel;
295  break;
296  }
297  }
298 
299  /*we already see all the adj triangle so we can do nothing*/
300  if (iel == 0) {
301  iter = mesh->nt+1;
302  }
303  } while ( iter<=mesh->nt );
304 
305  return iel;
306 }
