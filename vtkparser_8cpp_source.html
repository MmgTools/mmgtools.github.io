98 static
99 int MMG5_count_vtkEntities ( vtkDataSet *dataset, MMG5_pMesh mesh,
100  int8_t *ptMeditRef, int8_t *eltMeditRef, int *nsols ) {
101 
102  static int8_t mmgWarn0 = 0;
103  static int8_t mmgWarn1 = 0;
104 
105  // Count the number of entities in the mesh
106  assert ( !mesh->np );
107  assert ( !mesh->nt );
108  assert ( !mesh->na );
109  assert ( !mesh->nquad );
110  assert ( !mesh->nprism );
111 
112  mesh->np = dataset->GetNumberOfPoints();
113  vtkIdType numCells = dataset->GetNumberOfCells();
114 
115  assert ( !mesh->npi );
116  for ( vtkIdType i = 0; i < numCells; i++ ) {
117  int typ = dataset->GetCellType(i);
118  switch ( typ ) {
119  case ( VTK_VERTEX ):
120  ++mesh->npi;
121  break;
122  case ( VTK_POLY_LINE ):
123  printf( "polylin %lld \n", dataset->GetCell(i)->GetNumberOfPoints() - 1 );
124 
125  printf( "%lld %lld %lld\n", dataset->GetCell(i)->GetPointId(0)+1,
126  dataset->GetCell(i)->GetPointId(1)+1,dataset->GetCell(i)->GetPointId(2)+1);
127  mesh->na += dataset->GetCell(i)->GetNumberOfPoints() - 1;
128  break;
129  case ( VTK_LINE ):
130  ++mesh->na;
131  break;
132  case ( VTK_TRIANGLE ):
133  ++mesh->nt;
134  break;
135  case ( VTK_QUAD):
136  ++mesh->nquad;
137  break;
138  case ( VTK_TETRA):
139  ++mesh->ne;
140  break;
141  case ( VTK_WEDGE):
142  ++mesh->nprism;
143  break;
144  default:
145  if ( !mmgWarn1 ) {
146  printf( " ## Warning:%s: unexpected element type (%d).",__func__,typ);
147  mmgWarn1 = 1;
148  }
149  }
150  }
151  assert ( mesh->npi == mesh->np || !mesh->npi );
152  mesh->npi = 0;
153 
154  // Count the number of point data and detect point references
155  auto *pd = dataset->GetPointData();
156  int npointData = 0;
157  int npointRef = 0;
158  int nmetricField = 0;
159 
160  *ptMeditRef = -1;
161  if ( pd ) {
162  npointData = pd->GetNumberOfArrays();
163  for (int k = 0; k < npointData; k++) {
164  if ( strstr(pd->GetArrayName(k),":metric" ) ) {
165  ++nmetricField;
166  }
167  else if ( strstr(pd->GetArrayName(k),"medit:ref" ) ) {
168  (*ptMeditRef) = k;
169  ++npointRef;
170  }
171  }
172 
173  if ( npointRef > 1 ) {
174  printf( " ## Warning:%s: %d reference fields detected (labelled 'medit:ref')."
175  " Only the last is used, others are ignored.", __func__, npointRef);
176  }
177  if ( nmetricField > 1 ) {
178  printf(" ## Error:%s: %d metric fields detected (labelled with a string"
179  " containing the 'metric' keyword).\n"
180  " Exit Program.\n",__func__,nmetricField);
181  return -1;
182  }
183  }
184 
185  // Count the number of cell data and detect cell references
186  auto *cd = dataset->GetCellData();
187  int ncellData = 0;
188  int ncellRef = 0;
189 
190  *eltMeditRef = -1;
191  if ( cd ) {
192  ncellData = cd->GetNumberOfArrays();
193  for (int k = 0; k < ncellData; k++) {
194  if ( strstr(cd->GetArrayName(k),"medit:ref" ) ) {
195  *eltMeditRef = k;
196  ++ncellRef;
197  }
198  }
199  }
200 
201  // Count the number of field data
202  auto *fd = dataset->GetFieldData();
203  if ( fd->GetNumberOfArrays() ) {
204  printf( " ## Warning:%s: VTK field data not used in Mmg."
205  " Ignored.\n",__func__ );
206  }
207 
208  *nsols = npointData + ncellData - npointRef - ncellRef;
209 
210  return 1;
211 }

339  // Point transfers in Mmg data structure

363  if ( ppt->ref < 0 ) {
364  ppt->ref = -ppt->ref;
365  ++nref;
366  }
367  }
368  }
369  else {
370  // read vertices only
371  for ( vtkIdType k = 0; k < (*dataset)->GetNumberOfPoints(); k++ ) {
372  MMG5_pPoint ppt = &mesh->point[k+1];
373  (*dataset)->GetPoint(k,mesh->point[k+1].c);
374  ppt->tag = MG_NUL;
375  ppt->tmp = 0;
376  ppt->ref = 0;
377  }

383  assert ( !mesh->nai );
384  assert ( (mesh->npi == mesh->np) || !mesh->npi );
385

391  int nt = 0;
392  int nbl_t = 0;

395  vtkIdType numCells = (*dataset)->GetNumberOfCells();

398  // If present, read elements references
399  car = (*dataset)->GetCellData()->GetArray(eltMeditRef);

402  assert ( strstr( car->GetName(), "medit:ref" ) );
403  // Check that we get 1 data only
404  assert ( car->GetNumberOfComponents() == 1 );
405 
406  int ne = car->GetNumberOfTuples();
407  if ( ne != numCells ) {
408  printf( " ## Error: Cell data size (%d) differs from the number of"
409  " cells (%lld)\n",ne,numCells);
410  return 0;
411  }
412  }
413 
414  // Transfer cells to Mmg
415  for ( vtkIdType k = 0; k < numCells; k++ ) {
416  MMG5_pPoint ppt = NULL;
417  MMG5_pEdge pa = NULL;
418  MMG5_pTria ptt = NULL;

425

428  ppt = &mesh->point[++mesh->npi];
429  ppt->ref = car ? car->GetTuple1(k) : 0;
430  if ( ppt->ref < 0 ) {
431  ppt->ref = -ppt->ref;
432  ++nref;
433  }
434  break;
435  case ( VTK_POLY_LINE ):
436  int n;
437  n = (*dataset)->GetCell(k)->GetNumberOfPoints() - 1;
438 
439  for ( int i=0; i<n; ++i ) {
440  ++mesh->nai;
441  ref = car ? car->GetTuple1(k) : 0;
442  }
443  /* Skip edges with iso ref */
444  if ( mesh->info.iso && abs(ref) == MG_ISO ) {
445  /* Skip this edge */
446  ++nbl_a;
447  }
448  else {
449  for ( int i=0; i < n; ++i ) {
450  pa = &mesh->edge[++na];
451  pa->a = (*dataset)->GetCell(k)->GetPointId(i) +1;
452  pa->b = (*dataset)->GetCell(k)->GetPointId(i+1)+1;
453  }
454  pa->ref = ref;
455  pa->tag |= MG_REF;
456  if ( pa->ref < 0 ) {
457  pa->ref = -pa->ref;
458  ++nref;
459  }
460  }
461 
462  break;
463  case ( VTK_LINE ):
464 
465  ++mesh->nai;
466  ref = car ? car->GetTuple1(k) : 0;
467 
468  // Skip edges with iso ref
469  if ( mesh->info.iso && abs(ref) == MG_ISO ) {
470  /* Skip this edge */
471  ++nbl_a;
472  }
473  else {
474  pa = &mesh->edge[++na];
475  pa->a = (*dataset)->GetCell(k)->GetPointId(0)+1;
476  pa->b = (*dataset)->GetCell(k)->GetPointId(1)+1;
477  pa->ref = ref;
478  pa->tag |= MG_REF;
479  if ( pa->ref < 0 ) {
480  pa->ref = -pa->ref;
481  ++nref;
482  }
483  }
484  assert( na+nbl_a <= mesh->na );
485  break;
486

488  ++mesh->nti;
489  ref = car ? car->GetTuple1(k) : 0;
490 
491  // skip tria with iso ref in 3D
492  if ( mesh->info.iso && abs(ref) == MG_ISO && mesh->dim == 3 ) {
493  /* Skip this edge */
494  ++nbl_t;
495  }
496  else {
497  ptt = &mesh->tria[++nt];
498  for ( int i=0; i<3; ++i ) {
499  ptt->v[i] = (*dataset)->GetCell(k)->GetPointId(i)+1;

502 
503  if ( ptt->ref < 0 ) {
504  ptt->ref = -ptt->ref;
505  ++nref;
506  }
507  }
508  break;

511  pq = &mesh->quadra[++nqi];
512  pq->ref = car ? car->GetTuple1(k) : 0;
513  for ( int i=0; i<4; ++i ) {
514  pq->v[i] = (*dataset)->GetCell(k)->GetPointId(i)+1;
515  }
516  if ( pq->ref < 0 ) {
517  pq->ref = -pq->ref;
518  ++nref;
519  }
520  break;
521 
522  case ( VTK_TETRA ):
523  // Skip volume elts if called from mmgs
524  if ( !mesh->tetra ) continue;
525 
526  pt = &mesh->tetra[++mesh->nei];
527  pt->ref = car ? car->GetTuple1(k) : 0;
528  for ( int i=0; i<4; ++i ) {

532  pt->ref = -pt->ref;
533  ++nref;
534  }
535  break;

538  // Skip volume elts if called from mmgs
539  if ( !mesh->prism ) continue;
540 
541  ppr = &mesh->prism[++npri];

544  ppr->v[i] = (*dataset)->GetCell(k)->GetPointId(i)+1;
545  }

557  }
558  }
559  }

696  }
697  break;
698

702

718  }
719  }
