31 static int MMG2D_correction_iso(MMG5_pMesh mesh,int ip,int *list,int ilist,int nedep) {
32  MMG5_pTria pt;
33  MMG5_pPoint ppt,p1,p2;
34  double dd,ux,uy,vx,vy;
35  int *adja,i,ipil,iel,lon,iadr,adj,ib,ic,base,ncor,nei[3];
36 
37  ppt = &mesh->point[ip];
38  if ( !MG_VOK(ppt) ) return ilist;
39  base = mesh->base;
40  lon = ilist;
41  do {
42  ipil = lon-1;
43  ncor = 0;
44 
45  while ( ipil >= 0 ) {
46  iel = list[ipil];
47  iadr = (iel-1)*3 + 1;
48  adja = &mesh->adja[iadr];
49  nei[0] = adja[0] /3;
50  nei[1] = adja[1] /3;
51  nei[2] = adja[2] /3;
52  pt = &mesh->tria[iel];
53 
54  for (i=0; i<3; i++) {
55  adj = nei[i];
56  /* Consider only the external faces of the cavity */
57  if ( adj && mesh->tria[adj].base == base ) continue;
58 
59  ib = pt->v[ MMG5_inxt2[i] ];
60  ic = pt->v[ MMG5_iprv2[i] ];
61 
62  p1 = &mesh->point[ib];
63  p2 = &mesh->point[ic];
64 
65  ux = p2->c[0] - p1->c[0];
66  uy = p2->c[1] - p1->c[1];
67 
68  vx = ppt->c[0] - p1->c[0];
69  vy = ppt->c[1] - p1->c[1];
70 
71  /* area PBC */
72  dd = ux*vy - uy*vx;
73  if ( dd < MMG2D_AREAMIN ) break;
74 
75  }
76 
77  /* Remove triangle iel from the cavity if it leads to a degenerate triangle after insertion of ppt */
78  if ( i < 3 /*|| pt->tag & MG_REQ*/ ) {
79  /* remove iel from list */
80  pt->base = base-1;
81  list[ipil] = list[--lon];
82  ncor = 1;
83  break;
84  }
85  else
86  ipil--;
87  }
88  }
89  while ( ncor > 0 && lon >= nedep );
90  return lon;
91 }

106 static int MMG2D_hashEdgeDelone(MMG5_pMesh mesh,MMG5_Hash *hash,int iel,int i) {
107  MMG5_pTria pt;
108  int *adja,iadr,jel,j,ip1,ip2;
109  int16_t i1,i2;
110 
111  pt = &mesh->tria[iel];
112  i1 = MMG5_inxt2[i];
113  i2 = MMG5_iprv2[i];
114  ip1 = pt->v[i1];
115  ip2 = pt->v[i2];
116 
117  /* Search if the face is already hashed */
118  jel = MMG5_hashGet(hash,ip1,ip2);
119 
120  if ( !jel ) {
121  /* if not, hash the edge and store its unique key */
122  jel = MMG5_hashEdge(mesh,hash,ip1,ip2,3*iel+i);
123  if ( !jel ) {
124  printf(" # Error: %s: Unable to add edge %d %d within the hash table\n",
125  __func__,MMG2D_indPt(mesh,ip1),MMG2D_indPt(mesh,ip2));
126  return 0;
127  }
128  }
129  else {
130  /* otherwise, update the adjacency array */
131  iadr = (iel-1)*3 + 1;
132  adja = &mesh->adja[iadr];
133  adja[i] = jel;
134 
135  j = jel % 3;
136  jel /= 3;
137  iadr = (jel-1)*3 + 1;
138  adja = &mesh->adja[iadr];
139  adja[j] = iel*3 + i;
140  }
141 
142  return 1;
143 }

147 int MMG2D_cavity(MMG5_pMesh mesh,MMG5_pSol sol,int ip,int *list) {
148  MMG5_pTria pt,pt1,ptc;
149  MMG5_pPoint ppt;
150  double c[2],crit,dd,eps,rad,ct[6];
151  int *adja,*adjb,adj,adi,voy,i,j,ilist,ipil,jel,iadr,base,nei[3],l,tref; //isreq;
152  static int8_t mmgWarn0=0;
153 
154  ppt = &mesh->point[ip];
155  base = ++mesh->base;
156  //isreq = 0;
157  tref = mesh->tria[list[0]].ref;
158  mesh->tria[list[0]].base = base;
159 
160  /* Pile up cavity by adjacency */
161  eps = 1. + MMG5_EPSOK;
162  ilist = 1;
163  ipil = 0;
164 
165  do {
166  jel = list[ipil];
167  iadr = (jel-1)*3 + 1;
168  adja = &mesh->adja[iadr];
169  nei[0] = adja[0];
170  nei[1] = adja[1];
171  nei[2] = adja[2];
172  ptc = &mesh->tria[jel];
173 
174  for (i=0; i<3; i++) {
175  adj = nei[i] /3;
176  voy = nei[i] % 3;
177  if ( !adj ) continue;
178  pt = &mesh->tria[adj];
179 
180  /* Case where the triangle has already been piled, or a boundary face is hit */
181  if ( pt->base == base || pt->ref != ptc->ref ) continue;
182 
183  /* Store the 6 coordinates of the vertices of pt */
184  for (j=0,l=0; j<3; j++,l+=2) {
185  memcpy(&ct[l],mesh->point[pt->v[j]].c,2*sizeof(double));
186  }
187 
188  if ( !MMG2D_cenrad_iso(mesh,ct,c,&rad) ) continue;
189  crit = eps * rad;
190 
191  /* Delaunay criterion */
192  dd = (ppt->c[0] - c[0]) * (ppt->c[0] - c[0]) + (ppt->c[1] - c[1]) * (ppt->c[1] - c[1]);
193  if ( dd > crit ) continue;
194 
195  /* Lost face(s); I don't understand this test: the algorithm is supposed to stop when changing references */
196  iadr = (adj-1)*3 + 1;
197  adjb = &mesh->adja[iadr];
198 
199  for (j=0; j<3; j++) {
200  if ( j == voy ) continue;
201  adi = adjb[j] /3;
202  if ( !adi ) continue;
203  pt1 = &mesh->tria[adi];
204  if ( pt1->base == base && adi != jel ) {
205  if ( pt1->ref != tref ) {
206  break;
207  }
208  }
209  }
210  /* store tria */
211  if ( j == 3 ) {
212  //if ( pt->tag & MG_REQ ) isreq = 1;
213  pt->base = base;
214  list[ilist++] = adj;
215  }
216  else {
217  if ( !mmgWarn0 ) {
218  mmgWarn0 = 1;
219  fprintf(stderr,"\n ## Error: %s: we pass here at least one time but one"
220  " should never go through here.\n",__func__);
221  }
222  }
223  }
224  if ( ilist > MMG2D_LONMAX - 3 ) return -1;
225 
226  ++ipil;
227  }
228  while ( ipil < ilist );
229 
230  ilist = MMG2D_correction_iso(mesh,ip,list,ilist,1);
231  //if ( isreq ) ilist = -fabs(ilist);
232  return ilist;
233 }

247 int MMG2D_delone(MMG5_pMesh mesh,MMG5_pSol sol,int ip,int *list,int ilist) {
248  MMG5_pTria pt,pt1;
249  MMG5_pPoint ppt;
250  int *adja,*adjb,i,j,k,iel,jel,old,iadr,base,size,nei[3],iadrold;
251  int tref,ielnum[3*MMG2D_LONMAX+1];
252  int8_t ier;
253  short i1;
254  int8_t alert;
255  MMG5_Hash hedg;
256  static int8_t mmgWarn0=0,mmgWarn1=0;
257 
258  /* Reset tagdel field */
259  for (k=1; k<ilist; k++)
260  mesh->point[k].tagdel = 0;
261 
262  /* Triangles in the cavity are those s.t. pt->base == base */
263  base = mesh->base;
264  /* Count the number of external faces in the cavity, and tag the corresponding vertices */
265  size = 0;
266  for (k=0; k<ilist; k++) {
267  old = list[k];
268  pt1 = &mesh->tria[old];
269  iadr = (old-1)*3 + 1;
270  adja = &mesh->adja[iadr];
271  nei[0] = adja[0]/3 ;
272  nei[1] = adja[1]/3 ;
273  nei[2] = adja[2]/3 ;
274  for (i=0; i<3; i++) {
275  jel = nei[i];
276  if ( (!jel) || (mesh->tria[jel].base != base) ) {
277  for (j=0; j<2; j++) {
278  i1 = MMG2D_iare[i][j];
279  ppt = &mesh->point[ pt1->v[i1] ];
280  ppt->tagdel = 1;
281  }
282  size++;
283  }
284  }
285  }
286 
287  /* Check for an isolated vertex (the cavity should ne contain any internal vertex) */
288  alert = 0;
289  for (k=0; k<ilist; k++) {
290  old = list[k];
291  pt1 = &mesh->tria[old];
292  for (i=0; i<3; i++) {
293  ppt = &mesh->point[ pt1->v[i] ];
294  if ( !ppt->tagdel ) {
295  alert = 1;
296  }
297  }
298  }
299  /* Reset tagdel field */
300  for (k=0; k<ilist; k++) {
301  old = list[k];
302  pt1 = &mesh->tria[old];
303  for (i=0; i<3; i++) {
304  ppt = &mesh->point[ pt1->v[i] ];
305  ppt->tagdel = 0;
306  }
307  }
308  if ( alert ) return 0;
309 
310  /* Hash table parameters */
311  if ( size >= 3*MMG2D_LONMAX ) return 0;
312  if ( !MMG5_hashNew(mesh,&hedg,size,3*size) ) {
313  fprintf(stderr,"\n ## Warning: %s: unable to allocate hash table.\n",__func__);
314  return -1;
315  }
316 
317  /* Allocate memory for "size" new triangles */
318  ielnum[0] = size;
319  for (k=1; k<=size; k++) {
320  ielnum[k] = MMG2D_newElt(mesh);
321  if ( !ielnum[k] ) {
322  MMG2D_TRIA_REALLOC(mesh,ielnum[k],mesh->gap,
323  fprintf(stderr,"\n ## Error: %s: unable to allocate"
324  " a new element.\n",__func__);
325  MMG5_INCREASE_MEM_MESSAGE();
326  printf(" Exit program.\n");return -1);
327  }
328  }
329 
330  size = 1;
331  for (k=0; k<ilist; k++) {
332  old = list[k];
333  pt = &mesh->tria[old];
334 
335  iadrold = (old-1)*3 + 1;
336  adja = &mesh->adja[iadrold];
337  nei[0] = adja[0];
338  nei[1] = adja[1];
339  nei[2] = adja[2];
340 
341  for (i=0; i<3; i++) {
342  jel = nei[i] / 3;
343  j = nei[i] % 3;
344 
345  /* Catch the associated external face */
346  if ( (!jel) || (mesh->tria[jel].base != base) ) {
347  assert ( size <= ielnum[0] );
348  iel = ielnum[size++];
349  assert(iel);
350 
351  pt1 = &mesh->tria[iel];
352  memcpy(pt1,pt,sizeof(MMG5_Tria));
353  pt1->v[i] = ip;
354  pt1->qual = MMG2D_caltri_iso(mesh,sol,pt1);
355  pt1->ref = pt->ref;
356 
357  if ( (!mmgWarn0) && (pt1->qual < MMG2D_AREAMIN) ) {
358  mmgWarn0 = 1;
359  fprintf(stderr," ## Warning: %s: creation of a very bad element.\n",
360  __func__);
361  }
362 
363  /* Update adjacency via the external face */
364  iadr = (iel-1)*3 + 1;
365  adjb = &mesh->adja[iadr];
366  adjb[i] = adja[i];
367 
368  if ( jel ) {
369  iadr = (jel-1)*3 + 1;
370  adjb = &mesh->adja[iadr];
371  adjb[j] = iel*3 + i;
372  }
373  /* Update adjacency via the internal faces */
374  for (j=0; j<3; j++) {
375  if ( j != i ) {
376  ier = MMG2D_hashEdgeDelone(mesh,&hedg,iel,j);
377  if ( !ier ) {
378  fprintf(stderr," ## Warning: %s: unable to update adjacency"
379  " relationship (elt %d, edge %d).\n",
380  __func__,MMG2D_indElt(mesh,iel),j);
381  return -1;
382  }
383  }
384  }
385  }
386  }
387  }
388 
389  /* Remove the old triangles */
390  tref = mesh->tria[list[0]].ref;
391  for (k=0; k<ilist; k++) {
392  if ( (!mmgWarn1) && (tref != mesh->tria[list[k]].ref) ) {
393  mmgWarn1 = 1;
394  fprintf(stderr,"\n ## Warning: %s: sud-domain ignored.\n",__func__);
395  }
396  MMG2D_delElt(mesh,list[k]);
397  }
398 
399  //ppt = &mesh->point[ip];
400  // ppt->flag = mesh->flag;
401  MMG5_SAFE_FREE(hedg.item);
402  return 1;
403 }
