447 int MMGS_update_normalAndTangent(MMG5_pMesh mesh,MMG5_pPoint p0,MMG5_pPoint p,

474  ps2 *= (2.0/llold);

482  return 0;

484

486  nn1[0] = dd1*nn1[0];
487  nn1[1] = dd1*nn1[1];
488  nn1[2] = dd1*nn1[2];
489 
490  dd2 = 1.0 / sqrt(dd2);
491  nn2[0] = dd2*nn2[0];
492  nn2[1] = dd2*nn2[1];
493  nn2[2] = dd2*nn2[2];
494

496  have been computed .*/

498  nn1[1] = lam0*no1[1] + lam1*nn1[1] + lam2*np1[1];

504

506  to[1] = nn1[2]*nn2[0]-nn1[0]*nn2[2];
507  to[2] = nn1[0]*nn2[1]-nn1[1]*nn2[0];

510  dd1 = nn1[0]*nn1[0] + nn1[1]*nn1[1] + nn1[2]*nn1[2];
511  dd2 = nn2[0]*nn2[0] + nn2[1]*nn2[1] + nn2[2]*nn2[2];
512 
513  if ( (dd1 < MMG5_EPSD2) || (dd2<MMG5_EPSD2) || (ddt < MMG5_EPSD2) ) {
514  return 0;
515  }
516 
517  dd1 = 1.0 / sqrt(dd1);
518  nn1[0] = dd1*nn1[0];
519  nn1[1] = dd1*nn1[1];
520  nn1[2] = dd1*nn1[2];
521 
522  dd2 = 1.0 / sqrt(dd2);
523  nn2[0] = dd2*nn2[0];
524  nn2[1] = dd2*nn2[1];
525  nn2[2] = dd2*nn2[2];
526 
527  ddt = 1.0 / sqrt(ddt);
528  to[0] = ddt*to[0];
529  to[1] = ddt*to[1];
530  to[2] = ddt*to[2];
531 
532  return 1;
533 }
