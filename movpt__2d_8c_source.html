52 int MMG2D_movedgpt(MMG5_pMesh mesh,MMG5_pSol met,int ilist,int *list, int8_t improve) {

54  MMG5_pPoint p0,p1,p2,ppt;
55  double step,ll1,ll2,o[2],no[2],calold,calnew;
56  int k,iel,ip0,ip1,ip2,it1,it2;
57  int8_t i,i1,i2;
58  static int8_t mmgWarn0=0,mmgWarn1=0;
59 
60  pt0 = &mesh->tria[0];
61  step = 0.1;
62  ip1 = ip2 = it1 = it2 = 0;
63  calold = calnew = DBL_MAX;
64 
65  /* First step: retrieve the two boundary points connected to p0 */
66  for (k=0; k<ilist; k++) {
67  iel = list[k] / 3;
68  i = list[k] % 3;
69  i1 = MMG5_inxt2[i];
70  i2 = MMG5_iprv2[i];
71 
72  pt = &mesh->tria[iel];
73  calold = MG_MIN(MMG2D_caltri(mesh,met,pt),calold);
74 
75  if ( MG_EDG(pt->tag[i1]) ) {
76  if ( ip1 == 0 ) {
77  ip1 = pt->v[i2];
78  it1 = 3*iel+i1;
79  }
80  else if ( ip1 != pt->v[i2] ) {
81  if ( ip2 == 0 ) {
82  ip2 = pt->v[i2];
83  it2 = 3*iel+i1;
84  }
85  else if ( ip2 != pt->v[i2] ) {
86  if ( !mmgWarn0 ) {
87  mmgWarn0 = 1;
88  fprintf(stderr,"\n ## Warning: %s: at least 1 point at the "
89  "intersection of 3 edges. abort.\n",__func__);
90  }
91  return 0;
92  }
93  }
94  }
95 
96  if ( MG_EDG(pt->tag[i2]) ) {
97  if ( ip1 == 0 ) {
98  ip1 = pt->v[i1];
99  it1 = 3*iel+i2;
100  }
101  else if ( ip1 != pt->v[i1] ) {
102  if ( ip2 == 0 ) {
103  ip2 = pt->v[i1];
104  it2 = 3*iel+i2;
105  }
106  else if ( ip2 != pt->v[i1] ) {
107  if ( !mmgWarn0 ) {
108  mmgWarn0 = 1;
109  fprintf(stderr,"\n ## Warning: %s: at least 1 point at the "
110  "intersection of 3 edges. abort.\n",__func__);
111  }
112  return 0;
113  }
114  }
115  }
116  }
117 
118  /* Check that there are exactly two boundary points connected at p0 */
119  if ( ip1 == 0 || ip2 == 0 ) {
120  if ( !mmgWarn1 ) {
121  mmgWarn1 = 1;
122  fprintf(stderr,"\n ## Warning: %s: non singular point at end of edge.\n",
123  __func__);
124  }
125  return 0;
126  }
127 
128  ip0 = pt->v[i];
129  p0 = &mesh->point[ip0];
130  p1 = &mesh->point[ip1];
131  p2 = &mesh->point[ip2];
132 
133  /* Calculate length of both edges */
134  /* Anisotropic case: ll1 and ll2 = anisotropic edge lengths */
135  if ( met->m && met->size == 3 ) {
136  ll1 = MMG2D_lencurv_ani(mesh,met,ip0,ip1);
137  ll2 = MMG2D_lencurv_ani(mesh,met,ip0,ip2);
138  }
139  /* In all the remaining cases, ll1 and ll2 = squared edge lengths;
140  inconsistency between both cases is not problematic since these values serve only for comparison */
141  else {
142  ll1 = (p1->c[0]-p0->c[0])*(p1->c[0]-p0->c[0]) + (p1->c[1]-p0->c[1])*(p1->c[1]-p0->c[1]);
143  ll2 = (p2->c[0]-p0->c[0])*(p2->c[0]-p0->c[0]) + (p2->c[1]-p0->c[1])*(p2->c[1]-p0->c[1]);
144  }
145 
146  /* Relocate p0 slightly towards p1 */
147  if ( ll1 > ll2 ) {
148  iel = it1 / 3;
149  i = it1 % 3;
150  }
151  /* Relocate p0 slightly towards p2 */
152  else {
153  iel = it2 / 3;
154  i = it2 % 3;
155  }
156 
157  i1 = MMG5_inxt2[i];
158 
159  pt = &mesh->tria[iel];
160 
161  /* step = distance of the relocated position from ip0
162  bezierCurv = distance s from inxt2[i] */
163  if ( pt->v[i1] == ip0 ) MMG2D_bezierCurv(mesh,iel,i,step,o,no);
164  else MMG2D_bezierCurv(mesh,iel,i,1.0-step,o,no);
165 
166  /* Evaluate resulting configuration */
167  ppt = &mesh->point[0];
168  ppt->c[0] = o[0];
169  ppt->c[1] = o[1];
170  ppt->n[0] = no[0];
171  ppt->n[1] = no[1];
172 
173  for (k=0; k<ilist; k++) {
174  iel = list[k] / 3;
175  i = list[k] % 3;
176  pt = &mesh->tria[iel];
177  memcpy(pt0,pt,sizeof(MMG5_Tria));
178  pt0->v[i] = 0;
179 
180  calnew = MG_MIN(MMG2D_caltri(mesh,met,pt0),calnew);
181  }
182 
183  if ( calold < MMG2D_NULKAL && calnew <= calold ) return 0;
184  else if ( calnew < MMG2D_NULKAL ) return 0;
185  else if ( improve && calnew < 1.02 * calold ) return 0;
186  else if ( calnew < 0.3 * calold ) return 0;
187 
188  /* Update of the coordinates and normal of the point */
189  p0 = &mesh->point[pt->v[i]];
190  p0->c[0] = o[0];
191  p0->c[1] = o[1];
192 
193  p0->n[0] = no[0];
194  p0->n[1] = no[1];
195 
196  return 1;
197 }

224  calold = calnew = DBL_MAX;
225 
226  /* Calculate the now position for vertex, as well as the quality of the previous configuration */
227  for (k=0; k<ilist; k++) {
228  iel = list[k] / 3;
229  i = list[k] % 3;
230  i1 = MMG5_inxt2[i];
231  i2 = MMG5_iprv2[i];
232

234 
235  /* Volume of iel */
236  p0 = &mesh->point[pt->v[i]];
237  p1 = &mesh->point[pt->v[i1]];
238  p2 = &mesh->point[pt->v[i2]];
239  vol = 0.5* fabs((p1->c[0]-p0->c[0])*(p2->c[1]-p0->c[1]) - (p1->c[1]-p0->c[1])*(p2->c[0]-p0->c[0]));
240 
241  volbal += vol;
242 
243  /* Add coordinates of the centre of mass of iel, weighted by its volume */
244  b[0] += MMG5_ATHIRD*vol*(p0->c[0]+p1->c[0]+p2->c[0]);
245  b[1] += MMG5_ATHIRD*vol*(p0->c[1]+p1->c[1]+p2->c[1]);
246 
247  /* Quality of pt */
248  calold = MG_MIN(MMG2D_caltri_iso(mesh,NULL,pt),calold);
249  }
250

253  b[0] *= volbal;
254  b[1] *= volbal;
255

258  ppt0->c[1] = b[1];
259

261  iel = list[k] / 3;
262  i = list[k] % 3;
263  pt = &mesh->tria[iel];
264  memcpy(pt0,pt,sizeof(MMG5_Tria));
265  pt0->v[i] = 0;

268  }
269

271  else if (calnew < MMG2D_NULKAL) return 0;
272  else if ( improve && calnew < 1.02 * calold ) return 0;
273  else if ( calnew < 0.3 * calold ) return 0;

280  p0->c[1] = b[1];
