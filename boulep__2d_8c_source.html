30 /* Find all triangles sharing P
31  in: ifirst : triangle containing p
32  iploc : index of p in start
33  out: list : list of triangles */
34 int MMG2D_boulep(MMG5_pMesh mesh, int ifirst, int iploc, int * list) {
35  MMG5_pTria pt;
36  MMG5_pPoint ppt;
37  int ip,voy,ilist,iel,*adja,i,iadr;
38 
39  if ( ifirst < 1 ) return 0;
40  pt = &mesh->tria[ifirst];
41  if ( !MG_EOK(pt) ) return 0;
42  ip = pt->v[iploc];
43  ppt = &mesh->point[ip];
44  if ( !MG_VOK(ppt) ) return 0;
45 
46  /* init list */
47  ilist = 1;
48  list[ilist] = 3*ifirst + iploc;
49 
50  iadr = 3*(ifirst-1) + 1;
51  adja = &mesh->adja[iadr];
52  iel = adja[inxt[iploc]]/3;
53  voy = adja[inxt[iploc]]%3;
54  i = inxt[voy];
55 
56  while ( iel && (iel != ifirst) && mesh->tria[iel].v[0]){
57  if(ilist==MMG2D_LMAX-1) return 0;
58  list[++ilist] = 3*iel + i;
59  assert( ip==(&mesh->tria[iel])->v[i] );
60  iadr = 3*(iel-1) + 1;
61  adja = &mesh->adja[iadr];
62  iel = adja[inxt[i]]/3;
63  voy = adja[inxt[i]]%3;
64  i = inxt[voy];
65  }
66 
67  if ( iel!=ifirst ) {
68  iadr = 3*(ifirst-1) + 1;
69  adja = &mesh->adja[iadr];
70  iel = adja[iprev[iploc]]/3;
71  voy = adja[iprev[iploc]]%3;
72  i = iprev[voy];
73 
74  while ( iel && (iel != ifirst) && mesh->tria[iel].v[0]) {
75  if(ilist==MMG2D_LMAX-1) return 0;
76  list[++ilist] = 3*iel + i;
77  assert( ip==(&mesh->tria[iel])->v[i] );
78  iadr = 3*(iel-1) + 1;
79  adja = &mesh->adja[iadr];
80  iel = adja[iprev[i]]/3;
81  if (!iel) break;
82  voy = adja[iprev[i]]%3;
83  i = iprev[voy];
84 
85  }
86  }
87 
88  return ilist;
89 }

112 int MMG2D_boulen(MMG5_pMesh mesh, int start,int8_t ip, int *pleft, int *pright, double *nn) {
113  MMG5_pTria pt;
114  MMG5_pPoint p1,p2;
115  double ux,uy,dd,n1[2],n2[2];
116  int *adja,k,kk,refs;
117  int8_t notedg;
118  int8_t i,ii,i1,i2;
119 
120  /* First travel of the ball of ip; initialization */
121  kk = start;
122  ii = MMG5_iprv2[ip];
123  refs = mesh->tria[start].ref;
124 
125  do {
126  k = kk;
127  i = MMG5_iprv2[ii];
128  adja = &mesh->adja[3*(k-1)+1];
129  kk = adja[i] / 3;
130  ii = adja[i] % 3;
131 
132  notedg = mesh->info.opnbdy ?
133  (!mesh->tria[k].tag[i]) : (mesh->tria[kk].ref == refs);
134  }
135  while ( kk && (kk != start) && notedg );
136 
137  if ( kk == start ) {
138  fprintf(stderr," ## Error: %s: Unable to find a boundary edge in"
139  " the ball of point %d.\n",__func__,MMG2D_indPt(mesh,mesh->tria[start].v[ip]));
140  return 0;
141  }
142 
143  /* Calculation of the first normal vector */
144  pt = &mesh->tria[k];
145  i1 = MMG5_iprv2[i];
146  i2 = MMG5_inxt2[i];
147 
148  p1 = &mesh->point[pt->v[i1]];
149  p2 = &mesh->point[pt->v[i2]];
150  ux = p2->c[0] - p1->c[0];
151  uy = p2->c[1] - p1->c[1];
152  dd = ux*ux + uy*uy;
153 
154  if ( dd < MMG5_EPSD ) {
155  fprintf(stderr,"\n ## Error: %s: Null edge"
156  " length (%e).\n",__func__,dd);
157  return 0;
158  }
159 
160  dd = 1.0 / sqrt(dd);
161  n1[0] = -uy*dd;
162  n1[1] = ux*dd;
163 
164  *pright = 3*k+i1;
165 
166  /* Second travel */
167  kk = start;
168  ii = MMG5_inxt2[ip];
169 
170  do {
171  k = kk;
172  i = MMG5_inxt2[ii];
173  adja = &mesh->adja[3*(k-1)+1];
174  kk = adja[i] / 3;
175  ii = adja[i] % 3;
176 
177  notedg = mesh->info.opnbdy ?
178  !mesh->tria[k].tag[i] : mesh->tria[kk].ref == refs;
179  }
180  while ( kk && (kk != start) && notedg );
181 
182  /* Calculation of the second normal vector */
183  pt = &mesh->tria[k];
184  i1 = MMG5_inxt2[i];
185  i2 = MMG5_iprv2[i];
186 
187  p1 = &mesh->point[pt->v[i1]];
188  p2 = &mesh->point[pt->v[i2]];
189  ux = p2->c[0] - p1->c[0];
190  uy = p2->c[1] - p1->c[1];
191  dd = ux*ux + uy*uy;
192 
193  if ( dd < MMG5_EPSD ) {
194  fprintf(stderr,"\n ## Error: %s: Null edge length"
195  " (%e).\n",__func__,dd);
196  return 0;
197  }
198 
199  dd = 1.0 / sqrt(dd);
200  n2[0] = uy*dd;
201  n2[1] = -ux*dd;
202 
203  *pleft = 3*k+i1;
204 
205  /* Averaging of normal vectors */
206  nn[0] = n1[0] + n2[0];
207  nn[1] = n1[1] + n2[1];
208  dd = nn[0]*nn[0] + nn[1]*nn[1];
209  if ( dd > MMG5_EPSD ){
210  dd = 1.0 / sqrt(dd);
211  nn[0] *= dd;
212  nn[1] *= dd;
213  }
214 
215  return 1;
216 }

229 int MMG2D_boulet(MMG5_pMesh mesh,int start,int8_t ip,int *list) {
230  int *adja,k,ilist;
231  int8_t i,i1,i2;
232 
233  ilist = 0;
234 
235  /* store neighbors */
236  k = start;
237  i = ip;
238  do {
239  if ( ilist > MMG2D_LONMAX-2 ) return -ilist;
240  list[ilist] = 3*k + i;
241  ++ilist;
242 
243  adja = &mesh->adja[3*(k-1)+1];
244  i1 = MMG5_inxt2[i];
245  k = adja[i1] / 3;
246  i = adja[i1] % 3;
247  i = MMG5_inxt2[i];
248  }
249  while ( k && k != start );
250  if ( k > 0 ) return ilist;
251 
252  /* check if boundary hit */
253  k = start;
254  i = ip;
255  do {
256  adja = &mesh->adja[3*(k-1)+1];
257  i2 = MMG5_iprv2[i];
258  k = adja[i2] / 3;
259  if ( k == 0 ) break;
260  i = adja[i2] % 3;
261  i = MMG5_iprv2[i];
262 
263  if ( ilist > MMG2D_LONMAX-2 ) return -ilist;
264  list[ilist] = 3*k + i;
265  ilist++;
266  }
267  while ( k );
268 
269  return ilist;
270 }
