42 int MMG2D_removeBBtriangles(MMG5_pMesh mesh) {
43  MMG5_pTria pt;
44  int ip1,ip2,ip3,ip4,k,iadr,*adja,iadr2,*adja2,iel,nd;
45  int8_t i,ii;
46  static int8_t mmgWarn0=0;
47 
48  /* Bounding Box vertices */
49  ip1 = mesh->np-3;
50  ip2 = mesh->np-2;
51  ip3 = mesh->np-1;
52  ip4 = mesh->np;
53 
54  nd = 0;
55  for(k=1; k<=mesh->nt; k++) {
56  pt = &mesh->tria[k];
57  if ( !MG_EOK(pt) ) continue;
58 
59  if ( pt->base < 0 ) {
60  iadr = 3*(k-1) + 1;
61  adja = &mesh->adja[iadr];
62  for(i=0; i<3; i++) {
63  if ( !adja[i] ) continue;
64  iel = adja[i] / 3;
65  ii = adja[i] % 3;
66  iadr2 = 3*(iel-1) + 1;
67  adja2 = &mesh->adja[iadr2];
68  adja2 [ii] = 0;
69  }
70  MMG2D_delElt(mesh,k);
71  continue;
72  }
73  else if ( !pt->base ) {
74  if ( !mmgWarn0 ) {
75  mmgWarn0 = 1;
76  fprintf(stderr,"\n ## Warning: %s: at least 1 undetermined"
77  " triangle.\n",__func__);
78  }
79  nd++;
80  }
81  }
82 
83  if ( !nd ) {
84  MMG2D_delPt(mesh,ip1);
85  MMG2D_delPt(mesh,ip2);
86  MMG2D_delPt(mesh,ip3);
87  MMG2D_delPt(mesh,ip4);
88  }
89  else {
90  fprintf(stderr,"\n ## Error: %s: procedure failed :"
91  " %d indetermined triangles.\n",__func__,nd);
92  return 0;
93  }
94  return 1;
95 }

309 int MMG2D_markSD(MMG5_pMesh mesh) {

311  MMG5_pEdge ped;
312  MMG5_pPoint ppt;
313  int k,l,iadr,*adja,ped0,ped1,*list,ipil,ncurc,nref;
314  int kinit,nt,nsd,ip1,ip2,ip3,ip4,ned,iel,voy;
315  int8_t i,i1,i2;
316 
317  /* Reset flag field for triangles */
318  for(k=1 ; k<=mesh->nt ; k++)
319  mesh->tria[k].flag = mesh->mark;
320 
321  MMG5_SAFE_CALLOC(list,mesh->nt,int,return 0);
322  kinit = 0;
323  nref = 0;
324  ip1 = mesh->np;
325 
326  /* Catch first triangle with vertex ip1 */
327  for(k=1; k<=mesh->nt; k++) {
328  pt = &mesh->tria[k];
329  if ( !MG_EOK(pt) ) continue;
330  pt->flag = mesh->mark;
331  pt->ref = 0;
332  if ( (!kinit) && ( pt->v[0]==ip1 || pt->v[1]==ip1 || pt->v[2]==ip1) ) kinit = k;
333  }
334 
335  /* Travel mesh by adjacencies to set references on triangles as long as no boundary is met */
336  do {
337  nref++;
338  list[0] = kinit;
339  ipil = 0;
340  ncurc = 0;
341  do {
342  k = list[ipil];
343  pt = &mesh->tria[k];
344  pt->ref = nref;
345  iadr = 3*(k-1) + 1;
346  adja = &mesh->adja[iadr];
347  for(i=0; i<3; i++) {
348  iel = adja[i] / 3;
349  pt1 = &mesh->tria[iel];
350 
351  if( !iel || pt1->ref == nref ) continue;
352 
353  i1 = MMG5_inxt2[i];
354  i2 = MMG5_iprv2[i];
355  ped0 = pt->v[i1];
356  ped1 = pt->v[i2];
357 
358  /* WARNING: exhaustive search among edges, to be optimized with a hashing structure */
359  for(l=1; l<=mesh->na; l++) {
360  ped = &mesh->edge[l];
361  if( ( ped->a == ped0 && ped->b == ped1 ) || ( ped->b == ped0 && ped->a == ped1 ) ) break;
362  }
363  if ( l <= mesh->na ) continue;
364 
365  pt1->ref = nref;
366  list[++ncurc] = iel;
367  }
368  ++ipil ;
369  }
370  while ( ipil <= ncurc );
371 
372  kinit = 0;
373  for(k=1; k<=mesh->nt; k++) {
374  pt = &mesh->tria[k];
375  if ( !MG_EOK(pt) ) continue;
376  pt->flag = mesh->mark;
377  if ( !kinit && !(mesh->tria[k].ref) ) kinit = k;
378  }
379  }
380  while ( kinit );
381 
382  if ( mesh->info.imprim > 0 ) {
383  /* nref - 1 subdomains because Bounding Box triangles have been counted */
384  fprintf(stdout," %8d sub-domains\n",nref-1);
385  }
386 
387  MMG5_SAFE_FREE(list);
388 
389  /* Remove BB triangles and vertices */
390  /*BB vertex*/
391  ip1=(mesh->np-3);
392  ip2=(mesh->np-2);
393  ip3=(mesh->np-1);
394  ip4=(mesh->np);
395 
396  /* Case when there are inner and outer triangles */
397  if ( nref != 1 ) {
398  nt = mesh->nt;
399  for(k=1; k<=nt; k++) {
400  pt = &mesh->tria[k];
401  if ( !MG_EOK(pt) ) continue;
402 
403  if ( pt->ref != 1 ) continue;
404  /*update adjacencies*/
405  iadr = 3*(k-1)+1;
406  adja = &mesh->adja[iadr];
407  for(i=0; i<3; i++) {
408  if ( !adja[i] ) continue;
409  iel = adja[i] / 3;
410  voy = adja[i] % 3;
411  (&mesh->adja[3*(iel-1)+1])[voy] = 0;
412  }
413  MMG2D_delElt(mesh,k);
414  }
415  }
416  /* Remove only the triangles containing one of the BB vertex*/
417  else {
418  nt = mesh->nt;
419  for(k=1 ; k<=nt ; k++) {
420  pt = &mesh->tria[k];
421  if ( !MG_EOK(pt) ) continue;
422 
423  if( !(pt->v[0]==ip1 || pt->v[1]==ip1 || pt->v[2]==ip1 ||
424  pt->v[0]==ip2 || pt->v[1]==ip2 || pt->v[2]==ip2 ||
425  pt->v[0]==ip3 || pt->v[1]==ip3 || pt->v[2]==ip3 ||
426  pt->v[0]==ip4 || pt->v[1]==ip4 || pt->v[2]==ip4 ) ) continue;
427 
428  /*update adjacencies*/
429  iadr = 3*(k-1)+1;
430  adja = &mesh->adja[iadr];
431  for(i=0 ; i<3 ; i++) {
432  if(!adja[i]) continue;
433  iel = adja[i]/3;
434  voy = adja[i]%3;
435  (&mesh->adja[3*(iel-1)+1])[voy] = 0;
436  }
437  MMG2D_delElt(mesh,k);
438  }
439  }
440 
441  MMG2D_delPt(mesh,ip1);
442  MMG2D_delPt(mesh,ip2);
443  MMG2D_delPt(mesh,ip3);
444  MMG2D_delPt(mesh,ip4);
445 
446  if(mesh->info.nsd) {
447  nsd = mesh->info.nsd;
448  nt = mesh->nt;
449  for(k=1 ; k<=nt ; k++) {
450  pt = &mesh->tria[k];
451  if ( !MG_EOK(pt) ) continue;
452  pt->ref--;
453  if ( pt->ref == nsd ) continue;
454  MMG2D_delElt(mesh,k);
455  }
456  }
457 
458  /* Remove vertex*/
459  for (k=1; k<=mesh->nt; k++) {
460  pt = &mesh->tria[k];
461  if ( !MG_EOK(pt) ) continue;
462  for (i=0; i<3; i++) {
463  ppt = &mesh->point[ pt->v[i] ];
464  ppt->tag &= ~MG_NUL;
465  }
466  }
467  /* Remove edge*/
468  ned = mesh->na;
469  for (k=1; k<=ned; k++) {
470  ped = &mesh->edge[k];
471  if ( !ped->a ) continue;
472  ppt = &mesh->point[ ped->a ];
473  if ( !MG_VOK(ppt) ) {
474  MMG5_delEdge(mesh,k);
475  continue;
476  }
477  ppt = &mesh->point[ ped->b ];
478  if ( !MG_VOK(ppt) ) {
479  MMG5_delEdge(mesh,k);
480  continue;
481  }
482  }
483 
484  return 1;
485 }
