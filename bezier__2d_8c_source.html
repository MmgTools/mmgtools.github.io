27 /* Check if triangle k should be split based on geometric and rough edge length considerations */
28 int MMG2D_chkedg(MMG5_pMesh mesh, int k) {
29  MMG5_pTria pt;
30  MMG5_pPoint p1,p2;
31  double hausd,hmax,ps,cosn,ux,uy,ll,li,t1[2],t2[2];
32  int8_t i,i1,i2;
33 
34  pt = &mesh->tria[k];
35  hausd = mesh->info.hausd;
36  hmax = mesh->info.hmax;
37 
38  /* Analyze the three edges of k */
39  for (i=0; i<3; i++) {
40  i1 = MMG5_inxt2[i];
41  i2 = MMG5_iprv2[i];
42 
43  p1 = &mesh->point[pt->v[i1]];
44  p2 = &mesh->point[pt->v[i2]];
45 
46  /* Check the lengths of the edges */
47  ux = p2->c[0] - p1->c[0];
48  uy = p2->c[1] - p1->c[1];
49  ll = ux*ux + uy*uy;
50 
51  /* Long edges should be split */
52  if ( ll > hmax*hmax ) {
53  MG_SET(pt->flag,i);
54  continue;
55  }
56  /* Do not split very short edges */
57  else if ( ll < MMG5_EPSD ) continue;
58 
59  /* Split non geometric edges connecting two parts of the border */
60  else if ( mesh->info.fem &&
61  ( (!MG_EDG(pt->tag[i])) && (p1->tag > MG_NOTAG) && (p2->tag > MG_NOTAG)) ) {
62  MG_SET(pt->flag,i);
63  continue;
64  }
65 
66  /* Test the distance with respect to the continuous support in the case of a geometric edge */
67  if ( !MG_EDG(pt->tag[i]) ) continue;
68 
69  /* Collect tangent vectors at both endpoints; remark t1 and t2 need not be oriented in the same fashion */
70  if ( (MG_CRN & p1->tag) || (p1->tag & MG_NOM) ) {
71  li = 1.0 / sqrt(ll);
72  t1[0] = li*ux;
73  t1[1] = li*uy;
74  }
75  else {
76  t1[0] = -p1->n[1];
77  t1[1] = p1->n[0];
78  }
79 
80  if ( (MG_CRN & p2->tag) || (p2->tag & MG_NOM) ) {
81  li = 1.0 / sqrt(ll);
82  t2[0] = li*ux;
83  t2[1] = li*uy;
84  }
85  else {
86  t2[0] = -p2->n[1];
87  t2[1] = p2->n[0];
88  }
89 
90  /* Evaluate Hausdorff distance with respect to the geometric support */
91  ps = t1[0]*ux + t1[1]*uy;
92  ps *= ps;
93  cosn = ps/ll ;
94  cosn *= (1.0-cosn);
95  cosn *= ll;
96  if ( cosn > 9.0*hausd*hausd ) { // Not so sure about that 9.0
97  MG_SET(pt->flag,i);
98  continue;
99  }
100 
101  ps = -(t2[0]*ux + t2[1]*uy);
102  ps *= ps;
103  cosn = ps/ll ;
104  cosn *= (1.0-cosn);
105  cosn *= ll;
106  if ( cosn > 9.0*hausd*hausd ) {
107  MG_SET(pt->flag,i);
108  continue;
109  }
110  }
111 
112  return pt->flag;
113 }

116 /* Calculate coordinates o[2] and interpolated normal vector no[2] of a new point
117  situated at parametric distance s from i1 = inxt2[i] */
118 int MMG2D_bezierCurv(MMG5_pMesh mesh,int k,int8_t i,double s,double *o,double *no) {
119  MMG5_pTria pt;
120  MMG5_pPoint p1,p2;
121  double b1[2],b2[2],t1[2],t2[2],n1[2],n2[2],bn[2],ux,uy,ll,li,ps;
122  int8_t i1,i2;
123 
124  pt = &mesh->tria[k];
125  if ( !MG_EOK(pt) ) return 0;
126 
127  i1 = MMG5_inxt2[i];
128  i2 = MMG5_iprv2[i];
129  p1 = &mesh->point[pt->v[i1]];
130  p2 = &mesh->point[pt->v[i2]];
131 
132  /* When the edge is not geometric, simply take the midpoint */
133  if ( !MG_EDG(pt->tag[i]) ) {
134  o[0] = (1-s)*p1->c[0]+s*p2->c[0];
135  o[1] = (1-s)*p1->c[1]+s*p2->c[1];
136  memset(no,0,2*sizeof(double));
137  return 1;
138  }
139 
140  ux = p2->c[0] - p1->c[0];
141  uy = p2->c[1] - p1->c[1];
142  ll = ux*ux + uy*uy;
143 
144  if ( ll < MMG5_EPSD ) return 0;
145 
146  /* Recover normal and tangent vectors */
147  if ( (MG_CRN & p1->tag) || (p1->tag & MG_NOM) ) {
148  li = 1.0 / sqrt(ll);
149  t1[0] = li*ux;
150  t1[1] = li*uy;
151 
152  n1[0] = t1[1];
153  n1[1] = - t1[0];
154  }
155  else {
156  n1[0] = p1->n[0];
157  n1[1] = p1->n[1];
158 
159  t1[0] = -p1->n[1];
160  t1[1] = p1->n[0];
161  }
162 
163  if ( (MG_CRN & p2->tag) || (p2->tag & MG_NOM) ) {
164  li = 1.0 / sqrt(ll);
165  t2[0] = li*ux;
166  t2[1] = li*uy;
167 
168  n2[0] = t2[1];
169  n2[1] = - t2[0];
170  }
171  else {
172  n2[0] = p2->n[0];
173  n2[1] = p2->n[1];
174 
175  t2[0] = -p2->n[1];
176  t2[1] = p2->n[0];
177  }
178 
179  /* When either p1 or p2 is singular, make orientation of both normal vectors consistent
180  (otherwise, it is already the case) */
181  if ( (MG_CRN & p1->tag) || (p1->tag & MG_NOM) ){
182  ps = n1[0]*n2[0] + n1[1]*n2[1];
183  if ( ps < 0.0 ) {
184  n1[0] *= -1.0;
185  n1[1] *= -1.0;
186  }
187  }
188  else if ( (MG_CRN & p2->tag) || (p2->tag & MG_NOM) ) {
189  ps = n1[0]*n2[0] + n1[1]*n2[1];
190  if ( ps < 0.0 ) {
191  n2[0] *= -1.0;
192  n2[1] *= -1.0;
193  }
194  }
195 
196  /* Calculation of control points */
197  ps = ux*t1[0]+uy*t1[1];
198  b1[0] = p1->c[0] + MMG5_ATHIRD*ps*t1[0];
199  b1[1] = p1->c[1] + MMG5_ATHIRD*ps*t1[1];
200 
201  ps = ux*t2[0]+uy*t2[1];
202  b2[0] = p2->c[0] - MMG5_ATHIRD*ps*t2[0];
203  b2[1] = p2->c[1] - MMG5_ATHIRD*ps*t2[1];
204 
205  ps = ux*(n1[0]+n2[0]) + uy*(n1[1]+n2[1]);
206  ps = 2.0*ps/ll;
207 
208  bn[0] = n1[0]+n2[0] - ps*ux;
209  bn[1] = n1[1]+n2[1] - ps*uy;
210  ps = bn[0]*bn[0] + bn[1]*bn[1];
211  if ( ps > MMG5_EPSD2 ) {
212  ps = 1.0 / sqrt(ps);
213  bn[0] *= ps;
214  bn[1] *= ps;
215  }
216 
217  /* Interpolation of the position and normal vector */
218  o[0] = (1.0-s)*(1.0-s)*(1.0-s)*p1->c[0] + 3.0*(1.0-s)*(1.0-s)*s*b1[0] + 3.0*(1.0-s)*s*s*b2[0] + s*s*s*p2->c[0];
219  o[1] = (1.0-s)*(1.0-s)*(1.0-s)*p1->c[1] + 3.0*(1.0-s)*(1.0-s)*s*b1[1] + 3.0*(1.0-s)*s*s*b2[1] + s*s*s*p2->c[1];
220 
221  no[0] = (1.0-s)*(1.0-s)*n1[0] + 2*(1.0-s)*s*bn[0] + s*s*n2[0];
222  no[1] = (1.0-s)*(1.0-s)*n1[1] + 2*(1.0-s)*s*bn[1] + s*s*n2[1];
223 
224  return 1;
225 }
