63 static int MMG5_parsop(MMG5_pMesh mesh,MMG5_pSol met) {
64  float fp1,fp2,hausd;
65  int ref,i,j,ret,npar;
66  char *ptr,buf[256],data[256];
67  FILE *in;
68 
69  /* check for parameter file */
70  strcpy(data,mesh->namein);
71 
72  ptr = MMG5_Get_filenameExt(data);
73 
74  if ( ptr ) *ptr = '\0';
75  strcat(data,".mmgs");
76 
77  in = fopen(data,"rb");
78  if ( !in ) {
79  sprintf(data,"%s","DEFAULT.mmgs");
80  in = fopen(data,"rb");
81  if ( !in ) return 1;
82  }
83  fprintf(stdout,"\n %%%% %s OPENED\n",data);
84 
85  /* read parameters */
86  mesh->info.npar = 0;
87  while ( !feof(in) ) {
88  /* scan line */
89  ret = fscanf(in,"%255s",data);
90  if ( !ret || feof(in) ) break;
91  for (i=0; i<strlen(data); i++) data[i] = tolower(data[i]);
92 
93  /* check for condition type */
94  if ( !strcmp(data,"parameters") ) {
95  MMG_FSCANF(in,"%d",&npar);
96 
97  if ( !MMGS_Set_iparameter(mesh,met,MMGS_IPARAM_numberOfLocalParam,npar) )
98  return 0;
99 
100  for (i=0; i<mesh->info.npar; i++) {
101  MMG_FSCANF(in,"%d %255s ",&ref,buf);
102  ret = fscanf(in,"%f %f %f",&fp1,&fp2,&hausd);
103 
104  if ( !ret ) {
105  fprintf(stderr," %%%% Wrong format: %s\n",buf);
106  return 0;
107  }
108 
109  for (j=0; j<strlen(buf); j++) buf[j] = tolower(buf[j]);
110 
111  if ( !strcmp(buf,"triangles") || !strcmp(buf,"triangle") ) {
112  if ( !MMGS_Set_localParameter(mesh,met,MMG5_Triangle,ref,fp1,fp2,hausd) ) {
113  return 0;
114  }
115  }
116  else {
117  fprintf(stdout," %%%% Wrong format: %s\n",buf);
118  return 0;
119  }
120  }
121  }
122  }
123  fclose(in);
124  return 1;
125 }

219

223

227  met->np = 0;

236  chrono(OFF,&(ctim[1]));
237  printim(ctim[1].gdif,stim);

242  chrono(ON,&(ctim[2]));

244 
245  if ( mesh->info.imprim > 0 ) {
246  fprintf(stdout,"\n %s\n MODULE MMGS: IMB-LJLL : %s (%s)\n %s\n",

248  fprintf(stdout,"\n -- DEFAULT PARAMETERS COMPUTATION\n");

262

265  if ( !MMG5_unscaleMesh(mesh,met,sol) ) _LIBMMG5_RETURN(mesh,met,sol,MMG5_STRONGFAILURE);
266  _LIBMMG5_RETURN(mesh,met,sol,MMG5_STRONGFAILURE);
267  }
268  }
269 
270  /* unscaling mesh */
271  if ( !MMG5_unscaleMesh(mesh,met,sol) ) _LIBMMG5_RETURN(mesh,met,sol,MMG5_STRONGFAILURE);
272 
273  /* Save the local parameters file */
274  mesh->mark = 0;
275  if ( !MMGS_writeLocalParam(mesh) ) {
276  fprintf(stderr,"\n ## Error: %s: unable to save the local parameters file.\n"
277  " Exit program.\n",__func__);
278  _LIBMMG5_RETURN(mesh,met,sol,MMG5_LOWFAILURE);
279  }
280 
281  _LIBMMG5_RETURN(mesh,met,sol,MMG5_SUCCESS);
282 }

344

407 
408  chrono(OFF,&MMG5_ctim[1]);
409 
410  if ( mesh->info.imprim >= 0 ) {

419  }

447  break;
448  case ( MMG5_FMT_VtkVtu ):
449  ierSave = MMGS_saveVtuMesh(mesh,met,mesh->nameout);
450  break;
451  case ( MMG5_FMT_VtkVtk ):
452  ierSave = MMGS_saveVtkMesh(mesh,met,mesh->nameout);
453  break;
454  default:
455  ierSave = MMGS_saveMesh(mesh,mesh->nameout);
456  if ( !ierSave ) {
457  MMGS_RETURN_AND_FREE(mesh,met,ls,MMG5_STRONGFAILURE);
458  }
459  if ( met && met->np ) {
460  ierSave = MMGS_saveSol(mesh,met,met->nameout);

462  break;
463  }
464 
465  if ( !ierSave )

467 
468  chrono(OFF,&MMG5_ctim[1]);
469  if ( mesh->info.imprim > 0 ) fprintf(stdout," -- WRITING COMPLETED\n");
470  }
471 
472  /* release memory */
473  /* free mem */
474  MMGS_RETURN_AND_FREE(mesh,met,ls,ier);
475 
476  return 0;
