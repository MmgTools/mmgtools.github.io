87 /* args of InitMesh:
88 * MMG5_ARG_start: we start to give the args of a variadic func
89 * MMG5_ARG_ppMesh: next arg will be a pointer over a MMG5_pMesh
90 * &mmgMesh: pointer toward your MMG5_pMesh (that store your mesh)
91 * MMG5_ARG_ppLs: next arg will be a pointer over a MMG5_pSol storing a level-set

94 mmgMesh = NULL;
95 mmgLs = NULL;

101 /* Ask for level set discretization: note that it is important to do this step
102 * here because in iso mode, some filters are applied at mesh loading */
103 if ( MMG2D_Set_iparameter(mmgMesh,NULL,MMG2D_IPARAM_iso, 1) != 1 )
104 exit(EXIT_FAILURE);
105
106 /* Ask for optim mode: compute the mean of input edge lengths */
107 if ( MMG2D_Set_iparameter(mmgMesh,NULL,MMG2D_IPARAM_optim, 1) != 1 )
108 exit(EXIT_FAILURE);
109
110 /* Ask to do this with anisotropic metric */
111 if ( MMG2D_Set_iparameter(mmgMesh,NULL,MMG2D_IPARAM_anisosize, 1) != 1 )
112 exit(EXIT_FAILURE);
113
119 if ( MMG2D_loadMesh(mmgMesh,inname) != 1 ) exit(EXIT_FAILURE);
120
125 if ( MMG2D_loadSol(mmgMesh,mmgLs,lsname) != 1 )
126 exit(EXIT_FAILURE);
127
129 if ( MMG2D_Chk_meshData(mmgMesh,mmgLs) != 1 ) exit(EXIT_FAILURE);
130
140 ier = MMG2D_mmg2dls(mmgMesh,mmgLs,NULL);
141
142 if ( ier == MMG5_STRONGFAILURE ) {
143 fprintf(stdout,"BAD ENDING OF MMG2DLS: UNABLE TO SAVE MESH\n");
144 return(ier);
145 } else if ( ier == MMG5_LOWFAILURE )
146 fprintf(stdout,"BAD ENDING OF MMG2DLS\n");
147
148 /* (Not mandatory) Automatically save the mesh */
149 if ( MMG2D_saveMesh(mmgMesh,outname) != 1 )
150 exit(EXIT_FAILURE);
151
152 /* 9) free the MMG2D5 structures */
153 MMG2D_Free_all(MMG5_ARG_start,
154 MMG5_ARG_ppMesh,&mmgMesh,MMG5_ARG_ppLs,&mmgLs,
155 MMG5_ARG_end);
156
157 free(inname);
158 inname = NULL;
159
160 free(outname);
161 outname = NULL;
162
163 free(lsname);
164 outname = NULL;
165
166 return(ier);
