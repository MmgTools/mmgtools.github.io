64

66  ii = adja[i] % 3;
67  jj = MMG5_inxt2[ii];

75  hausd = mesh->info.hausd;
76  isloc = 0;
77  for (l=0; l<mesh->info.npar; l++) {
78  par = &mesh->info.par[l];
79  if ( ((par->elt == MMG5_Triangle) &&
80  ( (pt->ref == par->ref) || (pt1->ref == par->ref) )) /* ||
81  ( (par->elt == MMG5_Vertex) &&
82  ((p[1]->ref == par->ref ) || (p[2]->ref == par->ref)) ) */ ) {
83  if ( !isloc ) {
84  hausd = par->hausd;
85  isloc = 1;
86  }
87  else {
88  // take the minimum value between the two local hausdorff number asked
89  // by the user.
90  hausd = MG_MIN(hausd,par->hausd);
91  }
92  }
93  }
94 
95  /* check length */
96  if ( typchk == 2 && met->m ) {
97  loni = MMG5_lenSurfEdg(mesh,met,ip1,ip2,0);
98  lona = MMG5_lenSurfEdg(mesh,met,ip0,iq,0);
99  if ( loni > 1.0 ) loni = MG_MIN(1.0 / loni,MMGS_LSHRT);
100  if ( lona > 1.0 ) lona = 1.0 / lona;
101  if ( lona < loni || !loni ) return 0;
102  }
103 
104  /* check non convexity */
105  MMG5_norpts(mesh,ip0,ip1,iq,c1);
106  MMG5_norpts(mesh,ip0,iq,ip2,c2);
107  ps = c1[0]*c2[0] + c1[1]*c2[1] + c1[2]*c2[2];
108  if ( ps < MMG5_ANGEDG ) return 0;
109 
110  /* normal recovery at points p[0],p[1],p[2],q */
111  for (j=0; j<3; j++) {
112  if ( MS_SIN(p[j]->tag) ) {
113  MMG5_nortri(mesh,pt,np[j]);
114  }

123  else

125  }

132  }

136  nr2 = &mesh->xpoint[q->xp].n2[0];
137  ps = nr1[0]*nt[0] + nr1[1]*nt[1] + nr1[2]*nt[2];
138  ps2 = nr2[0]*nt[0] + nr2[1]*nt[1] + nr2[2]*nt[2];
139  if ( fabs(ps) > fabs(ps2) )
140  memcpy(&nq,nr1,3*sizeof(double));
141  else
142  memcpy(&nq,nr2,3*sizeof(double));
143  }
144  else
145  memcpy(&nq,q->n,3*sizeof(double));

147  /* Estimate of the Hausdorff distance between approximation and underlying surface
148  when using the 'natural' edge [i1,i2] */
149  ux = p[2]->c[0] - p[1]->c[0];
150  uy = p[2]->c[1] - p[1]->c[1];
151  uz = p[2]->c[2] - p[1]->c[2];
152 
153  ll = ux*ux + uy*uy + uz*uz;
154  if ( ll < MMG5_EPS ) return 0; /* no change for short edge */
155 
156  n1 = np[1];
157  n2 = np[2];
158 
159  ps = ux*n1[0] + uy*n1[1] + uz*n1[2];
160  c1[0] = (2.0*p[1]->c[0] + p[2]->c[0] - ps*n1[0]) / 3.0 - p[1]->c[0];
161  c1[1] = (2.0*p[1]->c[1] + p[2]->c[1] - ps*n1[1]) / 3.0 - p[1]->c[1];
162  c1[2] = (2.0*p[1]->c[2] + p[2]->c[2] - ps*n1[2]) / 3.0 - p[1]->c[2];
163 
164  ps = -(ux*n2[0] + uy*n2[1] + uz*n2[2]);
165  c2[0] = (2.0*p[2]->c[0] + p[1]->c[0] - ps*n2[0]) / 3.0 - p[2]->c[0];
166  c2[1] = (2.0*p[2]->c[1] + p[1]->c[1] - ps*n2[1]) / 3.0 - p[2]->c[1];
167  c2[2] = (2.0*p[2]->c[2] + p[1]->c[2] - ps*n2[2]) / 3.0 - p[2]->c[2];
168 
169  /* squared cosines */
170  ps = c1[0]*ux + c1[1]*uy + c1[2]*uz;
171  ps *= ps;
172  dd = c1[0]*c1[0] + c1[1]*c1[1] + c1[2]*c1[2];
173  cosn1 = ps / (dd*ll);
174  cosn1 *= (1.0-cosn1);
175  cosn1 *= (0.25*ll);
176 
177  ps = -c2[0]*ux - c2[1]*uy - c2[2]*uz;
178  ps *= ps;
179  dd = c2[0]*c2[0]+c2[1]*c2[1]+c2[2]*c2[2];
180  cosn2 = ps / (dd*ll);
181  cosn2 *= (1.0-cosn2);
182  cosn2 *= (0.25*ll);
183 
184  cosnat = MG_MAX(fabs(cosn1),fabs(cosn2));
185  cosnat = cosnat < MMG5_EPS ? 0.0 : cosnat;
186 
187  /* Estimate of the Hausdorff distance between approximation and underlying surface
188  when using the 'swapped' edge [i0,q] */
189  ux = q->c[0] - p[0]->c[0];
190  uy = q->c[1] - p[0]->c[1];
191  uz = q->c[2] - p[0]->c[2];
192 
193  ll = ux*ux + uy*uy + uz*uz;
194  if ( ll < MMG5_EPS ) return 0;
195 
196  n1 = np[0];
197  n2 = nq;
198 
199  ps = ux*n1[0] + uy*n1[1] + uz*n1[2];
200  c1[0] = (2.0*p[0]->c[0] + q->c[0] - ps*n1[0]) / 3.0 - p[0]->c[0];
201  c1[1] = (2.0*p[0]->c[1] + q->c[1] - ps*n1[1]) / 3.0 - p[0]->c[1];
202  c1[2] = (2.0*p[0]->c[2] + q->c[2] - ps*n1[2]) / 3.0 - p[0]->c[2];

205  c2[0] = (2.0*q->c[0] + p[0]->c[0] - ps*n2[0]) / 3.0 - q->c[0];
206  c2[1] = (2.0*q->c[1] + p[0]->c[1] - ps*n2[1]) / 3.0 - q->c[1];
207  c2[2] = (2.0*q->c[2] + p[0]->c[2] - ps*n2[2]) / 3.0 - q->c[2];
208 
209  /* squared cosines */
210  ps = c1[0]*ux + c1[1]*uy + c1[2]*uz;
211  ps *= ps;
212  dd = c1[0]*c1[0] + c1[1]*c1[1] + c1[2]*c1[2];
213  cosn1 = ps / (dd*ll);
214  cosn1 *= (1.0-cosn1);
215  cosn1 *= (0.25*ll);
216 
217  ps = -c2[0]*ux - c2[1]*uy - c2[2]*uz;
218  ps *= ps;
219  dd = c2[0]*c2[0]+c2[1]*c2[1]+c2[2]*c2[2];
220  cosn2 = ps / (dd*ll);
221  cosn2 *= (1.0-cosn2);
222  cosn2 *= (0.25*ll);
223 
224  coschg = MG_MAX(fabs(cosn1),fabs(cosn2));
225  coschg = coschg < MMG5_EPS ? 0.0 : coschg;
226 
227  /* swap if Hausdorff contribution of the swapped edge is less than existing one */
228  if ( coschg > hausd*hausd ) return 0;
229 
230  if ( typchk == 2 && met->m ) {
231  /* initial quality */
232  pt0->v[0]= ip0; pt0->v[1]= ip1; pt0->v[2]= ip2;
233  pt0->tag[0] = pt->tag[i];

296 int swapar(MMG5_pMesh mesh,int k,int i) {
297  MMG5_pTria pt,pt1;
298  int *adja,adj,k11,k21,ip1,ip2,i2save,j2save;
299  int8_t i1,i2,j,jj,j2,v11,v21;
300 
301  pt = &mesh->tria[k];
302  if ( MG_EDG(pt->tag[i]) || MS_SIN(pt->tag[i]) ) return 0;
303 
304  adja = &mesh->adja[3*(k-1)+1];
305  assert(adja[i]);
306 
307  adj = adja[i] / 3;
308  j = adja[i] % 3;
309  pt1 = &mesh->tria[adj];
310 
311  /* simulation */
312  i1 = MMG5_inxt2[i];
313  i2 = MMG5_iprv2[i];
314 
315  /* update structure */
316  k11 = adja[i1] / 3;
317  v11 = adja[i1] % 3;
318  if ( k11 < 1 ) return 0;
319  ip1 = mesh->tria[k11].v[v11];
320 
321  adja = &mesh->adja[3*(adj-1)+1];
322  jj = MMG5_inxt2[j];
323  j2 = MMG5_iprv2[j];
324  k21 = adja[jj] / 3;
325  v21 = adja[jj] % 3;
326  if ( k21 < 1 ) return 0;
327  ip2 = mesh->tria[k21].v[v21];
328 
329  i2save = pt->v[i2];
330  pt->v[i2] = pt1->v[j];
331  j2save = pt1->v[j2];
332  pt1->v[j2] = pt->v[i];
333 
334  /* Check that the edge swap doesn't create a lost face. Revert the swap in
335  * this case */
336  if ( pt->v[i] == ip2 ) {
337  pt->v[i2] = i2save;
338  pt1->v[j2] = j2save;
339  return 0;
340  }
341 
342  if ( pt1->v[j] == ip1 ) {
343  pt->v[i2] = i2save;
344  pt1->v[j2] = j2save;
345  return 0;
346  }
347 
348 
349  /* update info */
350  pt->tag[i] = pt1->tag[jj];
351  pt->edg[i] = pt1->edg[jj];
352  pt->base = mesh->base;
353  pt1->tag[j] = pt->tag[i1];
354  pt1->edg[j] = pt->edg[i1];
355  pt->tag[i1] = 0;
356  pt->edg[i1] = 0;
357  pt1->tag[jj] = 0;
358  pt1->edg[jj] = 0;
359  pt1->base = mesh->base;
360 
361  /* update adjacent */
362  mesh->adja[3*(k-1)+1+i] = 3*k21+v21;
363  mesh->adja[3*(k21-1)+1+v21] = 3*k+i;
364  mesh->adja[3*(k-1)+1+i1] = 3*adj+jj;
365  mesh->adja[3*(adj-1)+1+jj] = 3*k+i1;
366  mesh->adja[3*(k11-1)+1+v11] = 3*adj+j;
367  mesh->adja[3*(adj-1)+1+j] = 3*k11+v11;
368 
369  return 1;
370 }

385  i2 = MMG5_iprv2[i];

390  adja = &mesh->adja[3*(k-1)+1];

410  kalf = MMG5_calelt(mesh,NULL,pt0);
411  kalf = MG_MIN(kalf,kalt);
