193 int MMG5_countLocalParamAtTri( MMG5_pMesh mesh,MMG5_iNode **bdryRefs) {
194  int npar,k,ier;
195 
197  (*bdryRefs) = NULL;
198 
199  k = mesh->nt? mesh->tria[1].ref : 0;
200 
201  /* Try to alloc the first node */
202  ier = MMG5_Add_inode( mesh, bdryRefs, k );
203  if ( ier < 0 ) {
204  fprintf(stderr,"\n ## Error: %s: unable to allocate the first boundary"
205  " reference node.\n",__func__);
206  return 0;
207  }
208  else {
209  assert(ier);
210  npar = 1;
211  }
212 
213  for ( k=1; k<=mesh->nt; ++k ) {
214  ier = MMG5_Add_inode( mesh, bdryRefs, mesh->tria[k].ref );
215 
216  if ( ier < 0 ) {
217  printf(" ## Warning: %s: unable to list the tria references."
218  " Uncomplete parameters file.\n",__func__ );
219  break;
220  }
221  else if ( ier ) ++npar;
222  }
223 
224  return npar;
225 }

264 void MMG5_solTruncatureForOptim(MMG5_pMesh mesh, MMG5_pSol met) {
265  MMG5_pTetra pt;
266  MMG5_pPoint ppt;
267  double isqhmin, isqhmax;
268  int i,k,iadr,sethmin,sethmax;
269 
270  assert ( mesh->info.optim );
271 
272  /* Detect the point used only by prisms */
273  if ( mesh->nprism ) {
274  for (k=1; k<=mesh->np; k++) {
275  mesh->point[k].flag = 1;
276  }
277  for (k=1; k<=mesh->ne; k++) {
278  pt = &mesh->tetra[k];
279  if ( !MG_EOK(pt) ) continue;
280 
281  for (i=0; i<4; i++) {
282  mesh->point[pt->v[i]].flag = 0;
283  }
284  }
285  }
286 
287  /* If not provided by the user, compute hmin/hmax from the metric computed by
288  * the DoSol function. */
289  sethmin = sethmax = 1;
290  if ( mesh->info.hmin < 0 ) {
291  sethmin = 0;
292  if ( met->size == 1 ) {
293  mesh->info.hmin = FLT_MAX;
294  for (k=1; k<=mesh->np; k++) {
295  ppt = &mesh->point[k];
296  if ( !MG_VOK(ppt) || ppt->flag ) continue;
297  mesh->info.hmin = MG_MIN(mesh->info.hmin,met->m[k]);
298  }
299  }
300  else if ( met->size == 6 ){
301  mesh->info.hmin = 0.;
302  for (k=1; k<=mesh->np; k++) {
303  ppt = &mesh->point[k];
304  if ( !MG_VOK(ppt) || ppt->flag ) continue;
305  iadr = met->size*k;
306  mesh->info.hmin = MG_MAX(mesh->info.hmin,met->m[iadr]);
307  mesh->info.hmin = MG_MAX(mesh->info.hmin,met->m[iadr+3]);
308  mesh->info.hmin = MG_MAX(mesh->info.hmin,met->m[iadr+5]);
309  }
310  mesh->info.hmin = 1./sqrt(mesh->info.hmin);
311  }
312  }
313 
314  if ( mesh->info.hmax < 0 ) {
315  sethmax = 0;
316  if ( met->size == 1 ) {
317  mesh->info.hmax = 0.;
318  for (k=1; k<=mesh->np; k++) {
319  ppt = &mesh->point[k];
320  if ( !MG_VOK(ppt) || ppt->flag ) continue;
321  mesh->info.hmax = MG_MAX(mesh->info.hmax,met->m[k]);
322  }
323  }
324  else if ( met->size == 6 ){
325  mesh->info.hmax = FLT_MAX;
326  for (k=1; k<=mesh->np; k++) {
327  ppt = &mesh->point[k];
328  if ( !MG_VOK(ppt) || ppt->flag ) continue;
329  iadr = met->size*k;
330  mesh->info.hmax = MG_MIN(mesh->info.hmax,met->m[iadr]);
331  mesh->info.hmax = MG_MIN(mesh->info.hmax,met->m[iadr+3]);
332  mesh->info.hmax = MG_MIN(mesh->info.hmax,met->m[iadr+5]);
333  }
334  mesh->info.hmax = 1./sqrt(mesh->info.hmax);
335  }
336  }
337 
338  /* Check the compatibility between the user settings and the automatically
339  * computed values */
340  MMG5_check_hminhmax(mesh,sethmin,sethmax);
341 
342  /* vertex size */
343  if ( met->size == 1 ) {
344  for (k=1; k<=mesh->np; k++) {
345  ppt = &mesh->point[k];
346  if ( !MG_VOK(ppt) ) continue;
347  met->m[k] = MG_MIN(mesh->info.hmax,MG_MAX(mesh->info.hmin,met->m[k]));
348  }
349  }
350  else if ( met->size == 6 ) {
351  isqhmin = 1./(mesh->info.hmin*mesh->info.hmin);
352  isqhmax = 1./(mesh->info.hmax*mesh->info.hmax);
353 
354  for (k=1; k<=mesh->np; k++) {
355  ppt = &mesh->point[k];
356  if ( !MG_VOK(ppt) ) continue;
357  iadr = 6*k;
358  met->m[iadr] = MG_MAX(isqhmax,MG_MIN(isqhmin,met->m[iadr]));
359  met->m[iadr+3] = met->m[iadr];
360  met->m[iadr+5] = met->m[iadr];
361  }
362  }
363 
364  return;
365 }

455 char *MMG5_Remove_ext (char* path,char *ext) {
456  int len;
457  char *retpath, *lastext, *lastpath;
458  char *extloc;
459 
460  /* Default extension if not provided */
461  if ( (!ext) || !*ext ) {
462  extloc = ".";
463  }
464  else {
465  extloc = ext;
466  }
467 
468  /* Error checks and string allocation. */
469  if ( path == NULL) return NULL;
470 
471  /* Find the relevant characters and the length of the string without
472  * extension */
473  lastext = strstr (path, extloc);
474  lastpath = (MMG5_PATHSEP == 0) ? NULL : strrchr (path, MMG5_PATHSEP);
475 
476  if ( lastext == NULL || (lastpath != NULL && lastpath > lastext) ) {
477  /* No extension or the extension is left from a separator (i.e. it is not an
478  * extension) */
479  len = strlen(path);
480  }
481  else {
482  /* An extension is found */
483  len = 0;
484  while ( path+len != lastext ) {
485  ++len;
486  }
487  }
488 
489  MMG5_SAFE_MALLOC(retpath,len+1,char,return NULL);
490 
491  /* Copy the string without the extension and add \0 */
492  strncpy ( retpath, path, len );
493  retpath[len] = '\0';
494 
495  return retpath;
496 }
