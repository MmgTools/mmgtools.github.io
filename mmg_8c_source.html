193int MMG5_countLocalParamAtTri( MMG5_pMesh mesh,MMG5_iNode **bdryRefs) {
194 int npar,k,ier;
195
197 (*bdryRefs) = NULL;
198
199 k = mesh->nt? mesh->tria[1].ref : 0;
200
201 /* Try to alloc the first node */
202 ier = MMG5_Add_inode( mesh, bdryRefs, k );
203 if ( ier < 0 ) {
204 fprintf(stderr,"\n ## Error: %s: unable to allocate the first boundary"
205 " reference node.\n",__func__);
206 return 0;
207 }
208 else {
209 assert(ier);
210 npar = 1;
211 }
212
213 for ( k=1; k<=mesh->nt; ++k ) {
214 ier = MMG5_Add_inode( mesh, bdryRefs, mesh->tria[k].ref );
215
216 if ( ier < 0 ) {
217 printf(" ## Warning: %s: unable to list the tria references."
218 " Uncomplete parameters file.\n",__func__ );
219 break;
220 }
221 else if ( ier ) ++npar;
222 }
223
224 return npar;
225}

262char *MMG5_Get_filenameExt( char *filename ) {
263 const char pathsep='/';
264 char *dot,*lastpath;
265
266 if ( !filename ) {
267 return NULL;
268 }
269
270 dot = strrchr(filename, '.');
271 lastpath = (pathsep == 0) ? NULL : strrchr (filename, pathsep);
272
273 if ( (!dot) || dot == filename || (lastpath>dot) || (!strcmp(dot,".o")) ) {
274 /* No extension */
275 return filename + strlen(filename);
276 }
277
278 return dot;
279}

306char *MMG5_Get_path(char *path) {
307 char *lastpath,*retpath;
308 int len;
309
310 if ( path == NULL) return NULL;
311
312 lastpath = (MMG5_PATHSEP == 0) ? NULL : strrchr (path, MMG5_PATHSEP);
313
314 if ( !lastpath ) {
315 return NULL;
316 }
317
318
319 len = 0;
320 while ( path+len != lastpath ) {
321 ++len;
322 }
323
324 MMG5_SAFE_MALLOC(retpath,len+1,char,return NULL);
325
326 /* Copy the string without the extension and add \0 */
327 strncpy ( retpath, path, len );
328 retpath[len] = '\0';
329
330 return retpath;
331}

341char *MMG5_Remove_ext (char* path,char *ext) {
342 int len;
343 char *retpath, *lastext, *lastpath;
344 char *extloc;
345
346 /* Default extension if not provided */
347 if ( (!ext) || !*ext ) {
348 extloc = ".";
349 }
350 else {
351 extloc = ext;
352 }
353
354 /* Error checks and string allocation. */
355 if ( path == NULL) return NULL;
356
357 /* Find the relevant characters and the length of the string without
358 * extension */
359 lastext = strstr (path, extloc);
360 lastpath = (MMG5_PATHSEP == 0) ? NULL : strrchr (path, MMG5_PATHSEP);
361
362 if ( lastext == NULL || (lastpath != NULL && lastpath > lastext) ) {
363 /* No extension or the extension is left from a separator (i.e. it is not an
364 * extension) */
365 len = strlen(path);
366 }
367 else {
368 /* An extension is found */
369 len = 0;
370 while ( path+len != lastext ) {
371 ++len;
372 }
373 }
374
375 MMG5_SAFE_MALLOC(retpath,len+1,char,return NULL);
376
377 /* Copy the string without the extension and add \0 */
378 strncpy ( retpath, path, len );
379 retpath[len] = '\0';
380
381 return retpath;
382}

395int MMG5_Get_format( char *ptr, int fmt ) {
396 /* Default is the Medit file format or a format given as input */
397 int defFmt = fmt;
398
399 if ( !ptr ) return defFmt;
400
401 if ( !strncmp ( ptr,".meshb",strlen(".meshb") ) ) {
402 return MMG5_FMT_MeditBinary;
403 }
404 else if ( !strncmp( ptr,".mesh",strlen(".mesh") ) ) {
405 return MMG5_FMT_MeditASCII;
406 }
407 else if ( !strncmp( ptr,".mshb",strlen(".mshb") ) ) {
408 return MMG5_FMT_GmshBinary;
409 }
410 else if ( !strncmp( ptr,".msh",strlen(".msh") ) ) {
411 return MMG5_FMT_GmshASCII;
412 }
413 else if ( !strncmp ( ptr,".pvtu",strlen(".pvtu") ) ) {
414 return MMG5_FMT_VtkPvtu;
415 }
416 else if ( !strncmp ( ptr,".vtu",strlen(".vtu") ) ) {
417 return MMG5_FMT_VtkVtu;
418 }
419 else if ( !strncmp ( ptr,".pvtp",strlen(".pvtu") ) ) {
420 return MMG5_FMT_VtkPvtp;
421 }
422 else if ( !strncmp ( ptr,".vtp",strlen(".vtp") ) ) {
423 return MMG5_FMT_VtkVtp;
424 }
425 else if ( !strncmp ( ptr,".vtk",strlen(".vtk") ) ) {
426 return MMG5_FMT_VtkVtk;
427 }
428 else if ( !strncmp ( ptr,".node",strlen(".node") ) ) {
429 return MMG5_FMT_Tetgen;
430 }
431
432 return defFmt;
433}
