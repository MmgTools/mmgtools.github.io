55 int MMG5_chkswpgen(MMG5_pMesh mesh,MMG5_pSol met,int start,int ia,
56  int *ilist,int *list,double crit,int8_t typchk) {
57  MMG5_pTetra pt,pt0;
58  MMG5_pPoint p0;
59  double calold,calnew,caltmp;
60  int na,nb,np,adj,piv,npol,refdom,k,l,iel;
61  int *adja,pol[MMG3D_LMAX+2];
62  int8_t i,ip,ier,ifac;
63 
64  pt = &mesh->tetra[start];
65  refdom = pt->ref;
66 
67  pt0 = &mesh->tetra[0];
68  na = pt->v[MMG5_iare[ia][0]];
69  nb = pt->v[MMG5_iare[ia][1]];
70  calold = pt->qual;
71 
72  /* Store shell of ia in list, and associated pseudo polygon in pol */
73  (*ilist) = 0;
74  npol = 0;
75  list[(*ilist)] = 6*start+ia;
76  (*ilist)++;
77  adja = &mesh->adja[4*(start-1)+1];
78  adj = adja[MMG5_ifar[ia][0]]; // start travelling by face (ia,0)
79  ifac = adj%4;
80  piv = pt->v[MMG5_ifar[ia][1]];
81  pol[npol] = 4*start + MMG5_ifar[ia][1];
82  npol++;
83 
84  /* Edge is on a boundary between two different domains */
85  if ( mesh->info.opnbdy )
86  if ( pt->xt && (mesh->xtetra[pt->xt].ftag[MMG5_ifar[ia][1]] & MG_BDY) )
87  return 0;
88 
89  while ( adj ) {
90  adj /= 4;
91  if ( adj ==start ) break;
92 
93  pt = &mesh->tetra[adj];
94  if ( pt->tag & MG_REQ ) return 0;
95 
96  /* Edge is on a boundary between two different domains */
97  if ( pt->ref != refdom ) return 0;
98  else if ( mesh->info.opnbdy ) {
99  if ( pt->xt && (mesh->xtetra[pt->xt].ftag[ifac] & MG_BDY) ) return 0;
100  }
101 
102  calold = MG_MIN(calold, pt->qual);
103 
104  /* identification of edge number in tetra adj */
105  if ( !MMG3D_findEdge(mesh,pt,adj,na,nb,1,NULL,&i) ) return -1;
106 
107  list[(*ilist)] = 6*adj +i;
108  (*ilist)++;
109  /* overflow */
110  if ( (*ilist) > MMG3D_LMAX-3 ) return 0;
111 
112  /* set new triangle for travel */
113  adja = &mesh->adja[4*(adj-1)+1];
114  if ( pt->v[ MMG5_ifar[i][0] ] == piv ) {
115  pol[npol] = 4*adj + MMG5_ifar[i][1];
116  npol++;
117  adj = adja[ MMG5_ifar[i][0] ];
118  piv = pt->v[ MMG5_ifar[i][1] ];
119  }
120  else {
121  assert(pt->v[ MMG5_ifar[i][1] ] == piv);
122  pol[npol] = 4*adj + MMG5_ifar[i][0];
123  npol++;
124  adj = adja[ MMG5_ifar[i][1] ];
125  piv = pt->v[ MMG5_ifar[i][0] ];
126  }
127  ifac = adj%4;
128  }
129 
130  //CECILE : je vois pas pourquoi ca ameliore de faire ce test
131  //plus rapide mais du coup on elimine des swap...
132  //4/01/14 commentaire
133  // if ( calold*MMG3D_ALPHAD > 0.5 ) return 0;
134 
135  /* Prevent swap of an external boundary edge */
136  if ( !adj ) return 0;
137 
138  assert(npol == (*ilist)); // du coup, apres on pourra virer npol
139 
140  /* Find a configuration that enhances the worst quality within the shell */
141  for (k=0; k<npol; k++) {
142  iel = pol[k] / 4;
143  ip = pol[k] % 4;
144  np = mesh->tetra[iel].v[ip];
145  calnew = 1.0;
146  ier = 1;
147 
148  if ( mesh->info.fem ) {
149  /* Do not create internal edges between boundary points */
150  p0 = &mesh->point[np];
151  if ( p0->tag & MG_BDY ) {
152  /* One of the vertices of the pseudo polygon is boundary */
153  for (l=0; l<npol;l++) {
154  if ( k < npol-1 ) {
155  /* Skip the two elts of the pseudo polygon that contains p0 */
156  if ( l == k || l == k+1 ) continue;
157  }
158  else {
159  /* Skip the two elts of the pseudo polygon that contains p0 (for k==npol-1) */
160  if ( l == npol-1 || l == 0 ) continue;
161  }
162  iel = pol[l] / 4;
163  ip = pol[l] % 4;
164  pt = &mesh->tetra[iel];
165  p0 = &mesh->point[pt->v[ip]];
166  if ( p0->tag & MG_BDY ) {
167  /* Another vertex is boundary */
168  ier = 0;
169  break;
170  }
171  }
172  }
173  if ( !ier ) continue;
174  ier = 1;
175  }
176 
177  for (l=0; l<(*ilist); l++) {
178  /* Do not consider tets of the shell of collapsed edge */
179  if ( k < npol-1 ) {
180  /* Skip the two elts of the pseudo polygon that contains np */
181  if ( l == k || l == k+1 ) continue;
182  }
183  else {
184  /* Skip the two elts of the pseudo polygon that contains np for the last polygon elt */
185  if ( l == npol-1 || l == 0 ) continue;
186  }
187  iel = list[l] / 6;
188  i = list[l] % 6;
189  pt = &mesh->tetra[iel];
190 
191  /* Check that we will not insert a node that we will fail to collapse
192  * (recreation of an existing element) */
193  adja = &mesh->adja[4*(iel-1)+1];
194  adj = adja[MMG5_iare[i][0]]/4;
195  piv = adja[MMG5_iare[i][0]]%4;
196  if ( adj && mesh->tetra[adj].v[piv]==np ) {
197  ier = 0;
198  break;
199  }
200  adj = adja[MMG5_iare[i][1]]/4;
201  piv = adja[MMG5_iare[i][1]]%4;
202  if ( adj && mesh->tetra[adj].v[piv]==np ) {
203  ier = 0;
204  break;
205  }
206 
207  /* Prevent from creating a tetra with 4 bdy vertices */
208  if ( mesh->point[np].tag & MG_BDY ) {
209  if ( ( mesh->point[pt->v[MMG5_ifar[i][0]]].tag & MG_BDY ) &&
210  ( mesh->point[pt->v[MMG5_ifar[i][1]]].tag & MG_BDY ) ) {
211  if ( ( mesh->point[pt->v[MMG5_iare[i][0]]].tag & MG_BDY ) ||
212  ( mesh->point[pt->v[MMG5_iare[i][1]]].tag & MG_BDY ) ) {
213  ier = 0;
214  break;
215  }
216  }
217  }
218 
219  /* First tetra obtained from iel */
220  memcpy(pt0,pt,sizeof(MMG5_Tetra));
221  pt0->v[MMG5_iare[i][0]] = np;
222 
223 
224  if ( typchk==1 && met->size > 1 && met->m )
225  caltmp = MMG5_caltet33_ani(mesh,met,pt0);
226  else
227  caltmp = MMG5_orcal(mesh,met,0);
228 
229  calnew = MG_MIN(calnew,caltmp);
230 
231  ier = (calnew > crit*calold);
232  if ( !ier ) break;
233 
234  /* Second tetra obtained from iel */
235  memcpy(pt0,pt,sizeof(MMG5_Tetra));
236  pt0->v[MMG5_iare[i][1]] = np;
237 
238  if ( typchk==1 && met->size > 1 && met->m )
239  caltmp = MMG5_caltet33_ani(mesh,met,pt0);
240  else
241  caltmp = MMG5_orcal(mesh,met,0);
242 
243  calnew = MG_MIN(calnew,caltmp);
244 
245  ier = (calnew > crit*calold);
246  if ( !ier ) break;
247  }
248  if ( ier ) return pol[k];
249  }
250  return 0;
251 }
