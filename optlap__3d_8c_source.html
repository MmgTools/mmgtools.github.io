44 int MMG3D_optlap(MMG5_pMesh mesh,MMG5_pSol sol) {

52  double LLAMBDA = 0.33;
53  double LMU = 0.331;
54

56  MMG5_ADD_MEM(mesh,(3*mesh->np+1)*sizeof(double),"nv",
57  return 0);
58  MMG5_ADD_MEM(mesh,(3*mesh->np+1)*sizeof(double),"pos",
59  return 0);

61  return 0);

70  /*initialisation*/

72  compt[i] = 0;
73  pos[3*(i-1) + 1 + 0] = 0.;
74  pos[3*(i-1) + 1 + 1] = 0.;

79  for(k = 1 ; k<=mesh->ne ; k++) {
80  pt = &mesh->tetra[k];

82 
83  for(i=0 ; i<6 ; i++) {
84  ipta = pt->v[MMG5_iare[i][0]];
85  ppta = &mesh->point[ipta];
86 
87  iptb = pt->v[MMG5_iare[i][1]];
88  pptb = &mesh->point[iptb];
89 
90  if ( !( (ppta->tag & MG_BDY) || (ppta->tag & MG_REQ) ) ) {
91  pos[3*(ipta-1) + 1 + 0] += pptb->c[0];
92  pos[3*(ipta-1) + 1 + 1] += pptb->c[1];
93  pos[3*(ipta-1) + 1 + 2] += pptb->c[2];
94  compt[ipta]++;
95  }
96  if ( !( (pptb->tag & MG_BDY) || (pptb->tag & MG_REQ) ) ) {
97  pos[3*(iptb-1) + 1 + 0] += ppta->c[0];
98  pos[3*(iptb-1) + 1 + 1] += ppta->c[1];
99  pos[3*(iptb-1) + 1 + 2] += ppta->c[2];
100  compt[iptb]++;
101  }
102  }
103  }
104 
105  for ( i=1 ; i<=mesh->np ; i++) {
106 
107  ppt = &mesh->point[i];
108  if ( !MG_VOK(ppt) ) {
109  assert ( !compt[i] );
110  continue;
111  }
112 
113  if ( compt[i]) {
114  dd = 1./(double) compt[i];
115  pos[3*(i-1) + 1 + 0] *= dd;
116  pos[3*(i-1) + 1 + 1] *= dd;
117  pos[3*(i-1) + 1 + 2] *= dd;
118  nv[3*(i-1) + 1] = ppt->c[0] + LLAMBDA * (ppt->c[0] - pos[3*(i-1) + 1 + 0]);
119  nv[3*(i-1) + 2] = ppt->c[1] + LLAMBDA * (ppt->c[1] - pos[3*(i-1) + 1 + 1]);
120  nv[3*(i-1) + 3] = ppt->c[2] + LLAMBDA * (ppt->c[2] - pos[3*(i-1) + 1 + 2]);
121  } else {
122  nv[3*(i-1) + 1] = ppt->c[0];
123  nv[3*(i-1) + 2] = ppt->c[1];
124  nv[3*(i-1) + 3] = ppt->c[2];
125 
126  }
127  compt[i] = 0;
128  pos[3*(i-1) + 1 + 0] = 0.;
129  pos[3*(i-1) + 1 + 1] = 0.;
130  pos[3*(i-1) + 1 + 2] = 0.;
131 
132  }
133 
134  /* 2nd stage : anti-laplacian */
135  for(k = 1 ; k<=mesh->ne ; k++) {
136  pt = &mesh->tetra[k];
137  if ( (!MG_EOK(pt)) || MG_REQ & pt->tag || pt->qual > declic ) continue;
138 
139  for(i=0 ; i<6 ; i++) {
140  ipta = pt->v[MMG5_iare[i][0]];
141  ppta = &mesh->point[ipta];
142 
143  iptb = pt->v[MMG5_iare[i][1]];
144  pptb = &mesh->point[iptb];
145 
146  if ( !( (ppta->tag & MG_BDY) || (ppta->tag & MG_REQ) ) ) {
147  pos[3*(ipta-1) + 1 + 0] += nv[3*(iptb-1) + 1];
148  pos[3*(ipta-1) + 1 + 1] += nv[3*(iptb-1) + 2];
149  pos[3*(ipta-1) + 1 + 2] += nv[3*(iptb-1) + 3];
150  compt[ipta]++;
151  }
152  if ( !( (pptb->tag & MG_BDY) || (pptb->tag & MG_REQ) ) ) {
153  pos[3*(iptb-1) + 1 + 0] += nv[3*(ipta-1) + 1];
154  pos[3*(iptb-1) + 1 + 1] += nv[3*(ipta-1) + 2];
155  pos[3*(iptb-1) + 1 + 2] += nv[3*(ipta-1) + 3];
156  compt[iptb]++;
157  }
158  }
159  }
160 
161  res= 0.;
162  for(i=1 ; i<=mesh->np ; i++) {
163  if ( compt[i] ) {
164  dd = 1./(double) compt[i];
165  pos[3*(i-1) + 1 + 0] *= dd;
166  pos[3*(i-1) + 1 + 1] *= dd;
167  pos[3*(i-1) + 1 + 2] *= dd;
168  ox = nv[3*(i-1) + 1];
169  oy = nv[3*(i-1) + 2];
170  oz = nv[3*(i-1) + 3];
171  nv[3*(i-1) + 1] = nv[3*(i-1) + 1] - LMU * (nv[3*(i-1) + 1] - pos[3*(i-1) + 1 + 0]);
172  nv[3*(i-1) + 2] = nv[3*(i-1) + 2] - LMU * (nv[3*(i-1) + 2] - pos[3*(i-1) + 1 + 1]);
173  nv[3*(i-1) + 3] = nv[3*(i-1) + 3] - LMU * (nv[3*(i-1) + 3] - pos[3*(i-1) + 1 + 2]);
174 
175  dd = (nv[3*(i-1) + 1]-ox)*(nv[3*(i-1) + 1]-ox)
176  + (nv[3*(i-1) + 2]-oy)*(nv[3*(i-1) + 2]-oy)
177  + (nv[3*(i-1) + 3]-oz)*(nv[3*(i-1) + 3]-oz);
178  res +=dd;
179 
180  }
181 
182  compt[i] = 0;
183  pos[3*(i-1) + 1 + 0] = 0.;
184  pos[3*(i-1) + 1 + 1] = 0.;
185  pos[3*(i-1) + 1 + 2] = 0.;
186  }
187 
188  /* check new coor */

190  pt = &mesh->tetra[k];
191 
192  if ( !MG_EOK(pt) ) continue;
193 
194  for(i=0 ; i<4 ; i++) {
195  ipt = pt->v[i];
196  ppt = &mesh->point[ipt];
197 
198  if ( (ppt->tag & MG_BDY) || (ppt->tag & MG_REQ) ) continue;
199 
200  //if(ppt->tmp) continue;
201  //ppt->tmp = 1;
202  lon = MMG5_boulevolp(mesh,k,i,&list[0]);
203 
204  for ( l=0; l<lon; l++ ) {
205  iel = list[l] /4;
206  pt1 = &mesh->tetra[iel];
207  ipt0 = 3*(pt1->v[0] - 1);
208  ipt1 = 3*(pt1->v[1] - 1);
209  ipt2 = 3*(pt1->v[2] - 1);
210  ipt3 = 3*(pt1->v[3] - 1);
211 
212  ax = nv[ipt2 + 1] - nv[ipt0 + 1];
213  ay = nv[ipt2 + 2] - nv[ipt0 + 2];
214  az = nv[ipt2 + 3] - nv[ipt0 + 3];
215 
216  bx = nv[ipt3 + 1] - nv[ipt0 + 1];
217  by = nv[ipt3 + 2] - nv[ipt0 + 2];
218  bz = nv[ipt3 + 3] - nv[ipt0 + 3];
219

221  + (nv[ipt1 + 2] - nv[ipt0 + 2]) * (az*bx - ax*bz) \
222  + (nv[ipt1 + 3] - nv[ipt0 + 3]) * (ax*by - ay*bx);
223 
224  if ( vol < 0 ) {
225  break;
226  }
227  }
228  if ( l<=lon ) {
229  memcpy(&pos[3*(ipt-1) + 1],ppt->c,3*sizeof(double));
230  for ( l=0; l<lon; l++ ) {
231  iel = list[l] / 4;
232  pt1 = &mesh->tetra[iel];
233  ipt0 = 3*(pt1->v[0] - 1);
234  ipt1 = 3*(pt1->v[1] - 1);
235  ipt2 = 3*(pt1->v[2] - 1);
236  ipt3 = 3*(pt1->v[3] - 1);
237 
238  ax = nv[ipt2 + 1] - nv[ipt0 + 1];

240  az = nv[ipt2 + 3] - nv[ipt0 + 3];
241 
242  bx = nv[ipt3 + 1] - nv[ipt0 + 1];
243  by = nv[ipt3 + 2] - nv[ipt0 + 2];
244  bz = nv[ipt3 + 3] - nv[ipt0 + 3];
245 
246  vol = (nv[ipt1 + 1] - nv[ipt0 + 1]) * (ay*bz - az*by) \
247  + (nv[ipt1 + 2] - nv[ipt0 + 2]) * (az*bx - ax*bz) \
248  + (nv[ipt1 + 3] - nv[ipt0 + 3]) * (ax*by - ay*bx);
249 
250  if ( vol < 0 ) {
251  break;
252  }
253  }
254 
255  if ( l<lon ) break;
256  }
257  }
258  if ( i<4 ) break;
259  }
260 
261 
262  if(k > mesh->ne) {
263  /*update coor*/
264  for(i=1 ; i<=mesh->np ; i++) {
265  ppt = &mesh->point[i];
266  ppt->c[0] = nv[3*(i-1) + 1];
267  ppt->c[1] = nv[3*(i-1) + 2];
268  ppt->c[2] = nv[3*(i-1) + 3];
269  }
270  for(k=1 ; k<=mesh->ne ; k++) {
271  pt = &mesh->tetra[k];
272  if ( !MG_EOK(pt) ) continue;
273 
274  pt->qual = MMG5_caltet(mesh,sol,pt);
275  }
276  if( mesh->info.imprim > 5) fprintf(stdout," LAPLACIAN : %8f\n",res);
277  } else {
278  if( mesh->info.imprim > 5) fprintf(stdout," NO LAPLACIAN\n");
279  break;
280  }
281  if(res<1e-5) break;
282 
283  } while(it++ < maxiter);
284
