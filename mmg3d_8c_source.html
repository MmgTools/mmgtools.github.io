61int MMG5_countLocalParamAtTet( MMG5_pMesh mesh,MMG5_iNode **bdyRefs) {
62 int npar,ier;
63 MMG5_int k;
64
66 (*bdyRefs) = NULL;
67
68 k = mesh->ne? mesh->tetra[1].ref : 0;
69
70 /* Try to alloc the first node */
71 ier = MMG5_Add_inode( mesh, bdyRefs, k );
72 if ( ier < 0 ) {
73 fprintf(stderr,"\n ## Error: %s: unable to allocate the first boundary"
74 " reference node.\n",__func__);
75 return 0;
76 }
77 else {
78 assert(ier);
79 npar = 1;
80 }
81
82 for ( k=1; k<=mesh->ne; ++k ) {
83 ier = MMG5_Add_inode( mesh, bdyRefs, mesh->tetra[k].ref );
84
85 if ( ier < 0 ) {
86 fprintf(stderr, "\n ## Warning: %s: unable to list the tetra references.\n"
87 " Uncomplete parameters file.\n",__func__ );
88 break;
89 }
90 else if ( ier ) ++npar;
91 }
92
93 return npar;
94}

130int MMG3D_writeLocalParam( MMG5_pMesh mesh ) {
131 MMG5_iNode *triRefs,*tetRefs;
132 int nparTri,nparTet;
133 char *ptr,data[MMG5_FILESTR_LGTH];
134 FILE *out;
135
137 strcpy(data,mesh->namein);
138
139 ptr = MMG5_Get_filenameExt(data);
140
141 if ( ptr ) *ptr = '\0';
142 strcat(data,".mmg3d");
143
144 if ( !(out = fopen(data,"wb")) ) {
145 fprintf(stderr,"\n ** UNABLE TO OPEN %s.\n",data);
146 return 0;
147 }
148
149 fprintf(stdout,"\n %%%% %s OPENED\n",data);
150
151 nparTri = MMG5_countLocalParamAtTri( mesh, &triRefs);
152 if ( !nparTri ) {
153 fclose(out);
154 return 0;
155 }
156
157 nparTet = MMG5_countLocalParamAtTet( mesh, &tetRefs);
158 if ( !nparTet ) {
159 fclose(out);
160 return 0;
161 }
162
163 fprintf(out,"parameters\n %d\n",nparTri+nparTet);
164
166 if (! MMG5_writeLocalParamAtTri(mesh,triRefs,out) ) {
167 fclose(out);
168 return 0;
169 }
170
172 if (! MMG5_writeLocalParamAtTet(mesh,tetRefs,out) ) {
173 fclose(out);
174 return 0;
175 }
176
177 fclose(out);
178 fprintf(stdout," -- WRITING COMPLETED\n");
179
180 return 1;
181}

197static inline
198int MMG3D_defaultOption(MMG5_pMesh mesh,MMG5_pSol met,MMG5_pSol sol) {
199 mytime ctim[TIMEMAX];
200 double hsiz;
201 char stim[32];
202
203 MMG3D_Set_commonFunc();
204
205 signal(SIGABRT,MMG5_excfun);
206 signal(SIGFPE,MMG5_excfun);
207 signal(SIGILL,MMG5_excfun);
208 signal(SIGSEGV,MMG5_excfun);
209 signal(SIGTERM,MMG5_excfun);
210 signal(SIGINT,MMG5_excfun);
211
212 tminit(ctim,TIMEMAX);
213 chrono(ON,&(ctim[0]));
214
215 if ( mesh->info.npar ) {
216 fprintf(stderr,"\n ## Error: %s: "
217 "unable to save of a local parameter file with"
218 " the default parameters values because local parameters"
219 " are provided.\n",__func__);
220 _LIBMMG5_RETURN(mesh,met,sol,MMG5_LOWFAILURE);
221 }
222
223
224 if ( mesh->info.imprim > 0 ) fprintf(stdout,"\n -- INPUT DATA\n");
225
226 /* load data */
227 chrono(ON,&(ctim[1]));
228
229 if ( met && met->np && (met->np != mesh->np) ) {
230 fprintf(stderr,"\n ## WARNING: WRONG SOLUTION NUMBER. IGNORED\n");
231 MMG5_DEL_MEM(mesh,met->m);
232 met->np = 0;
233 }
234
235 if ( sol && sol->np && (sol->np != mesh->np) ) {
236 fprintf(stderr,"\n ## WARNING: WRONG SOLUTION NUMBER. IGNORED\n");
237 MMG5_DEL_MEM(mesh,sol->m);
238 sol->np = 0;
239 }
240
241 chrono(OFF,&(ctim[1]));
242 printim(ctim[1].gdif,stim);
243 if ( mesh->info.imprim > 0 )
244 fprintf(stdout," -- INPUT DATA COMPLETED. %s\n",stim);
245
246 /* analysis */
247 chrono(ON,&(ctim[2]));
248 MMG3D_setfunc(mesh,met);
249
250 if ( mesh->info.imprim > 0 ) {
251 fprintf(stdout,"\n -- DEFAULT PARAMETERS COMPUTATION\n");
252 }
253
254 /* scaling mesh + hmin/hmax computation*/
255 if ( !MMG5_scaleMesh(mesh,met,sol) ) _LIBMMG5_RETURN(mesh,met,sol,MMG5_STRONGFAILURE);
256
257 /* specific meshing + hmin/hmax update */
258 if ( mesh->info.optim ) {
259 if ( !MMG3D_doSol(mesh,met) ) {
260 if ( !MMG5_unscaleMesh(mesh,met,sol) ) _LIBMMG5_RETURN(mesh,met,sol,MMG5_STRONGFAILURE);
261 _LIBMMG5_RETURN(mesh,met,sol,MMG5_LOWFAILURE);
262 }
263 }
264
265 if ( mesh->info.hsiz > 0. ) {
266 if ( !MMG5_Compute_constantSize(mesh,met,&hsiz) ) {
267 if ( !MMG5_unscaleMesh(mesh,met,sol) ) _LIBMMG5_RETURN(mesh,met,sol,MMG5_STRONGFAILURE);
268 _LIBMMG5_RETURN(mesh,met,sol,MMG5_STRONGFAILURE);
269 }
270 }
271
272 /* unscaling mesh */
273 if ( !MMG5_unscaleMesh(mesh,met,sol) ) _LIBMMG5_RETURN(mesh,met,sol,MMG5_STRONGFAILURE);
274
275 /* Save the local parameters file */
276 mesh->mark = 0;
277 if ( !MMG3D_writeLocalParam(mesh) ) {
278 fprintf(stderr,"\n ## Error: %s: Unable to save the local parameters file.\n"
279 " Exit program.\n",__func__);
280 _LIBMMG5_RETURN(mesh,met,sol,MMG5_LOWFAILURE);
281 }
282
283 _LIBMMG5_RETURN(mesh,met,sol,MMG5_SUCCESS);
284}

297int main(int argc,char *argv[]) {
298
299 MMG5_pMesh mesh;
300 MMG5_pSol sol,met,disp,ls;
301 int ier,ierSave,fmtin,fmtout;
302 char stim[32],*ptr;
303
304 /* Select line buffering even if the output is not a terminal and force stderr
305 * and stdout to print in the same order as the events */
306 setvbuf(stdout, NULL, _IOLBF, 1024);
307 setvbuf(stderr, NULL, _IOLBF, 1024);
308
309 /* Version info */
310#ifndef MMG_COMPARABLE_OUTPUT
311 fprintf(stdout," -- MMG3D, Release %s (%s) \n",MMG_VERSION_RELEASE,MMG_RELEASE_DATE);
312 fprintf(stdout," %s\n",MMG_COPYRIGHT);
313 fprintf(stdout," %s %s\n",__DATE__,__TIME__);
314#endif
315
316 MMG3D_Set_commonFunc();
317
318 /* Print timer at exit */
319 atexit(MMG5_endcod);
320
321 tminit(MMG5_ctim,TIMEMAX);
322 chrono(ON,&MMG5_ctim[0]);
323
324 /* assign default values */
325 mesh = NULL;
326 met = NULL;
327 disp = NULL;
328 ls = NULL;
329
330 MMG3D_Init_mesh(MMG5_ARG_start,
331 MMG5_ARG_ppMesh,&mesh,MMG5_ARG_ppMet,&met,
332 MMG5_ARG_ppDisp,&disp,
333 MMG5_ARG_ppLs,&ls,
334 MMG5_ARG_end);
335
336 /* reset default values for file names */
337 if ( !MMG3D_Free_names(MMG5_ARG_start,
338 MMG5_ARG_ppMesh,&mesh,MMG5_ARG_ppMet,&met,
339 MMG5_ARG_ppDisp,&disp,
340 MMG5_ARG_ppLs,&ls,
341 MMG5_ARG_end) )
342 return MMG5_STRONGFAILURE;
343
344
345 /* Set default metric size */
346 if ( !MMG3D_Set_solSize(mesh,met,MMG5_Vertex,0,MMG5_Scalar) )
347 MMG5_RETURN_AND_FREE(mesh,met,ls,disp,MMG5_STRONGFAILURE);
348
349 /* Read command line */
350 if ( !MMG3D_parsar(argc,argv,mesh,met,ls) ) return MMG5_STRONGFAILURE;
351
352 /* load data */
353 if ( mesh->info.imprim >= 0 )
354 fprintf(stdout,"\n -- INPUT DATA\n");
355 chrono(ON,&MMG5_ctim[1]);
356
357 /* For each mode: pointer over the solution structure to load */
358 if ( mesh->info.lag >= 0 ) {
359 sol = disp;
360 }
361 else if ( mesh->info.iso || mesh->info.isosurf ) {
362 sol = ls;
363 }
364 else {
365 sol = met;
366 }
367
368 /* read mesh/sol files */
369 ptr = MMG5_Get_filenameExt(mesh->namein);
370 fmtin = MMG5_Get_format(ptr,MMG5_FMT_MeditASCII);
371
372 switch ( fmtin ) {
373
374 case ( MMG5_FMT_GmshASCII ): case ( MMG5_FMT_GmshBinary ):
375 ier = MMG3D_loadMshMesh(mesh,sol,mesh->namein);
376 break;
377
378 case ( MMG5_FMT_VtkVtu ):
379 ier = MMG3D_loadVtuMesh(mesh,sol,mesh->namein);
380 break;
381
382 case ( MMG5_FMT_VtkVtk ):
383 ier = MMG3D_loadVtkMesh(mesh,sol,mesh->namein);
384 break;
385
386 case ( MMG5_FMT_MeditASCII ): case ( MMG5_FMT_MeditBinary ):
387 ier = MMG3D_loadMesh(mesh,mesh->namein);
388 if ( ier < 1 ) { break; }
389
390 /* In Lagrangian mode, the name of the displacement file has been parsed in ls */
391 if ( mesh->info.lag >= 0 ) {
392 if ( !MMG3D_Set_inputSolName(mesh,disp,ls->namein) ) {
393 MMG5_RETURN_AND_FREE(mesh,met,ls,disp,MMG5_STRONGFAILURE);
394 }
395 MMG5_DEL_MEM(mesh,ls->namein);
396 }
397
398 if ( mesh->info.lag >= 0 || mesh->info.iso || mesh->info.isosurf ) {
399 /* displacement or isovalue are mandatory */
400 if ( MMG3D_loadSol(mesh,sol,sol->namein) < 1 ) {
401 fprintf(stdout," ## ERROR: UNABLE TO LOAD SOLUTION FILE.\n");
402 MMG5_RETURN_AND_FREE(mesh,met,ls,disp,MMG5_STRONGFAILURE);
403 }
404 }
405 else {
406 /* Facultative metric */
407 if ( MMG3D_loadSol(mesh,met,met->namein) == -1 ) {
408 fprintf(stderr,"\n ## ERROR: WRONG DATA TYPE OR WRONG SOLUTION NUMBER.\n");
409 MMG5_RETURN_AND_FREE(mesh,met,ls,disp,MMG5_STRONGFAILURE);
410 }
411 }
412 /* In iso mode: read metric if any */
413 if ( mesh->info.iso && met->namein ) {
414 if ( MMG3D_loadSol(mesh,met,met->namein) < 1 ) {
415 fprintf(stdout," ## ERROR: UNABLE TO LOAD METRIC.\n");
416 MMG5_RETURN_AND_FREE(mesh,met,ls,disp,MMG5_STRONGFAILURE);
417 }
418 }
419 break;
420
421 default:
422 fprintf(stderr," ** I/O AT FORMAT %s NOT IMPLEMENTED.\n",MMG5_Get_formatName(fmtin) );
423 MMG5_RETURN_AND_FREE(mesh,met,ls,disp,MMG5_STRONGFAILURE);
424 }
425
426 if ( ier < 1 ) {
427 if ( ier == 0 ) {
428 fprintf(stderr," ** %s NOT FOUND.\n",mesh->namein);
429 fprintf(stderr," ** UNABLE TO OPEN INPUT FILE.\n");
430 }
431 MMG5_RETURN_AND_FREE(mesh,met,ls,disp,MMG5_STRONGFAILURE);
432 }
433
434 /* Check input data */
435 if ( mesh->info.lag > -1 ) {
436 if ( met->namein ) {
437 fprintf(stdout," ## WARNING: MESH ADAPTATION UNAVAILABLE IN"
438 " LAGRANGIAN MODE. METRIC IGNORED.\n");
439 MMG5_RETURN_AND_FREE(mesh,met,ls,disp,MMG5_STRONGFAILURE);
440 }
441 }
442 else if ( mesh->info.iso || mesh->info.isosurf ) {
443 if ( ls == NULL || ls->m == NULL ) {
444 fprintf(stderr,"\n ## ERROR: NO ISOVALUE DATA.\n");
445 MMG5_RETURN_AND_FREE(mesh,met,ls,disp,MMG5_STRONGFAILURE);
446 }
447 }
448
449 /* Read parameter file */
450 if ( !MMG3D_parsop(mesh,met) )
451 MMG5_RETURN_AND_FREE(mesh,met,ls,disp,MMG5_LOWFAILURE);
452
453 chrono(OFF,&MMG5_ctim[1]);
454 if ( mesh->info.imprim >= 0 ) {
455 printim(MMG5_ctim[1].gdif,stim);
456 fprintf(stdout," -- DATA READING COMPLETED. %s\n",stim);
457 }
458
459 if ( mesh->mark ) {
460 /* Save a local parameters file containing the default parameters */
461 ier = MMG3D_defaultOption(mesh,met,disp);
462 MMG5_RETURN_AND_FREE(mesh,met,ls,disp,ier);
463 }
464 /* Lagrangian mode */
465 else if ( mesh->info.lag > -1 ) {
466 ier = MMG3D_mmg3dmov(mesh,met,disp);
467 }
468 /* Level Set mode */
469 else if ( mesh->info.iso || mesh->info.isosurf ) {
470 ier = MMG3D_mmg3dls(mesh,ls,met);
471 }
472 /* Remeshing mode */
473 else {
474 if ( met && ls && met->namein && ls->namein ) {
475 fprintf(stdout,"\n ## ERROR: IMPOSSIBLE TO PROVIDE BOTH A METRIC"
476 " AND A SOLUTION IN ADAPTATION MODE.\n");
477 MMG5_RETURN_AND_FREE(mesh,met,ls,disp,MMG5_STRONGFAILURE);
478 }
479 ier = MMG3D_mmg3dlib(mesh,met);
480 }
481
482 if ( ier != MMG5_STRONGFAILURE ) {
484 chrono(ON,&MMG5_ctim[1]);
485 if ( mesh->info.imprim > 0 )
486 fprintf(stdout,"\n -- WRITING DATA FILE %s\n",mesh->nameout);
487
488 ptr = MMG5_Get_filenameExt(mesh->nameout);
489 fmtout = MMG5_Get_format(ptr,fmtin);
490 switch ( fmtout ) {
491 case ( MMG5_FMT_GmshASCII ): case ( MMG5_FMT_GmshBinary ):
492 ierSave = MMG3D_saveMshMesh(mesh,met,mesh->nameout);
493 break;
494 case ( MMG5_FMT_VtkVtu ):
495 ierSave = MMG3D_saveVtuMesh(mesh,met,mesh->nameout);
496 break;
497 case ( MMG5_FMT_VtkVtk ):
498 ierSave = MMG3D_saveVtkMesh(mesh,met,mesh->nameout);
499 break;
500 case ( MMG5_FMT_Tetgen ):
501 ierSave = MMG3D_saveTetgenMesh(mesh,mesh->nameout);
502 /* This format dont allow to save a solution: use a .sol file */
503 if ( !ierSave ) {
504 MMG5_RETURN_AND_FREE(mesh,met,ls,disp,MMG5_STRONGFAILURE);
505 }
506 if ( met && met->np ) {
507 ierSave = MMG3D_saveSol(mesh,met,met->nameout);
508 }
509 break;
510
511 default:
512 ierSave = MMG3D_saveMesh(mesh,mesh->nameout);
513 if ( !ierSave ) {
514 MMG5_RETURN_AND_FREE(mesh,met,ls,disp,MMG5_STRONGFAILURE);
515 }
516 if ( met && met->np ) {
517 ierSave = MMG3D_saveSol(mesh,met,met->nameout);
518 }
519 break;
520 }
521
522 if ( !ierSave )
523 MMG5_RETURN_AND_FREE(mesh,met,ls,disp,MMG5_STRONGFAILURE);
524
525 chrono(OFF,&MMG5_ctim[1]);
526 if ( mesh->info.imprim > 0 )
527 fprintf(stdout," -- WRITING COMPLETED\n");
528 }
529
530 /* free mem */
531 MMG5_RETURN_AND_FREE(mesh,met,ls,disp,ier);
532}
