42 double MMG5_surftri_iso(MMG5_pMesh mesh,MMG5_pSol met,MMG5_pTria ptt) {
43  double *a,*b,*c,abx,aby,abz,acx,acy,acz,det,n[3];
44 
45  a = mesh->point[ptt->v[0]].c;
46  b = mesh->point[ptt->v[1]].c;
47  c = mesh->point[ptt->v[2]].c;
48 
49  /* area */
50  abx = b[0] - a[0];
51  aby = b[1] - a[1];
52  abz = b[2] - a[2];
53 
54  acx = c[0] - a[0];
55  acy = c[1] - a[1];
56  acz = c[2] - a[2];
57 
58  n[0] = aby*acz - abz*acy;
59  n[1] = abz*acx - abx*acz;
60  n[2] = abx*acy - aby*acx;
61  det = n[0]*n[0] + n[1]*n[1] + n[2]*n[2];
62 
63  return 0.5*sqrt(det) ;
64 }

129 int MMG5_sum_reqEdgeLengthsAtPoint ( MMG5_pMesh mesh,MMG5_pSol met,int ip0,int ip1 ) {
130  MMG5_pPoint p0,p1;
131  int j;
132  double len,dist;
133 
134  /* Compute the euclidean edge length */
135  p0 = &mesh->point[ip0];
136  p1 = &mesh->point[ip1];
137 
138  len = 0.;
139  for ( j=0; j<mesh->dim; ++j ) {
140  dist = p1->c[j]-p0->c[j];
141  len += dist*dist;
142  }
143  len = sqrt(len);
144 
145  /* Add the length to the point's metric and increment the number of
146  * times the point has been seen */
147  met->m[met->size*ip0] += len;
148  met->m[met->size*ip1] += len;
149  ++p0->s;
150  ++p1->s;
151 
152  return 1;
153 }

277 int MMG5_gradsiz_iso(MMG5_pMesh mesh,MMG5_pSol met) {
278  MMG5_pTria pt;
279  MMG5_pPoint p1,p2;
280  double hgrad,ll,h1,h2,hn,val;
281  int k,it,ip1,ip2,maxit,nup,nu;
282  int8_t i,j,i1,i2;
283 
284  if ( abs(mesh->info.imprim) > 5 || mesh->info.ddebug ) {
285  fprintf(stdout," ** Grading mesh\n");
286  }
287 
288  MMG5_mark_pointsOnReqEdge_fromTria ( mesh );
289 
290  for ( k=1; k<=mesh->np; k++ ) {
291  mesh->point[k].flag = mesh->base;
292  }
293 
294 
295  hgrad = mesh->info.hgrad;
296  it = nup = 0;
297  maxit = 100;
298 
299  do {
300  mesh->base++;
301  nu = 0;
302  for (k=1; k<=mesh->nt; k++) {
303  pt = &mesh->tria[k];
304  if ( !MG_EOK(pt) ) continue;
305 
306  for (i=0; i<3; i++) {
307  i1 = MMG5_inxt2[i];
308  i2 = MMG5_iprv2[i];
309  ip1 = pt->v[i1];
310  ip2 = pt->v[i2];
311  p1 = &mesh->point[ip1];
312  p2 = &mesh->point[ip2];
313  if ( p1->flag < mesh->base-1 && p2->flag < mesh->base-1 ) continue;
314 
315  /* Skip points belonging to a required edge */
316  if ( p1->s || p2->s ) continue;
317 
318  ll = 0.;
319  for ( j=0; j<mesh->dim; ++j ) {
320  val = p2->c[j]-p1->c[j];
321  ll += val * val;
322  }
323  ll = sqrt(ll);
324 
325  h1 = met->m[ip1];
326  h2 = met->m[ip2];
327  if ( h1 < h2 ) {
328  if ( h1 < MMG5_EPSD ) continue;
329  hn = h1 + hgrad*ll;
330  if ( h2 > hn ) {
331  met->m[ip2] = hn;
332  p2->flag = mesh->base;
333  nu++;
334  }
335  }
336  else {
337  if ( h2 < MMG5_EPSD ) continue;
338  hn = h2 + hgrad*ll;
339  if ( h1 > hn ) {
340  met->m[ip1] = hn;
341  p1->flag = mesh->base;
342  nu++;
343  }
344  }
345  }
346  }
347  nup += nu;
348  }
349  while ( ++it < maxit && nu > 0 );
350 
351  if ( abs(mesh->info.imprim) > 4 ) {
352  fprintf(stdout," gradation: %7d updated, %d iter.\n",nup,it);
353  }
354 
355  return 1;

368 int MMG5_gradsizreq_iso(MMG5_pMesh mesh,MMG5_pSol met) {
369  MMG5_pTria pt;
370  MMG5_pPoint p1,p2;
371  double hgrad,ll,h1,h2,hn,ux,uy;
372  int k,it,ip1,ip2,ipmaster,ipslave,maxit,nup,nu;
373  uint8_t i,i1,i2;
374 
375 
376  if ( abs(mesh->info.imprim) > 5 || mesh->info.ddebug ) {
377  fprintf(stdout," ** Grading required points.\n");
378  }
379 
380  if ( mesh->info.hgrad < 0. ) {
382  MMG5_mark_pointsOnReqEdge_fromTria ( mesh );
383  }
384 
386  hgrad = mesh->info.hgradreq;
387  it = nup = 0;
388  maxit = 100;
389 
390  do {
391  nu = 0;
392  for (k=1; k<=mesh->nt; k++) {
393  pt = &mesh->tria[k];
394  if ( !MG_EOK(pt) ) {
395  continue;
396  }
397 
398  for (i=0; i<3; i++) {
399  i1 = MMG5_inxt2[i];
400  i2 = MMG5_iprv2[i];
401  ip1 = pt->v[i1];
402  ip2 = pt->v[i2];
403  p1 = &mesh->point[ip1];
404  p2 = &mesh->point[ip2];
405 
406  if ( abs ( p1->s - p2->s ) < 2 ) {
407  /* No size to propagate */
408  continue;
409  }
410  else if ( p1->s > p2->s ) {
411  ipmaster = ip1;
412  ipslave = ip2;
413  }
414  else {
415  assert ( p2->s > p1->s );
416  ipmaster = ip2;
417  ipslave = ip1;
418  }
419 
420  ux = p2->c[0]-p1->c[0];
421  uy = p2->c[1]-p1->c[1];
422  ll = ux*ux + uy*uy;
423  ll = sqrt(ll);
424 
425  h1 = met->m[ipmaster];
426  h2 = met->m[ipslave];
427  if ( h1 < h2 ) {
428  if ( h1 < MMG5_EPSD ) {
429  continue;
430  }
431  hn = h1 + hgrad*ll;
432  if ( h2 <= hn ) {
433  continue;
434  }
435  }
436  else {
437  hn = h1 - hgrad*ll;
438  if ( h2 >= hn ) {
439  continue;
440  }
441  }
442  met->m[ipslave] = hn;
443  mesh->point[ipslave].s = mesh->point[ipmaster].s - 1;
444  nu++;
445  }
446  }
447  nup += nu;
448  }
449  while ( ++it < maxit && nu > 0 );
450 
451  if ( abs(mesh->info.imprim) > 4 && nup ) {
452  fprintf(stdout," gradation (required): %7d updated, %d iter.\n",nup,it);
453  }
454 
455  return nup;
456 }
