25 /* Compute isotropic edge length */
26 double long_iso(double *ca,double *cb,double *ma,double *mb) {
27  double ha,hb,ux,uy,dd,rap,len;
28 
29  ha = *ma;
30  hb = *mb;
31  ux = cb[0] - ca[0];
32  uy = cb[1] - ca[1];
33  dd = sqrt(ux*ux + uy*uy);
34 
35  rap = (hb - ha) / ha;
36  if ( fabs(rap) < MMG2D_EPSD )
37  len = dd / ha;
38  else
39  len = dd * (1.0/ha + 1.0/hb + 8.0 / (ha+hb)) / 6.0;
40 
41  return len;
42 }

45 /* compute aniso edge length */
46 double long_ani(double *ca,double *cb,double *ma,double *mb) {
47  double ux,uy,dd1,dd2,len;
48  ux = cb[0] - ca[0];
49  uy = cb[1] - ca[1];
50 
51  dd1 = ma[0]*ux*ux + ma[2]*uy*uy + 2.0*ma[1]*ux*uy;
52  if ( dd1 <= 0.0 ) dd1 = 0.0;
53  dd2 = mb[0]*ux*ux + mb[2]*uy*uy + 2.0*mb[1]*ux*uy;
54  if ( dd2 <= 0.0 ) dd2 = 0.0;
55 
56  len = (sqrt(dd1)+sqrt(dd2)+4.0*sqrt(0.5*(dd1+dd2))) / 6.0;
57 
58  return len;
59 }

62 double MMG2D_lencurv_iso(MMG5_pMesh mesh,MMG5_pSol met,int ip1,int ip2) {
63  MMG5_pPoint p1,p2;
64  double h1,h2,len,l,r;
65 
66  p1 = &mesh->point[ip1];
67  p2 = &mesh->point[ip2];
68 
69  h1 = met->m[ip1];
70  h2 = met->m[ip2];
71 
72  l = (p2->c[0]-p1->c[0])*(p2->c[0]-p1->c[0]) + (p2->c[1]-p1->c[1])*(p2->c[1]-p1->c[1]);
73  l = sqrt(l);
74  r = h2 / h1 - 1.0;
75  len = ( fabs(r) < MMG5_EPS ) ? ( l/h1 ) : ( l / (h2-h1) * log1p(r) );
76 
77  return len;
78 }

80 /* Calculate length of a curve in the considered anisotropic metric by using a two-point quadrature formula */
81 double MMG2D_lencurv_ani(MMG5_pMesh mesh,MMG5_pSol met,int ip1,int ip2) {
82  MMG5_pPoint p1,p2;
83  double len,*m1,*m2,ux,uy,l1,l2;
84  static int8_t mmgWarn0=0,mmgWarn1=0;
85 
86  p1 = &mesh->point[ip1];
87  p2 = &mesh->point[ip2];
88 
89  m1 = &met->m[3*ip1];
90  m2 = &met->m[3*ip2];
91 
92  ux = p2->c[0] - p1->c[0];
93  uy = p2->c[1] - p1->c[1];
94 
95  l1 = m1[0]*ux*ux + 2.0*m1[1]*ux*uy + m1[2]*uy*uy;
96  l2 = m2[0]*ux*ux + 2.0*m2[1]*ux*uy + m2[2]*uy*uy;
97 
98  if ( l1 < 0.0 ) {
99  if ( !mmgWarn0 ) {
100  mmgWarn0 = 1;
101  fprintf(stderr,"\n ## Error: %s: at least 1 negative edge length"
102  " (l1: %e).\n",__func__,l1);
103  }
104  return 0.;
105  }
106  if ( l2 < 0.0 ) {
107  if ( !mmgWarn1 ) {
108  mmgWarn1 = 1;
109  fprintf(stderr,"\n ## Error: %s: at least 1 negative edge length"
110  " (l2: %e)\n",__func__,l2);
111  }
112  return 0.;
113  }
114 
115  l1 = sqrt(l1);
116  l2 = sqrt(l2);
117 
118  len = 0.5*(l1+l2);
119 
120  return len;
121 }

124 int MMG2D_prilen(MMG5_pMesh mesh,MMG5_pSol sol) {
125  MMG5_pTria pt;
126  double lavg,len,ecart,som,lmin,lmax;
127  int k,l,navg,ia,ipa,ipb,iamin,ibmin,iamax,ibmax,nullEdge,hl[9];
128  static double bd[9] = {0.0, 0.3, 0.6, 0.7071, 0.9, 1.3, 1.4142, 2.0, 5.0};
129 
130  navg = 0;
131  lavg = 0.0;
132  lmin = 1.e20;
133  lmax = 0.0;
134  som = 0.0;
135  iamin = 0;
136  ibmin = 0;
137  iamax = 0;
138  ibmax = 0;
139  nullEdge = 0;
140 
141  for (k=0; k<9; k++) hl[k] = 0;
142 
143  for (k=1; k<=mesh->nt; k++) {
144  pt = &mesh->tria[k];
145  if ( !MG_EOK(pt) ) continue;
146 
147  for (ia=0; ia<3; ia++) {
148  l = (&mesh->adja[3*(k-1)+1])[ia];
149  if ( l < 3*k ) continue;
150 
151  ipa = MMG2D_iare[ia][0];
152  ipb = MMG2D_iare[ia][1];
153 
154  if ( sol->m )
155  len = MMG2D_lencurv(mesh,sol,pt->v[ipa],pt->v[ipb]);
156  else
157  len = MMG2D_lencurv_iso(mesh,sol,pt->v[ipa],pt->v[ipb]);
158 
159  navg++;
160  ecart = len;
161  lavg += len;
162 
163  /* update efficiency index */
164  if ( ecart > 1.0 ) ecart = 1.0 / ecart;
165 
166  som += (ecart - 1.0);
167 
168  /* find largest, smallest edge */
169  if (len < lmin) {
170  lmin = len;
171  iamin = pt->v[ipa];
172  ibmin = pt->v[ipb];
173  }
174  if (len > lmax) {
175  lmax = len;
176  iamax = pt->v[ipa];
177  ibmax = pt->v[ipb];
178  }
179 
180  /* update histogram */
181  if (len < bd[3]) {
182  if (len > bd[2]) hl[2]++;
183  else if (len > bd[1]) hl[1]++;
184  else hl[0]++;
185  }
186  else if (len < bd[5]) {
187  if (len > bd[4]) hl[4]++;
188  else if (len > bd[3]) hl[3]++;
189  }
190  else if (len < bd[6]) hl[5]++;
191  else if (len < bd[7]) hl[6]++;
192  else if (len < bd[8]) hl[7]++;
193  else hl[8]++;
194  }
195  }
196  MMG5_displayLengthHisto(mesh, navg, &lavg, iamin, ibmin, lmin,
197  iamax, ibmax, lmax,nullEdge, &bd[0], &hl[0],0);
198 
199 
200  return 1;
201 }
