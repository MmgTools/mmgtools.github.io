51 void MMG5_Init_parameters(MMG5_pMesh mesh) {
52 
53  memset(&mesh->info,0, sizeof(MMG5_Info));
54 
55  /* default values for integers */
56  /* [-1..10], Tune level of imprim */
57  mesh->info.imprim = 1;
58  /* [0/1] ,Turn on/off levelset meshing */
59  mesh->info.iso = MMG5_OFF;
60  /* [n/-1] ,Set memory size to n Mbytes/keep the default value */
61  mesh->info.mem = MMG5_NONSET_MEM;
62  /* [0/1] ,Turn on/off debug mode */
63  mesh->info.ddebug = MMG5_OFF;
64  /* [n] ,number of local parameters */
65  mesh->info.npar = MMG5_OFF;
66  /* [0/1] ,avoid/allow point insertion/deletion */
67  mesh->info.noinsert = MMG5_OFF;
68  /* [0/1] ,avoid/allow edge or face flipping */
69  mesh->info.noswap = MMG5_OFF;
70  /* [0/1] ,avoid/allow point relocation */
71  mesh->info.nomove = MMG5_OFF;
72  /* [n] ,number of user-defined references */
73  mesh->info.nmat = MMG5_OFF;
74  /* [-1/val] ,Turn off/on the removal of small bubles in levelset meshing */
75  mesh->info.rmc = MMG5_NONSET;
76  /* [0/1] ,avoid/allow */
77  mesh->info.nosizreq = MMG5_OFF;
78 
79 
80  /* default values for doubles */
81  /* angle detection */
82  mesh->info.dhd = MMG5_ANGEDG;
83  /* minimal mesh size */
84  mesh->info.hmin = MMG5_NONSET_HMIN;
85  /* maximal mesh size */
86  mesh->info.hmax = MMG5_NONSET_HMAX;
87  /* constant mesh size */
88  mesh->info.hsiz = MMG5_NONSET_HSIZ;
89  /* control Hausdorff */
90  mesh->info.hausd = MMG5_HAUSD;
91  /* control gradation */
92  mesh->info.hgrad = MMG5_HGRAD;
93  /* control gradation on required entities */
94  mesh->info.hgradreq = MMG5_HGRADREQ;
95 
96  /* default values for pointers */
97  /* list of user-defined references */
98  mesh->info.mat = NULL;
99 
102  mesh->info.lag = MMG5_LAG;
103 
104  /* initial value for memMax and gap */
105  mesh->gap = MMG5_GAP;
106  mesh->memMax = MMG5_memSize();
107  if ( mesh->memMax ) {
108  /* maximal memory = 50% of total physical memory */
109  mesh->memMax = (size_t)(mesh->memMax*MMG5_MEMPERCENT);
110  } else {

112  printf(" Maximum memory set to default value: %d MB.\n",MMG5_MEMMAX);

393 void MMG5_Set_constantSize(MMG5_pMesh mesh,MMG5_pSol met,double hsiz) {
394  MMG5_pPoint ppt;
395  int k,iadr;
396 
397  if ( met->size == 1 ) {
398  for (k=1; k<=mesh->np; k++) {
399  ppt = &mesh->point[k];
400  if ( !MG_VOK(ppt) ) continue;
401  met->m[k] = hsiz;
402  }
403  }
404  else {
405  hsiz = 1./(hsiz*hsiz);
406 
407  if ( mesh->dim==2 ) {
408  for (k=1; k<=mesh->np; k++) {
409  ppt = &mesh->point[k];
410  if ( !MG_VOK(ppt) ) continue;
411 
412  iadr = 3*k;
413  met->m[iadr] = hsiz;
414  met->m[iadr+1] = 0.;
415  met->m[iadr+2] = hsiz;
416  }
417  }
418  else {
419  assert ( mesh->dim==3 );
420  for (k=1; k<=mesh->np; k++) {
421  ppt = &mesh->point[k];
422  if ( !MG_VOK(ppt) ) continue;
423 
424  iadr = 6*k;
425  met->m[iadr] = hsiz;
426  met->m[iadr+1] = 0.;
427  met->m[iadr+2] = 0.;
428  met->m[iadr+3] = hsiz;
429  met->m[iadr+4] = 0.;
430  met->m[iadr+5] = hsiz;
431  }
432  }
433  }
434 
435  return;

562 int MMG5_Compute_constantSize(MMG5_pMesh mesh,MMG5_pSol met,double *hsiz) {
563  int8_t sethmin,sethmax;
564 
565 
566  if ( mesh->info.hmin > mesh->info.hsiz ) {
567  fprintf(stderr,"\n ## Error: %s: Mismatched options: hmin (%e) is greater"
568  " than hsiz (%e). Exit Program.\n",__func__,
569  mesh->info.hmin,mesh->info.hsiz);
570  return 0;
571  }
572 
573  if ( mesh->info.hmax > 0. && mesh->info.hmax < mesh->info.hsiz ) {
574  fprintf(stderr,"\n ## Error: %s: Mismatched options: hmax (%e) is lower"
575  " than hsiz (%e). Exit Program.\n",__func__,
576  mesh->info.hmax,mesh->info.hsiz);
577  return 0;
578  }
579 
580  *hsiz = mesh->info.hsiz;
581 
582  sethmin = sethmax = 0;
583  if ( mesh->info.hmin > 0. ) {
584  sethmin = 1;
585  *hsiz = MG_MAX(mesh->info.hmin,*hsiz);
586  }
587 
588  if ( mesh->info.hmax > 0. ) {
589  sethmax = 1;
590  *hsiz = MG_MIN(mesh->info.hmax,*hsiz);
591  }
592 
593  /* Set hmin */
594  if ( !sethmin ) {
595  if ( sethmax ) {
596  mesh->info.hmin = MG_MIN(0.1*(*hsiz),0.1*mesh->info.hmax);
597  } else {
598  mesh->info.hmin = 0.1*(*hsiz);
599  }
600  }
601 
602  /* Set hmax */
603  if ( !sethmax ) {
604  if ( sethmin ) {
605  mesh->info.hmax = MG_MAX(10.*(*hsiz),10.*mesh->info.hmin);
606  } else {
607  mesh->info.hmax = 10.*(*hsiz);
608  }
609  }
610 
611  return 1;
612 }

659 int MMG5_Set_multiMat(MMG5_pMesh mesh,MMG5_pSol sol,int ref,
660  int split,int rin,int rex){
661  MMG5_pMat mat;
662  int k;
663 
664  if ( !mesh->info.nmat ) {
665  fprintf(stderr,"\n ## Error: %s: You must set the number of material",__func__);
666  fprintf(stderr," with the MMG2D_Set_iparameters function before setting");
667  fprintf(stderr," values in multi material structure. \n");
668  return 0;
669  }
670  if ( mesh->info.nmati >= mesh->info.nmat ) {
671  fprintf(stderr,"\n ## Error: %s: unable to set a new material.\n",
672  __func__);
673  fprintf(stderr," max number of materials: %d\n",mesh->info.nmat);
674  return 0;
675  }
676  if ( ref < 0 ) {
677  fprintf(stderr,"\n ## Error: %s: negative references are not allowed.\n",
678  __func__);
679  return 0;
680  }
681 
682  for (k=0; k<mesh->info.nmati; k++) {
683  mat = &mesh->info.mat[k];
684 
685  if ( mat->ref == ref ) {
686  mat->dospl = split;
687  if ( split ) {
688  mat->rin = rin;
689  mat->rex = rex;
690  }
691  else {
692  mat->rin = mat->ref;
693  mat->rex = mat->ref;
694  }
695  if ( (mesh->info.imprim > 5) || mesh->info.ddebug ) {
696  fprintf(stderr,"\n ## Warning: %s: new materials (interior, exterior)",
697  __func__);
698  fprintf(stderr," for material of ref %d\n",ref);
699  }
700  return 1;
701  }
702  }
703 
704  if ( ( split != MMG5_MMAT_Split ) && ( split != MMG5_MMAT_NoSplit ) ) {
705  fprintf(stderr,"\n ## Error: %s: unexpected value for the 'split' argument."
706  " You must use the MMG5_MMAT_Split or MMG5_MMAT_NpSplit keywords \n",
707  __func__);
708  return 0;
709  }
710 
711  mesh->info.mat[mesh->info.nmati].ref = ref;
712  mesh->info.mat[mesh->info.nmati].dospl = split;
713  mesh->info.mat[mesh->info.nmati].rin = rin;
714  mesh->info.mat[mesh->info.nmati].rex = rex;
715 
716  mesh->info.nmati++;
717 
718  /* Invert the table if all materials have been set */
