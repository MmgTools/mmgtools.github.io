52 int MMG5_kPartBoxCompute(SCOTCH_Graph *graf, int vertNbr, int boxVertNbr,
53  SCOTCH_Num *permVrtTab,MMG5_pMesh mesh) {
54  int boxNbr, vertIdx;
55  SCOTCH_Num logMaxVal, SupMaxVal, InfMaxVal, maxVal;
56  char s[200];
57  SCOTCH_Num *sortPartTb;
58  SCOTCH_Strat strat ;
59  SCOTCH_Arch arch;
60 
61  /* Computing the number of boxes */
62  boxNbr = vertNbr / boxVertNbr;
63  if (boxNbr * boxVertNbr != vertNbr) {
64  boxNbr = boxNbr + 1;
65  }
66 
67 
68  /* Initializing SCOTCH functions */
69  CHECK_SCOTCH(SCOTCH_stratInit(&strat), "scotch_stratInit", 0) ;
70  if ( SCOTCH_6 ) {
71  CHECK_SCOTCH(SCOTCH_archCmplt(&arch, boxNbr), "scotch_archCmplt", 0) ;
72  }
73  else {
74  CHECK_SCOTCH(SCOTCH_archVcmplt(&arch), "scotch_archVcmplt", 0) ;
75  }
76  sprintf(s, "m{vert=%d,low=r{job=t,map=t,poli=S,sep=m{vert=80,low=h{pass=10}f{bal=0.0005,move=80},asc=f{bal=0.005,move=80}}}}", vertNbr / boxVertNbr);
77  CHECK_SCOTCH(SCOTCH_stratGraphMap(&strat, s), "scotch_stratGraphMap", 0) ;
78 
79  MMG5_ADD_MEM(mesh,2*vertNbr*sizeof(SCOTCH_Num),"sortPartTb",return 1);
80  MMG5_SAFE_CALLOC(sortPartTb,2*vertNbr,SCOTCH_Num,return 0);
81 
82  /* Partionning the graph */
83  if ( 0!=SCOTCH_graphMap(graf, &arch, &strat, sortPartTb) ) {
84  perror("scotch_graphMap");
85  MMG5_DEL_MEM(mesh,sortPartTb);
86  return 0;
87  }
88 
89 
90  if ( SCOTCH_6 ) {
91  // Looking for the max value in sortPartTb and computing sortPartTb as
92  // followed :
93  // - sortPartTb[2i] is the box value
94  // - sortPartTb[2i+1] is the vertex number
95  maxVal = sortPartTb[0];
96  }
97  for (vertIdx = vertNbr - 1 ; vertIdx >= 0 ; vertIdx--) {
98  sortPartTb[2*vertIdx] = sortPartTb[vertIdx];
99  sortPartTb[2*vertIdx+1] = vertIdx + 1;
100  if ( SCOTCH_5 ) {
101  if (sortPartTb[vertIdx] > maxVal)
102  maxVal = sortPartTb[vertIdx];
103  }
104  }
105 
106  if ( SCOTCH_5 ) {
107  // Determining the log of MaxVal
108  logMaxVal = 0;
109  while ( maxVal > 0) {
110  logMaxVal++;
111  maxVal >>= 1;
112  }
113 
114  // Infering the interval in which box values will be
115  InfMaxVal = logMaxVal << logMaxVal;
116  SupMaxVal = (logMaxVal << (logMaxVal + 1)) - 1;
117 
118  // Increasing box values until they are in the previous interval
119  for (vertIdx = 0 ; vertIdx < vertNbr ; vertIdx++) {
120  while (!(sortPartTb[2*vertIdx] >= InfMaxVal && sortPartTb[2*vertIdx] <= SupMaxVal)) {
121  sortPartTb[2*vertIdx] <<= 1;
122  }
123  }
124  }
125 
126  // Sorting the tabular, which contains box values and vertex numbers
127  _SCOTCHintSort2asc1(sortPartTb, vertNbr);
128 
129 
130  /* Infering the new numbering */
131  for (vertIdx = 0; vertIdx < vertNbr ; vertIdx++) {
132  permVrtTab[sortPartTb[2*vertIdx + 1]] = vertIdx + 1;
133  }
134 
135  SCOTCH_stratExit(&strat) ;
136  SCOTCH_archExit(&arch) ;
137 
138  MMG5_DEL_MEM(mesh,sortPartTb);
139 
140  return 0;
141 }

157 void MMG5_swapNod(MMG5_pMesh mesh,MMG5_pPoint points, double* sols,
158  MMG5_pSol field,int* perm,int ind1, int ind2, int solsiz) {
159  MMG5_Point ptttmp;
160  MMG5_pSol psl;
161  MMG5_Sol soltmp;

163 
164  /* swap the points */
165  memcpy(&ptttmp ,&points[ind2],sizeof(MMG5_Point));
166  memcpy(&points[ind2],&points[ind1],sizeof(MMG5_Point));
167  memcpy(&points[ind1],&ptttmp ,sizeof(MMG5_Point));
168 
169  /* swap the sols */
170  if ( sols ) {
171  addr1 = ind1*solsiz;
172  addr2 = ind2*solsiz;
173  memcpy(&soltmp ,&sols[addr2],solsiz*sizeof(double));

175  memcpy(&sols[addr1],&soltmp ,solsiz*sizeof(double));
176  }
177

179  if ( field ) {
180  if( mesh->nsols ) { /* swap solution fields (for ParMmg) */
181  for ( i=0; i<mesh->nsols; ++i ) {
182  psl = field+i;
183  assert ( psl && psl->m );
184 
185  pslsiz = psl->size;
186  addr1 = ind1*pslsiz;
187  addr2 = ind2*pslsiz;
188 
189  memcpy(&soltmp , psl->m + addr2,pslsiz*sizeof(double));
190  memcpy(psl->m + addr2, psl->m + addr1,pslsiz*sizeof(double));
191  memcpy(psl->m + addr1, &soltmp ,pslsiz*sizeof(double));
192  }
193  } else { /* swap a single displacement field (for Lagrangian motion) */
194  psl = field;
195  assert ( psl );
196 
197  if( psl->m ) { /* it is null after Lagrangian step */
198  pslsiz = psl->size;
199  addr1 = ind1*pslsiz;
200  addr2 = ind2*pslsiz;
201 
202  memcpy(&soltmp , psl->m + addr2,pslsiz*sizeof(double));
203  memcpy(psl->m + addr2, psl->m + addr1,pslsiz*sizeof(double));
204  memcpy(psl->m + addr1, &soltmp ,pslsiz*sizeof(double));
205  }
206  }
207  }
208 
209  /* swap the permutaion table */
210  tmp = perm[ind2];
211  perm[ind2] = perm[ind1];
212  perm[ind1] = tmp;
213 }
