35 int MMG2D_hashTria(MMG5_pMesh mesh) {
36  MMG5_pTria pt,pt1;
37  int k,kk,pp,l,ll,mins,mins1,maxs,maxs1;
38  int *hcode,*link,inival,hsize,iadr;
39  uint8_t i,ii,i1,i2;
40  unsigned int key;
41 
42  if ( mesh->adja ) return 1;
43  if ( !mesh->nt ) return 0;
44 
45  /* memory alloc */
46  MMG5_SAFE_CALLOC(hcode,mesh->nt+1,int,return 0);
47 
48  /* memory alloc */
49  MMG5_ADD_MEM(mesh,(3*mesh->ntmax+5)*sizeof(int),"adjacency table",
50  printf(" Exit program.\n");
51  return 0;);
52  MMG5_SAFE_CALLOC(mesh->adja,3*mesh->ntmax+5,int,return 0);
53 
54  link = mesh->adja;
55  hsize = mesh->nt;
56 
57  /* init */
58  inival = INT_MAX;
59  for (k=0; k<=mesh->nt; k++)
60  hcode[k] = -inival;
61 
62  /* build hash table */
63  for (k=1; k<=mesh->nt; k++) {
64  pt = &mesh->tria[k];
65 
66  if ( !pt->v[0] ) continue;
67  for (i=0; i<3; i++) {
68  i1 = MMG2D_idir[i+1];
69  i2 = MMG2D_idir[i+2];
70  if ( pt->v[i1] < pt->v[i2] ) {
71  mins = pt->v[i1];
72  maxs = pt->v[i2];
73  }
74  else {
75  mins = pt->v[i2];
76  maxs = pt->v[i1];
77  }
78 
79  /* compute key */
80  key = KTA*mins + KTB*maxs;
81  key = key % hsize + 1;
82 
83  /* insert */
84  iadr = 3*(k-1) + i+1;
85  link[iadr] = hcode[key];
86  hcode[key] = -iadr;
87  }
88  }
89 
90  /* set adjacency */
91  for (l=3*mesh->nt; l>0; l--) {
92  if ( link[l] >= 0 ) continue;
93  k = (l-1) / 3 + 1;
94  i = (l-1) % 3;
95  i1 = MMG2D_idir[i+1];
96  i2 = MMG2D_idir[i+2];
97  pt = &mesh->tria[k];
98 
99  mins = M_MIN(pt->v[i1],pt->v[i2]);
100  maxs = M_MAX(pt->v[i1],pt->v[i2]);
101 
102  /* accross link */
103  ll = -link[l];
104  pp = 0;
105  link[l] = 0;
106 
107  while ( ll != inival ) {
108  kk = (ll-1) / 3 + 1;
109  ii = (ll-1) % 3;
110  i1 = MMG2D_idir[ii+1];
111  i2 = MMG2D_idir[ii+2];
112  pt1 = &mesh->tria[kk];
113  if ( pt1->v[i1] < pt1->v[i2] ) {
114  mins1 = pt1->v[i1];
115  maxs1 = pt1->v[i2];
116  }
117  else {
118  mins1 = pt1->v[i2];
119  maxs1 = pt1->v[i1];
120  }
121 
122  if ( mins1 == mins && maxs1 == maxs ) {
123  /* adjacent found */
124  if ( pp != 0 ) link[pp] = link[ll];
125  link[l] = 3*kk + ii;
126  link[ll]= 3*k + i;
127  break;
128  }
129  pp = ll;
130  ll = -link[ll];
131  }
132  }
133  MMG5_SAFE_FREE(hcode);
134 
135  return 1;
136 }

304

306  continue;

311  i2 = MMG2D_idir_q[i][1];

316  hcode[i] = -kk;
317  }
318  }
319  }
320  MMG5_DEL_MEM(mesh,hash.item);
321 
322  return 1;

333 int MMG2D_assignEdge(MMG5_pMesh mesh) {
334  MMG5_Hash hash;
335  MMG5_pTria pt;
336  MMG5_pQuad pq;
337  MMG5_pEdge pa;
338  int k,ia;
339  int8_t ier;
340  uint8_t i,i1,i2;
341 
342  /* Try to clean triangle structure (in case where mmg2dlib is called after
343  * mmg2dmesh) */
344  for (k=1; k<=mesh->nt; k++) {
345  pt = &mesh->tria[k];
346  if ( !MG_EOK(pt) ) continue;
347 
348  /* If all edges are required, the triangle is maybe required by the user */
349  if ( (pt->tag[0] & MG_REQ) && (pt->tag[1] & MG_REQ) && (pt->tag[2] & MG_REQ) ) {
350  continue;
351  }
352 
353  /* Otherwise there is no reason to have required tags on edges */
354  for ( ia = 0; ia < 3; ++ia ) {
355  pt->tag[ia] &= ~MG_REQ;
356  }
357  }
358 
359  if ( !mesh->na ) return 1;
360 
361  /* Temporarily allocate a hash structure for storing edges */
362  ier = MMG5_hashNew ( mesh,&hash, mesh->na,3*mesh->na );
363  if ( !ier ) {
364  printf(" ## Error: %s: Unable to allocate edge hash table\n.",__func__);
365  return 0;
366  }
367 
368  /* hash mesh edges */
369  for (k=1; k<=mesh->na; k++) {
370  ier = MMG5_hashEdge(mesh,&hash,mesh->edge[k].a,mesh->edge[k].b,k);
371  if ( !ier ) {
372  fprintf(stderr,"\n ## Error: %s: unable to hash edge %d %d.\n",__func__,
373  MMG2D_indPt(mesh,mesh->edge[k].a),MMG2D_indPt(mesh,mesh->edge[k].b));
374  return 0;
375  }
376  }
377 
378  /* set references to triangles */
379  for (k=1; k<=mesh->nt; k++) {
380  pt = &mesh->tria[k];
381  if ( !MG_EOK(pt) ) continue;
382 
383  for (i=0; i<3; i++) {
384  i1 = MMG5_inxt2[i];
385  ia = MMG5_hashGet(&hash,pt->v[i],pt->v[i1]);
386  if ( ia ) {
387  i2 = MMG5_inxt2[i1];
388  pa = &mesh->edge[ia];
389  pt->edg[i2] = pa->ref;
390  pt->tag[i2] |= pa->tag;
391 
392  }
393  }
394  }
395 
396  /* set references to quadrangles */
397  for (k=1; k<=mesh->nquad; k++) {
398  pq = &mesh->quadra[k];
399  if ( !MG_EOK(pq) ) continue;
400 
401  for (i=0; i<4; i++) {
402  i1 = MMG2D_idir_q[i][0];
403  i2 = MMG2D_idir_q[i][1];
404  ia = MMG5_hashGet(&hash,pq->v[i1],pq->v[i2]);
405  if ( ia ) {
406  pa = &mesh->edge[ia];
407  pq->edg[i] = pa->ref;
408  pq->tag[i] |= pa->tag;
409  }
410  }
411  }
412 
413  /* Delete the hash for edges */
414  MMG5_DEL_MEM(mesh,hash.item);
415  MMG5_DEL_MEM(mesh,mesh->edge);
416  mesh->na = 0;
417 
418  return 1;
419 }
