49 int MMG5_mmgIntmet33_ani(double *m,double *n,double *mr,double s) {
50  int order;
51  double lambda[3],vp[3][3],mu[3],is[6],isnis[6],mt[9],P[9],dd;
52  int8_t i;
53  static int8_t mmgWarn=0;
54 
55  /* Compute inverse of square root of matrix M : is =
56  * P*diag(1/sqrt(lambda))*{^t}P */
57  order = MMG5_eigenv(1,m,lambda,vp);
58  if ( !order ) {
59  if ( !mmgWarn ) {
60  fprintf(stderr,"\n ## Warning: %s: unable to diagonalize at least"
61  " 1 metric.\n",__func__);
62  mmgWarn = 1;
63  }
64  return 0;
65  }
66 
67  for (i=0; i<3; i++) {
68  if ( lambda[i] < MMG5_EPSD ) return 0;
69  lambda[i] = sqrt(lambda[i]);
70  lambda[i] = 1.0 / lambda[i];
71  }
72 
73  is[0] = lambda[0]*vp[0][0]*vp[0][0] + lambda[1]*vp[1][0]*vp[1][0]
74  + lambda[2]*vp[2][0]*vp[2][0];
75  is[1] = lambda[0]*vp[0][0]*vp[0][1] + lambda[1]*vp[1][0]*vp[1][1]
76  + lambda[2]*vp[2][0]*vp[2][1];
77  is[2] = lambda[0]*vp[0][0]*vp[0][2] + lambda[1]*vp[1][0]*vp[1][2]
78  + lambda[2]*vp[2][0]*vp[2][2];
79  is[3] = lambda[0]*vp[0][1]*vp[0][1] + lambda[1]*vp[1][1]*vp[1][1]
80  + lambda[2]*vp[2][1]*vp[2][1];
81  is[4] = lambda[0]*vp[0][1]*vp[0][2] + lambda[1]*vp[1][1]*vp[1][2]
82  + lambda[2]*vp[2][1]*vp[2][2];
83  is[5] = lambda[0]*vp[0][2]*vp[0][2] + lambda[1]*vp[1][2]*vp[1][2]
84  + lambda[2]*vp[2][2]*vp[2][2];
85 
86  mt[0] = n[0]*is[0] + n[1]*is[1] + n[2]*is[2];
87  mt[1] = n[0]*is[1] + n[1]*is[3] + n[2]*is[4];
88  mt[2] = n[0]*is[2] + n[1]*is[4] + n[2]*is[5];
89  mt[3] = n[1]*is[0] + n[3]*is[1] + n[4]*is[2];
90  mt[4] = n[1]*is[1] + n[3]*is[3] + n[4]*is[4];
91  mt[5] = n[1]*is[2] + n[3]*is[4] + n[4]*is[5];
92  mt[6] = n[2]*is[0] + n[4]*is[1] + n[5]*is[2];
93  mt[7] = n[2]*is[1] + n[4]*is[3] + n[5]*is[4];
94  mt[8] = n[2]*is[2] + n[4]*is[4] + n[5]*is[5];
95 
96  isnis[0] = is[0]*mt[0] + is[1]*mt[3] + is[2]*mt[6];
97  isnis[1] = is[0]*mt[1] + is[1]*mt[4] + is[2]*mt[7];
98  isnis[2] = is[0]*mt[2] + is[1]*mt[5] + is[2]*mt[8];
99  isnis[3] = is[1]*mt[1] + is[3]*mt[4] + is[4]*mt[7];
100  isnis[4] = is[1]*mt[2] + is[3]*mt[5] + is[4]*mt[8];
101  isnis[5] = is[2]*mt[2] + is[4]*mt[5] + is[5]*mt[8];
102 
103  order = MMG5_eigenv(1,isnis,lambda,vp);
104  if ( !order ) {
105  if ( !mmgWarn ) {
106  fprintf(stderr,"\n ## Warning: %s: unable to diagonalize at least"
107  " 1 metric.\n",__func__);
108  mmgWarn = 1;
109  }
110  return 0;
111  }
112 
113  /* P = is * (vp) */
114  P[0] = is[0]*vp[0][0] + is[1]*vp[0][1] + is[2]*vp[0][2];
115  P[1] = is[0]*vp[1][0] + is[1]*vp[1][1] + is[2]*vp[1][2];
116  P[2] = is[0]*vp[2][0] + is[1]*vp[2][1] + is[2]*vp[2][2];
117  P[3] = is[1]*vp[0][0] + is[3]*vp[0][1] + is[4]*vp[0][2];
118  P[4] = is[1]*vp[1][0] + is[3]*vp[1][1] + is[4]*vp[1][2];
119  P[5] = is[1]*vp[2][0] + is[3]*vp[2][1] + is[4]*vp[2][2];
120  P[6] = is[2]*vp[0][0] + is[4]*vp[0][1] + is[5]*vp[0][2];
121  P[7] = is[2]*vp[1][0] + is[4]*vp[1][1] + is[5]*vp[1][2];
122  P[8] = is[2]*vp[2][0] + is[4]*vp[2][1] + is[5]*vp[2][2];
123 
124  /* At this point, theory states that ^tPMP = I, {^t}PNP=\Lambda */
125  /* Linear interpolation between sizes */
126  for(i=0; i<3; i++) {
127  if ( lambda[i] < 0.0 ) return 0;
128  dd = s*sqrt(lambda[i]) + (1.0-s);
129  dd = dd*dd;
130  if ( dd < MMG5_EPSD ) return 0;
131  mu[i] = lambda[i]/dd;
132  }
133 
134  if ( !MMG5_invmatg(P,mt) ) return 0;
135 
136  /* Resulting matrix = ^tP^{-1} diag(mu) P^{-1} */
137  mr[0] = mu[0]*mt[0]*mt[0] + mu[1]*mt[3]*mt[3] + mu[2]*mt[6]*mt[6];
138  mr[1] = mu[0]*mt[0]*mt[1] + mu[1]*mt[3]*mt[4] + mu[2]*mt[6]*mt[7];
139  mr[2] = mu[0]*mt[0]*mt[2] + mu[1]*mt[3]*mt[5] + mu[2]*mt[6]*mt[8];
140  mr[3] = mu[0]*mt[1]*mt[1] + mu[1]*mt[4]*mt[4] + mu[2]*mt[7]*mt[7];
141  mr[4] = mu[0]*mt[1]*mt[2] + mu[1]*mt[4]*mt[5] + mu[2]*mt[7]*mt[8];
142  mr[5] = mu[0]*mt[2]*mt[2] + mu[1]*mt[5]*mt[5] + mu[2]*mt[8]*mt[8];
143 
144  return 1;
145 }

173 
174  /* Case when both endpoints are singular */
175  if ( (MG_SIN(p1->tag) || (p1->tag & MG_NOM)) &&
176  (MG_SIN(p2->tag) || (p2->tag & MG_NOM)) ) {
177  /* m1 and m2 are isotropic metrics */
178  dd = (1-s)*sqrt(m2[0]) + s*sqrt(m1[0]);
179  dd *= dd;
180  if ( dd < MMG5_EPSD ) {
181  if ( s < 0.5 ) {
182  mr[0] = m1[0];
183  mr[1] = m1[0];
184  mr[2] = m1[0];
185  mr[3] = m1[0];
186  mr[4] = m1[0];
187  }
188  else {
189  mr[0] = m2[0];
190  mr[1] = m2[0];
191  mr[2] = m2[0];
192  mr[3] = m2[0];
193  mr[4] = m2[0];
194  }
195  }
196  else {
197  mr[0] = m1[0]*m2[0] / dd;
198  mr[1] = mr[0];
199  mr[2] = mr[0];
200  mr[3] = mr[0];
201  mr[4] = mr[0];
202  }
203  }
204  /* vertex p1 is singular, p2 is regular */
205  else if ( (MG_SIN(p1->tag) || (p1->tag & MG_NOM)) &&
206  !(MG_SIN(p2->tag) || (p2->tag & MG_NOM)) ) {
207  /* m1 is an isotropic metric and m2 is a "ridge" metric that respect our
208  * storage convention. */
209  go2 = &mesh->xpoint[p2->xp];
210  n21 = &go2->n1[0];
211  n22 = &go2->n2[0];
212

214  dd = (1-s)*sqrt(m2[0]) + s*sqrt(m1[0]);
215  dd *= dd;
216  if ( dd < MMG5_EPSD ) {
217  mr[0] = s < 0.5 ? m1[0] : m2[0];
218  }
219  else {
220  mr[0] = m1[0]*m2[0] / dd;
221  }
222 
223  /* Interpolation of the two other eigenvalues for each configuration. */
224  /* 1. For the surface ruled by n1. */
225  dd = (1-s)*sqrt(m2[1]) + s*sqrt(m1[0]);
226  dd *= dd;
227  if ( dd < MMG5_EPSD ) {
228  hu1 = s < 0.5 ? m1[0] : m2[1];
229  }
230  else {
231  hu1 = m1[0]*m2[1] / dd;
232  }
233  dd = (1-s)*sqrt(m2[3]) + s*sqrt(m1[0]);
234  dd *= dd;
235  if ( dd < MMG5_EPSD ) {
236  hn1 = s < 0.5 ? m1[0] : m2[3];
237  }
238  else {
239  hn1 = m1[0]*m2[3] / dd;
240  }
241 
242  /* 2. For the surface ruled by n2. */
243  dd = (1-s)*sqrt(m2[2]) + s*sqrt(m1[0]);
244  dd *= dd;
245  if ( dd < MMG5_EPSD ) {
246  hu2 = s < 0.5 ? m1[0] : m2[2];
247  }
248  else {
249  hu2 = m1[0]*m2[2] / dd;
250  }
251  dd = (1-s)*sqrt(m2[4]) + s*sqrt(m1[0]);
252  dd *= dd;
253  if ( dd < MMG5_EPSD ) {
254  hn2 = s < 0.5 ? m1[0] : m2[4];
255  }
256  else {
257  hn2 = m1[0]*m2[4] / dd;
258  }
259 
260  /* Decision of the ordering of hu1 and hu2 */
261  ps11 = n21[0]*v[0] + n21[1]*v[1] + n21[2]*v[2];
262  ps12 = n22[0]*v[0] + n22[1]*v[1] + n22[2]*v[2];
263  if ( fabs(ps11) > fabs(ps12) ) {
264  mr[1] = hu1;
265  mr[2] = hu2;
266  mr[3] = hn1;
267  mr[4] = hn2;
268  }
269  else {
270  mr[1] = hu2;
271  mr[2] = hu1;
272  mr[3] = hn2;
273  mr[4] = hn1;
274  }
275  }
276  /* vertex p2 is singular, p1 is regular */
277  else if ( ( MG_SIN(p2->tag) || (p2->tag & MG_NOM)) &&
278  !(MG_SIN(p1->tag) || (p1->tag & MG_NOM)) ) {
279  /* m2 is an isotropic metric and m1 is a "ridge" metric that respect our
280  * storage convention. */
281  go1 = &mesh->xpoint[p1->xp];
282  n11 = &go1->n1[0];
283  n12 = &go1->n2[0];
284 
285  /* Interpolation of the eigenvalue associated to tangent vector */
286  dd = (1-s)*sqrt(m2[0]) + s*sqrt(m1[0]);
287  dd *= dd;
288  if ( dd < MMG5_EPSD ) {
289  mr[0] = s < 0.5 ? m1[0] : m2[0];
290  }
291  else {
292  mr[0] = m1[0]*m2[0] / dd;
293  }
294  /* Interpolation of the two other eigenvalues for each configuration. */
295  /* 1. For the surface ruled by n1. */
296  dd = (1-s)*sqrt(m2[0]) + s*sqrt(m1[1]);
297  dd *= dd;
298  if ( dd < MMG5_EPSD ) {
299  hu1 = s < 0.5 ? m1[1] : m2[0];
300  }
301  else {
302  hu1 = m1[1]*m2[0] / dd;
303  }
304  dd = (1-s)*sqrt(m2[0]) + s*sqrt(m1[3]);
305  dd *= dd;
306  if ( dd < MMG5_EPSD ) {
307  hn1 = s < 0.5 ? m1[3] : m2[0];
308  }
309  else {
310  hn1 = m1[3]*m2[0] / dd;
311  }
312 
313  /* 2. For the surface ruled by n2. */
314  dd = (1-s)*sqrt(m2[0]) + s*sqrt(m1[2]);
315  dd *= dd;
316  if ( dd < MMG5_EPSD ) {
317  hu2 = s < 0.5 ? m1[2] : m2[0];
318  }
319  else {
320  hu2 = m1[2]*m2[0] / dd;
321  }
322  dd = (1-s)*sqrt(m2[0]) + s*sqrt(m1[4]);
323  dd *= dd;
324  if ( dd < MMG5_EPSD ) {
325  hn2 = s < 0.5 ? m1[4] : m2[0];
326  }
327  else {
328  hn2 = m1[4]*m2[0] / dd;
329  }
330 
331  /* Decision of the ordering of hu1 and hu2 */
332  ps11 = n11[0]*v[0] + n11[1]*v[1] + n11[2]*v[2];
333  ps12 = n12[0]*v[0] + n12[1]*v[1] + n12[2]*v[2];
334  if ( fabs(ps11) > fabs(ps12) ) {
335  mr[1] = hu1;
336  mr[2] = hu2;
337  mr[3] = hn1;
338  mr[4] = hn2;
339  }
340  else {
341  mr[1] = hu2;
342  mr[2] = hu1;
343  mr[3] = hn2;
344  mr[4] = hn1;
345  }
346  }
347  /* p1,p2 : nonsingular vertices */
348  else {
349  go1 = &mesh->xpoint[p1->xp];
350  go2 = &mesh->xpoint[p2->xp];
351 
352  /* Interpolation of the eigenvalue associated to tangent vector */
353  dd = (1-s)*sqrt(m2[0]) + s*sqrt(m1[0]);
354  dd *= dd;
355  if ( dd < MMG5_EPSD ) {
356  mr[0] = s < 0.5 ? m1[0] : m2[0];
357  }
358  else {
359  mr[0] = m1[0]*m2[0] / dd;

361

363  n11 = &go1->n1[0];
364  n12 = &go1->n2[0];
365  n21 = &go2->n1[0];

370  /* 1. For the surface ruled by n1. */
371  dd = (1-s)*sqrt(m2[1]) + s*sqrt(m1[1]);
372  dd *= dd;
373  if ( dd < MMG5_EPSD ) {
374  hu1 = s < 0.5 ? m1[1] : m2[1];
375  }

378  }

380  dd *= dd;
381  if ( dd < MMG5_EPSD ) {
382  hn1 = s < 0.5 ? m1[3] : m2[3];
383  }
384  else {
385  hn1 = m1[3]*m2[3] / dd;

389  dd *= dd;

391  hu2 = s < 0.5 ? m1[2] : m2[2];
392  }

398  if ( dd < MMG5_EPSD ) {

403  }
404 
405  }
406  else {
407  /* 1. */

411  hu1 = s < 0.5 ? m1[1] : m2[2];

460  mr[4] = hn1;
461  }
462  }
463  mr[5] = 0.0;
464 
465  return 1;
466 }

498 int MMG5_interpreg_ani(MMG5_pMesh mesh,MMG5_pSol met,MMG5_pTria pt,int8_t i,
499  double s,double mr[6]) {

530  cold[0] = p1->c[0];
531  cold[1] = p1->c[1];
532  cold[2] = p1->c[2];
533 
534  nold[0] = n1[0];

536  nold[2] = n1[2];
537 
538  if ( MG_SIN(p1->tag) || (p1->tag & MG_NOM) || (ddbn < MMG5_EPSD) ) {
539  memcpy(m1,&met->m[6*ip1],6*sizeof(double));
540  }
541  else {

545  }
546  else {

548  }

550 
551  /* Go from point (l-1)step, to point l step */
552  for (l=1; l<=nstep; l++) {
553  u = l*step;

563  n[2] = (1.0-u)*(1.0-u)*n1[2] + 2.0*u*(1.0-u)*bn[2] + u*u*n2[2];
564  dd = n[0]*n[0] + n[1]*n[1] + n[2]*n[2];
565  if ( dd < MMG5_EPSD ) return 0;
566  dd = 1.0 / sqrt(dd);
567  n[0] *= dd;
568  n[1] *= dd;
569  n[2] *= dd;

573  memcpy(cold,c,3*sizeof(double));
574  memcpy(nold,n,3*sizeof(double));
575  memcpy(m1old,m1,6*sizeof(double));
576  }
577  }
578

580  step = (1.0-s) / nstep;
581  cold[0] = p2->c[0];
582  cold[1] = p2->c[1];
583  cold[2] = p2->c[2];
584 
585  nold[0] = n2[0];
586  nold[1] = n2[1];
587  nold[2] = n2[2];

590  memcpy(m2,&met->m[6*ip2],6*sizeof(double));
591 
592  /* In this pathological case, n is empty */
593  if ( MG_SIN(p1->tag) || (p1->tag & MG_NOM))

595  }

607  for (l=1; l<=nstep; l++) {
608  u = 1.0 - l*step;
609  c[0] = p1->c[0]*(1.0-u)*(1.0-u)*(1.0-u) + 3.0*(1.0-u)*(1.0-u)*u*b1[0]\

617  n[1] = (1.0-u)*(1.0-u)*n1[1] + 2.0*u*(1.0-u)*bn[1] + u*u*n2[1];
618  n[2] = (1.0-u)*(1.0-u)*n1[2] + 2.0*u*(1.0-u)*bn[2] + u*u*n2[2];
619  dd = n[0]*n[0] + n[1]*n[1] + n[2]*n[2];

630  memcpy(m2old,m2,6*sizeof(double));
631  }
632  }
633  /* At this point, c is point p(s), n is the normal at p(s), m1 and m2 are the 3*3
634  transported metric tensors from p1 and p2 to p(s) */
635 
636  /* Rotate both matrices to the tangent plane */
637  if ( !MMG5_rotmatrix(n,r) ) return 0;
638  MMG5_rmtr(r,m1,m1old);
639  MMG5_rmtr(r,m2,m2old);
640

642  if ( !MMG5_mmgIntmet33_ani(m1old,m2old,mr,s) ) {
643  if ( !warn ) {

646  " interpolation.\n", __func__);
647  }

649  }

652  {^t}R*mr*R m1old serve for nothing, let it be the temporary resulting
653  matrix. */
654  m1old[0] = mr[0]*r[0][0]*r[0][0] + mr[3]*r[1][0]*r[1][0] + mr[5]*r[2][0]*r[2][0]
655  + 2.*( mr[1]*r[0][0]*r[1][0] + mr[2]*r[0][0]*r[2][0] + mr[4]*r[1][0]*r[2][0] );

665  + mr[1]*r[1][0]*r[0][2] + mr[3]*r[1][0]*r[1][2] + mr[4]*r[1][0]*r[2][2]
666  + mr[2]*r[2][0]*r[0][2] + mr[4]*r[2][0]*r[1][2] + mr[5]*r[2][0]*r[2][2];
