48 int MMG3D_tetraQual(MMG5_pMesh mesh, MMG5_pSol met,int8_t metRidTyp) {
49  MMG5_pTetra pt;
50  double minqual;
51  int k,iel;
52 
53  minqual = 2./MMG3D_ALPHAD;
54 
55  /*compute tet quality*/
56  iel = 1;
57  for (k=1; k<=mesh->ne; k++) {
58  pt = &mesh->tetra[k];
59  if( !MG_EOK(pt) ) continue;
60 
61  if ( !metRidTyp && met->size == 6 && met->m ) {
62  pt->qual = MMG5_caltet33_ani(mesh,met,pt);
63  }
64  else if ( !(met && met->m) ) {
65  pt->qual = MMG5_caltet_iso(mesh,NULL,pt);
66  }
67  else {
68  pt->qual = MMG5_orcal(mesh,met,k);
69  }
70 
71 
72  if ( pt->qual < minqual ) {
73  minqual = pt->qual;
74  iel = k;
75  }
76  }
77 
78  /* Here the quality is not normalized by alpha, thus we need to
79  * normalized it */
80  return MMG5_minQualCheck(iel,minqual,MMG3D_ALPHAD);
81 }

95 inline double MMG5_caltet33_ani(MMG5_pMesh mesh,MMG5_pSol met,MMG5_pTetra pt) {
96  double cal,abx,aby,abz,acx,acy,acz,adx,ady,adz;
97  double bcx,bcy,bcz,bdx,bdy,bdz,cdx,cdy,cdz;
98  double h1,h2,h3,h4,h5,h6,det,vol,rap,v1,v2,v3,num;
99  double *a,*b,*c,*d;
100  double mm[6];
101  int ip[4],iad0,iad1,iad2,iad3,k;
102 
103  ip[0] = pt->v[0];
104  ip[1] = pt->v[1];
105  ip[2] = pt->v[2];
106  ip[3] = pt->v[3];
107 
108  iad0 = met->size * ip[0];
109  iad1 = met->size * ip[1];
110  iad2 = met->size * ip[2];
111  iad3 = met->size * ip[3];
112 
113  /* average metric */
114  for (k=0; k<6; k++)
115  mm[k] = 0.25 * (met->m[iad0+k]+met->m[iad1+k]+met->m[iad2+k]+met->m[iad3+k]);
116 
117  a = mesh->point[ip[0]].c;
118  b = mesh->point[ip[1]].c;
119  c = mesh->point[ip[2]].c;
120  d = mesh->point[ip[3]].c;
121 
122  /* volume */
123  abx = b[0] - a[0];
124  aby = b[1] - a[1];
125  abz = b[2] - a[2];
126 
127  acx = c[0] - a[0];
128  acy = c[1] - a[1];
129  acz = c[2] - a[2];
130 
131  adx = d[0] - a[0];
132  ady = d[1] - a[1];
133  adz = d[2] - a[2];
134 
135  bcx = c[0] - b[0];
136  bcy = c[1] - b[1];
137  bcz = c[2] - b[2];
138 
139  bdx = d[0] - b[0];
140  bdy = d[1] - b[1];
141  bdz = d[2] - b[2];
142 
143  cdx = d[0] - c[0];
144  cdy = d[1] - c[1];
145  cdz = d[2] - c[2];
146 
147  v1 = acy*adz - acz*ady;
148  v2 = acz*adx - acx*adz;
149  v3 = acx*ady - acy*adx;
150  vol = abx * v1 + aby * v2 + abz * v3;
151  if ( vol <= 0. ) return 0.0;
152 
153  det = mm[0] * ( mm[3]*mm[5] - mm[4]*mm[4]) \
154  - mm[1] * ( mm[1]*mm[5] - mm[2]*mm[4]) \
155  + mm[2] * ( mm[1]*mm[4] - mm[2]*mm[3]);
156  if ( det < MMG5_EPSD2 ) {
157  return 0.0;
158  }
159  det = sqrt(det) * vol;
160 
161  /* edge lengths */
162  h1 = mm[0]*abx*abx + mm[3]*aby*aby + mm[5]*abz*abz
163  + 2.0*(mm[1]*abx*aby + mm[2]*abx*abz + mm[4]*aby*abz);
164  h2 = mm[0]*acx*acx + mm[3]*acy*acy + mm[5]*acz*acz
165  + 2.0*(mm[1]*acx*acy + mm[2]*acx*acz + mm[4]*acy*acz);
166  h3 = mm[0]*adx*adx + mm[3]*ady*ady + mm[5]*adz*adz
167  + 2.0*(mm[1]*adx*ady + mm[2]*adx*adz + mm[4]*ady*adz);
168  h4 = mm[0]*bcx*bcx + mm[3]*bcy*bcy + mm[5]*bcz*bcz
169  + 2.0*(mm[1]*bcx*bcy + mm[2]*bcx*bcz + mm[4]*bcy*bcz);
170  h5 = mm[0]*bdx*bdx + mm[3]*bdy*bdy + mm[5]*bdz*bdz
171  + 2.0*(mm[1]*bdx*bdy + mm[2]*bdx*bdz + mm[4]*bdy*bdz);
172  h6 = mm[0]*cdx*cdx + mm[3]*cdy*cdy + mm[5]*cdz*cdz
173  + 2.0*(mm[1]*cdx*cdy + mm[2]*cdx*cdz + mm[4]*cdy*cdz);
174 
175  /* quality */
176  rap = h1 + h2 + h3 + h4 + h5 + h6;
177  num = sqrt(rap) * rap;
178 
179  cal = det / num;
180 
181  return cal;
182 }

206 int MMG3D_computePrilen( MMG5_pMesh mesh, MMG5_pSol met, double* avlen,
207  double* lmin, double* lmax, int* ned, int* amin, int* bmin, int* amax,
208  int* bmax, int* nullEdge, int8_t metRidTyp, double** bd_in, int hl[9] )
209 {
210  MMG5_pTetra pt;
211  MMG5_pPoint ppt;
212  MMG5_Hash hash;
213  double len;
214  int k,np,nq,n;
215  int8_t ia,i0,i1,ier,i;
216  static double bd[9]= {0.0, 0.3, 0.6, 0.7071, 0.9, 1.3, 1.4142, 2.0, 5.0};
217 
218  *bd_in = bd;
219  memset(hl,0,9*sizeof(int));
220  *ned = 0;
221  *avlen = 0.0;
222  *lmax = 0.0;
223  *lmin = 1.e30;
224  *amin = *amax = *bmin = *bmax = 0;
225  *nullEdge = 0;
226 
227  /* Hash all edges in the mesh */
228  if ( !MMG5_hashNew(mesh,&hash,mesh->np,7*mesh->np) ) return 0;
229 
230  for(k=1; k<=mesh->ne; k++) {
231  pt = &mesh->tetra[k];
232  if ( !MG_EOK(pt) ) continue;
233 
234  for(ia=0; ia<6; ia++) {
235  i0 = MMG5_iare[ia][0];
236  i1 = MMG5_iare[ia][1];
237  np = pt->v[i0];
238  nq = pt->v[i1];
239 
240  if(!MMG5_hashEdge(mesh,&hash,np,nq,0)){
241  fprintf(stderr," ## Error: %s: function MMG5_hashEdge return 0\n",
242  __func__);
243  return 0;
244  }
245  }
246  }
247 
248  /* Pop edges from hash table, and analyze their length */
249  for(k=1; k<=mesh->ne; k++) {
250  pt = &mesh->tetra[k];
251  if ( !MG_EOK(pt) ) continue;
252  n = 0;
253  for(i=0 ; i<4 ; i++) {
254  ppt = &mesh->point[pt->v[i]];
255  if(!(MG_SIN(ppt->tag) || MG_NOM & ppt->tag) && (ppt->tag & MG_GEO)) continue;
256  n++;
257  }
258  if(!n) {
259  continue;
260  }
261  for(ia=0; ia<6; ia++) {
262  i0 = MMG5_iare[ia][0];
263  i1 = MMG5_iare[ia][1];
264  np = pt->v[i0];
265  nq = pt->v[i1];
266 
267  /* Remove edge from hash ; ier = 1 if edge has been found */
268  ier = MMG5_hashPop(&hash,np,nq);
269  if( ier ) {
270  if ( (!metRidTyp) && met->size==6 && met->m ) {
271  len = MMG5_lenedg33_ani(mesh,met,ia,pt);
272  }
273  else
274  len = MMG5_lenedg(mesh,met,ia,pt);
275 
276 
277  if ( !len ) {
278  ++(*nullEdge);
279  }
280  else {
281  *avlen += len;
282  (*ned)++;
283 
284  if( len < (*lmin) ) {
285  *lmin = len;
286  *amin = np;
287  *bmin = nq;
288  }
289 
290  if ( len > (*lmax) ) {
291  *lmax = len;
292  *amax = np;
293  *bmax = nq;
294  }
295 
296  /* Locate size of edge among given table */
297  for(i=0; i<8; i++) {
298  if ( bd[i] <= len && len < bd[i+1] ) {
299  hl[i]++;
300  break;
301  }
302  }
303  if( i == 8 ) hl[8]++;
304  }
305  }
306  }
307  }
308 
309  MMG5_DEL_MEM(mesh,hash.item);
310  return 1;
311 }

359 void MMG3D_computeLESqua(MMG5_pMesh mesh,MMG5_pSol met,int *ne,double *max,double *avg,
360  double *min,int *iel,int *good,int *med,int his[5],int imprim) {
361  MMG5_pTetra pt;
362  double rap;
363  int k,ok,nex;
364  static int8_t mmgWarn0=0;
365 
366  /*compute tet quality*/
367  for (k=1; k<=mesh->ne; k++) {
368  pt = &mesh->tetra[k];
369  if( !MG_EOK(pt) ) continue;
370 
371  pt->qual = MMG5_orcal(mesh,met,k);
372  }
373 
374  if ( imprim <= 0 )
375  return;
376 
377  (*min) = (*avg) = 0.0;
378  (*max) = 1.0;
379  (*iel) = 0;
380  (*med) = (*good) = 0;
381 
382  for (k=0; k<5; k++) his[k] = 0;
383 
384  nex = ok = 0;
385  for (k=1; k<=mesh->ne; k++) {
386  pt = &mesh->tetra[k];
387  if( !MG_EOK(pt) ) {
388  nex++;
389  continue;
390  }
391  ok++;
392  if ( (!mmgWarn0) && (MMG5_orvol(mesh->point,pt->v) < 0.0) ) {
393  mmgWarn0 = 1;
394  fprintf(stderr," ## Warning: %s: at least 1 negative volume.\n",
395  __func__);
396  }
397  rap = 1. - MMG3D_ALPHAD * pt->qual;
398  if ( rap > (*min) ) {
399  (*min) = rap;
400  (*iel) = ok;
401  }
402  if ( rap < 0.9 ) (*med)++;
403  if ( rap < 0.6 ) (*good)++;
404  // if ( rap < MMG3D_BADKAL ) mesh->info.badkal = 1;
405  (*avg) += rap;
406  (*max) = MG_MIN((*max),rap);
407  if(rap < 0.6)
408  his[0] += 1;
409  else if(rap < 0.9)
410  his[1] += 1;
411  else if(rap < 0.93)
412  his[2] += 1;
413  else if(rap < 0.99)
414  his[3] += 1;
415  else
416  his[4] += 1;
417  }
418 
419  (*ne) = mesh->ne-nex;
420 
421  return;
422 }

477 int MMG3D_displayQualHisto_internal(int ne,double max,double avg,double min,int iel,
478  int good,int med,int his[5],int nrid,int optimLES,
479  int imprim)
480 {
481  const double les_ticks[6] = {0,0.6,0.9,0.93,0.99,1};
482  int i,imax;
483 
484  if ( abs(imprim) >= 3 ){
485  if ( optimLES ) {
486  /* print histo */
487  fprintf(stdout," HISTOGRAMM:");
488  fprintf(stdout," %6.2f %% < 0.6\n",100.0*((float)good/(float)ne));
489  if ( abs(imprim) > 3 ) {
490  fprintf(stdout," %6.2f %% < 0.9\n",100.0*( (float)med/(float)ne));
491 
492  assert ( min >= max );
493  for ( i=0; i<5; ++i ) {
494  if ( max < les_ticks[i+1] && min >= les_ticks[i] ) {
495  fprintf(stdout," %5.2f < Q < %5.2f %7d %6.2f %%\n",
496  les_ticks[i],les_ticks[i+1],his[i],
497  100.*((float)his[i]/(float)ne));
498  }
499  }
500  }
501  return 1;
502  }
503  else {
504  /* print histo */
505  fprintf(stdout," HISTOGRAMM:");
506  fprintf(stdout," %6.2f %% > 0.12\n",100.0*((float)good/(float)ne));
507  if ( abs(imprim) > 3 ) {
508  fprintf(stdout," %6.2f %% > 0.5\n",100.0*( (float)med/(float)ne));
509  imax = MG_MIN(4,(int)(5.*max));
510  for (i=imax; i>=(int)(5*min); i--) {
511  fprintf(stdout," %5.1f < Q < %5.1f %7d %6.2f %%\n",
512  i/5.,i/5.+0.2,his[i],100.*((float)his[i]/(float)ne));
513  }
514  if ( nrid ) {
515  fprintf(stdout,"\n ## WARNING: %d TETRA WITH 4 RIDGES POINTS:"
516  " UNABLE TO COMPUTE ANISO QUALITY.\n",nrid);
517  }
518  }
519  }
520  }
521 
522  return MMG5_minQualCheck(iel,min,1.);
523 }
