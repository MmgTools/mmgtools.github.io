143 void MMG3D_compute_tangent(MMG5_pMesh mesh,int nump,int ip0,int ip1,double t[3]) {
144  MMG5_pPoint ppt,p0,p1;
145  double l0,l1,dd;
146  int8_t i;
147 
148  ppt = &mesh->point[nump];
149  p0 = &mesh->point[ip0];
150  p1 = &mesh->point[ip1];
151 
152  l0 = (ppt->c[0] - p0->c[0])*(ppt->c[0] - p0->c[0]) \
153  + (ppt->c[1] - p0->c[1])*(ppt->c[1] - p0->c[1]) + (ppt->c[2] - p0->c[2])*(ppt->c[2] - p0->c[2]);
154  l1 = (ppt->c[0] - p1->c[0])*(ppt->c[0] - p1->c[0]) \
155  + (ppt->c[1] - p1->c[1])*(ppt->c[1] - p1->c[1]) + (ppt->c[2] - p1->c[2])*(ppt->c[2] - p1->c[2]);
156  l0 = sqrt(l0);
157  l1 = sqrt(l1);
158 
159  if ( (l0 < MMG5_EPSD2) || (l1 < MMG5_EPSD2) ) {
160  for ( i=0; i<3; ++i ) {
161  t[i] = p1->c[i] - p0->c[i];
162  }
163  }
164  else if ( l0 < l1 ) {
165  dd = l0 / l1;
166  for ( i=0; i<3; ++i ) {
167  t[i] = dd*(p1->c[i] - ppt->c[i]) + ppt->c[i] - p0->c[i];
168  }
169  }
170  else {
171  dd = l1 / l0;
172  for ( i=0; i<3; ++i ) {
173  t[i] = dd*(p0->c[i] - ppt->c[i]) + ppt->c[i] - p1->c[i];
174  }
175  }
176 
177  return;
178 }

193 int MMG5_boulenm(MMG5_pMesh mesh,int start,int ip,int iface,

195  MMG5_pTetra pt;
196  double dd,nt[3];
197  int base,nump,nr,nnm,k,piv,na,nb,adj,nvstart,fstart,aux,ip0,ip1;
198  int *adja;
199  int16_t tag;
200  int8_t iopp,ipiv,indb,inda,i,isface;
201  int8_t indedg[4][4] = { {-1,0,1,2}, {0,-1,3,4}, {1,3,-1,5}, {2,4,5,-1} };
202 
203  base = ++mesh->base;
204  nr = nnm = 0;
205  ip0 = ip1 = 0;
206 
207  memset(n,0x00,3*sizeof(double));
208  memset(t,0x00,3*sizeof(double));
209 
210  pt = &mesh->tetra[start];
211  nump = pt->v[ip];
212  k = start;
213 
214  na = pt->v[ip];
215  nb = pt->v[MMG5_idir[iface][MMG5_inxt2[MMG5_idirinv[iface][ip]]]];
216  piv = pt->v[MMG5_idir[iface][MMG5_iprv2[MMG5_idirinv[iface][ip]]]];
217 
218  iopp = iface;
219  fstart = 4*k+iopp;
220  do {
221  /* computation of normal and tangent at nump */
222  if ( MMG5_norface(mesh,k,iopp,nt) ) {
223  n[0] += nt[0];
224  n[1] += nt[1];
225  n[2] += nt[2];
226  }
227 
228  if ( pt->xt ) {
229  for ( inda=0; inda<4; inda++ ){
230  if ( pt->v[inda]==na ) break;
231  }
232  for ( indb=0; indb<4; indb++ ){
233  if ( pt->v[indb]==nb ) break;
234  }
235  assert( (inda < 4) && (indb < 4));
236  tag = mesh->xtetra[pt->xt].tag[indedg[inda][indb]];
237  }
238 
239  else tag = 0;
240 
241  if ( MG_EDG(tag) && !(tag & MG_NOM) )
242  nr++;
243  else if ( tag & MG_NOM ) {
244  nnm++;
245  if ( !ip0 )
246  ip0 = nb;
247  else
248  ip1 = nb;
249  }
250 
251  /* A boundary face has been hit : change travel edge */
252  aux = nb;
253  nb = piv;
254  piv = aux;
255  nvstart = k;
256  adj = k;
257 
258  /* Now unfold shell of edge (na,nb) starting from k (included) */
259  do {
260  k = adj;
261  pt = &mesh->tetra[k];
262  adja = &mesh->adja[4*(k-1)+1];
263  if ( pt->flag != base ) {
264  for (i=0; i<4; i++)
265  if ( pt->v[i] == nump ) break;
266  assert(i<4);
267  pt->flag = base;
268  }
269 
270  /* identification of edge number in tetra k */
271  if ( !MMG3D_findEdge(mesh,pt,k,na,nb,1,NULL,&i) ) return -1;
272 
273  /* set sense of travel */
274  if ( pt->v[ MMG5_ifar[i][0] ] == piv ) {
275  adj = adja[ MMG5_ifar[i][0] ] / 4;
276  ipiv = MMG5_ifar[i][1];
277  iopp = MMG5_ifar[i][0];
278  piv = pt->v[ipiv];
279  }
280  else {
281  adj = adja[ MMG5_ifar[i][1] ] / 4;
282  ipiv = MMG5_ifar[i][0];
283  iopp = MMG5_ifar[i][1];
284  piv = pt->v[ipiv];
285  }
286  isface = (adja[iopp] == 0);
287  }
288  while ( adj && (adj != nvstart) && !isface );
289  }
290  while ( 4*k+iopp != fstart );
291 
292  if ( (nr > 0 && nnm > 0) || nnm != 2 ) return 0;
293 
294  dd = n[0]*n[0] + n[1]*n[1] + n[2]*n[2];
295  if ( dd > MMG5_EPSD2 ) {
296  dd = 1.0 / sqrt(dd);
297  n[0] *= dd;
298  n[1] *= dd;
299  n[2] *= dd;
300  }
301  assert( ip0 && ip1 );
302  if ( ip0 == ip1 ) return 0;
303 
304  MMG3D_compute_tangent(mesh,nump,ip0,ip1,t);
305 
306  dd = t[0]*n[0] + t[1]*n[1] + t[2]*n[2];
307  t[0] -= dd*n[0];
308  t[1] -= dd*n[1];
309  t[2] -= dd*n[2];
310 
311  dd = t[0]*t[0] + t[1]*t[1] + t[2]*t[2];
312  if ( dd > MMG5_EPSD2 ) {
313  dd = 1.0 / sqrt(dd);
314  t[0] *= dd;
315  t[1] *= dd;
316  t[2] *= dd;
317  }
318 
319  return 1;
320 }
