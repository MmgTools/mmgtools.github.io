253static inline
254void * myrealloc(void * ptr_in, size_t s, size_t oldsize) {
255 char *ptr;
256 char *ptr_in_c = (char*)ptr_in;
257
258 if ( !ptr_in ) {
259 assert ( !oldsize );
260 return mymalloc( s );
261 }
262
263 ptr_in_c -= sizeof(size_t);
264 if (oldsize != *((size_t*)ptr_in_c)) {
265 fprintf(stderr, "myrealloc: Error: freed memory mismatch\n");
266 assert(0);
267 }
268 ptr = (char *)realloc(ptr_in_c, s+sizeof(size_t));
269 if (ptr == NULL)
270 return NULL;
271 else {
272 *((size_t*)ptr)=s;
273 ptr+=sizeof(size_t);
274 return (void*)ptr;
275 }
