39 int MMG3D_newPt(MMG5_pMesh mesh,double c[3],int16_t tag,int src) {
40  MMG5_pPoint ppt;
41  int curpt;
42 
43  if ( !mesh->npnil ) return 0;
44  curpt = mesh->npnil;
45  if ( mesh->npnil > mesh->np ) mesh->np = mesh->npnil;
46  ppt = &mesh->point[curpt];
47  memcpy(ppt->c,c,3*sizeof(double));
48  mesh->npnil = ppt->tmp;
49  ppt->tmp = 0;
50 
51  ppt->ref = 0;
52  ppt->xp = 0;
53  ppt->flag = 0;
54  /* point on geometry */
55  if ( tag & MG_BDY ) {
56  mesh->xp++;
57  if(mesh->xp > mesh->xpmax){
58  /* reallocation of xpoint table */
59  MMG5_TAB_RECALLOC(mesh,mesh->xpoint,mesh->xpmax,MMG5_GAP,MMG5_xPoint,
60  "larger xpoint table",
61  return 0);
62  }
63  ppt->xp = mesh->xp;
64  }
65  assert(tag < 24704);
66  assert(tag >= 0);
67  ppt->n[0] = 0;
68  ppt->n[1] = 0;
69  ppt->n[2] = 0;
70  ppt->tag = tag;
71  ppt->tagdel = 0;
72 #ifdef USE_POINTMAP
73  assert( src );
74  ppt->src = src;
75 #endif
76  return curpt;
77 }

172 int MMG3D_memOption_memRepartition(MMG5_pMesh mesh) {
173  size_t usedMem,avMem,reservedMem,npadd;
174  int ctri,bytes;
175 
176  /* init allocation need MMG5_MEMMIN B */
177  reservedMem = MMG5_MEMMIN +
178  (mesh->nprism*sizeof(MMG5_Prism)+mesh->nquad*sizeof(MMG5_Quad));
179 
180  /* Compute the needed initial memory */
181  usedMem = reservedMem + (mesh->np+1)*sizeof(MMG5_Point)
182  + (mesh->nt+1)*sizeof(MMG5_Tria) + (mesh->ne+1)*sizeof(MMG5_Tetra)
183  + (3*mesh->nt+1)*sizeof(int) + (4*mesh->ne+1)*sizeof(int)
184  + (mesh->np+1)*sizeof(double);
185 
186  if ( usedMem > mesh->memMax ) {
187  fprintf(stderr,"\n ## Error: %s: %zu MB of memory ",__func__,mesh->memMax/MMG5_MILLION);
188  fprintf(stderr,"is not enough to load mesh. You need to ask %zu MB minimum\n",
189  usedMem/MMG5_MILLION+1);
190  return 0;
191  }
192 
193  ctri = 2;
194  /* Euler-poincare: ne = 6*np; nt = 2*np; na = np/5 *
195  * point+tria+tets+adja+adjt+aniso sol+item */
196  bytes = sizeof(MMG5_Point) + sizeof(MMG5_xPoint) +
197  6*sizeof(MMG5_Tetra) + ctri*sizeof(MMG5_xTetra) +
198  4*6*sizeof(int) + ctri*3*sizeof(int) +
199  4*sizeof(MMG5_hedge)+6*sizeof(double);
200 
201 #ifdef USE_SCOTCH
202  /* bytes = bytes + vertTab + edgeTab + PermVrtTab *
203  * + vertOldTab + sortPartTab - adja */
204  bytes = bytes + 3*6*sizeof(int);
205 #endif
206 
207  avMem = mesh->memMax-usedMem;
208 
209  /* If npadd is exactly the maximum memory available, we will use all the
210  * memory and the analysis step will fail. As arrays may be reallocated, we
211  * can have smaller values for npmax,ntmax and nemax (npadd/2). */
212  npadd = avMem/(2*bytes);
213  mesh->npmax = MG_MIN(mesh->npmax,mesh->np+npadd);
214  mesh->ntmax = MG_MIN(mesh->ntmax,ctri*npadd+mesh->nt);
215  mesh->nemax = MG_MIN(mesh->nemax,6*npadd+mesh->ne);
216 
217  /* check if the memory asked is enough to load the mesh*/
218  if ( abs(mesh->info.imprim) > 4 || mesh->info.ddebug ) {
219  fprintf(stdout," MAXIMUM MEMORY AUTHORIZED (MB) %zu\n",
220  mesh->memMax/MMG5_MILLION);
221  }
222  if ( abs(mesh->info.imprim) > 5 || mesh->info.ddebug ) {
223  fprintf(stdout," MMG3D_NPMAX %d\n",mesh->npmax);
224  fprintf(stdout," MMG3D_NTMAX %d\n",mesh->ntmax);
225  fprintf(stdout," MMG3D_NEMAX %d\n",mesh->nemax);
226  }
227 
228  return 1;
229 }

256 int MMG3D_setMeshSize_alloc( MMG5_pMesh mesh ) {
257  int k;
258 
259  MMG5_ADD_MEM(mesh,(mesh->npmax+1)*sizeof(MMG5_Point),"initial vertices",
260  fprintf(stderr," Exit program.\n");
261  return 0);
262  MMG5_SAFE_CALLOC(mesh->point,mesh->npmax+1,MMG5_Point,return 0);
263 
264  MMG5_ADD_MEM(mesh,(mesh->nemax+1)*sizeof(MMG5_Tetra),"initial tetrahedra",
265  fprintf(stderr," Exit program.\n");
266  return 0);
267  MMG5_SAFE_CALLOC(mesh->tetra,mesh->nemax+1,MMG5_Tetra,return 0);
268 
269  if ( mesh->nprism ) {
270  MMG5_ADD_MEM(mesh,(mesh->nprism+1)*sizeof(MMG5_Prism),"initial prisms",return 0);
271  MMG5_SAFE_CALLOC(mesh->prism,(mesh->nprism+1),MMG5_Prism,return 0);
272  }
273 
274  if ( mesh->nt ) {
275  MMG5_ADD_MEM(mesh,(mesh->nt+1)*sizeof(MMG5_Tria),"initial triangles",return 0);
276  MMG5_SAFE_CALLOC(mesh->tria,mesh->nt+1,MMG5_Tria,return 0);
277  memset(&mesh->tria[0],0,sizeof(MMG5_Tria));
278  }
279 
280  if ( mesh->nquad ) {
281  MMG5_ADD_MEM(mesh,(mesh->nquad+1)*sizeof(MMG5_Quad),"initial quadrilaterals",return 0);
282  MMG5_SAFE_CALLOC(mesh->quadra,(mesh->nquad+1),MMG5_Quad,return 0);
283  }
284 
285  mesh->namax = mesh->na;
286  if ( mesh->na ) {
287  MMG5_ADD_MEM(mesh,(mesh->na+1)*sizeof(MMG5_Edge),"initial edges",return 0);
288  MMG5_SAFE_CALLOC(mesh->edge,(mesh->na+1),MMG5_Edge,return 0);
289  }
290 
291  /* keep track of empty links */
292  mesh->npnil = mesh->np + 1;
293  mesh->nenil = mesh->ne + 1;
294 
295  for (k=mesh->npnil; k<mesh->npmax-1; k++) {
296  /* link */
297  mesh->point[k].tmp = k+1;
298  }
299 
300  for (k=mesh->nenil; k<mesh->nemax-1; k++)
301  mesh->tetra[k].v[3] = k+1;
302 
303  return 1;
304 }
