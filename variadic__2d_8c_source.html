55 int MMG2D_Alloc_mesh(MMG5_pMesh *mesh, MMG5_pSol *met, MMG5_pSol *ls,
56  MMG5_pSol *disp) {
57 
58  /* mesh allocation */
59  if ( *mesh ) MMG5_SAFE_FREE(*mesh);
60  MMG5_SAFE_CALLOC(*mesh,1,MMG5_Mesh,return 0);
61 
62  /* metric allocation */
63  if ( met ) {
64  if ( *met ) MMG5_DEL_MEM(*mesh,*met);
65  MMG5_SAFE_CALLOC(*met,1,MMG5_Sol,return 0);
66  }
67 
68  /* level-set allocation in ls mode */
69  if ( ls ) {
70  if ( *ls )
71  MMG5_DEL_MEM(*mesh,*ls);
72  MMG5_SAFE_CALLOC(*ls,1,MMG5_Sol,return 0);
73  }
74 
75  /* Displacement allocation */
76  if ( disp ) {
77  if ( *disp )
78  MMG5_DEL_MEM(*mesh,*disp);
79  MMG5_SAFE_CALLOC(*disp,1,MMG5_Sol,return 0);
80  }
81 
82  return 1;
83 }
