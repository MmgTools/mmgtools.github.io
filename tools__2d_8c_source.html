46MMG5_int MMG2D_indElt(MMG5_pMesh mesh, MMG5_int kel) {
47 MMG5_pTria pt;
48 MMG5_int ne, k;
49
50 ne = 0;
51 for (k=1; k<=mesh->nt; k++) {
52 pt = &mesh->tria[k];
53 if ( MG_EOK(pt) ) {
54 ne++;
55 if ( k == kel ) return ne;
56 }
57 }
58 return 0;
59}
