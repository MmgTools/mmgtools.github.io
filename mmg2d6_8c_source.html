45 inline double MMG2D_voltri(MMG5_pMesh mesh,int ip0,int ip1,int ip2) {
46  MMG5_pPoint p0,p1,p2;
47  double vol;
48 
49  p0 = &mesh->point[ip0];
50  p1 = &mesh->point[ip1];
51  p2 = &mesh->point[ip2];
52 
53  vol = (p1->c[0]-p0->c[0])*(p2->c[1]-p0->c[1]) - (p1->c[1]-p0->c[1])*(p2->c[0]-p0->c[0]);
54  vol = 0.5*fabs(vol);
55 
56  return vol;

69 double MMG2D_vfrac(MMG5_pMesh mesh,MMG5_pSol sol,int k,int pm) {
70  MMG5_pTria pt;
71  MMG5_pPoint ppt[3];
72  double v[3],vfp,vfm,lam,area,eps,o1[2],o2[2];
73  int ip[3],nplus,nminus,nzero;
74  int8_t i,i0,i1,i2,imin1,iplus1,iz;
75 
76  eps = MMG5_EPS*MMG5_EPS;
77  pt = &mesh->tria[k];
78 
79  ip[0] = pt->v[0];
80  ip[1] = pt->v[1];
81  ip[2] = pt->v[2];
82 
83  ppt[0] = &mesh->point[ip[0]];
84  ppt[1] = &mesh->point[ip[1]];
85  ppt[2] = &mesh->point[ip[2]];
86 
87  v[0] = sol->m[ip[0]];
88  v[1] = sol->m[ip[1]];
89  v[2] = sol->m[ip[2]];
90 
91  /* Identify number of zero, positive and negative vertices, and corresponding indices */
92  nplus = nminus = nzero = 0;
93  imin1 = iplus1 = iz = -1;
94 
95  for (i=0; i<3; i++) {
96  if ( fabs(v[i]) < eps ) {
97  nzero++;
98  if ( iz < 0 ) iz = i;
99  }
100  else if ( v[i] >= eps ) {
101  nplus++;
102  if ( iplus1 < 0 ) iplus1 = i;
103  }
104  else {
105  nminus++;
106  if ( imin1 < 0 ) imin1 = i;
107  }
108  }
109 
110  /* Degenerate case */
111  if ( nzero == 3 ) return 0.0;
112 
113  /* Whole triangle is positive */
114  if ( nminus == 0 ) {
115  vfp = (ppt[1]->c[0]-ppt[0]->c[0])*(ppt[2]->c[1]-ppt[0]->c[1]) - (ppt[1]->c[1]-ppt[0]->c[1])*(ppt[2]->c[0]-ppt[0]->c[0]);
116  vfp = 0.5*fabs(vfp);
117  if ( pm == 1 ) return vfp;
118  else return 0.0;
119  }
120 
121  /* Whole triangle is negative */
122  if ( nplus == 0 ) {
123  vfm = (ppt[1]->c[0]-ppt[0]->c[0])*(ppt[2]->c[1]-ppt[0]->c[1]) - (ppt[1]->c[1]-ppt[0]->c[1])*(ppt[2]->c[0]-ppt[0]->c[0]);
124  vfm = 0.5*fabs(vfm);
125  if ( pm == -1 ) return vfm;
126  else return 0.0;
127  }
128 
129  /* Exactly one vertex is negative */
130  if ( nminus == 1 ) {
131  i0 = imin1;
132  i1 = MMG5_inxt2[i0];
133  i2 = MMG5_iprv2[i0];
134 
135  lam = v[i0] / (v[i0]-v[i1]);
136  o1[0] = ppt[i0]->c[0] + lam*(ppt[i1]->c[0]-ppt[i0]->c[0]);
137  o1[1] = ppt[i0]->c[1] + lam*(ppt[i1]->c[1]-ppt[i0]->c[1]);
138 
139  lam = v[i0] / (v[i0]-v[i2]);
140  o2[0] = ppt[i0]->c[0] + lam*(ppt[i2]->c[0]-ppt[i0]->c[0]);
141  o2[1] = ppt[i0]->c[1] + lam*(ppt[i2]->c[1]-ppt[i0]->c[1]);
142 
143  vfm = (o1[0]-ppt[i0]->c[0])*(o2[1]-ppt[i0]->c[1]) - (o1[1]-ppt[i0]->c[1])*(o2[0]-ppt[i0]->c[0]);
144  vfm = 0.5*fabs(vfm);
145 
146  if ( pm == -1 ) return vfm;
147  else {
148  area = (ppt[1]->c[0]-ppt[0]->c[0])*(ppt[2]->c[1]-ppt[0]->c[1]) - (ppt[1]->c[1]-ppt[0]->c[1])*(ppt[2]->c[0]-ppt[0]->c[0]);
149  area = 0.5*fabs(area);
150  vfp = area-vfm;
151  return vfp;
152  }
153  }
154 
155  /* Exactly one vertex is positive */
156  if ( nplus == 1 ) {
157  i0 = iplus1;
158  i1 = MMG5_inxt2[i0];
159  i2 = MMG5_iprv2[i0];
160 
161  lam = v[i0] / (v[i0]-v[i1]);
162  o1[0] = ppt[i0]->c[0] + lam*(ppt[i1]->c[0]-ppt[i0]->c[0]);
163  o1[1] = ppt[i0]->c[1] + lam*(ppt[i1]->c[1]-ppt[i0]->c[1]);
164 
165  lam = v[i0] / (v[i0]-v[i2]);
166  o2[0] = ppt[i0]->c[0] + lam*(ppt[i2]->c[0]-ppt[i0]->c[0]);
167  o2[1] = ppt[i0]->c[1] + lam*(ppt[i2]->c[1]-ppt[i0]->c[1]);
168 
169  vfp = (o1[0]-ppt[i0]->c[0])*(o2[1]-ppt[i0]->c[1]) - (o1[1]-ppt[i0]->c[1])*(o2[0]-ppt[i0]->c[0]);
170  vfp = 0.5*fabs(vfp);
171 
172  if ( pm == 1 ) return vfp;
173  else {
174  area = (ppt[1]->c[0]-ppt[0]->c[0])*(ppt[2]->c[1]-ppt[0]->c[1]) - (ppt[1]->c[1]-ppt[0]->c[1])*(ppt[2]->c[0]-ppt[0]->c[0]);
175  area = 0.5*fabs(area);
176  vfm = area-vfp;
177  return vfm;
178  }
179  }
180 
181  /* Should not pass here */
182  return(0.0);
183 }

221 int MMG2D_ismaniball(MMG5_pMesh mesh, MMG5_pSol sol, int start, int8_t istart) {
222  MMG5_pTria pt;
223  double v1, v2;
224  int *adja,k,ip1,ip2,end1,refstart;
225  int8_t i,i1,smsgn;
226  static int8_t mmgWarn=0;
227 
228  k = start;
229  refstart = mesh->tria[k].ref;
230  i = MMG5_inxt2[istart];
231 
232  /* First loop: stop if an external boundary, or a change in signs (or a 0) is met
233  recall that MG_SMGSGN(a,b) = 1 provided a*b >0 */
234  do{
235  adja = &mesh->adja[3*(k-1)+1];
236  k = adja[i] / 3;
237  i1 = adja[i] % 3;
238  i = MMG5_iprv2[i1];
239 
240  if ( k==0 ) break;
241 
242  pt = &mesh->tria[k];
243 
244  ip1 = pt->v[i1];
245  ip2 = pt->v[i];
246 
247  v1 = sol->m[ip1];
248  v2 = sol->m[ip2];
249 
250  /* Authorize change of references only provided the boundary reference is MG_ISO */
251  if ( pt->ref != refstart && pt->edg[i1] != MG_ISO ) {
252  smsgn = 0;
253  k = 0;
254  } else
255  smsgn = (fabs(v1) < MMG5_EPS) || ( (fabs(v2) > MMG5_EPS) && MG_SMSGN(v1,v2) ) ? 1 : 0;
256  // smsgn = MG_SMSGN(v1,v2) ? 1 : 0;
257  }
258  while ( smsgn );
259 
260  end1 = k;
261  k = start;
262  i = MMG5_iprv2[istart];
263 
264  /* Second loop: same travel in the opposite sense */
265  do{
266  adja = &mesh->adja[3*(k-1)+1];
267  k = adja[i] / 3;
268  i1 = adja[i] % 3;
269  i = MMG5_inxt2[i1];
270 
271  if ( k==0 ) break;
272 
273  pt = &mesh->tria[k];
274  ip1 = pt->v[i1];
275  ip2 = pt->v[i];
276 
277  v1 = sol->m[ip1];
278  v2 = sol->m[ip2];
279 
280  if ( pt->ref != refstart && pt->edg[i1] != MG_ISO ) {
281  smsgn = 0;
282  k = 0;
283  } else
284  smsgn = (fabs(v2) < MMG5_EPS) || ( (fabs(v1) > MMG5_EPS) && MG_SMSGN(v1,v2) ) ? 1 : 0;
285  // smsgn = MG_SMSGN(v1,v2) ? 1 : 0;
286  }
287  while ( smsgn );
288 
289  /* If first stop was due to an external boundary, the second one must too;
290  else, the final triangle for the first travel must be that of the second one */
291  if ( k != end1 ) {
292  if ( !mmgWarn ) {
293  mmgWarn = 1;
294  fprintf(stderr,"\n ## Warning: %s: unsnap at least 1 point "
295  "(point %d in tri %d).\n",__func__,MMG2D_indElt(mesh,start),
296  MMG2D_indPt(mesh,mesh->tria[start].v[istart]));
297  }
298  return 0;
299  }
300  return 1;
301 }

312 int MMG2D_snapval(MMG5_pMesh mesh, MMG5_pSol sol) {
313  MMG5_pTria pt,pt1;
314  MMG5_pPoint p0;
315  double v1,v2,*tmp;
316  int k,kk,iel,ns,nc,ip,ip1,ip2,npl,nmn,ilist,list[MMG2D_LONMAX+2];
317  int8_t i,j,j1,j2;
318 
319  /* Allocate memory for tmp */
320  MMG5_ADD_MEM(mesh,(mesh->npmax+1)*sizeof(double),"temporary table",
321  printf(" Exit program.\n");
322  return 0);
323  MMG5_SAFE_CALLOC(tmp,mesh->npmax+1,double,return 0);
324 
325  /* Reset point flags */
326  for (k=1; k<=mesh->np; k++)
327  mesh->point[k].flag = 0;
328 
329  /* Snap values of sol that are close to 0 to 0 exactly */
330  ns = nc = 0;
331  for (k=1; k<=mesh->np; k++) {
332  p0 = &mesh->point[k];
333  if ( !MG_VOK(p0) ) continue;
334  if ( fabs(sol->m[k]) < MMG5_EPS ) {
335  tmp[k] = sol->m[k];
336  p0->flag = 1;
337  sol->m[k] = 0.0;
338  ns++;
339  }
340  }
341 
342  /* Check that the snapping process has not led to a nonmanifold situation */
343  for (k=1; k<=mesh->nt; k++) {
344  pt = &mesh->tria[k];
345  if ( !MG_EOK(pt) ) continue;
346  for (i=0; i<3; i++) {
347  ip = pt->v[i];
348  ip1 = pt->v[MMG5_inxt2[i]];
349  ip2 = pt->v[MMG5_iprv2[i]];
350 
351  p0 = &mesh->point[ip];
352  v1 = sol->m[ip1];
353  v2 = sol->m[ip2];
354 
355  /* Catch a snapped point by a triangle where there is a sign change */
356  if ( p0->flag && !(MG_SMSGN(v1,v2)) ) {
357  if ( !MMG2D_ismaniball(mesh,sol,k,i) ) {
358  sol->m[ip] = tmp[ip];
359  nc++;
360  }
361  p0->flag = 0;
362  }
363  }
364  }
365 
366  /* Check that the ls function does not show isolated spots with 0 values (without sign changes) */
367  for (k=1; k<=mesh->nt; k++) {
368  pt = &mesh->tria[k];
369  if ( !MG_EOK(pt) ) continue;
370  for (i=0; i<3; i++) {
371  ip = pt->v[i];
372  if ( fabs(sol->m[ip]) >= MMG5_EPS ) continue;
373  npl = nmn = 0;
374  ilist = MMG2D_boulet(mesh,k,i,list);
375  for(kk=0; kk<ilist; kk++) {
376  iel = list[kk] / 3;
377  j = list[kk] % 3;
378  j1 = MMG5_inxt2[j];
379  j2 = MMG5_iprv2[i];
380  pt1 = &mesh->tria[iel];
381  ip1 = pt1->v[j1];
382  ip2 = pt1->v[j2];
383  if ( sol->m[ip1] >= MMG5_EPS ) npl = 1;
384  else if ( sol->m[ip1] <= -MMG5_EPS ) nmn = 1;
385 
386  if ( sol->m[ip2] >= MMG5_EPS ) npl = 1;
387  else if ( sol->m[ip2] <= -MMG5_EPS ) nmn = 1;
388  }
389 
390  if ( npl == 1 && nmn == 0 )
391  sol->m[ip] = 100.0*MMG5_EPS;
392  else if ( npl == 0 && nmn == 1 )
393  sol->m[ip] = -100.0*MMG5_EPS;
394  }
395  }
396 
397  MMG5_DEL_MEM ( mesh, tmp );
398 
399  if ( (abs(mesh->info.imprim) > 5 || mesh->info.ddebug) && ns+nc > 0 )
400  fprintf(stdout," %8d points snapped, %d corrected\n",ns,nc);
401 
402  return 1;
403 }

406  by assumption, i inxt[i] is one edge of the implicit boundary */
407 int MMG2D_chkmaniball(MMG5_pMesh mesh, int start, int8_t istart) {
408  MMG5_pTria pt;
409  int *adja,k,refstart;
410  int8_t i,i1;
411 
412  pt = &mesh->tria[start];
413  k = start;
414  i = istart;
415  refstart = pt->ref;
416 
417  /* First travel, while another part of the implicit boundary is not met */
418  do {
419  adja = &mesh->adja[3*(k-1)+1];
420  i1 = MMG5_inxt2[i];
421 
422  k = adja[i1] / 3;
423  i = adja[i1] % 3;
424  i = MMG5_inxt2[i];
425  }
426  while ( k && ( mesh->tria[k].ref == refstart ) );
427 
428  /* Case where a boundary is hit: travel in the other sense from start, and make sure
429  that a boundary is hit too */
430  if ( k == 0 ) {
431  k = start;
432  i = istart;
433 
434  adja = &mesh->adja[3*(k-1)+1];
435  i1 = MMG5_iprv2[i];
436  k = adja[i1] / 3;
437  i = adja[i1] % 3;
438  i = MMG5_iprv2[i];
439 
440  /* Tested point is connected to two external edges */
441  if ( k == 0 ) return 1;
442 
443  do {
444  adja = &mesh->adja[3*(k-1)+1];
445  i1 = MMG5_iprv2[i];
446 
447  k = adja[i1] / 3;
448  i = adja[i1] % 3;
449  i = MMG5_iprv2[i];
450  }
451  while ( k && ( mesh->tria[k].ref != refstart ) );
452 
453  if ( k == 0 ) return 1;
454  else return 0;
455 
456  }
457 
458  /* General case: go on travelling until another implicit boundary is met */
459  do {
460  adja = &mesh->adja[3*(k-1)+1];
461  i1 = MMG5_inxt2[i];
462 
463  k = adja[i1] / 3;
464  i = adja[i1] % 3;
465  i = MMG5_inxt2[i];
466  }
467  while ( k && ( mesh->tria[k].ref != refstart ) );
468 
469  /* At least 3 boundary segments meeting at p */
470  if ( k != start )
471  return 0;
472 
473  return 1;
474 }

477  * discretization are manifold */
478 int MMG2D_chkmanimesh(MMG5_pMesh mesh) {
479  MMG5_pTria pt,pt1;
480  int *adja,k,cnt,iel;
481  int8_t i,i1;
482  static int8_t mmgWarn=0;
483 
484  /* First check: check whether one triangle in the mesh has 3 boundary faces */
485  for (k=1; k<=mesh->nt; k++) {
486  pt = &mesh->tria[k];
487  if ( !MG_EOK(pt) ) continue;
488 
489  adja = &mesh->adja[3*(k-1)+1];
490  cnt = 0;
491  for (i=0; i<3; i++) {
492  iel = adja[i] / 3;
493 
494  if (!iel ) {
495  cnt++;
496  continue;
497  }
498  else {
499  pt1 = &mesh->tria[iel];
500  if ( pt1->ref != pt->ref ) cnt++;
501  }
502  }
503  if( cnt == 3 ) {
504  if ( !mmgWarn ) {
505  mmgWarn = 1;
506  fprintf(stderr,"\n ## Warning: %s: at least 1 triangle with 3 boundary"
507  " edges.\n",__func__);
508  }
509  }
510  }
511 
512  /* Second check: check whether the configuration is manifold in the ball of
513  each point; each vertex on the implicit boundary is caught in such a way
514  that i1 inxt[i1] is one edge of the implicit boundary */
515  for (k=1; k<=mesh->nt; k++) {
516  pt = &mesh->tria[k];
517  if ( !MG_EOK(pt) ) continue;
518 
519  for (i=0; i<3; i++) {
520  adja = &mesh->adja[3*(k-1)+1];
521  iel = adja[i] / 3;
522 
523  if (! iel ) continue;
524  pt1 = &mesh->tria[iel];
525  if ( pt->ref == pt1->ref || pt->edg[i]!= MG_ISO ) continue;
526 
527  i1 = MMG5_inxt2[i];
528  if ( !MMG2D_chkmaniball(mesh,k,i1) ) {
529  fprintf(stderr," *** Topological problem\n");
530  fprintf(stderr," non manifold curve at point %d %d\n",pt->v[i1], MMG2D_indPt(mesh,pt->v[i1]));
531  fprintf(stderr," non manifold curve at tria %d (ip %d)\n", MMG2D_indElt(mesh,k),i1);
532  return 0;
533  }
534  }
535  }
536 
537  if ( mesh->info.imprim > 0 || mesh->info.ddebug )
538  fprintf(stdout," *** Manifold implicit surface.\n");
539  return 1;
540 }

552 int MMG2D_rmc(MMG5_pMesh mesh, MMG5_pSol sol){
553  MMG5_pTria pt,pt1,pt2;
554  double volc,voltot,v0,v1,v2;
555  int k,kk,l,ll,ncp,ncm,ip0,ip1,ip2,base,cur,ipile,*pile,*adja;
556  int8_t i,i1,i2;
557 
558  ncp = 0;
559  ncm = 0;
560 
561  /* Erase triangle flags */
562  for (k=1; k<=mesh->nt; k++) mesh->tria[k].flag = 0;
563 
564  /* Calculate volume of the total mesh */
565  voltot = 0.0;
566  for (k=1; k<=mesh->nt; k++) {
567  pt = &mesh->tria[k];
568  if ( !MG_EOK(pt) ) continue;
569  ip0 = pt->v[0];
570  ip1 = pt->v[1];
571  ip2 = pt->v[2];
572  voltot += MMG2D_voltri(mesh,ip0,ip1,ip2);
573  }
574 
575  /* Memory allocation for pile */
576  MMG5_ADD_MEM(mesh,(mesh->nt+1)*sizeof(int),"temporary table",
577  printf(" Exit program.\n");
578  return 0);
579  MMG5_SAFE_CALLOC(pile,mesh->nt+1,int,return 0);
580 
581  /* Investigate only positive connected components */
582  base = ++mesh->base;
583 
584  for (k=1; k<=mesh->nt; k++) {
585  ipile = 0;
586  volc = 0.0;
587  pt = &mesh->tria[k];
588  if ( !MG_EOK(pt) ) continue;
589  if ( pt->flag == base ) continue;
590 
591  /* Checks signs of the LS function at the 3 vertices of pt */
592  ip0 = pt->v[0];
593  ip1 = pt->v[1];
594  ip2 = pt->v[2];
595 
596  v0 = sol->m[ip0];
597  v1 = sol->m[ip1];
598  v2 = sol->m[ip2];
599 
600  if ( v0 <= 0.0 && v1 <= 0.0 && v2 <= 0.0 ) continue;
601 
602  /* Add triangle to pile if one vertex is > 0 */
603  pt->flag = base;
604  pile[ipile] = k;
605  ipile++;
606  if ( ipile > mesh->nt ) {
607  fprintf(stderr,"\n ## Problem in length of pile; function rmc.\n"
608  " Check that the level-set intersect the mesh.\n"
609  " Exit program.\n");
610 
611  return 0;
612  }
613 
614  /* Pile up all the positive connected component attached to the first triangle */
615  cur = 0;
616  do {
617  kk = pile[cur];
618  pt1 = &mesh->tria[kk];
619 
620  /* Add local volume fraction of the positive subdomain to volc */
621  volc += MMG2D_vfrac(mesh,sol,kk,1);
622 
623  /* Add adjacent triangles to kk via positive vertices to the pile, if need be */
624  adja = &mesh->adja[3*(kk-1)+1];
625  for (i=0; i<3; i++) {
626  ip0 = pt1->v[i];
