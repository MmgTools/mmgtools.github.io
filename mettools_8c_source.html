51 inline int
52 MMG5_buildridmetfic(MMG5_pMesh mesh,double t[3],double n[3],double dtan,
53  double dv,double dn,double m[6]) {
54  double u[3],r[3][3];
55 
56  u[0] = n[1]*t[2] - n[2]*t[1];
57  u[1] = n[2]*t[0] - n[0]*t[2];
58  u[2] = n[0]*t[1] - n[1]*t[0];
59 
60  /* If u = n1 ^ t, matrix of the desired metric in (t,u,n1) = diag(dtan,dv,dn)*/
61  r[0][0] = t[0]; r[0][1] = u[0]; r[0][2] = n[0];
62  r[1][0] = t[1]; r[1][1] = u[1]; r[1][2] = n[1];
63  r[2][0] = t[2]; r[2][1] = u[2]; r[2][2] = n[2];
64 
65  m[0] = dtan*r[0][0]*r[0][0] + dv*r[0][1]*r[0][1] + dn*r[0][2]*r[0][2];
66  m[1] = dtan*r[0][0]*r[1][0] + dv*r[0][1]*r[1][1] + dn*r[0][2]*r[1][2];
67  m[2] = dtan*r[0][0]*r[2][0] + dv*r[0][1]*r[2][1] + dn*r[0][2]*r[2][2];
68  m[3] = dtan*r[1][0]*r[1][0] + dv*r[1][1]*r[1][1] + dn*r[1][2]*r[1][2];
69  m[4] = dtan*r[1][0]*r[2][0] + dv*r[1][1]*r[2][1] + dn*r[1][2]*r[2][2];
70  m[5] = dtan*r[2][0]*r[2][0] + dv*r[2][1]*r[2][1] + dn*r[2][2]*r[2][2];
71 
72  return 1;
73 }

254 int MMG5_intersecmet22(MMG5_pMesh mesh, double *m,double *n,double *mr) {
255  double det,imn[4],dd,sqDelta,trimn,lambda[2],vp0[2],vp1[2],dm[2],dn[2],vnorm,d0,d1,ip[4];
256  double isqhmin,isqhmax;
257  static int8_t mmgWarn = 0, mmgWarn1 = 0;
258 
259  isqhmin = 1.0 / (mesh->info.hmin*mesh->info.hmin);
260  isqhmax = 1.0 / (mesh->info.hmax*mesh->info.hmax);
261 
262  /* Compute imn = M^{-1}N */
263  det = m[0]*m[2] - m[1]*m[1];
264  if ( fabs(det) < MMG5_EPS*MMG5_EPS ) {
265  if ( !mmgWarn ) {
266  fprintf(stderr,"\n ## Warning: %s: null metric det : %E \n",__func__,det);
267  mmgWarn = 1;
268  }
269  return 0;
270  }
271  det = 1.0 / det;
272 
273  imn[0] = det * ( m[2]*n[0] - m[1]*n[1]);
274  imn[1] = det * ( m[2]*n[1] - m[1]*n[2]);
275  imn[2] = det * (-m[1]*n[0] + m[0]*n[1]);
276  imn[3] = det * (-m[1]*n[1] + m[0]*n[2]);
277  dd = imn[0] - imn[3];
278  sqDelta = sqrt(fabs(dd*dd + 4.0*imn[1]*imn[2]));
279  trimn = imn[0] + imn[3];
280 
281  lambda[0] = 0.5 * (trimn - sqDelta);
282  if ( lambda[0] < 0.0 ) {
283  if ( !mmgWarn1 ) {
284  fprintf(stderr,"\n ## Warning: %s: negative eigenvalue (%f).\n",
285  __func__,lambda[0]);
286  mmgWarn1 = 1;
287  }
288  return 0;
289  }
290 
291  /* First case : matrices m and n are homothetic : n = lambda0*m */
292  if ( sqDelta < MMG5_EPS ) {
293  /* Diagonalize m and truncate eigenvalues : trimn, det, etc... are reused */
294  if ( fabs(m[1]) < MMG5_EPS ) {
295  dm[0] = m[0];
296  dm[1] = m[2];
297  vp0[0] = 1;
298  vp0[1] = 0;
299  vp1[0] = 0;
300  vp1[1] = 1;
301  }
302  else {
303  dd = m[0] - m[2];
304  trimn = m[0] + m[2];
305 
306  sqDelta = sqrt(fabs(dd*dd +4*0*m[1]*m[1]));
307  dm[0] = 0.5 * (trimn + sqDelta);
308  dm[1] = 0.5 * (trimn - sqDelta);
309 
310  if(fabs(dm[0]-dm[1]) < MMG5_EPS) {
311  vp0[1] = 0;
312  vp1[0] = 0;
313  vp1[1] = 1;
314  vp0[0] = 1;
315  vp0[1] = 0;
316  vp1[0] = 0;
317  vp1[1] = 1;
318  } else {
319  vp0[0] = m[1];
320  vp0[1] = (dm[0]-m[0]);
321  vnorm = sqrt(vp0[0]*vp0[0] + vp0[1]*vp0[1]);
322  if ( vnorm < MMG5_EPS ) {
323  vp0[0] = (dm[0] - m[2]);
324  vp0[1] = m[1];
325  vnorm = sqrt(vp0[0]*vp0[0] + vp0[1]*vp0[1]);
326 
327  if ( vnorm < MMG5_EPS ) return 0;
328  }
329 
330  vnorm = 1.0 / vnorm;
331  vp0[0] *= vnorm;
332  vp0[1] *= vnorm;
333 
334  vp1[0] = m[1];
335  vp1[1] = (dm[1]-m[0]);
336  vnorm = sqrt(vp1[0]*vp1[0] + vp1[1]*vp1[1]);
337 
338  if ( vnorm < MMG5_EPS ) {
339  vp1[0] = (dm[1] - m[2]);
340  vp1[1] = m[1];
341  vnorm = sqrt(vp1[0]*vp1[0] + vp1[1]*vp1[1]);
342 
343  if ( vnorm < MMG5_EPS ) return 0;
344  }
345 
346  vnorm = 1.0 / vnorm;
347  vp1[0] *= vnorm;
348  vp1[1] *= vnorm;
349  }
350  }
351  /* Eigenvalues of the resulting matrix*/
352  dn[0] = MG_MAX(dm[0],lambda[0]*dm[0]);
353  dn[0] = MG_MIN(isqhmin,MG_MAX(isqhmax,dn[0]));
354  dn[1] = MG_MAX(dm[1],lambda[0]*dm[1]);
355  dn[1] = MG_MIN(isqhmin,MG_MAX(isqhmax,dn[1]));
356 
357  /* Intersected metric = P diag(d0,d1){^t}P, P = (vp0, vp1) stored in columns */
358  mr[0] = dn[0]*vp0[0]*vp0[0] + dn[1]*vp1[0]*vp1[0];
359  mr[1] = dn[0]*vp0[0]*vp0[1] + dn[1]*vp1[0]*vp1[1];
360  mr[2] = dn[0]*vp0[1]*vp0[1] + dn[1]*vp1[1]*vp1[1];
361 
362  return 1;
363  }
364 
365  /* Second case : both eigenvalues of imn are distinct ; theory says qf associated to m and n
366  are diagonalizable in basis (vp0, vp1) - the coreduction basis */
367  else {
368  lambda[1] = 0.5 * (trimn + sqDelta);
369  assert(lambda[1] >= 0.0);
370 
371  vp0[0] = imn[1];
372  vp0[1] = (lambda[0] - imn[0]);
373  vnorm = sqrt(vp0[0]*vp0[0] + vp0[1]*vp0[1]);
374 
375  if ( vnorm < MMG5_EPS ) {
376  vp0[0] = (lambda[0] - imn[3]);
377  vp0[1] = imn[2];
378  vnorm = sqrt(vp0[0]*vp0[0] + vp0[1]*vp0[1]);
379  }
380 
381  vnorm = 1.0 / vnorm;
382  vp0[0] *= vnorm;
383  vp0[1] *= vnorm;
384 
385  vp1[0] = imn[1];
386  vp1[1] = (lambda[1] - imn[0]);
387  vnorm = sqrt(vp1[0]*vp1[0] + vp1[1]*vp1[1]);
388 
389  if ( vnorm < MMG5_EPS ) {
390  vp1[0] = (lambda[1] - imn[3]);
391  vp1[1] = imn[2];
392  vnorm = sqrt(vp1[0]*vp1[0] + vp1[1]*vp1[1]);
393  }
394 
395  vnorm = 1.0 / vnorm;
396  vp1[0] *= vnorm;
397  vp1[1] *= vnorm;
398 
399  /* Compute diagonal values in simultaneous reduction basis */
400  dm[0] = m[0]*vp0[0]*vp0[0] + 2.0*m[1]*vp0[0]*vp0[1] + m[2]*vp0[1]*vp0[1];
401  dm[1] = m[0]*vp1[0]*vp1[0] + 2.0*m[1]*vp1[0]*vp1[1] + m[2]*vp1[1]*vp1[1];
402  dn[0] = n[0]*vp0[0]*vp0[0] + 2.0*n[1]*vp0[0]*vp0[1] + n[2]*vp0[1]*vp0[1];
403  dn[1] = n[0]*vp1[0]*vp1[0] + 2.0*n[1]*vp1[0]*vp1[1] + n[2]*vp1[1]*vp1[1];
404 
405  /* Diagonal values of the intersected metric */
406  d0 = MG_MAX(dm[0],dn[0]);
407  d0 = MG_MIN(isqhmin,MG_MAX(d0,isqhmax));
408 
409  d1 = MG_MAX(dm[1],dn[1]);
410  d1 = MG_MIN(isqhmin,MG_MAX(d1,isqhmax));
411 
412  /* Intersected metric = tP^-1 diag(d0,d1)P^-1, P = (vp0, vp1) stored in columns */
413  det = vp0[0]*vp1[1] - vp0[1]*vp1[0];
414  if ( fabs(det) < MMG5_EPS ) return 0;
415  det = 1.0 / det;
416 
417  ip[0] = vp1[1]*det;
418  ip[1] = -vp1[0]*det;
419  ip[2] = -vp0[1]*det;
420  ip[3] = vp0[0]*det;
421 
422  mr[0] = d0*ip[0]*ip[0] + d1*ip[2]*ip[2];
423  mr[1] = d0*ip[0]*ip[1] + d1*ip[2]*ip[3];
424  mr[2] = d0*ip[1]*ip[1] + d1*ip[3]*ip[3];
425  }
426  return 1;
427 }

443 int MMG5_mmgIntextmet(MMG5_pMesh mesh,MMG5_pSol met,int np,double me[6],
444  double n[3]) {
445  MMG5_pPoint p0;
446  MMG5_pxPoint go;
447  double hu,isqhmin,isqhmax,dd,alpha1,alpha2,alpha3,u[3];
448  double lambda[3],vp[3][3];
449  double *m,*n1,*n2,*t,r[3][3],mrot[6],mr[3],mtan[3],metan[3];
450  int order;
451  int8_t i;
452  static int8_t mmgWarn=0, mmgWarn1=0, mmgWarn2=0;
453 
454  isqhmin = 1.0 / (mesh->info.hmin*mesh->info.hmin);
455  isqhmax = 1.0 / (mesh->info.hmax*mesh->info.hmax);
456 
457  p0 = &mesh->point[np];
458  m = &met->m[6*np];
459 
460  /* Case of a singular point : take the physical metric */
461  if ( MG_SIN(p0->tag) || (p0->tag & MG_NOM) ) {
462  for(i=0 ; i<6; i++) {
463  m[i] = me[i];
464  }
465  }
466  /* Case of a ridge point : take sizes in 3 directions t,n1,u */
467  else if ( p0->tag & MG_GEO ) {
468  /* Size prescribed by metric me in direction t */
469  t = n;
470  hu = me[0]*t[0]*t[0] + me[3]*t[1]*t[1] + me[5]*t[2]*t[2] \
471  + 2.0*me[1]*t[0]*t[1] + 2.0*me[2]*t[0]*t[2] + 2.0*me[4]*t[1]*t[2];
472 
473  hu = MG_MIN(isqhmin,hu);
474  hu = MG_MAX(isqhmax,hu);
475  m[0] = MG_MAX(m[0],hu);
476 
477  /* Size prescribed by metric me in direction u1 = n1 ^ t */
478  assert ( p0->xp );
479  go = &mesh->xpoint[p0->xp];
480  n1 = &go->n1[0];
481  n2 = &go->n2[0];
482 
483  u[0] = n1[1]*t[2] - n1[2]*t[1];
484  u[1] = n1[2]*t[0] - n1[0]*t[2];
485  u[2] = n1[0]*t[1] - n1[1]*t[0];
486  dd = u[0]*u[0] + u[1]*u[1] + u[2]*u[2];
487  if ( dd > MMG5_EPSD ) {
488  dd = 1.0 / sqrt(dd);
489 
490  u[0] *= dd;
491  u[1] *= dd;
492  u[2] *= dd;
493 
494  hu = me[0]*u[0]*u[0] + me[3]*u[1]*u[1] + me[5]*u[2]*u[2] \
495  + 2.0*me[1]*u[0]*u[1] + 2.0*me[2]*u[0]*u[2] + 2.0*me[4]*u[1]*u[2];
496 
497  hu = MG_MIN(isqhmin,hu);
498  hu = MG_MAX(isqhmax,hu);
499  m[1] = MG_MAX(m[1],hu);
500  }
501  /* Size prescribed by metric me in direction u2 = n2 ^ t */
502  u[0] = n2[1]*t[2] - n2[2]*t[1];
503  u[1] = n2[2]*t[0] - n2[0]*t[2];
504  u[2] = n2[0]*t[1] - n2[1]*t[0];
505  dd = u[0]*u[0] + u[1]*u[1] + u[2]*u[2];
506  if ( dd > MMG5_EPSD ) {
507  dd = 1.0 / sqrt(dd);
508 
509  u[0] *= dd;
510  u[1] *= dd;
511  u[2] *= dd;
512 
513  hu = me[0]*u[0]*u[0] + me[3]*u[1]*u[1] + me[5]*u[2]*u[2] \
514  + 2.0*me[1]*u[0]*u[1] + 2.0*me[2]*u[0]*u[2] + 2.0*me[4]*u[1]*u[2];
515 
516  hu = MG_MIN(isqhmin,hu);
517  hu = MG_MAX(isqhmax,hu);
518  m[2] = MG_MAX(m[2],hu);
519  }
520 
521  /* Size prescribed by metric me in direction n1 */
522  hu = me[0]*n1[0]*n1[0] + me[3]*n1[1]*n1[1] + me[5]*n1[2]*n1[2]
523  + 2.0*me[1]*n1[0]*n1[1] + 2.0*me[2]*n1[0]*n1[2] + 2.0*me[4]*n1[1]*n1[2];
524 
525  hu = MG_MIN(isqhmin,hu);
526  hu = MG_MAX(isqhmax,hu);
527  m[3] = hu;
528 
529  /* Size prescribed by metric me in direction n2 */
530  hu = me[0]*n2[0]*n2[0] + me[3]*n2[1]*n2[1] + me[5]*n2[2]*n2[2]

654

656  if ( !MMG5_rotmatrix(n0,r) ) return 0;

668  u[2] = r[0][2]*vp[0][0] + r[1][2]*vp[0][1];

672  u[0] -= ps*n1[0];
673  u[1] -= ps*n1[1];
674  u[2] -= ps*n1[2];
