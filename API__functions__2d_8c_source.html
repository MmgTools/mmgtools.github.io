324  return 0;
325  }

342  __func__);
343  return 0;
344  }
345

347  par = &mesh->info.par[k];
348 
349  if ( par->elt == typ && par->ref == ref ) {
350  par->hausd = hausd;
351  par->hmin = hmin;
352  par->hmax = hmax;
353  if ( (mesh->info.imprim > 5) || mesh->info.ddebug ) {
354  fprintf(stderr,"\n ## Warning: %s: new parameters (hausd, hmin and hmax)",
355  __func__);
356  fprintf(stderr," for entities of type %d and of ref %d\n",typ,ref);
357  }
358  return 1;
359  }
360  }
361 
362  mesh->info.par[mesh->info.npari].elt = typ;
363  mesh->info.par[mesh->info.npari].ref = ref;
364  mesh->info.par[mesh->info.npari].hmin = hmin;
365  mesh->info.par[mesh->info.npari].hmax = hmax;
366  mesh->info.par[mesh->info.npari].hausd = hausd;

371  mesh->info.parTyp |= MG_Tria;
372  break;
373  case ( MMG5_Edg ):

375  break;
376  default:
377  fprintf(stderr,"\n ## Error: %s: unexpected entity type: %s.\n",
378  __func__,MMG5_Get_entitiesName(typ));
379  return 0;
380  }
381 
382  mesh->info.npari++;
383 
384  return 1;
385 }
