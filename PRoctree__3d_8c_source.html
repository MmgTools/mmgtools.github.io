181 int64_t MMG3D_getPROctreeCoordinate(MMG3D_pPROctree q, double* ver, int dim)
182 {
183  int64_t s = 1<<20;
184  double prec = 1./(1<<30);
185  int place = 0;
186  int ix = (int)floor((ver[0]-prec)*s);
187  int iy = (int)floor((ver[1]-prec)*s);
188  int iz = (int)floor((ver[2]-prec)*s);
189  ix = (ix > 0) ? ix:0;
190  iy = (iy > 0) ? iy:0;
191  iz = (iz > 0) ? iz:0;
192  int64_t i=0;
193  int j;
194  for(j=19; j>=0; j--)
195  {
196  s=s>>1;
197  i += ((ix & s) >> j)<<place;
198  place++;
199  i += ((iy & s) >> j)<<place;
200  place++;
201  i += ((iz & s) >> j)<<place;
202  place++;
203  }
204  return i;
205 }

404  rectinout[3] = rect1Temp[3]<rect2Temp[3] ? rect1Temp[3]:rect2Temp[3];
405  rectinout[4] = rect1Temp[4]<rect2Temp[4] ? rect1Temp[4]:rect2Temp[4];
406  rectinout[5] = rect1Temp[5]<rect2Temp[5] ? rect1Temp[5]:rect2Temp[5];
407

409  rectinout[4] = rectinout[4] - rectinout[1];
410  rectinout[5] = rectinout[5] - rectinout[2];
411

441 int MMG3D_getListSquareRec(MMG3D_PROctree_s* q, double* center, double* rect,
442  MMG3D_PROctree_s*** qlist, double* dist, double* ani,
443  double l0, int nc, int dim, int* index)
444 {
445  double recttemp[6];
446  double centertemp[3];
447  int recCenter[6];
448  double l = 1./(1<<(q->depth+1));
449  double distTemp;
450  double x,y,z;
451  int indexTemp,i,j,k,nBranch;
452 
453  // number max of PROctree cells listed for one search
454  if ((*index)>nc-4)
455  return 1;
456 
457  // check if the current cell is included in the search zone, can avoid
458  // the loop over the vertices. Never occured in tests unless nc==4, in that
459  // case, there is no gain in computing time.
460  //~ if (q->nbVer>0 && MMG3D_isCellIncluded(center, l, &(dist[nc-3]), l0))
461  //~ {
462  //~ (*index)=nc-3;
463  //~ fprintf(stdout,"Included cell found\n");
464  //~ return;
465  //~ }
466 
467  assert ( nc>= 3 );
468  if (q->branches==NULL && q->v != NULL)
469  {
470  // the vector dist is of size nc whereas qlist allows nc-3 inputs
471  // so the 3 last values can contain the coordinates of the center
472  // of the search volume.
473  x = dist[nc-3] - center[0];
474  y = dist[nc-2] - center[1];
475  z = dist[nc-1] - center[2];
476 
477  // Should be replaced with distance in metric?
478  distTemp = x*x+y*y+z*z;
479 
480  // Here the anisotropic distance not tested (not so important, this only
481  // reorders the cells)
482  //~ distTemp = ani[0]*x*x+ani[3]*y*y+ani[5]*z*z+
483  //~ 2*(ani[1]*x*y+ani[2]*x*z+ani[4]*y*z);
484  if (*index > 0)
485  {
486  indexTemp = MMG3D_seekIndex(dist,distTemp,0, *index-1);
487  if (indexTemp+1<*index)
488  {
489  MMG3D_placeInListDouble(dist, distTemp, indexTemp, *index);
490  MMG3D_placeInListPROctree((*qlist), q, indexTemp, *index);
491  }else
492  {
493  dist[*index]=distTemp;
494  (*qlist)[*index]=q;
495  }
496  }
497  else
498  {
499  dist[*index]=distTemp;
500  (*qlist)[*index]=q;
501  }
502 
503  (*index)++;
504 
505  }else if (q->branches!=NULL)
506  {
507 
508  // check the position of the search zone in the current cell
509  for (i=0;i<3;i++)
510  {
511  recCenter[i] = (rect[i]>center[i]);
512  recCenter[i+3] = ((rect[i]+rect[i+3])>center[i]);
513  }
514 
515  // three loop describing the 8 branches in binary (k,j,i):(0,0,0),(0,0,1)....(1,1,1)
516  for(i=0;i<2;i++)
517  {
518  for(j=0;j<2;j++)
519  {
520  for(k=0;k<2;k++)
521  {
522  // test if that branch intersects the rectangle
523  if (((i && recCenter[3]) || (!i && !recCenter[0])) &&
524  ((j && recCenter[4]) || (!j && !recCenter[1]))&&
525  ((k && recCenter[5]) || (!k && !recCenter[2])))
526  {
527  // set the branch number
528  nBranch = i+2*j+4*k;
529 
530  // set recttemp to the cell size of the branch nBranch
531  recttemp[0] = center[0]-l*(1-i);
532  recttemp[1] = center[1]-l*(1-j);
533  recttemp[2] = center[2]-l*(1-k);
534  recttemp[3] = recttemp[4] = recttemp[5] = l;
535  // intersect the rectangle and the cell and store it in recttemp
536  if ( !MMG3D_intersectRect(rect,recttemp) ) return 0;
537 
538  // set the new center
539  centertemp[0] = center[0]-l/2+i*l;
540  centertemp[1] = center[1]-l/2+j*l;
541  centertemp[2] = center[2]-l/2+k*l;
542 
543  // recursive call in the branch
544  if ( !MMG3D_getListSquareRec(&(q->branches[nBranch]),
545  centertemp, recttemp, qlist, dist,
546  ani, l0, nc, dim, index) )
547  return 0;
548  }
549  }
550  }
551  }
552  }
553  return 1;
554 }

572 int MMG3D_getListSquare(MMG5_pMesh mesh, double* ani, MMG3D_pPROctree q, double* rect,
573  MMG3D_PROctree_s*** qlist)
574 {
575  double rect2[6], center[3], *dist;
576  double l0;
577  int i,index,dim;
578 
579  dim = mesh->dim;
580  index = 0;
581 
582  memcpy(&rect2, rect, sizeof(double)*dim*2);
583 
584  //instead of counting exactly the number of cells to be listed, the
585  //maximum size is set to nc-3 (so the list dist can have nc-3 values + 3 coordinates of
586  //the center of the rectangle)
587  assert(q->nc>=3);
588  index = q->nc-3;
589 
590  MMG5_ADD_MEM(mesh,index*sizeof(MMG3D_PROctree_s*),"PROctree cell",return -1);
591  MMG5_SAFE_MALLOC(*qlist,index,MMG3D_PROctree_s*, return -1);
592 
593  MMG5_ADD_MEM(mesh,q->nc*sizeof(double),"dist array",return -1);
594  MMG5_SAFE_MALLOC(dist,q->nc,double,return -1);
595 
596  // Set the center of the zone search
597  dist[q->nc-3] = rect[0]+rect[3]/2;
598  dist[q->nc-2] = rect[1]+rect[4]/2;
599  dist[q->nc-1] = rect[2]+rect[5]/2;
600 
601  // Set the radius of the zone search (used for cell inclusion test if activated)
602  l0 = rect[3]/2;
603 
604  // Initialization of the PROctree cell list
605  for (i = 0; i<index; i++)
606  (*qlist)[i] = NULL;
607 
608  index = 0;
609 
610  // Set center of the first PROctree cell
611  for (i = 0; i < dim; ++i)
612  center[i] = 0.5;
613 
614  // Avoid modification of input parameter rect
615  memcpy(&rect2, rect, sizeof(double)*dim*2);
616 
617  if ( !MMG3D_getListSquareRec(q->q0, center, rect2, qlist, dist, ani, l0,
618  q->nc, dim, &index) ) {
619  MMG5_DEL_MEM(mesh,dist);
620  return 0;
621  }
622 
623 
624  if (index>q->nc-4)
625  {
626  MMG5_DEL_MEM(mesh,dist);
627  return 0;
628  }
629 
630  MMG5_DEL_MEM(mesh,dist);
631 
632  return index;
633 }

648 int MMG3D_addPROctreeRec(MMG5_pMesh mesh, MMG3D_PROctree_s* q, double* ver,
649  const int no, int nv)
650 {
651  double pt[3],quadrant;
652  int dim, nbBitsInt,depthMax,i,j,k;
653  int sizBr;
654  int sizeRealloc;
655 
656  nbBitsInt = sizeof(int64_t)*8;
657  dim = mesh->dim;
658  depthMax = nbBitsInt/dim - 1; // maximum depth is to allow integer coordinates
659  sizBr = 1<<dim;
660 
661  if ( q->depth < depthMax ) // not at the maximum depth of the tree
662  {
663  if (q->nbVer < nv) // not at the maximum number of vertice in the cell
664  {
665 
666  if(q->nbVer == 0) // first vertex list allocation
667  {
668  MMG5_ADD_MEM(mesh,sizeof(int),"PROctree vertice table", return 0);
669  MMG5_SAFE_MALLOC(q->v,1,int,return 0);
670  }
671  else if(!(q->nbVer & (q->nbVer - 1))) //is a power of 2 -> reallocation of the vertex list
672  {
673  sizeRealloc = q->nbVer;
674  sizeRealloc<<=1;
675  MMG5_ADD_MEM(mesh,(sizeRealloc-sizeRealloc/2)*sizeof(int),"PROctree realloc",
676  return 0);
677  MMG5_SAFE_REALLOC(q->v,q->nbVer,sizeRealloc,int,"PROctree",return 0);
678  }
679 
680  q->v[q->nbVer] = no;
681  q->nbVer++;
682  return 1;
683  }
684  else if (q->nbVer == nv && q->branches==NULL) //vertex list at maximum -> cell subdivision
685  {
686  /* creation of sub-branch and relocation of vertices in the sub-branches */
687  MMG5_ADD_MEM(mesh,sizBr*sizeof(MMG3D_PROctree_s),"PROctree branches",
688  return 0);
689  MMG5_SAFE_MALLOC(q->branches,sizBr,MMG3D_PROctree_s,return 0);
690 
691  for ( i = 0; i<sizBr; i++)
692  {
693  MMG3D_initPROctree_s(&(q->branches[i]));
694  q->branches[i].depth = q->depth+1;
695  }
696  q->nbVer++;
697  for (i = 0; i<nv; i++)
698  {
699 
700  memcpy(&pt, mesh->point[q->v[i]].c ,dim*sizeof(double));
701  for ( j =0; j < q->depth; j++)
702  {
703  for (k = 0; k<dim; k++)
704  {
705  pt[k] -= ((double) (pt[k]>0.5))*0.5;
706  pt[k] *= 2;
707  }
708  }
709  if (!MMG3D_addPROctreeRec(mesh, q, pt, q->v[i],nv))
710  return 0;
711  q->nbVer--;
712  }
713  if (!MMG3D_addPROctreeRec(mesh, q, ver, no, nv))
714  return 0;
715  q->nbVer--;
716  MMG5_DEL_MEM(mesh,q->v);
717 
718  }else // Recursive call in the corresponding sub cell
719  {
720  quadrant = 0.;
721  for ( i = 0; i<dim; i++)
722  {
723  quadrant += ((double) (ver[i]>0.5))*(1<<i);
724  ver[i] -= ((double) (ver[i]>0.5))*0.5;
725  ver[i] *= 2;
726  }
727 
728  q->nbVer++;
729  if (!MMG3D_addPROctreeRec(mesh, &(q->branches[(int)quadrant]), ver, no, nv))
730  return 0;
731  }
732  }else // maximum PROctree depth reached
733  {
734  if (q->nbVer < nv)
735  {
736  if(q->nbVer == 0) // first allocation
737  {
738  MMG5_ADD_MEM(mesh,sizeof(int),"PROctree vertices table",
739  return 0);
740  MMG5_SAFE_MALLOC(q->v,1,int,return 0);
741  }
742  else if(!(q->nbVer & (q->nbVer - 1))) //is a power of 2 -> normal reallocation
743  {
744  sizeRealloc = q->nbVer;
745  sizeRealloc <<= 1;
746  MMG5_ADD_MEM(mesh,(sizeRealloc-sizeRealloc/2)*sizeof(int),"PROctree realloc",
747  return 0);
748  MMG5_SAFE_REALLOC(q->v,q->nbVer,sizeRealloc,int,"PROctree",return 0);
749  }
750  }
751  else if (q->nbVer%nv == 0) // special reallocation of the vertex list because it is at maximum depth
752  {
753  MMG5_ADD_MEM(mesh,nv*sizeof(int),"PROctree realloc",
754  return 0);
755  MMG5_SAFE_REALLOC(q->v,q->nbVer,q->nbVer+nv,int,"PROctree",return 0);
756  }
757 
758  q->v[q->nbVer] = no;
759  q->nbVer++;
760  }
761 
762  return 1;
763 }
