218  static int8_t mmgWarn0 = 0, mmgWarn1 = 0;
219 
220  pt = &mesh->tria[iel];
221  p[0] = &mesh->point[pt->v[0]];
222  p[1] = &mesh->point[pt->v[1]];
223  p[2] = &mesh->point[pt->v[2]];
224 
225  /* normal recovery */
226  for (i=0; i<3; i++) {
227  if ( MS_SIN(p[i]->tag) ) {
228  MMG5_nortri(mesh,pt,n[i]);
229  }
230  else if ( MG_EDG(p[i]->tag) ) {
231  MMG5_nortri(mesh,pt,nt);
232  n1 = &mesh->xpoint[p[i]->xp].n1[0];
233  n2 = &mesh->xpoint[p[i]->xp].n2[0];
234  ps = n1[0]*nt[0] + n1[1]*nt[1] + n1[2]*nt[2];
235  ps2 = n2[0]*nt[0] + n2[1]*nt[1] + n2[2]*nt[2];
236  if ( fabs(ps) > fabs(ps2) )
237  memcpy(&n[i],n1,3*sizeof(double));
238  else
239  memcpy(&n[i],n2,3*sizeof(double));
240  memcpy(&t[i],p[i]->n,3*sizeof(double));
241  }
242  else
243  memcpy(&n[i],p[i]->n,3*sizeof(double));
244  }
245 
246  /* analyze edges */
247  for (i=0; i<3; i++) {
248  if ( pt->tag[i] & MG_REQ) continue;
249 
250  i1 = MMG5_inxt2[i];
251  i2 = MMG5_iprv2[i];
252 
253  /* local parameters */
254  hmax = mesh->info.hmax;
255  hausd = mesh->info.hausd;
256  isloc = 0;
257  for (l=0; l<mesh->info.npar; l++) {
258  par = &mesh->info.par[l];
259  if ( /*((par->elt == MMG5_Vertex) &&
260  ( (p[i1]->ref == par->ref ) || (p[i2]->ref == par->ref) ))

262  if ( !isloc ) {
263  hmax = par->hmax;
264  hausd = par->hausd;
265  isloc = 1;
266  }
267  else {
268  hausd = MG_MIN(par->hausd,hausd);
269  hmax = MG_MIN(par->hmax,hmax);
270  }
271  }
272  }
273 
274  /* check length */
275  ux = p[i2]->c[0] - p[i1]->c[0];
276  uy = p[i2]->c[1] - p[i1]->c[1];
277  uz = p[i2]->c[2] - p[i1]->c[2];
278  ll = ux*ux + uy*uy + uz*uz;
279  if ( ll > hmax*hmax ) {
280  MG_SET(pt->flag,i);
281  continue;
282  }
283  else if ( !MG_EDG(pt->tag[i]) && p[i1]->tag > MG_NOTAG && p[i2]->tag > MG_NOTAG ) {
284  MG_SET(pt->flag,i);
285  continue;
286  }
287 
288  /* Hausdorff w/r tangent direction */
289  if ( MG_EDG(pt->tag[i]) ) {
290  if ( MS_SIN(p[i1]->tag) ) {
291  li = 1.0 / sqrt(ll);
292  t1[0] = li*ux;
293  t1[1] = li*uy;
294  t1[2] = li*uz;
295  }
296  else{
297  if(!((p[i1]->tag & MG_NOM) || MG_EDG(p[i1]->tag) ) ) {
298  // if(t[i1][0] > 10) {
299  if ( !mmgWarn0 ) {
300  fprintf(stderr,"\n ## Warning: %s: a- at least 1 geometrical"
301  " problem\n",__func__);
302  mmgWarn0 = 1;
303  }
304  return 0;
305  }
306  memcpy(t1,t[i1],3*sizeof(double));
307  }
308 
309  if ( MS_SIN(p[i2]->tag) ) {
310  li = 1.0 / sqrt(ll);
311  t2[0] = li*ux;
312  t2[1] = li*uy;
313  t2[2] = li*uz;
314  }
315  else{
316  if(!((p[i2]->tag & MG_NOM) || MG_EDG(p[i2]->tag) ) ) {
317  if ( !mmgWarn1 ) {
318  fprintf(stderr,"\n ## Warning: %s: b- at least 1 geometrical"
319  " problem\n",__func__);
320  mmgWarn1 = 1;
321  }
322  return 0;
323  }
324  memcpy(t2,t[i2],3*sizeof(double));
325  }
326 
327  ps = t1[0]*ux + t1[1]*uy + t1[2]*uz;
328  ps *= ps;
329  cosn = ps/ll ;
330  cosn *= (1.0-cosn);
331  cosn *= (0.25*ll);
332  if ( cosn > hausd*hausd ) {
333  MG_SET(pt->flag,i);
334  continue;
335  }
336 
337  ps = -(t2[0]*ux + t2[1]*uy + t2[2]*uz);
338  ps *= ps;
339  cosn = ps/ll ;
340  cosn *= (1.0-cosn);
341  cosn *= (0.25*ll);
342  if ( cosn > hausd*hausd ) {
343  MG_SET(pt->flag,i);
344  continue;
345  }
346  }
347  else {
348  n1 = n[i1];
349  n2 = n[i2];
350 
351  ps = ux*n1[0] + uy*n1[1] + uz*n1[2];
352  c1[0] = (2.0*p[i1]->c[0] + p[i2]->c[0] - ps*n1[0]) / 3.0 - p[i1]->c[0];
353  c1[1] = (2.0*p[i1]->c[1] + p[i2]->c[1] - ps*n1[1]) / 3.0 - p[i1]->c[1];
354  c1[2] = (2.0*p[i1]->c[2] + p[i2]->c[2] - ps*n1[2]) / 3.0 - p[i1]->c[2];
355 
356  ps = -(ux*n2[0] + uy*n2[1] + uz*n2[2]);
357  c2[0] = (2.0*p[i2]->c[0] + p[i1]->c[0] - ps*n2[0]) / 3.0 - p[i2]->c[0];
358  c2[1] = (2.0*p[i2]->c[1] + p[i1]->c[1] - ps*n2[1]) / 3.0 - p[i2]->c[1];
359  c2[2] = (2.0*p[i2]->c[2] + p[i1]->c[2] - ps*n2[2]) / 3.0 - p[i2]->c[2];
360 
361  /* squared cosines */
362  ps = c1[0]*ux + c1[1]*uy + c1[2]*uz;
363  ps *= ps;
364  dd = c1[0]*c1[0] + c1[1]*c1[1] + c1[2]*c1[2];
365  cosn = ps / (dd*ll);
366  cosn *= (1.0-cosn);
367  cosn *= (0.25*ll);
368  if ( cosn > hausd*hausd ) {
369  MG_SET(pt->flag,i);
370  continue;
371  }
372 
373  ps = -c2[0]*ux - c2[1]*uy - c2[2]*uz;
374  ps *= ps;
375  dd = c2[0]*c2[0]+c2[1]*c2[1]+c2[2]*c2[2];
376  cosn = ps / (dd*ll);
377  cosn *= (1.0-cosn);
378  cosn *= (0.25*ll);
379  if ( cosn > hausd*hausd ) {
380  MG_SET(pt->flag,i);
381  continue;
382  }
383  }
384  }
385 
386  return pt->flag;
387 }
