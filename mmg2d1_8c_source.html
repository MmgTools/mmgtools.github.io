35 /* Mesh adaptation routine for the first stages of the algorithm: intertwine splitting
36  based on patterns, collapses and swaps.
37  typchk = 1 -> adaptation based on edge lengths
38  typchk = 2 -> adaptation based on lengths calculated in metric met */

40  int it,maxit,ns,nc,nsw,nns,nnc,nnsw;
41 
42  nns = nnc = nnsw = 0;
43  it = 0;
44  maxit = 5;
45 
46  /* Main routine; intertwine split, collapse and swaps */
47  do {
48  if ( typchk == 2 && it == 0 ) {
49 // #warning Luca: check consistency with 3D
50  }
51  if ( !mesh->info.noinsert ) {
52  /* Memory free */
53  MMG5_DEL_MEM(mesh,mesh->adja);

76  }

80  }
81

109 /* Travel triangles and split long edges according to patterns */
110 int MMG2D_anaelt(MMG5_pMesh mesh,MMG5_pSol met,int typchk) {
111  MMG5_pTria pt;
112  MMG5_pPoint ppt,p1,p2;
113  MMG5_Hash hash;
114  double len,s,o[2],no[2];
115  int ns,nc,npinit,ni,k,nt,ip1,ip2,ip,it,vx[3];
116  int8_t i,ic,i1,i2,ier;
117  static int8_t mmgWarn0=0;
118 
119  s = 0.5;
120  ns = 0;
121  npinit = mesh->np;
122 
123  if ( !MMG5_hashNew(mesh,&hash,mesh->np,3*mesh->np) ) return 0;
124 
125  /* Step 1: travel mesh, check edges, and tag those to be split; create the new vertices in hash */
126  for (k=1; k<=mesh->nt; k++) {
127  pt = &mesh->tria[k];
128  if ( !MG_EOK(pt) || (pt->ref < 0) ) continue;
129  if ( MG_SIN(pt->tag[0]) || MG_SIN(pt->tag[1]) || MG_SIN(pt->tag[2]) ) continue;
130 
131  /* Check if pt should be cut */
132  pt->flag = 0;
133 
134  /* typchk=1 : split on geometric basis and rough size considerations */
135  if ( typchk == 1) {
136  if ( !MMG2D_chkedg(mesh,k) ) continue;
137  }
138  /* typchk =2 : split on basis of edge lengths in the metric */
139  else if ( typchk ==2 ) {
140  for (i=0; i<3; i++) {
141  i1 = MMG5_inxt2[i];
142  i2 = MMG5_iprv2[i];
143  len = MMG2D_lencurv(mesh,met,pt->v[i1],pt->v[i2]);
144  if ( len > MMG2D_LLONG ) MG_SET(pt->flag,i);
145  }
146  }
147 
148  /* mesh->info.fem : split edges which are not MG_BDY, but whose vertices are both MG_BDY */
149  if ( mesh->info.fem ) {
150  for (i=0; i<3; i++) {
151  i1 = MMG5_inxt2[i];
152  i2 = MMG5_iprv2[i];
153  p1 = &mesh->point[pt->v[i1]];
154  p2 = &mesh->point[pt->v[i2]];
155  if ( (p1->tag & MG_BDY) && (p2->tag & MG_BDY) && !(pt->tag[i] & MG_BDY) ) MG_SET(pt->flag,i);
156  }
157  }
158 
159  if ( !pt->flag ) continue;
160  ns++;
161 
162  /* Travel edges to split and create new points */
163  for (i=0; i<3; i++) {
164  if ( !MG_GET(pt->flag,i) ) continue;
165  i1 = MMG5_inxt2[i];
166  i2 = MMG5_iprv2[i];
167  ip1 = pt->v[i1];
168  ip2 = pt->v[i2];
169  ip = MMG5_hashGet(&hash,ip1,ip2);
170  if ( ip > 0 ) continue;
171 
172  /* Geometric attributes of the new point */
173  ier = MMG2D_bezierCurv(mesh,k,i,s,o,no);
174  if ( !ier ) {
175  MG_CLR(pt->flag,i);
176  continue;
177  }
178  ip = MMG2D_newPt(mesh,o,pt->tag[i]);
179  if ( !ip ) {
180  /* reallocation of point table */
181  MMG2D_POINT_REALLOC(mesh,met,ip,mesh->gap,
182  fprintf(stderr,"\n ## Error: %s: unable to"
183  " allocate a new point.\n",__func__);
184  MMG5_INCREASE_MEM_MESSAGE();
185  do {
186  MMG2D_delPt(mesh,mesh->np);
187  } while ( mesh->np>npinit );return -1;,
188  o,pt->tag[i]);
189  }
190  ppt = &mesh->point[ip];
191  if ( MG_EDG(pt->tag[i]) ) {
192  ppt->n[0] = no[0];
193  ppt->n[1] = no[1];
194  }
195 
196  /* If there is a metric in the mesh, interpolate it at the new point */
197  assert ( met );
198  if ( met->m )
199  MMG2D_intmet(mesh,met,k,i,ip,s);
200 
201  /* Add point to the hashing structure */
202  MMG5_hashEdge(mesh,&hash,ip1,ip2,ip);
203  }
204  }
205  if ( !ns ) {
206  MMG5_DEL_MEM(mesh,hash.item);
207  return ns;
208  }
209 
210  /* Step 2: Make flags at triangles consistent between themselves (check if adjacent triangle is split) */
211  for (k=1; k<=mesh->nt; k++) {
212  pt = &mesh->tria[k];
213  if ( !MG_EOK(pt) || pt->ref < 0 ) continue;
214  else if ( pt->flag == 7 ) continue;
215  nc = 0;
216 
217  for (i=0; i<3; i++) {
218  i1 = MMG5_iprv2[i];
219  i2 = MMG5_inxt2[i];
220  if ( !MG_GET(pt->flag,i) && !MG_SIN(pt->tag[i]) ) {
221  ip = MMG5_hashGet(&hash,pt->v[i1],pt->v[i2]);
222  if ( ip > 0 ) {
223  MG_SET(pt->flag,i);
224  nc++;
225  }
226  }
227  }
228  if ( nc > 0 ) ns++;
229  }
230  if ( mesh->info.ddebug && ns ) {
231  fprintf(stdout," %d analyzed %d proposed\n",mesh->nt,ns);
232  fflush(stdout);
233  }
234 
235  /* Step 3: Simulate splitting and delete points leading to an invalid configuration */
236  for (k=1; k<=mesh->np; k++)
237  mesh->point[k].flag = 0;
238 
239  it = 1;
240  nc = 0;
241  do {
242  ni = 0;
243  for ( k=1; k<= mesh->nt; k++) {
244  pt = &mesh->tria[k];
245  if ( !MG_EOK(pt) || pt->ref < 0 ) continue;
246  else if ( pt->flag == 0 ) continue;
247 
248  vx[0] = vx[1] =vx[2] = 0;
249  pt->flag = 0;
250  ic = 0;
251 
252  for (i=0; i<3; i++) {
253  i1 = MMG5_iprv2[i];
254  i2 = MMG5_inxt2[i];
255  vx[i] = MMG5_hashGet(&hash,pt->v[i1],pt->v[i2]);
256  if ( vx[i] > 0 ) {
257  MG_SET(pt->flag,i);
258  if ( mesh->point[vx[i]].flag > 2 ) ic = 1;
259  }
260  }
261 
262  if ( !pt->flag ) continue;
263  switch (pt->flag) {
264  case 1: case 2: case 4:
265  ier = MMG2D_split1_sim(mesh,met,k,vx);
266  break;
267  case 7:
268  ier = MMG2D_split3_sim(mesh,met,k,vx);
269  break;
270  default:
271  ier = MMG2D_split2_sim(mesh,met,k,vx);
272  break;
273  }
274  if ( ier ) continue;
275 
276  /* An edge is invalidated in the process */
277  ni++;
278  if ( ic == 0 && MMG2D_dichoto(mesh,met,k,vx) ) {
279  for (i=0; i<3; i++)
280  if ( vx[i] > 0 ) mesh->point[vx[i]].flag++;
281  }
282  /* Relocate point at the center of the edge */
283  else {
284  for (i=0; i<3; i++) {
285  if ( vx[i] > 0 ) {
286  p1 = &mesh->point[pt->v[MMG5_iprv2[i]]];
287  p2 = &mesh->point[pt->v[MMG5_inxt2[i]]];
288  ppt = &mesh->point[vx[i]];
289  ppt->c[0] = 0.5 * (p1->c[0] + p2->c[0]);
290  ppt->c[1] = 0.5 * (p1->c[1] + p2->c[1]);
291  }
292  }
293  }
294  }
295  nc += ni;
296  }
297  while ( ni >0 && ++it <20 );
298 
299  if ( mesh->info.ddebug && nc ) {
300  fprintf(stdout," %d corrected, %d invalid\n",nc,ni);
301  fflush(stdout);
302  }
303 
304  /* step 4: effective splitting */
305  ns = 0;
306  nt = mesh->nt;
307  for (k=1; k<=nt; k++) {
308  pt = &mesh->tria[k];
309  if ( !MG_EOK(pt) || pt->ref < 0 ) continue;
310  else if ( pt->flag == 0 ) continue;
311 
312  vx[0] = vx[1] = vx[2] = 0;
313  for (i=0; i<3; i++) {
314  i1 = MMG5_inxt2[i];
315  i2 = MMG5_inxt2[i1];
316  if ( MG_GET(pt->flag,i) ) {
317  vx[i] = MMG5_hashGet(&hash,pt->v[i1],pt->v[i2]);
318  if ( !vx[i] ) {
319  if ( !mmgWarn0 ) {
320  mmgWarn0 = 1;
321  fprintf(stderr,"\n ## Error: %s: unable to create point on"
322  " at least 1 edge.\n Exit program.\n",__func__);
323  }
324  return -1;
325  }
326  }
327  }
328  if ( pt->flag == 1 || pt->flag == 2 || pt->flag == 4 ) {
329  ier = MMG2D_split1(mesh,met,k,vx);
330  ns++;
331  }
332  else if ( pt->flag == 7 ) {
333  ier = MMG2D_split3(mesh,met,k,vx);
334  ns++;
335  }
336  else {
337  ier = MMG2D_split2(mesh,met,k,vx);
338  ns++;
339  }
340  if ( !ier ) return -1;
341  }
342  if ( (mesh->info.ddebug || abs(mesh->info.imprim) > 5) && ns > 0 )
343  fprintf(stdout," %7d splitted\n",ns);
344  MMG5_DEL_MEM(mesh,hash.item);
345 
346  return ns;
347 }

359 int MMG2D_dichoto(MMG5_pMesh mesh,MMG5_pSol met,int k,int *vx) {
360  MMG5_pTria pt;
361  MMG5_pPoint pa,pb,ps;
362  double o[3][2],p[3][2];
363  float to,tp,t;
364  int ia,ib,ier,it,maxit;
365  int8_t i,i1,i2;
366 
367  pt = &mesh->tria[k];
368 
369  /* Get point on curve and along segment for edge split */
370  for (i=0; i<3; i++) {
371  memset(p[i],0,2*sizeof(double));
372  memset(o[i],0,2*sizeof(double));
373  if ( vx[i] > 0 ) {
374  i1 = MMG5_inxt2[i];
375  i2 = MMG5_inxt2[i1];
376  ia = pt->v[i1];
377  ib = pt->v[i2];
378  pa = &mesh->point[ia];
379  pb = &mesh->point[ib];
380  ps = &mesh->point[vx[i]];
381  o[i][0] = 0.5 * (pa->c[0] + pb->c[0]);
382  o[i][1] = 0.5 * (pa->c[1] + pb->c[1]);
383  p[i][0] = ps->c[0];
384  p[i][1] = ps->c[1];
385  }
386  }
387  maxit = 4;
388  it = 0;
389  tp = 1.0;
390  to = 0.0;
391 
392  do {
393  /* compute new position */
394  t = 0.5 * (tp + to);
395  for (i=0; i<3; i++) {
396  if ( vx[i] > 0 ) {
397  ps = &mesh->point[vx[i]];
398  ps->c[0] = o[i][0] + t*(p[i][0] - o[i][0]);
399  ps->c[1] = o[i][1] + t*(p[i][1] - o[i][1]);
400  }
401  }
402  switch (pt->flag) {
403  case 1: case 2: case 4:
404  ier = MMG2D_split1_sim(mesh,met,k,vx);
405  break;
406  case 7:
407  ier = MMG2D_split3_sim(mesh,met,k,vx);
408  break;
409  default:
410  ier = MMG2D_split2_sim(mesh,met,k,vx);
411  break;
412  }
413  if ( ier )
414  to = t;
415  else
416  tp = t;
417  }
418  while ( ++it < maxit );
419 
420  /* restore coords of last valid pos. */
421  if ( !ier ) {
422  t = to;
423  for (i=0; i<3; i++) {
424  if ( vx[i] > 0 ) {
425  ps = &mesh->point[vx[i]];
426  ps->c[0] = o[i][0] + t*(p[i][0] - o[i][0]);
427  ps->c[1] = o[i][1] + t*(p[i][1] - o[i][1]);
428  }
429  }
430  }
431  return 1;
432 }

434 /* Travel triangles and collapse short edges */
435 int MMG2D_colelt(MMG5_pMesh mesh,MMG5_pSol met,int typchk) {
436  MMG5_pTria pt;
437  MMG5_pPoint p1,p2;
438  double ux,uy,ll,hmin2;
439  int list[MMG2D_LONMAX+2],ilist,nc,k;
440  uint8_t i,i1,i2,open;
441 
442  nc = 0;
443  hmin2 = mesh->info.hmin * mesh->info.hmin;
444 
445  for (k=1; k<=mesh->nt; k++) {
446  pt = &mesh->tria[k];
447  if ( !MG_EOK(pt) || pt->ref < 0 ) continue;
448 
449  /* Travel 3 edges of the triangle and decide whether to collapse p1->p2, based on length criterion */
450  pt->flag = 0; // was here before, but probably serves for nothing
451 
452  for (i=0; i<3; i++) {
453  if ( MG_SIN(pt->tag[i]) ) continue;
454  i1 = MMG5_inxt2[i];
455  i2 = MMG5_iprv2[i];
456  p1 = &mesh->point[pt->v[i1]];
457  p2 = &mesh->point[pt->v[i2]];
458  if ( MG_SIN(p1->tag) || p1->tag & MG_NOM ) continue;
459 
460  /* Impossible to collapse a surface point onto a non surface point -- impossible to
461  collapse a surface point along a non geometric edge */
462  else if ( p1->tag & MG_GEO ) {
463  if ( ! (p2->tag & MG_GEO) || !(pt->tag[i] & MG_GEO) ) continue;
464  }
465  /* Same test for REF points */
466  else if ( p1->tag & MG_REF ) {
467  if ( ! (p2->tag & MG_GEO || p2->tag & MG_REF) || !(pt->tag[i] & MG_REF) ) continue;
468  }
469 
470  open = (mesh->adja[3*(k-1)+1+i] == 0) ? 1 : 0;
471 
472  /* Check length */
473  if ( typchk == 1 ) {
474  ux = p2->c[0] - p1->c[0];
475  uy = p2->c[1] - p1->c[1];
476  ll = ux*ux + uy*uy;
477  if ( ll > hmin2 ) continue;
478  }
479  else {
480  ll = MMG2D_lencurv(mesh,met,pt->v[i1],pt->v[i2]);
481  if ( ll > MMG2D_LSHRT ) continue;
482  }
483 
484  /* Check whether the geometry is preserved */
485  ilist = MMG2D_chkcol(mesh,met,k,i,list,typchk);
486 
487  if ( ilist > 3 || ( ilist == 3 && open ) ) {
488  nc += MMG2D_colver(mesh,ilist,list);
489  break;
490  }
491  else if ( ilist == 3 ) {
492  nc += MMG2D_colver3(mesh,list);
493  break;
494  }
495  else if ( ilist == 2 ) {
496  nc += MMG2D_colver2(mesh,list);
497  break;
498  }
499  }
500  }
501 
502  if ( nc > 0 && (abs(mesh->info.imprim) > 5 || mesh->info.ddebug) )
503  fprintf(stdout," %8d vertices removed\n",nc);
504 
505  return nc;
506 }

615  }

618  if ( abs(mesh->info.imprim) < 5 && (nnc > 0 || nns > 0) )
619  fprintf(stdout," %8d splitted, %8d collapsed, %8d swapped, %8d moved, %d iter. \n",nns,nnc,nnsw,nnm,it);
620  }
621  return 1;
622 }

690 int MMG2D_adpcol(MMG5_pMesh mesh,MMG5_pSol met) {
691  MMG5_pTria pt;
692  MMG5_pPoint p1,p2;
693  double len;
694  int k,nc,ilist,list[MMG2D_LONMAX+2];
695  int8_t i,i1,i2,open;
696 
697  nc = 0;
698  for (k=1; k<=mesh->nt; k++) {
699  pt = &mesh->tria[k];
700  if ( !MG_EOK(pt) || pt->ref < 0 ) continue;
701 
702  /* Check edge length, and attempt collapse */
703  pt->flag = 0;
704  for (i=0; i<3; i++) {
705  if ( MG_SIN(pt->tag[i]) ) continue;
706 
707  open = ( mesh->adja[3*(k-1)+1+i] == 0 ) ? 1 : 0;
708 
709  i1 = MMG5_inxt2[i];
710  i2 = MMG5_iprv2[i];
711  p1 = &mesh->point[pt->v[i1]];
712  p2 = &mesh->point[pt->v[i2]];
713 
714  if ( MG_SIN(p1->tag) || p1->tag & MG_NOM ) continue;
715  else if ( p1->tag & MG_GEO ) {
716  if ( ! (p2->tag & MG_GEO) || !(pt->tag[i] & MG_GEO) ) continue;
717  }
718  else if ( p1->tag & MG_REF ) {
719  if ( ! (p2->tag & MG_GEO || p2->tag & MG_REF) || !(pt->tag[i] & MG_REF) ) continue;
720  }
721 
722  len = MMG2D_lencurv(mesh,met,pt->v[i1],pt->v[i2]);
723 
724  if ( len > MMG2D_LOPTS ) continue;
725 
726  ilist = MMG2D_chkcol(mesh,met,k,i,list,2);
727 
728  if ( ilist > 3 || ( ilist==3 && open ) ) {
729  nc += MMG2D_colver(mesh,ilist,list);
730  break;
731  }
732  else if ( ilist == 3 ) {
733  nc += MMG2D_colver3(mesh,list);
734  break;
735  }
736  else if ( ilist == 2 ) {
737  nc += MMG2D_colver2(mesh,list);
738  break;
739  }
740  }
741  }

746 /* Analyze points to relocate them according to a quality criterion */
747 int MMG2D_movtri(MMG5_pMesh mesh,MMG5_pSol met,int maxit,int8_t improve) {
748  MMG5_pTria pt;

751  int8_t i,ier;
752 
753  it = nnm = 0;
754  base = 0;
755 
756  for (k=1; k<=mesh->np; k++)
757  mesh->point[k].flag = base;
758 
759  do {
760  base++;
761  nm = ns = 0;
762  for (k=1; k<=mesh->nt; k++) {
763  pt = &mesh->tria[k];
764  if ( !MG_EOK(pt) || pt->ref < 0 ) continue;
765 
766  for (i=0; i<3; i++) {
767  p0 = &mesh->point[pt->v[i]];
768  if ( p0->flag == base || MG_SIN(p0->tag) || p0->tag & MG_NOM ) continue;
769 
770  ilist = MMG2D_boulet(mesh,k,i,list);
771 
772  if ( MG_EDG(p0->tag) ) {
773  ier = MMG2D_movedgpt(mesh,met,ilist,list,improve);
774  if ( ier ) ns++;
775  }
776  else {
777  if ( met->size == 3 && met->m )
778  ier = MMG2D_movintpt_ani(mesh,met,ilist,list,improve);
779  else
780  ier = MMG2D_movintpt(mesh,met,ilist,list,improve);
781  }
782 
783  if ( ier ) {
784  nm++;
785  p0->flag = base;
786  }
787  }
788  }
789  nnm += nm;
790  if ( mesh->info.ddebug ) fprintf(stdout," %8d moved, %d geometry\n",nm,ns);
791  }
792  while ( ++it < maxit && nm > 0 );
793 
794  if ( (abs(mesh->info.imprim) > 5 || mesh->info.ddebug) && nnm > 0 )
795  fprintf(stdout," %8d vertices moved, %d iter.\n",nnm,it);
796 
797  return nnm;
798 }
