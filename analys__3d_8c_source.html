96 int MMG5_setadj(MMG5_pMesh mesh){
97  MMG5_pTria pt,pt1;
98  int *adja,*adjb,adji1,adji2,*pile,iad,ipil,ip1,ip2,gen;
99  int k,kk,iel,jel,nvf,nf,nr,nt,nre,nreq,ncc,ned,ref;
100  int16_t tag;
101  int8_t i,ii,i1,i2,ii1,ii2,voy;
102 
103  nvf = nf = ncc = ned = 0;
104 
105  MMG5_SAFE_MALLOC(pile,mesh->nt+1,int,return 0);
106 
107  pile[1] = 1;
108  ipil = 1;
109 
110  while ( ipil > 0 ) {
111  ncc++;
112 
113  do {
114  k = pile[ipil--];
115  pt = &mesh->tria[k];
116  pt->flag = ncc;
117  if ( !MG_EOK(pt) ) continue;
118 
119  adja = &mesh->adjt[3*(k-1)+1];
120  for (i=0; i<3; i++) {
121  if( ((pt->tag[i] & MG_PARBDY) && !(pt->tag[i] & MG_PARBDYBDY)) ||
122  (pt->tag[i] & MG_BDY) ) continue;
123  i1 = MMG5_inxt2[i];
124  i2 = MMG5_iprv2[i];
125  ip1 = pt->v[i1];
126  ip2 = pt->v[i2];
127 
128  if ( !mesh->point[ip1].tmp ) mesh->point[ip1].tmp = ++nvf;
129  if ( !mesh->point[ip2].tmp ) mesh->point[ip2].tmp = ++nvf;
130 
131  if ( MG_EDG(pt->tag[i]) || pt->tag[i] & MG_REQ ) {
132  tag = mesh->point[ip1].tag;
133  mesh->point[ip1].tag |= pt->tag[i];
134  // Remove the MG_NOSURF tag if the vertex is really required.
135  if ( (tag & MG_REQ) && !(tag & MG_NOSURF) ) {
136  mesh->point[ip1].tag &= ~MG_NOSURF;
137  }
138  tag = mesh->point[ip2].tag;
139  mesh->point[ip2].tag |= pt->tag[i];
140  // Remove the MG_NOSURF tag if the vertex is really required.
141  if ( (tag & MG_REQ) && !(tag & MG_NOSURF) ) {
142  mesh->point[ip2].tag &= ~MG_NOSURF;
143  }
144  }
145 
146  /* open boundary */
147  tag = MG_GEO;
148  if ( mesh->info.opnbdy ) tag += MG_OPNBDY;
149  if ( !adja[i] ) {
150  tag += MG_NOM;
151  pt->tag[i] |= tag;
152  mesh->point[ip1].tag |= tag;
153  mesh->point[ip2].tag |= tag;
154  ned++;
155  continue;
156  }
157 
158  kk = adja[i] / 3;
159  ii = adja[i] % 3;
160  if ( kk > k ) ned++;
161 
162  /* store adjacent */
163  pt1 = &mesh->tria[kk];
164 
165  /* non manifold edge */
166  if ( pt->tag[i] & MG_NOM ) {
167  mesh->point[ip1].tag |= MG_NOM;
168  mesh->point[ip2].tag |= MG_NOM;
169  continue;
170  }
171 
172  if ( abs(pt1->ref) != abs(pt->ref) ) {
173  pt->tag[i] |= MG_REF;
174  pt1->tag[ii] |= MG_REF;
175  mesh->point[ip1].tag |= MG_REF;
176  mesh->point[ip2].tag |= MG_REF;
177  }
178 
179  /* store adjacent */
180  if ( !pt1->flag ) {
181  pt1->flag = ncc;
182  pile[++ipil] = kk;
183  }
184 
185  /* check orientation */
186  ii1 = MMG5_inxt2[ii];
187  ii2 = MMG5_iprv2[ii];
188  if ( pt1->v[ii1] == ip1 ) {
189  /* Moebius strip */
190  if ( pt1->base < 0 ) {
191  fprintf(stderr,"\n ## Error: %s: Triangle orientation problem (1):"
192  " Moebius strip?\n",__func__);
193  MMG5_SAFE_FREE(pile);
194  return 0;
195  }
196  /* flip orientation */
197  else {
198  pt1->base = -pt1->base;
199  pt1->v[ii1] = ip2;
200  pt1->v[ii2] = ip1;
201 
202  /* update adj */
203  iad = 3*(kk-1)+1;
204  adjb = &mesh->adjt[iad];
205  adji1 = mesh->adjt[iad+ii1];
206  adji2 = mesh->adjt[iad+ii2];
207  adjb[ii1] = adji2;
208  adjb[ii2] = adji1;
209 
210  /* modif tag + ref */
211  tag = pt1->tag[ii1];
212  pt1->tag[ii1] = pt1->tag[ii2];
213  pt1->tag[ii2] = tag;
214  ref = pt1->edg[ii1];
215  pt1->edg[ii1] = pt1->edg[ii2];
216  pt1->edg[ii2] = ref;
217 
218  /* modif voyeurs */
219  if ( adjb[ii1] ) {
220  iel = adjb[ii1] / 3;
221  voy = adjb[ii1] % 3;
222  mesh->adjt[3*(iel-1)+1+voy] = 3*kk + ii1;
223  }
224  if ( adjb[ii2] ) {
225  iel = adjb[ii2] / 3;
226  voy = adjb[ii2] % 3;
227  mesh->adjt[3*(iel-1)+1+voy] = 3*kk + ii2;
228  }
229  nf++;
230  }
231  }
232  else {
233  /* Mark triangles that have a consistent orientation with their
234  * neighbours */
235  pt1->base = -pt1->base;
236  }
237  }
238  }
239  while ( ipil > 0 );
240 
241  /* find next unmarked triangle */
242  ipil = 0;
243  for (kk=1; kk<=mesh->nt; kk++) {
244  pt = &mesh->tria[kk];
245  if ( MG_EOK(pt) && (pt->flag == 0) ) {
246  ipil = 1;
247  pile[ipil] = kk;
248  pt->flag = ncc+1;
249  break;
250  }
251  }
252  }
253 
254  /* bilan */
255  nr = nre = nreq = nt = 0;
256  for (k=1; k<=mesh->nt; k++) {
257  pt = &mesh->tria[k];
258  if ( !MG_EOK(pt) ) continue;
259  nt++;
260  adja = &mesh->adjt[3*(k-1)+1];
261  for (i=0; i<3; i++) {
262  if ( ( !MG_EDG(pt->tag[i]) ) && ( !(pt->tag[i] & MG_REQ) ) ) continue;
263 
264  jel = adja[i] / 3;
265  if ( !jel || jel > k ) {
266  if ( pt->tag[i] & MG_GEO ) nr++;
267  if ( pt->tag[i] & MG_REF ) nre++;
268  if ( pt->tag[i] & MG_REQ ) nreq++;
269  }
270  }
271  }
272 
273  if ( mesh->info.ddebug ) {
274  fprintf(stdout," a- ridges: %d found.\n",nr);
275  fprintf(stdout," a- requir: %d found.\n",nreq);
276  fprintf(stdout," a- connex: %d connected component(s)\n",ncc);
277  fprintf(stdout," a- orient: %d flipped\n",nf);
278  }
279  else if ( abs(mesh->info.imprim) > 3 ) {
280  gen = (2 - nvf + ned - nt) / 2;
281  fprintf(stdout," Connected component: %d, genus: %d, reoriented: %d\n",ncc,gen,nf);
282  fprintf(stdout," Edges: %d, tagged: %d, ridges: %d, required: %d, refs: %d\n",
283  ned,nr+nre+nreq,nr,nreq,nre);
284  }
285 
286  MMG5_SAFE_FREE(pile);
287  return 1;
288 }

292  MMG5_pTria pt,pt1;
293  double n1[3],n2[3],dhd;
294  int *adja,k,kk,ne,nr;
295  int8_t i,ii,i1,i2;
296 
297  ne = nr = 0;
298  for (k=1; k<=mesh->nt; k++) {
299  pt = &mesh->tria[k];
300  if ( !MG_EOK(pt) ) continue;
301 
302  /* triangle normal */
303  MMG5_nortri(mesh,pt,n1);
304  adja = &mesh->adjt[3*(k-1)+1];
305  for (i=0; i<3; i++) {
306  if ( ((pt->tag[i] & MG_PARBDY) && !(pt->tag[i] & MG_PARBDYBDY)) ||
307  (pt->tag[i] & MG_BDY) ) continue;
308 
309  kk = adja[i] / 3;
310  ii = adja[i] % 3;
311  if ( !kk ) {
312  pt->tag[i] |= MG_GEO;
313  i1 = MMG5_inxt2[i];
314  i2 = MMG5_inxt2[i1];
315  mesh->point[pt->v[i1]].tag |= MG_GEO;
316  mesh->point[pt->v[i2]].tag |= MG_GEO;
317  nr++;
318  }
319  else if ( k < kk ) {
320  pt1 = &mesh->tria[kk];
321  /* reference curve */
322  if ( pt1->ref != pt->ref ) {
323  pt->tag[i] |= MG_REF;
324  pt1->tag[ii] |= MG_REF;
325  i1 = MMG5_inxt2[i];
326  i2 = MMG5_inxt2[i1];
327  mesh->point[pt->v[i1]].tag |= MG_REF;
328  mesh->point[pt->v[i2]].tag |= MG_REF;
329  ne++;
330  }
331  /* check angle w. neighbor */
332  MMG5_nortri(mesh,pt1,n2);
333  dhd = n1[0]*n2[0] + n1[1]*n2[1] + n1[2]*n2[2];
334  if ( dhd <= mesh->info.dhd ) {
335  pt->tag[i] |= MG_GEO;
336  pt1->tag[ii] |= MG_GEO;
337  i1 = MMG5_inxt2[i];
338  i2 = MMG5_inxt2[i1];
339  mesh->point[pt->v[i1]].tag |= MG_GEO;
340  mesh->point[pt->v[i2]].tag |= MG_GEO;
341  nr++;
342  }
343  }
344  }
345  }
346  if ( abs(mesh->info.imprim) > 3 && nr > 0 )
347  fprintf(stdout," %d ridges, %d edges updated\n",nr,ne);
348 
349  return 1;
350 }

429 int MMG5_singul(MMG5_pMesh mesh) {
430  MMG5_pTria pt;
431  MMG5_pPoint ppt,p1,p2;
432  double ux,uy,uz,vx,vy,vz,dd;
433  int list[MMG3D_LMAX+2],listref[MMG3D_LMAX+2],k,nc,xp,nr,ns,nre;
434  int8_t i;
435 
436  nre = nc = 0;
437  for (k=1; k<=mesh->nt; k++) {
438  pt = &mesh->tria[k];
439  if ( !MG_EOK(pt) ) continue;
440 
441  for (i=0; i<3; i++) {
442  ppt = &mesh->point[pt->v[i]];
443  if ( !MG_VOK(ppt) || ( ppt->tag & MG_CRN ) || ( ppt->tag & MG_NOM ) ||
444  ( ppt->tag & MG_PARBDY ) ) continue;
445  else if ( MG_EDG(ppt->tag) ) {
446  /* Store the number of ridges passing through the point (xp) and the
447  * number of ref edges (nr) */
448  ns = MMG5_bouler(mesh,mesh->adjt,k,i,list,listref,&xp,&nr,MMG3D_LMAX);
449 
450  if ( !ns ) continue;
451  if ( (xp+nr) > 2 ) {
452  ppt->tag |= MG_CRN + MG_REQ;
453  ppt->tag &= ~MG_NOSURF;
454  nre++;
455  nc++;
456  }
457  else if ( (xp == 1) && (nr == 1) ) {
458  ppt->tag |= MG_REQ;
459  ppt->tag &= ~MG_NOSURF;
460  nre++;
461  }
462  else if ( xp == 1 && !nr ){
463  ppt->tag |= MG_CRN + MG_REQ;
464  ppt->tag &= ~MG_NOSURF;
465  nre++;
466  nc++;
467  }
468  else if ( nr == 1 && !xp ){
469  ppt->tag |= MG_CRN + MG_REQ;
470  ppt->tag &= ~MG_NOSURF;
471  nre++;
472  nc++;
473  }
474  /* check ridge angle */
475  else {
476  p1 = &mesh->point[list[1]];
477  p2 = &mesh->point[list[2]];
478  ux = p1->c[0] - ppt->c[0];
479  uy = p1->c[1] - ppt->c[1];
480  uz = p1->c[2] - ppt->c[2];
481  vx = p2->c[0] - ppt->c[0];
482  vy = p2->c[1] - ppt->c[1];
483  vz = p2->c[2] - ppt->c[2];
484  dd = (ux*ux + uy*uy + uz*uz) * (vx*vx + vy*vy + vz*vz);
485  if ( fabs(dd) > MMG5_EPSD ) {
486  dd = (ux*vx + uy*vy + uz*vz) / sqrt(dd);
487  if ( dd > -mesh->info.dhd ) {
488  ppt->tag |= MG_CRN;
489  nc++;
490  }
491  }
492  }
493  }
494  }
495  }
496 
497  if ( abs(mesh->info.imprim) > 3 && nre > 0 )
498  fprintf(stdout," %d corners, %d singular points detected\n",nc,nre);
499  return 1;
500 }

503 int MMG5_norver(MMG5_pMesh mesh) {
504  MMG5_pTria pt;
505  MMG5_pPoint ppt;
506  MMG5_xPoint *pxp;
507  double n[3],dd;
508  int *adja,k,kk,ng,nn,nt,nf,nnr;
509  int8_t i,ii,i1;
510 
511  /* recomputation of normals only if mesh->xpoint has been freed */
512  if ( mesh->xpoint ) {
513  if ( abs(mesh->info.imprim) > 3 || mesh->info.ddebug ) {
514  fprintf(stdout," ## Warning: %s: no research of boundary points"
515  " and normals of mesh. mesh->xpoint must be freed to enforce"
516  " analysis.\n",__func__);
517  }
518  return 1;
519  }
520 
521  /* identify boundary points */
522  ++mesh->base;
523  mesh->xp = 0;
524  nnr = 0;
525  for (k=1; k<=mesh->nt; k++) {
526  pt = &mesh->tria[k];
527  if ( !MG_EOK(pt) ) continue;
528 
529  for (i=0; i<3; i++) {
530  ppt = &mesh->point[pt->v[i]];
531  if ( ppt->flag == mesh->base ) continue;
532  else {
533  ++mesh->xp;
534  ppt->flag = mesh->base;
535  if ( mesh->nc1 ) {
536  if ( ppt->n[0]*ppt->n[0] + ppt->n[1]*ppt->n[1] + ppt->n[2]*ppt->n[2] > 0 ) {
537  if ( ppt->tag & MG_PARBDY || ppt->tag & MG_CRN || ppt->tag & MG_NOM || MG_EDG(ppt->tag) ) {
538  ++nnr;
539  continue;
540  }
541  ppt->xp = -1;
542  }
543  }
544  }
545  }
546  }
547 
548  /* memory to store normals for boundary points */
549  mesh->xpmax = MG_MAX( (long long)(1.5*mesh->xp),mesh->npmax);
550 
551  MMG5_ADD_MEM(mesh,(mesh->xpmax+1)*sizeof(MMG5_xPoint),"boundary points",return 0);
552  MMG5_SAFE_CALLOC(mesh->xpoint,mesh->xpmax+1,MMG5_xPoint,return 0);
553 
554  /* compute normals + tangents */
555  nn = ng = nt = nf = 0;
556  mesh->xp = 0;
557  ++mesh->base;
558  for (k=1; k<=mesh->nt; k++) {
559  pt = &mesh->tria[k];
560  if ( !MG_EOK(pt) ) continue;
561 
562  adja = &mesh->adjt[3*(k-1)+1];
563  for (i=0; i<3; i++) {
564  ppt = &mesh->point[pt->v[i]];
565  if ( ppt->tag & MG_PARBDY || ppt->tag & MG_CRN || ppt->tag & MG_NOM || ppt->flag == mesh->base ) continue;
566 
567  /* C1 point */
568  if ( !MG_EDG(ppt->tag) ) {
569 
570  if ( (!mesh->nc1) ||
571  ppt->n[0]*ppt->n[0]+ppt->n[1]*ppt->n[1]+ppt->n[2]*ppt->n[2]<=MMG5_EPSD2 ) {
572  if ( !MMG5_boulen(mesh,mesh->adjt,k,i,ppt->n) ) {
573  ++nf;
574  continue;
575  }
576  else ++nn;
577  }
578 
579  ++mesh->xp;
580  if(mesh->xp > mesh->xpmax){
581  MMG5_TAB_RECALLOC(mesh,mesh->xpoint,mesh->xpmax,MMG5_GAP,MMG5_xPoint,
582  "larger xpoint table",
583  mesh->xp--;return 0;);
584  }
585  ppt->xp = mesh->xp;
586  pxp = &mesh->xpoint[ppt->xp];
587  memcpy(pxp->n1,ppt->n,3*sizeof(double));
588  ppt->n[0] = ppt->n[1] = ppt->n[2] = 0.;
589  ppt->flag = mesh->base;
590 
591  }
592 
593  /* along ridge-curve */
594  i1 = MMG5_inxt2[i];
595  if ( !MG_EDG(pt->tag[i1]) ) continue;
596  else if ( !MMG5_boulen(mesh,mesh->adjt,k,i,n) ) {
597  ++nf;
598  continue;
599  }
600  ++mesh->xp;
601  if(mesh->xp > mesh->xpmax){
602  MMG5_TAB_RECALLOC(mesh,mesh->xpoint,mesh->xpmax,MMG5_GAP,MMG5_xPoint,
603  "larger xpoint table",
604  mesh->xp--;return 0;);
605  }
606  ppt->xp = mesh->xp;
607  pxp = &mesh->xpoint[ppt->xp];
608  memcpy(pxp->n1,n,3*sizeof(double));
609 
610  if ( pt->tag[i1] & MG_GEO && adja[i1] > 0 ) {
611  kk = adja[i1] / 3;
612  ii = adja[i1] % 3;
613  ii = MMG5_inxt2[ii];
614  if ( !MMG5_boulen(mesh,mesh->adjt,kk,ii,n) ) {
615  ++nf;
616  continue;
617  }
618  memcpy(pxp->n2,n,3*sizeof(double));
619 
620  /* compute tangent as intersection of n1 + n2 */
621  ppt->n[0] = pxp->n1[1]*pxp->n2[2] - pxp->n1[2]*pxp->n2[1];
622  ppt->n[1] = pxp->n1[2]*pxp->n2[0] - pxp->n1[0]*pxp->n2[2];
623  ppt->n[2] = pxp->n1[0]*pxp->n2[1] - pxp->n1[1]*pxp->n2[0];
624  dd = ppt->n[0]*ppt->n[0] + ppt->n[1]*ppt->n[1] + ppt->n[2]*ppt->n[2];
625  if ( dd > MMG5_EPSD2 ) {
626  dd = 1.0 / sqrt(dd);
627  ppt->n[0] *= dd;
628  ppt->n[1] *= dd;
629  ppt->n[2] *= dd;
630  }
631  ppt->flag = mesh->base;
632  ++nt;
633  continue;
634  }
635 
636  /* compute tgte */
637  ppt->flag = mesh->base;
638  ++nt;
639  if ( !MMG5_boulec(mesh,mesh->adjt,k,i,ppt->n) ) {
640  ++nf;
641  continue;
642  }
643  dd = pxp->n1[0]*ppt->n[0] + pxp->n1[1]*ppt->n[1] + pxp->n1[2]*ppt->n[2];
644  ppt->n[0] -= dd*pxp->n1[0];
645  ppt->n[1] -= dd*pxp->n1[1];
646  ppt->n[2] -= dd*pxp->n1[2];
647  dd = ppt->n[0]*ppt->n[0] + ppt->n[1]*ppt->n[1] + ppt->n[2]*ppt->n[2];
648  if ( dd > MMG5_EPSD2 ) {
649  dd = 1.0 / sqrt(dd);
650  ppt->n[0] *= dd;
651  ppt->n[1] *= dd;
652  ppt->n[2] *= dd;
653  }
654  }
655  }
656  mesh->nc1 = 0;
657 
658  if ( abs(mesh->info.imprim) > 3 && nn+nt > 0 ) {
659  if ( nnr )
660  fprintf(stdout," %d input normals ignored\n",nnr);
661  fprintf(stdout," %d normals, %d tangents updated (%d failed)\n",nn,nt,nf);
662  }
663  return 1;
664 }

766 int MMG3D_analys(MMG5_pMesh mesh) {
767  MMG5_Hash hash;
768  int ier;
769 
771  if ( abs(mesh->info.imprim) > 3 )
772  fprintf(stdout,"\n ** SURFACE ANALYSIS\n");
773 
774  /* create tetra adjacency */
775  if ( !MMG3D_hashTetra(mesh,1) ) {
776  fprintf(stderr,"\n ## Hashing problem (1). Exit program.\n");
777  return 0;
778  }
779 
780  if ( mesh->info.iso && mesh->info.opnbdy ) {
781  ier = MMG3D_update_xtetra ( mesh );
782  if ( !ier ) {
783  fprintf(stderr,"\n ## Problem when updating the xtetra data after ls discretization."
784  " Exit program.\n");
785  return 0;
786  }
787  }
788 
789  /* create prism adjacency */
790  if ( !MMG3D_hashPrism(mesh) ) {
791  fprintf(stderr,"\n ## Prism hashing problem. Exit program.\n");
792  return 0;
793  }
794 
795  /* compatibility triangle orientation w/r tetras */
796  if ( !MMG5_bdryPerm(mesh) ) {
797  fprintf(stderr,"\n ## Boundary orientation problem. Exit program.\n");
798  return 0;
799  }
800 
801  /* identify surface mesh */
802  if ( !MMG5_chkBdryTria(mesh) ) {
803  fprintf(stderr,"\n ## Boundary problem. Exit program.\n");
804  return 0;
805  }
806 
807  MMG5_freeXTets(mesh);
808  MMG5_freeXPrisms(mesh);
809 
810  /* Set surface triangles to required in nosurf mode or for parallel boundaries */
811  MMG3D_set_reqBoundaries(mesh);
812 
813  /* create surface adjacency */
814  memset ( &hash, 0x0, sizeof(MMG5_Hash));
815  if ( !MMG3D_hashTria(mesh,&hash) ) {
816  MMG5_DEL_MEM(mesh,hash.item);
817  fprintf(stderr,"\n ## Hashing problem (2). Exit program.\n");
818  return 0;
819  }
820 
821  /* build hash table for geometric edges */
822  if ( !MMG5_hGeom(mesh) ) {
823  fprintf(stderr,"\n ## Hashing problem (0). Exit program.\n");
824  MMG5_DEL_MEM(mesh,hash.item);
825  MMG5_DEL_MEM(mesh,mesh->htab.geom);
826  return 0;
827  }
828 
830  if ( abs(mesh->info.imprim) > 5 || mesh->info.ddebug )
831  fprintf(stdout," ** SETTING TOPOLOGY\n");
832 
833  /* identify connexity and flip orientation of faces if needed */
834  if ( !MMG5_setadj(mesh) ) {
835  fprintf(stderr,"\n ## Topology problem. Exit program.\n");
836  MMG5_DEL_MEM(mesh,hash.item);
837  return 0;
838  }
839 
840  /* check for ridges */
841  if ( mesh->info.dhd > MMG5_ANGLIM && !MMG5_setdhd(mesh) ) {
842  fprintf(stderr,"\n ## Geometry problem. Exit program.\n");
843  MMG5_DEL_MEM(mesh,hash.item);
844  return 0;
845  }
846 
847  /* identify singularities */
848  if ( !MMG5_singul(mesh) ) {
849  fprintf(stderr,"\n ## MMG5_Singularity problem. Exit program.\n");
850  MMG5_DEL_MEM(mesh,hash.item);
851  return 0;
852  }
853 
854  if ( abs(mesh->info.imprim) > 3 || mesh->info.ddebug )
855  fprintf(stdout," ** DEFINING GEOMETRY\n");
856 
857  /* define (and regularize) normals */
858  if ( !MMG5_norver(mesh) ) {
859  fprintf(stderr,"\n ## Normal problem. Exit program.\n");
860  MMG5_DEL_MEM(mesh,hash.item);
861  return 0;
862  }
863  if ( mesh->info.nreg && !MMG5_regnor(mesh) ) {
864  fprintf(stderr,"\n ## Normal regularization problem. Exit program.\n");
865  return 0;
866  }
867 
868  /* set bdry entities to tetra and fill the orientation field */
869  if ( !MMG5_bdrySet(mesh) ) {
870  fprintf(stderr,"\n ## Boundary problem. Exit program.\n");
871  MMG5_DEL_MEM(mesh,hash.item);
872  MMG5_DEL_MEM(mesh,mesh->xpoint);
873  return 0;
874  }
875 
876  /* set non-manifold edges sharing non-intersecting multidomains as required */
877  if ( abs(mesh->info.imprim) > 5 || mesh->info.ddebug )
878  fprintf(stdout," ** UPDATING TOPOLOGY AT NON-MANIFOLD POINTS\n");
879 
880  if ( !MMG5_setNmTag(mesh,&hash) ) {
881  fprintf(stderr,"\n ## Non-manifold topology problem. Exit program.\n");
882  MMG5_DEL_MEM(mesh,hash.item);
883  MMG5_DEL_MEM(mesh,mesh->xpoint);
884  return 0;
885  }
886 
887  /* check subdomains connected by a vertex and mark these vertex as corner and
888  required */
889  MMG5_chkVertexConnectedDomains(mesh);
890 
891  /* build hash table for geometric edges */
892  if ( !mesh->na && !MMG5_hGeom(mesh) ) {
893  fprintf(stderr,"\n ## Hashing problem (0). Exit program.\n");
894  MMG5_DEL_MEM(mesh,mesh->xpoint);
895  MMG5_DEL_MEM(mesh,mesh->htab.geom);
896  return 0;
897  }
898 
899  /* Update edges tags and references for xtetras */
900  if ( !MMG5_bdryUpdate(mesh) ) {
901  fprintf(stderr,"\n ## Boundary problem. Exit program.\n");
902  MMG5_DEL_MEM(mesh,mesh->xpoint);
903  return 0;
904  }
905 
906  /* define geometry for non manifold points */
907  if ( !MMG3D_nmgeom(mesh) ) return 0;
908 
909 #ifdef USE_POINTMAP
910  /* Initialize source point with input index */
911  int ip;
912  for( ip = 1; ip <= mesh->np; ip++ )
913  mesh->point[ip].src = ip;
914 #endif
915 
916  /* release memory */
917  MMG5_DEL_MEM(mesh,mesh->htab.geom);
918  MMG5_DEL_MEM(mesh,mesh->adjt);
919  MMG5_DEL_MEM(mesh,mesh->tria);
920  mesh->nt = 0;
921 
922  if ( mesh->nprism ) MMG5_DEL_MEM(mesh,mesh->adjapr);
923 
924  return 1;
925 }
