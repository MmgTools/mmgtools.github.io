165

250 int MMG3D_optbdry(MMG5_pMesh mesh,MMG5_pSol met,MMG3D_pPROctree PROctree,int k) {

252  MMG5_pxTetra pxt;
253  int ib,i,j,ipb,list[MMG3D_LMAX+2];
254  int iedg,ier,ilist,ied,ia,it1,it2,ret,imove;
255

257

260

262 
263  for(i=0 ; i<4 ; i++)
264  if ( pxt->ftag[i] & MG_BDY ) break;
265 
266  assert ( i< 4 );
267  if ( i== 4 ) return 0;
268 
269  ib = i;
270  ipb = pt->v[ib];
271 
272  /*check that the vertex is not a boundary one*/
273  if ( mesh->point[ipb].tag & MG_BDY ) return 0;
274 
275  /* try to move the vertex in order to improve the quality*/
276  ier = 0;
277  if ( !mesh->info.nomove ) {
278  for(j = 0 ; j<3 ; j++) {
279  imove = MMG3D_movetetrapoints(mesh,met,PROctree,k);
280  ier += imove;

282  }
283  if(ier) {
284  imove = 1;
285  }
286  }
287 
288  if(!mesh->info.noinsert) {
289  /*try to remove the non-bdry vertex*/
290  ier = MMG3D_coledges(mesh,met,k,ib);
291  if(ier) return 1;
292 
293  /* try to remove the non-bdry vertex : with all the edges containing the
294  * vertex */
295  ier = MMG3D_deletePoint(mesh,met,PROctree,k,i);
296  if(ier) return 1;
297  }
298 
299 
300  /*try to swap the 3 internal edges*/
301  if(!mesh->info.noswap) {
302  for(ied = 0 ; ied<3 ;ied++) {
303  iedg = MMG5_arpt[i][ied];
304  ier = MMG3D_swpItem(mesh,met,PROctree,k,iedg);
305  if(ier) {
306  return 1;
307  }
308  }
309  /*try to swap the bdry edges*/
310  for (j=0; j<3; j++) {
311  ia = MMG5_iarf[i][j];
312 
313  /* No swap of geometric edge */
314  if ( MG_EDG(pxt->tag[ia]) || (pxt->tag[ia] & MG_REQ) ||
315  (pxt->tag[ia] & MG_NOM) )
316  continue;
317 
318  ret = MMG5_coquilface(mesh,k,i,ia,list,&it1,&it2,0);
319  ilist = ret / 2;
320  if ( ret < 0 ) return -1;
321  /* CAUTION: trigger collapse with 2 elements */
322  if ( ilist <= 1 ) continue;
323  ier = MMG5_chkswpbdy(mesh,met,list,ilist,it1,it2,2);
324  if ( ier < 0 )
325  return -1;
326  else if ( ier ) {
327  ier = MMG5_swpbdy(mesh,met,list,ret,it1,PROctree,2);
328  if ( ier < 0 ) return -1;
329  else if(ier) {
330  return 1;
331  }
332  }
333  }
334  }
335 
336 
337  return imove;
