240 int chknor(MMG5_pMesh mesh) {
241  MMG5_pTria pt;
242  MMG5_pPoint p0;
243  MMG5_pxPoint go;
244  double dd,ps,*n,nt[3];
245  int k;
246  int8_t i;
247  static int8_t mmgWarn0=0, mmgWarn1=0;
248 
249  /* First test : check that all normal vectors at points are non 0 */
250  for (k=1; k<=mesh->np; k++) {
251  p0 = &mesh->point[k];
252  if ( !MG_VOK(p0) ) continue;
253  if ( MS_SIN(p0->tag) ) continue;
254  if ( !(p0->tag & MG_GEO) ) continue;
255 
256  assert( p0->xp );
257  go = &mesh->xpoint[p0->xp];
258  n = &go->n1[0];
259 
260  dd = n[0]*n[0] + n[1]*n[1] + n[2]*n[2];
261  if ( dd < 0.9 ) {
262  if ( !mmgWarn0 ) {
263  mmgWarn0 = 1;
264  fprintf(stderr,"\n ## Error: %s: at least 1 non unitary normal"
265  " (point: %d normal n1 = %f %f %f). exit program\n",
266  __func__,MMGS_indPt(mesh,k),n[0],n[1],n[2]);
267  }
268  return 0;
269  }
270 
271  n = &go->n2[0];
272 
273  dd = n[0]*n[0] + n[1]*n[1] + n[2]*n[2];
274  if ( dd < 0.9 ) {
275  if ( !mmgWarn0 ) {
276  mmgWarn0 = 1;
277  fprintf(stderr,"\n ## Error: %s: at least 1 non unitary normal"
278  " (point: %d normal n2 = %f %f %f). exit program\n",
279  __func__,MMGS_indPt(mesh,k),n[0],n[1],n[2]);
280  }
281  return 0;
282  }
283  }
284 
285  /* Second test : check that all normal vectors at points are consistently oriented with
286  respect to the underlying triangles */
287  for (k=1; k<=mesh->nt; k++) {
288  pt = &mesh->tria[k];
289  if ( !MG_EOK(pt) ) continue;
290 
291  MMG5_nortri(mesh,pt,nt);
292  for (i=0; i<3; i++) {
293  p0 = &mesh->point[pt->v[i]];
294  if ( MS_SIN(p0->tag) ) continue;
295  else if ( MG_EDG(p0->tag) ) {
296  assert ( p0->xp );
297  go = &mesh->xpoint[p0->xp];
298  if ( p0->tag & MG_GEO ) {
299  n = &go->n1[0];
300  ps = n[0]*nt[0] + n[1]*nt[1] + n[2]*nt[2];
301  if ( ps < -0.99 ) {
302  if ( !mmgWarn1 ) {
303  mmgWarn1 = 1;
304  fprintf(stderr,"\n ## Error: %s: at least 1"
305  " inconsistant normal (point %d in triangle %d):"
306  " ps = %f \n",__func__,MMGS_indPt(mesh,pt->v[i]),
307  MMGS_indElt(mesh,k),ps);
308  }
309  return 0;
310  }
311 
312  n = &go->n2[0];
313  ps = n[0]*nt[0] + n[1]*nt[1] + n[2]*nt[2];
314  if ( ps < -0.99 ) {
315  if ( !mmgWarn1 ) {
316  mmgWarn1 = 1;
317  fprintf(stderr,"\n ## Error: %s: at least 1"
318  " inconsistant normal (point %d in triangle %d):"
319  " ps = %f \n",__func__,MMGS_indPt(mesh,pt->v[i]),
320  MMGS_indElt(mesh,k),ps);
321  }
322  return 0;
323  }
324  }
325  else {
326  n = &go->n1[0];
327  ps = n[0]*nt[0] + n[1]*nt[1] + n[2]*nt[2];
328  if ( ps < -0.99 ) {
329  if ( !mmgWarn1 ) {
