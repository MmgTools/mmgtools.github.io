50  MMG5_pPoint p0,ppt0;
51  MMG5_Bezier pb;
52  double r[3][3],lispoi[3*MMGS_LMAX+1],*m0;//,m[6],mo[6];
53  double gv[2],area,detloc,step,lambda[3],o[3],no[3],to[3],uv[2];
54  double calold,calnew,caltmp;
55  int k,iel,kel,nump,nbeg,nend;
56  int8_t i0,i1,i2,ier;
57  static int warn=0;
58  static int8_t mmgErr0=0,mmgErr1=0;
59 
60  step = 0.1;
61 
62  /* Make sure ball of point is closed */
63  iel = list[0] / 3;
64  i0 = list[0] % 3;
65  i1 = MMG5_inxt2[i0];
66 
67  pt = &mesh->tria[iel];
68  nump = pt->v[i0];
69  nbeg = pt->v[i1];
70  p0 = &mesh->point[nump];
71  m0 = &met->m[6*nump];
72  assert( !p0->tag );
73 
74  iel = list[ilist-1] / 3;
75  i0 = list[ilist-1] % 3;
76  i2 = MMG5_iprv2[i0];
77 
78  pt = &mesh->tria[iel];
79  nend = pt->v[i2];
80  if ( nbeg != nend ) return 0;
81 
82  /* Rotation of the ball of p0 */
83  if ( !MMGS_surfballRotation(mesh,p0,list,ilist,r,lispoi) ) {
84  return 0;
85  }
86 
89  gv[0] = 0.0;
90  gv[1] = 0.0;
91 
92  for (k=0; k<ilist; k++) {
93  iel = list[k] / 3;
94  pt = &mesh->tria[iel];
95  if ( !MMG5_bezierCP(mesh,pt,&pb,1) ) return 0;
96 
97  /* Compute integral of sqrt(T^J(xi) M(P(xi)) J(xi)) * P(xi) over the triangle */
98  if ( !MMG5_elementWeight(mesh,met,pt,p0,&pb,r,gv) ) {
99  if ( !warn ) {
100  ++warn;
101  fprintf(stderr,"\n ## Warning: %s: unable to compute optimal position for at least"
102  " 1 point.\n",__func__ );
103  }
104  return 0;
105  }
106  }
107 
108  /* At this point : gv = - gradient of V = direction to follow */
110  area = lispoi[1]*gv[1] - lispoi[2]*gv[0];
111  kel = 0;
112  if ( area >= 0.0 ) {
113  for (k=0; k<ilist; k++) {
114  detloc = gv[0]*lispoi[3*(k+1)+2] - gv[1]*lispoi[3*(k+1)+1]; /*orientation with respect to p2 */
115  if ( detloc >= 0.0 ) {
116  kel = k;
117  break;
118  }
119  }
120  if ( k == ilist ) return 0;
121  }
122  else {
123  for (k=ilist-1; k>=0; k--) {
124  detloc = lispoi[3*k+1]*gv[1] - lispoi[3*k+2]*gv[0]; //orientation with respect to p2
125  if ( detloc >= 0.0 ) {
126  kel = k;
127  break;
128  }
129  }
130  if ( k == -1 ) return 0;
131  }
132 
133  /* Sizing of time step : make sure point does not go out corresponding triangle. */
134  area = - gv[1]*(lispoi[3*(kel+1)+1] - lispoi[3*(kel)+1]) \
135  + gv[0]*(lispoi[3*(kel+1)+2] - lispoi[3*(kel)+2]);
136  if ( fabs(area) < MMG5_EPSD2 ) return 0;
137  area = 1.0 / area;
138  step *= area;
139 
140  area = lispoi[3*(kel)+1]*(lispoi[3*(kel+1)+2] - lispoi[3*(kel)+2]) \
141  - lispoi[3*(kel)+2 ]*(lispoi[3*(kel+1)+1] - lispoi[3*(kel)+1]);
142  step *= area;
143  step = fabs(step);
144  gv[0] *= step;
145  gv[1] *= step;
146 
147  /* Computation of the barycentric coordinates of the new point in the corresponding triangle. */
148  area = lispoi[3*kel+1]*lispoi[3*(kel+1)+2] - lispoi[3*kel+2]*lispoi[3*(kel+1)+1];
149  if ( area < MMG5_EPSD2 ) return 0;
150  area = 1.0 / area;
151  lambda[1] = lispoi[3*(kel+1)+2]*gv[0] - lispoi[3*(kel+1)+1]*gv[1];
152  lambda[2] = -lispoi[3*(kel)+2]*gv[0] + lispoi[3*(kel)+1]*gv[1];
153  lambda[1]*= (area);
154  lambda[2]*= (area);
155  lambda[0] = 1.0 - lambda[1] - lambda[2];
156 
158  iel = list[kel] / 3;
159  i0 = list[kel] % 3;
160  pt = &mesh->tria[iel];
161 
162  ier = MMG5_bezierCP(mesh,pt,&pb,1);
163  if ( !ier ) {
164  if( !mmgErr0 ) {
165  mmgErr0 = 1;
166  fprintf(stderr,"\n ## Warning: %s: function MMG5_bezierCP return 0.\n",
167  __func__);
168  }
169  return 0;
170  }
171 
172  /* Now, for Bezier interpolation, one should identify which of i,i1,i2 is 0,1,2
173  recall uv[0] = barycentric coord associated to pt->v[1], uv[1] associated to pt->v[2] */
174  if ( i0 == 0 ) {
175  uv[0] = lambda[1];
176  uv[1] = lambda[2];
177  }
178  else if ( i0 == 1 ) {
179  uv[0] = lambda[0];
180  uv[1] = lambda[1];
181  }
182  else{
183  uv[0] = lambda[2];
184  uv[1] = lambda[0];
185  }
186 
187  ier = MMGS_bezierInt(&pb,uv,o,no,to);
188  if ( !ier ) {
189  if( !mmgErr1 ) {
190  mmgErr1 = 1;
191  fprintf(stderr," ## Warning: %s: function MMGS_bezierInt return 0.\n",
192  __func__);
193  }
194  return 0;
195  }
196 
197  /* Second test : check whether geometric approximation has not been too much degraded */
198  ppt0 = &mesh->point[0];
199  ppt0->c[0] = o[0];
200  ppt0->c[1] = o[1];
201  ppt0->c[2] = o[2];
202 
203  ppt0->n[0] = no[0];
204  ppt0->n[1] = no[1];
205  ppt0->n[2] = no[2];
206  ppt0->tag = 0;
207 
208  // parallel transport of metric at p0 to new point
209  MMG5_paratmet(p0->c,p0->n,m0,o,no,&met->m[0]);
210 
211  calold = calnew = DBL_MAX;
212  for (k= 0; k<ilist; k++) {
213  iel = list[k] / 3;
214  i0 = list[k] % 3;
215  pt = &mesh->tria[iel];
216  pt0 = &mesh->tria[0];
217  memcpy(pt0,pt,sizeof(MMG5_Tria));
218  pt0->v[i0] = 0;
219 
220  caltmp = caleltsig_ani(mesh,met,iel);
221  calold = MG_MIN(calold,caltmp);
222  caltmp = caleltsig_ani(mesh,met,0);
223  if ( caltmp < MMG5_EPSD2 ) {
224  /* We don't check the input triangle qualities, thus we may have a very
225  * bad triangle in our mesh */
226  return 0;
227  }
228  calnew = MG_MIN(calnew,caltmp);
229 
230  if ( calold < MMG5_EPSOK && calnew <= calold ) return 0;
231  else if (calnew < MMG5_EPSOK) return 0;
232  else if ( calnew < 0.3*calold ) return 0;
233  /* if ( chkedg(mesh,0) ) return 0; */
234  }
235 
236  /* Finally, update coordinates and normals of point, if new position is accepted :*/
237  p0->c[0] = o[0];
238  p0->c[1] = o[1];
239  p0->c[2] = o[2];
240 
241  p0->n[0] = no[0];
242  p0->n[1] = no[1];
243  p0->n[2] = no[2];
244 
245  memcpy(m0,&met->m[0],6*sizeof(double));
246 
247  return 1;
248 }

265  ip1 = ip2 = 0;

267 
268  /* First step : make sure 2 ridge edges pass through point, and recoved them,
269  along with neighbouring triangles */
270  for (k=0; k<ilist; k++) {
271  iel = list[k] / 3;
272  i0 = list[k] % 3;
273  i1 = MMG5_inxt2[i0];
274  i2 = MMG5_inxt2[i1];
275  pt = &mesh->tria[iel];
276 
277  if ( MG_EDG(pt->tag[i1]) ) {
278  if ( !it1 ) {
279  it1 = iel;
280  ip1 = pt->v[i2]; // edge(i1) = (p0p2)
281  voy1 = i1;
282  if ( pt->tag[i1] & MG_GEO ) isrid1 = 1;
283  }
284  else if ( it1 && !it2 ) {
285  if ( ip1 != pt->v[i2] ) {
286  it2 = iel;
287  ip2 = pt->v[i2]; // edge(i1) = (p0p2)
288  voy2 = i1;
289  if ( pt->tag[i1] & MG_GEO ) isrid2 = 1;
290  }
291  }
292  else if ( it1 && it2 && (pt->v[i2] != ip1) && (pt->v[i2] != ip2) ) {
293  if ( !mmgWarn0 ) {
294  mmgWarn0 = 1;
295  fprintf(stderr,"\n ## Warning: %s: at least 1 point at the"
296  " intersection of 3 ridge edges\n",__func__);
297  }
298  return 0;
299  }
300  }
301 
302  if ( MG_EDG(pt->tag[i2]) ) {
303  if ( !it1 ) {
304  it1 = iel;
305  ip1 = pt->v[i1]; // edge(i2) = (p0p1)
306  voy1 = i2;
307  if ( pt->tag[i2] & MG_GEO ) isrid1 = 1;
308  }
309  else if ( it1 && !it2 ) {
310  if ( ip1 != pt->v[i1] ) {
311  it2 = iel;
312  ip2 = pt->v[i1]; // edge(i1) = (p0p2)
313  voy2 = i2;
314  if ( pt->tag[i2] & MG_GEO ) isrid2 = 1;
315  }
316  }
317  else if ( it1 && it2 && (pt->v[i1] != ip1) && (pt->v[i1] != ip2) ) {
318  if ( !mmgWarn0 ) {
319  mmgWarn0 = 1;
320  fprintf(stderr,"\n ## Warning: %s: at least 1 point at the"
321  " intersection of 3 ridge edges\n",__func__);
322  }
323  return 0;
324  }
325  }
326  }
327 
328  /* Second step : compute lengths of edges (p0p1),(p0p2) */
329  iel = list[0] / 3;
330  i0 = list[0] % 3;
331  pt = &mesh->tria[iel];
332  ip0 = pt->v[i0];
333  p0 = &mesh->point[ip0];
334  p1 = &mesh->point[ip1];
335  p2 = &mesh->point[ip2];
336  m0 = &met->m[6*ip0];
337 
338  l1old = MMG5_lenSurfEdg(mesh,met,ip0,ip1,1);
339  l2old = MMG5_lenSurfEdg(mesh,met,ip0,ip2,1);
340 
341  if ( (!l1old) || (!l2old) ) return 0;
342 
343  if ( l1old < l2old ) { //move towards p2
344  ip = ip2;
345  isrid = isrid2;
346  it = it2;
347  }
348  else {
349  ip = ip1;
350  isrid = isrid1;
351  it = it1;
352  }
353 
354  /* Third step : infer arc length of displacement, parameterized over edges */
355  if ( !MMGS_paramDisp ( mesh,it,isrid,ip0,ip,step,o) ) {
356  return 0;
357  }
358 
359  /* check whether proposed move is admissible uner consideration of distances */
360  /* Normal and tangent vector updates */
361  lam0 = (1.0-step)*(1.0-step);
362  lam1 = 2.0*step*(1.0-step);
363  lam2 = step*step;
364 
365  /* Move is made towards p2 */
366  ll1old = l1old*l1old;
367  ll2old = l2old*l2old;
368 
369  if ( l2old > l1old ) {
370  if ( !MMGS_moveTowardPoint(mesh,p0,p2,ll2old,lam0,lam1,lam2,nn1,nn2,to) ) {
371  return 0;
372  }
373 
374  /* Interpolation of metric between ip0 and ip2 */
375  if ( isrid ) {
376  if(!MMG5_intridmet(mesh,met,mesh->tria[it2].v[MMG5_inxt2[voy2]],
377  mesh->tria[it2].v[MMG5_iprv2[voy2]],(1.0-step),nn1,mo))
378  return 0;
379  }
380  else {

383  }
384 
385  /* Move along p1 */
386  else {

416

429  l1new = MMG5_lenSurfEdg(mesh,met,0,ip1,1);
430  l2new = MMG5_lenSurfEdg(mesh,met,0,ip2,1);
