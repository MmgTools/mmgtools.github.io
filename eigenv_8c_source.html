85 static int newton3(double p[4],double x[3]) {
86  double b,c,d,da,db,dc,epsd;
87  double delta,fx,dfx,dxx;
88  double fdx0,fdx1,dx0,dx1,x1,x2,tmp,epsA,epsB;
89  int it,it2,n;
90  static int8_t mmgWarn=0;
91 
92  /* coeffs polynomial, a=1 */
93  if ( p[3] != 1. ) {
94  if ( !mmgWarn ) {
95  fprintf(stderr,"\n ## Warning: %s: bad use of newton3 function, polynomial"
96  " must be of type P(x) = x^3+bx^2+cx+d.\n",
97  __func__);
98  mmgWarn = 1;
99  }
100  return 0;
101  }
102 
103  b = p[2];
104  c = p[1];
105  d = p[0];
106 
107  /* 1st derivative of f */
108  da = 3.0;
109  db = 2.0*b;
110 
111  /* solve 2nd order eqn */
112  delta = db*db - 4.0*da*c;
113  epsd = db*db*MG_EIGENV_EPS10;
114 
115  /* inflexion (f'(x)=0, x=-b/2a) */
116  x1 = -db / 6.0f;
117 
118  n = 1;
119  if ( delta > epsd ) {
120  delta = sqrt(delta);
121  dx0 = (-db + delta) / 6.0;
122  dx1 = (-db - delta) / 6.0;
123  /* Horner */
124  fdx0 = d + dx0*(c+dx0*(b+dx0));
125  fdx1 = d + dx1*(c+dx1*(b+dx1));
126 
127 
128  x[2] = -b - 2.0*dx0;
129  tmp = -b - 2.0*dx1;
130  if ( fabs(fdx0) < MG_EIGENV_EPS27 ||
131  ( fabs(fdx0) < MG_EIGENV_EPS13 && (dx0 > 0. && x[2] > 0.) ) ) {
132  /* dx0: double root, compute single root */
133  n = 2;
134  x[0] = dx0;
135  x[1] = dx0;
136  /* check if P(x) = 0 */
137  fx = d + x[2]*(c+x[2]*(b+x[2]));
138  if ( fabs(fx) > MG_EIGENV_EPS10 ) {
139 #ifdef DEBUG
140  fprintf(stderr,"\n ## Error: %s: ERR 9100, newton3: fx= %E.\n",
141  __func__,fx);
142 #endif
143  return 0;
144  }
145  return n;
146  }
147  else if ( fabs(fdx1) < MG_EIGENV_EPS27 ||
148  ( fabs(fdx1) < MG_EIGENV_EPS13 && (dx1 > 0. && tmp > 0.) ) ) {
149  /* dx1: double root, compute single root */
150  n = 2;
151  x[0] = dx1;
152  x[1] = dx1;
153  x[2] = tmp;
154  /* check if P(x) = 0 */
155  fx = d + x[2]*(c+x[2]*(b+x[2]));
156  if ( fabs(fx) > MG_EIGENV_EPS10 ) {
157 #ifdef DEBUG
158  fprintf(stderr,"\n ## Error: %s: ERR 9100, newton3: fx= %E.\n",
159  __func__,fx);
160 #endif
161  return 0;
162  }
163  return n;
164  }
165  }
166 
167  else if ( fabs(delta) < db*db * 1.e-20 ||
168  ( fabs(delta) < epsd && x1 > 0. ) ) {
169  /* triple root */
170  n = 3;
171  x[0] = x1;
172  x[1] = x1;
173  x[2] = x1;
174  /* check if P(x) = 0 */
175  fx = d + x[0]*(c+x[0]*(b+x[0]));
176  if ( fabs(fx) > MG_EIGENV_EPS10 ) {
177 #ifdef DEBUG
178  fprintf(stderr,"\n ## Error: %s: ERR 9100, newton3: fx= %E.\n",
179  __func__,fx);
180 #endif
181  return 0;
182  }
183  return n;
184  }
185 
186  else {
187 #ifdef DEBUG
188  fprintf(stderr,"\n ## Error: %s: ERR 9101, newton3: no real roots.\n",
189  __func__);
190 #endif
191  return 0;
192  }
193 
194  /* Newton method: find one root (middle)
195  starting point: P"(x)=0 */
196  x1 = -b / 3.0;
197  dfx = c + b*x1;
198  fx = d + x1*(c -2.0*x1*x1);
199 
200  epsA = MG_EIGENV_EPS13;
201  epsB = MG_EIGENV_EPS10;
202 
203  it2 = 0;
204 newton :
205 
206  it = 0;
207  do {
208  x2 = x1 - fx / dfx;
209  fx = d + x2*(c+x2*(b+x2));
210  if ( fabs(fx) < epsA && x2 > 0. ) {
211  x[0] = x2;
212  break;
213  }
214  dfx = c + x2*(db + da*x2);
215 
216  /* check for break-off condition */
217  dxx = fabs((x2-x1) / x2);
218  if ( dxx < epsB && x2 > 0. ) {
219  x[0] = x2;
220 
221  /* Check accuracy for 1e-5 precision only (we don't want to fail for smaller
222  * precision) */
223  if ( fabs(fx) > MG_EIGENV_EPS10 ) {
224  fprintf(stderr,"\n ## Error: %s: ERR 9102, newton3, no root found"
225  " (fx %E).\n",
226  __func__,fx);
227  return 0;
228  }
229  break;
230  }
231  else
232  x1 = x2;
233  }
234  while ( ++it < MAXTOU );
235 
236  if ( it == MAXTOU ) {
237  x[0] = x1;
238  fx = d + x1*(c+(x1*(b+x1)));
239  /* Check accuracy for 1e-5 precision only (we don't want to fail for smaller
240  * precision) */
241  if ( fabs(fx) > MG_EIGENV_EPS10 ) {
242  fprintf(stderr,"\n ## Error: %s: ERR 9102, newton3, no root found"
243  " (fx %E).\n",
244  __func__,fx);
245  return 0;
246  }
247  }
248 
249  /* solve 2nd order equation
250  P(x) = (x-sol(1))* (x^2+bb*x+cc) */
251  db = b + x[0];
252  dc = c + x[0]*db;
253  delta = db*db - 4.0*dc;
254 
255  if ( delta <= 0.0 ) {
256  fprintf(stderr,"\n ## Error: %s: ERR 9103, newton3, det = 0.\n",__func__);
257  return 0;
258  }
259 
260  delta = sqrt(delta);
261  x[1] = 0.5 * (-db+delta);
262  x[2] = 0.5 * (-db-delta);
263 
264  while ( ++it2 < 5 && ( ( x[0] <= 0 && fabs(x[0]) <= MG_EIGENV_EPS5 ) ||
265  ( x[1] <= 0 && fabs(x[1]) <= MG_EIGENV_EPS5 ) ||
266  ( x[2] <= 0 && fabs(x[2]) <= MG_EIGENV_EPS5 ) ) ) {
267  /* Get back to the newton method with increased accuracy */
268  epsA /= 10;
269  epsB /= 10;
270  goto newton;
271  }
272 
273 #ifdef DEBUG
274  /* check for root accuracy */
275  fx = d + x[1]*(c+x[1]*(b+x[1]));
276  if ( fabs(fx) > MG_EIGENV_EPS10 ) {
277  fprintf(stderr,"\n ## Error: %s: ERR 9104, newton3: fx= %E x= %E.\n",
278  __func__,fx,x[1]);
279  return 0;
280  }
281  fx = d + x[2]*(c+x[2]*(b+x[2]));
282  if ( fabs(fx) > MG_EIGENV_EPS10 ) {
283  fprintf(stderr,"\n ## Error: %s: ERR 9104, newton3: fx= %E x= %E.\n",
284  __func__,fx,x[2]);
285  return 0;
286  }
287 #endif
288 
289  return n;
290 }

384 int MMG5_eigenv(int symmat,double *mat,double lambda[3],double v[3][3]) {

386  double aa,bb,cc,dd,ee,ii,vx1[3],vx2[3],vx3[3],dd1,dd2,dd3;

388  int k,n;
389 
390  epsd = MG_EIGENV_EPS13;
391 
392  w1[0] = w1[1] = w1[2] = 0;
393  w2[0] = w2[1] = w2[2] = 0;
394  w3[0] = w3[1] = w3[2] = 0;
395  /* default */
396  memcpy(v,Id,9*sizeof(double));
397  if ( symmat ) {
398  lambda[0] = (double)mat[0];
399  lambda[1] = (double)mat[3];
400  lambda[2] = (double)mat[5];
401 
402  maxm = fabs(mat[0]);
403  for (k=1; k<6; k++) {
404  valm = fabs(mat[k]);
405  if ( valm > maxm ) maxm = valm;
406  }
407  /* single float accuracy if sufficient, else double float accuracy */
408  if ( maxm < MG_EIGENV_EPS5e6 ) {
409  if ( lambda[0]>0. && lambda[1] > 0. && lambda[2] > 0. ) {
410  return 1;
411  }
412  else if ( maxm < MG_EIGENV_EPS13 ) {
413  return 0;
414  }
415  epsd = MG_EIGENV_EPS27;
416  }
417 
418  /* normalize matrix */
419  dd = 1.0 / maxm;
420  a11 = mat[0] * dd;
421  a12 = mat[1] * dd;
422  a13 = mat[2] * dd;
423  a22 = mat[3] * dd;
424  a23 = mat[4] * dd;
425  a33 = mat[5] * dd;
426 
427  /* diagonal matrix */
428  maxd = fabs(a12);
429  valm = fabs(a13);
430  if ( valm > maxd ) maxd = valm;
431  valm = fabs(a23);
432  if ( valm > maxd ) maxd = valm;
433  if ( maxd < epsd ) return 1;
434 
435  a21 = a12;
436  a31 = a13;
437  a32 = a23;
438 
439  /* build characteristic polynomial
440  P(X) = X^3 - trace X^2 + (somme des mineurs)X - det = 0 */
441  aa = a11*a22;
442  bb = a23*a32;
443  cc = a12*a21;
444  dd = a13*a31;
445  p[0] = a11*bb + a33*(cc-aa) + a22*dd -2.0*a12*a13*a23;
446  p[1] = a11*(a22 + a33) + a22*a33 - bb - cc - dd;
447  p[2] = -a11 - a22 - a33;
448  p[3] = 1.0;
449  }
450  else {
451  lambda[0] = (double)mat[0];
452  lambda[1] = (double)mat[4];
453  lambda[2] = (double)mat[8];
454

456  for (k=1; k<9; k++) {
457  valm = fabs(mat[k]);
458  if ( valm > maxm ) maxm = valm;
459  }
460 
461  /* single float accuracy if sufficient, else double float accuracy */
462  if ( maxm < MG_EIGENV_EPS5e6 ) {
463  if ( lambda[0]>0. && lambda[1] > 0. && lambda[2] > 0. ) {
464  return 1;
465  }
466  else if ( maxm < MG_EIGENV_EPS13 ) {
467  return 0;
468  }
469  epsd = MG_EIGENV_EPS27;
470  }
471 
472  /* normalize matrix */
473  dd = 1.0 / maxm;
474  a11 = mat[0] * dd;
475  a12 = mat[1] * dd;
476  a13 = mat[2] * dd;
477  a21 = mat[3] * dd;
478  a22 = mat[4] * dd;
479  a23 = mat[5] * dd;
480  a31 = mat[6] * dd;
481  a32 = mat[7] * dd;
482  a33 = mat[8] * dd;
483

492  valm = fabs(a31);
493  if ( valm > maxd ) maxd = valm;

499  P(X) = X^3 - trace X^2 + (somme des mineurs)X - det = 0 */
500  aa = a22*a33 - a23*a32;
501  bb = a23*a31 - a21*a33;
502  cc = a21*a32 - a31*a22;

584  w2[1] = a22 - lambda[2];
585  w3[2] = a33 - lambda[2];

611  else {

613  v[2][0] = vx3[0] * dd3;

615  v[2][2] = vx3[2] * dd3;
616  }
617  }
618  else {
619  if ( dd2 > dd3 ) {
620  dd2 = 1.0 / sqrt(dd2);
621  v[2][0] = vx2[0] * dd2;
622  v[2][1] = vx2[1] * dd2;
623  v[2][2] = vx2[2] * dd2;
624  }
625  else {
626  dd3 = 1.0 / sqrt(dd3);
627  v[2][0] = vx3[0] * dd3;
628  v[2][1] = vx3[1] * dd3;

631  }

635  dd2 = w2[0]*w2[0] + w2[1]*w2[1] + w2[2]*w2[2];
636  if ( dd1 > dd2 ) {
637  dd1 = 1.0 / sqrt(dd1);
638  v[0][0] = w1[0]*dd1;
639  v[0][1] = w1[1]*dd1;
640  v[0][2] = w1[2]*dd1;
641  }

644  v[0][0] = w2[0]*dd2;
645  v[0][1] = w2[1]*dd2;
646  v[0][2] = w2[2]*dd2;

651  v[1][1] = v[2][2]*v[0][0] - v[2][0]*v[0][2];
652  v[1][2] = v[2][0]*v[0][1] - v[2][1]*v[0][0];
653  dd1 = v[1][0]*v[1][0] + v[1][1]*v[1][1] + v[1][2]*v[1][2];
654  dd1 = 1.0 / sqrt(dd1);
655  v[1][0] *= dd1;
656  v[1][1] *= dd1;
657  v[1][2] *= dd1;
658  }
659 
660  lambda[0] *= maxm;
661  lambda[1] *= maxm;
662  lambda[2] *= maxm;
663 
664  /* check accuracy */
665  if ( getenv("MMG_EIGENV_DDEBUG") && symmat ) {
666  if ( !MMG5_check_accuracy ( mat, lambda, v, w1, w2, w3, maxm, n, symmat ) )
667  return 0;
668  }
669 
670  return n;

682 int MMG5_eigen2(double *mm,double *lambda,double vp[2][2]) {
683  double m[3],dd,a1,xn,ddeltb,rr1,rr2,ux,uy;
684 
685  /* normalize */
686  memcpy(m,mm,3*sizeof(double));
687  xn = fabs(m[0]);
688  if ( fabs(m[1]) > xn ) xn = fabs(m[1]);
689  if ( fabs(m[2]) > xn ) xn = fabs(m[2]);
690  if ( xn < MG_EIGENV_EPS10 ) {
691  lambda[0] = lambda[1] = 0.0;
692  vp[0][0] = 1.0;
693  vp[0][1] = 0.0;
694  vp[1][0] = 0.0;
695  vp[1][1] = 1.0;
696  return 1;
697  }
698  xn = 1.0 / xn;
699  m[0] *= xn;
700  m[1] *= xn;
701  m[2] *= xn;
702 
703  if ( egal(m[1],0.0) ) {
704  rr1 = m[0];
705  rr2 = m[2];
706  goto vect;
707  }
708 
709  /* eigenvalues of jacobian */
710  a1 = -(m[0] + m[2]);
711  ddeltb = a1*a1 - 4.0 * (m[0]*m[2] - m[1]*m[1]);
712 
713  if ( ddeltb < 0.0 ) {
714  fprintf(stderr,"\n ## Error: %s: Delta: %f\n",__func__,ddeltb);
715  ddeltb = 0.0;
716  }
717  ddeltb = sqrt(ddeltb);
718 
719  if ( fabs(a1) < MG_EIGENV_EPS6 ) {
720  rr1 = 0.5 * sqrt(ddeltb);
721  rr2 = -rr1;
722  }
723  else if ( a1 < 0.0 ) {
724  rr1 = 0.5 * (-a1 + ddeltb);
725  rr2 = (-m[1]*m[1] + m[0]*m[2]) / rr1;
726  }
727  else if ( a1 > 0.0 ) {
728  rr1 = 0.5 * (-a1 - ddeltb);
729  rr2 = (-m[1]*m[1] + m[0]*m[2]) / rr1;
730  }
731  else {
732  rr1 = 0.5 * ddeltb;
733  rr2 = -rr1;
734  }
735 
736 vect:
737  xn = 1.0 / xn;
738  lambda[0] = rr1 * xn;
739  lambda[1] = rr2 * xn;
740 
741  /* eigenvectors */
742  a1 = m[0] - rr1;
743  if ( fabs(a1)+fabs(m[1]) < MG_EIGENV_EPS6 ) {
744  if (fabs(lambda[1]) < fabs(lambda[0]) ) {
745  ux = 1.0;
746  uy = 0.0;
747  }
748  else {
749  ux = 0.0;
750  uy = 1.0;
751  }
752  }
753  else if ( fabs(a1) < fabs(m[1]) ) {
754  ux = 1.0;
755  uy = -a1 / m[1];
756  }
757  else if ( fabs(a1) > fabs(m[1]) ) {
758  ux = -m[1] / a1;
759  uy = 1.0;
760  }
761  else if ( fabs(lambda[1]) > fabs(lambda[0]) ) {
762  ux = 0.0;
763  uy = 1.0;
764  }
765  else {
766  ux = 1.0;
767  uy = 0.0;
768  }
769 
770  dd = sqrt(ux*ux + uy*uy);
771  dd = 1.0 / dd;
772  if ( fabs(lambda[0]) > fabs(lambda[1]) ) {
773  vp[0][0] = ux * dd;
774  vp[0][1] = uy * dd;
775  }
776  else {
777  vp[0][0] = uy * dd;
778  vp[0][1] = -ux * dd;
779  }
780 
781  /* orthogonal vector */
782  vp[1][0] = -vp[0][1];
783  vp[1][1] = vp[0][0];
784 
785  return 1;
786 }

797 inline int MMG5_eigensym(double m[3],double lambda[2],double vp[2][2]) {
798  double sqDelta,dd,trm,vnorm,maxm,a11,a12,a22;
799  static int ddebug = 0;
800 
801  lambda[0] = lambda[1] = 0.;
802  vp[0][0] = vp[0][1] = vp[1][0] = vp[1][1] = 0.;
803 
804  maxm = fabs(m[0]);
805  maxm = fabs( m[1] ) > maxm ? fabs ( m[1] ) : maxm;
806  maxm = fabs( m[2] ) > maxm ? fabs ( m[2] ) : maxm;
807 
808  if ( maxm < MG_EIGENV_EPS13 ) {
809  if ( ddebug ) printf(" ## Warning:%s: Quasi-null matrix.",__func__);
810  maxm = 1.;
811  }
812 
813  /* normalize matrix */
814  dd = 1.0 / maxm;
815  a11 = m[0] * dd;
816  a12 = m[1] * dd;
817  a22 = m[2] * dd;
818 
819  dd = a11 - a22;
820  trm = a11 + a22;
821  sqDelta = sqrt(dd*dd + 4.0*a12*a12);
822  lambda[0] = 0.5*(trm - sqDelta);
823 
824  /* Case when m = lambda[0]*I */
825  if ( sqDelta < MMG5_EPS ) {
826  lambda[0] *= maxm;
827  lambda[1] = lambda[0];
828  vp[0][0] = 1.0;
829  vp[0][1] = 0.0;
830 
831  vp[1][0] = 0.0;
832  vp[1][1] = 1.0;
833  return 2;
834  }
835  /* Remark: the computation of an independent basis of eigenvectors fail if the
836  * matrix is diagonal (we find twice the same eigenvector) */
837  vp[0][0] = a12;
838  vp[0][1] = (lambda[0] - a11);
839  vnorm = sqrt(vp[0][0]*vp[0][0] + vp[0][1]*vp[0][1]);
840 
841  if ( vnorm < MMG5_EPS ) {
842  vp[0][0] = (lambda[0] - a22);
843  vp[0][1] = a12;
844  vnorm = sqrt(vp[0][0]*vp[0][0] + vp[0][1]*vp[0][1]);
845  }
846  assert(vnorm > MMG5_EPSD);
847 
848  vnorm = 1.0/vnorm;
849  vp[0][0] *= vnorm;
850  vp[0][1] *= vnorm;
851 
852  vp[1][0] = -vp[0][1];
853  vp[1][1] = vp[0][0];
854 
855  lambda[1] = a11*vp[1][0]*vp[1][0] + 2.0*a12*vp[1][0]*vp[1][1]
856  + a22*vp[1][1]*vp[1][1];
857 
858  lambda[0] *= maxm;
859  lambda[1] *= maxm;
860 
861  /* Check orthogonality of eigenvectors. If they are not, we probably miss the
862  * dectection of a diagonal matrix. */
863  assert ( fabs(vp[0][0]*vp[1][0] + vp[0][1]*vp[1][1]) <= MG_EIGENV_EPS6 );
864 
865  return 1;
866 }
