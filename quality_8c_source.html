47 double MMG5_caltri33_ani(MMG5_pMesh mesh,MMG5_pSol met,MMG5_pTria pt) {
48  double anisurf,dd,abx,aby,abz,acx,acy,acz,bcx,bcy,bcz;
49  double *a,*b,*c,*ma,*mb,*mc,m[6],l0,l1,l2,rap;
50  int ia,ib,ic;
51  int8_t i;
52 
53  ia = pt->v[0];
54  ib = pt->v[1];
55  ic = pt->v[2];
56 
57  ma = &met->m[6*ia];
58  mb = &met->m[6*ib];
59  mc = &met->m[6*ic];
60 
61  /* 2*area */
62  anisurf = MMG5_surftri33_ani(mesh,pt,ma,mb,mc);
63  if ( anisurf <= MMG5_EPSD2 ) return 0.0;
64 
65  dd = 1.0 / 3.0;
66  for (i=0; i<6; i++)
67  m[i] = dd * (ma[i] + mb[i] + mc[i]);
68 
69  a = &mesh->point[ia].c[0];
70  b = &mesh->point[ib].c[0];
71  c = &mesh->point[ic].c[0];
72 
73  abx = b[0] - a[0];
74  aby = b[1] - a[1];
75  abz = b[2] - a[2];
76  acx = c[0] - a[0];
77  acy = c[1] - a[1];
78  acz = c[2] - a[2];
79  bcx = c[0] - b[0];
80  bcy = c[1] - b[1];
81  bcz = c[2] - b[2];
82 
83  /* length */
84  l0 = m[0]*abx*abx + m[3]*aby*aby + m[5]*abz*abz
85  + 2.0*(m[1]*abx*aby + m[2]*abx*abz + m[4]*aby*abz);
86 
87  l1 = m[0]*acx*acx + m[3]*acy*acy + m[5]*acz*acz
88  + 2.0*(m[1]*acx*acy + m[2]*acx*acz + m[4]*acy*acz);
89 
90  l2 = m[0]*bcx*bcx + m[3]*bcy*bcy + m[5]*bcz*bcz
91  + 2.0*(m[1]*bcx*bcy + m[2]*bcx*bcz + m[4]*bcy*bcz);
92 
93  rap = l0 + l1 + l2;
94 
95  /* quality = 2*area/length */
96  if ( rap > MMG5_EPSD2 ) {
97  return anisurf / rap;
98  }
99  else
100  return 0.0;
101 }

115 double MMG5_caltri_ani(MMG5_pMesh mesh,MMG5_pSol met,MMG5_pTria ptt) {
116  MMG5_pPoint p[3];
117  double rap,anisurf,l0,l1,l2,m[6],mm[6],rbasis[3][3];
118  double abx,aby,abz,acx,acy,acz,bcy,bcx,bcz;
119  int np[3],i,j;
120  int8_t i1,i2;
121 
122  for (i=0; i<3; i++) {
123  np[i] = ptt->v[i];
124  p[i] = &mesh->point[np[i]];
125  }
126 
127  /* Set metric tensors at vertices of tria iel */
128  for ( j=0; j<6; ++j) {
129  mm[j] = 0;
130  }
131 
132  for(i=0; i<3; i++) {
133 
134  if ( MG_SIN(p[i]->tag) || (MG_NOM & p[i]->tag) ) {
135  memcpy(&m[0],&met->m[6*np[i]],6*sizeof(double));
136  }
137  else if ( p[i]->tag & MG_GEO ) {
138  i1 = MMG5_inxt2[i];
139  i2 = MMG5_iprv2[i];
140  abx = 0.5*(p[i1]->c[0]+p[i2]->c[0]) - p[i]->c[0];
141  aby = 0.5*(p[i1]->c[1]+p[i2]->c[1]) - p[i]->c[1];
142  abz = 0.5*(p[i1]->c[2]+p[i2]->c[2]) - p[i]->c[2];
143  /* Note that rbasis is unused here */
144  if ( !MMG5_buildridmet(mesh,met,np[i],abx,aby,abz,&m[0],rbasis) ) {
145  return 0.0;
146  }
147  }
148  else {
149  memcpy(&m[0],&met->m[6*np[i]],6*sizeof(double));
150  }
151 
152  for ( j=0; j<6; ++j) {
153  mm[j] += MMG5_ATHIRD*m[j];
154  }
155  }
156 
157  anisurf = MMG5_surftri33_ani(mesh,ptt,mm,mm,mm);
158 
159  /* length */
160  abx = p[1]->c[0] - p[0]->c[0];
161  aby = p[1]->c[1] - p[0]->c[1];
162  abz = p[1]->c[2] - p[0]->c[2];
163  acx = p[2]->c[0] - p[0]->c[0];
164  acy = p[2]->c[1] - p[0]->c[1];
165  acz = p[2]->c[2] - p[0]->c[2];
166  bcx = p[2]->c[0] - p[1]->c[0];
167  bcy = p[2]->c[1] - p[1]->c[1];
168  bcz = p[2]->c[2] - p[1]->c[2];
169 
170 
171  l0 = mm[0]*abx*abx + mm[3]*aby*aby + mm[5]*abz*abz
172  + 2.0*(mm[1]*abx*aby + mm[2]*abx*abz + mm[4]*aby*abz);
173 
174  l1 = mm[0]*acx*acx + mm[3]*acy*acy + mm[5]*acz*acz
175  + 2.0*(mm[1]*acx*acy + mm[2]*acx*acz + mm[4]*acy*acz);
176 
177  l2 = mm[0]*bcx*bcx + mm[3]*bcy*bcy + mm[5]*bcz*bcz
178  + 2.0*(mm[1]*bcx*bcy + mm[2]*bcx*bcz + mm[4]*bcy*bcz);
179 
180  rap = l0 + l1 + l2;
181 
182  if ( rap < MMG5_EPSD2 ) return 0.0;
183 
184  /* quality = 2*area/length */
185  return (anisurf / rap);
186 }

198 inline double MMG5_caltri_iso(MMG5_pMesh mesh,MMG5_pSol met,MMG5_pTria ptt) {
199  double *a,*b,*c,cal,abx,aby,abz,acx,acy,acz,bcx,bcy,bcz,rap;
200 
201  a = &mesh->point[ptt->v[0]].c[0];
202  b = &mesh->point[ptt->v[1]].c[0];
203  c = &mesh->point[ptt->v[2]].c[0];
204 
205  /* area */
206  abx = b[0] - a[0];
207  aby = b[1] - a[1];
208  abz = b[2] - a[2];
209  acx = c[0] - a[0];
210  acy = c[1] - a[1];
211  acz = c[2] - a[2];
212  bcx = c[0] - b[0];
213  bcy = c[1] - b[1];
214  bcz = c[2] - b[2];
215 
216  cal = (aby*acz - abz*acy) * (aby*acz - abz*acy);
217  cal += (abz*acx - abx*acz) * (abz*acx - abx*acz);
218  cal += (abx*acy - aby*acx) * (abx*acy - aby*acx);
219 
220  if ( cal < MMG5_EPSD2 ) return 0.0;
221 
222  /* qual = 2.*surf / length */
223  rap = abx*abx + aby*aby + abz*abz;
224  rap += acx*acx + acy*acy + acz*acz;
225  rap += bcx*bcx + bcy*bcy + bcz*bcz;
226 
227  if ( rap < MMG5_EPSD2 ) return 0.0;
228 
229  return sqrt(cal) / rap;
230 }
