233 inline int MMG5_rotmatrix(double n[3],double r[3][3]) {
234  double aa,bb,ab,ll,l,cosalpha,sinalpha;
235 
236  aa = n[0]*n[0];
237  bb = n[1]*n[1];
238  ab = n[0]*n[1];
239  ll = aa+bb;
240  cosalpha = n[2];
241  sinalpha = sqrt(1.0- MG_MIN(1.0,cosalpha*cosalpha));
242 
243  /* No rotation needed in this case */
244  if ( ll < MMG5_EPS ) {
245  if ( n[2] > 0.0 ) {
246  r[0][0] = 1.0 ; r[0][1] = 0.0 ; r[0][2] = 0.0;
247  r[1][0] = 0.0 ; r[1][1] = 1.0 ; r[1][2] = 0.0;
248  r[2][0] = 0.0 ; r[2][1] = 0.0 ; r[2][2] = 1.0;
249  }
250  else {
251  r[0][0] = -1.0 ; r[0][1] = 0.0 ; r[0][2] = 0.0;
252  r[1][0] = 0.0 ; r[1][1] = 1.0 ; r[1][2] = 0.0;
253  r[2][0] = 0.0 ; r[2][1] = 0.0 ; r[2][2] = -1.0;
254  }
255  }
256  else {
257  l = sqrt(ll);
258 
259  r[0][0] = (aa*cosalpha + bb)/ll;
260  r[0][1] = ab*(cosalpha-1)/ll;
261  r[0][2] = -n[0]*sinalpha/l;
262  r[1][0] = r[0][1];
263  r[1][1] = (bb*cosalpha + aa)/ll;
264  r[1][2] = -n[1]*sinalpha/l;
265  r[2][0] = n[0]*sinalpha/l;
266  r[2][1] = n[1]*sinalpha/l;
267  r[2][2] = cosalpha;
268  }
269  return 1;
270 }

279 int MMG5_invmat(double *m,double *mi) {
280  double aa,bb,cc,det,vmax,maxx;
281  int k;
282 
283  /* check diagonal matrices */
284  vmax = fabs(m[1]);
285  maxx = fabs(m[2]);
286  if( maxx > vmax ) vmax = maxx;
287  maxx = fabs(m[4]);
288  if( maxx > vmax ) vmax = maxx;
289  if ( vmax < MMG5_EPS ) {
290  mi[0] = 1./m[0];
291  mi[3] = 1./m[3];
292  mi[5] = 1./m[5];
293  mi[1] = mi[2] = mi[4] = 0.0;
294  return 1;
295  }
296 
297  /* check ill-conditionned matrix */
298  vmax = fabs(m[0]);
299  for (k=1; k<6; k++) {
300  maxx = fabs(m[k]);
301  if ( maxx > vmax ) vmax = maxx;
302  }
303  if ( vmax == 0.0 ) return 0;
304 
305  /* compute sub-dets */
306  aa = m[3]*m[5] - m[4]*m[4];
307  bb = m[4]*m[2] - m[1]*m[5];
308  cc = m[1]*m[4] - m[2]*m[3];
309  det = m[0]*aa + m[1]*bb + m[2]*cc;
310  if ( fabs(det) < MMG5_EPSD2 ) return 0;
311  det = 1.0 / det;
312 
313  mi[0] = aa*det;
314  mi[1] = bb*det;
315  mi[2] = cc*det;
316  mi[3] = (m[0]*m[5] - m[2]*m[2])*det;
317  mi[4] = (m[1]*m[2] - m[0]*m[4])*det;
318  mi[5] = (m[0]*m[3] - m[1]*m[1])*det;
319 
320  return 1;
321 }

370 int MMG5_invmat33(double m[3][3],double mi[3][3]) {
371  double aa,bb,cc,det,vmax,maxx;
372  int k,l;
373 
374  /* check ill-conditionned matrix */
375  vmax = fabs(m[0][0]);
376  for (k=0; k<3; k++) {
377  for (l=0; l<3; l++) {
378  maxx = fabs(m[k][l]);
379  if ( maxx > vmax ) vmax = maxx;
380  }
381  }
382  if ( vmax == 0.0 ) return 0;
383 
384  /* check diagonal matrices */
385  /* lower */
386  vmax = fabs(m[1][0]);
387  maxx = fabs(m[2][0]);
388  if( maxx > vmax ) vmax = maxx;
389  maxx = fabs(m[2][1]);
390  if( maxx > vmax ) vmax = maxx;
391  /* upper */
392  maxx = fabs(m[0][1]);
393  if( maxx > vmax ) vmax = maxx;
394  maxx = fabs(m[0][2]);
395  if( maxx > vmax ) vmax = maxx;
396  maxx = fabs(m[1][2]);
397  if( maxx > vmax ) vmax = maxx;
398 
399  if ( vmax < MMG5_EPS ) {
400  mi[0][0] = 1./m[0][0];
401  mi[1][1] = 1./m[1][1];
402  mi[2][2] = 1./m[2][2];
403  mi[1][0] = mi[0][1] = mi[2][0] = mi[0][2] = mi[1][2] = mi[2][1] = 0.0;
404  return 1;
405  }
406 
407  /* compute sub-dets */
408  aa = m[1][1]*m[2][2] - m[2][1]*m[1][2];
409  bb = m[2][1]*m[0][2] - m[0][1]*m[2][2];
410  cc = m[0][1]*m[1][2] - m[1][1]*m[0][2];
411  det = m[0][0]*aa + m[1][0]*bb + m[2][0]*cc;
412  if ( fabs(det) < MMG5_EPSD ) return 0;
413  det = 1.0 / det;
414 
415  mi[0][0] = aa*det;
416  mi[0][1] = bb*det;
417  mi[0][2] = cc*det;
418  mi[1][0] = (m[2][0]*m[1][2] - m[1][0]*m[2][2])*det;
419  mi[1][1] = (m[0][0]*m[2][2] - m[2][0]*m[0][2])*det;
420  mi[1][2] = (m[1][0]*m[0][2] - m[0][0]*m[1][2])*det;
421  mi[2][0] = (m[1][0]*m[2][1] - m[2][0]*m[1][1])*det;
422  mi[2][1] = (m[2][0]*m[0][1] - m[0][0]*m[2][1])*det;
423  mi[2][2] = (m[0][0]*m[1][1] - m[1][0]*m[0][1])*det;
424 
425  /* Check results */
426 #ifndef NDEBUG
427  double res[3][3];
428 
429  res[0][0] = m[0][0] * mi[0][0] + m[0][1] * mi[1][0] + m[0][2] * mi[2][0];
430  res[0][1] = m[0][0] * mi[0][1] + m[0][1] * mi[1][1] + m[0][2] * mi[2][1];
431  res[0][2] = m[0][0] * mi[0][2] + m[0][1] * mi[1][2] + m[0][2] * mi[2][2];
432 
433  res[1][0] = m[1][0] * mi[0][0] + m[1][1] * mi[1][0] + m[1][2] * mi[2][0];
434  res[1][1] = m[1][0] * mi[0][1] + m[1][1] * mi[1][1] + m[1][2] * mi[2][1];
435  res[1][2] = m[1][0] * mi[0][2] + m[1][1] * mi[1][2] + m[1][2] * mi[2][2];
436 
437  res[2][0] = m[2][0] * mi[0][0] + m[2][1] * mi[1][0] + m[2][2] * mi[2][0];
438  res[2][1] = m[2][0] * mi[0][1] + m[2][1] * mi[1][1] + m[2][2] * mi[2][1];
439  res[2][2] = m[2][0] * mi[0][2] + m[2][1] * mi[1][2] + m[2][2] * mi[2][2];
440 
441 
442  assert ( ( fabs(res[0][0]-1.) < MMG5_EPS ) &&
443  ( fabs(res[1][1]-1.) < MMG5_EPS ) &&
444  ( fabs(res[2][2]-1.) < MMG5_EPS ) &&
445  ( fabs(res[0][1]) < MMG5_EPS ) && ( fabs(res[0][2]) < MMG5_EPS ) &&
446  ( fabs(res[1][2]) < MMG5_EPS ) &&
447  ( fabs(res[1][0]) < MMG5_EPS ) && ( fabs(res[2][0]) < MMG5_EPS ) &&
448  ( fabs(res[2][1]) < MMG5_EPS ) && "Matrix inversion" );
449 
450 #endif
451 
452  return 1;

464 inline int MMG5_sys33sym(double a[6], double b[3], double r[3]){
465  double ia[6],as[6],det,m;
466  int i;
467 
468  /* If possible : multiply matrix by a constant coefficient for stability
469  purpose */
470  m = fabs(a[0]);
471  m = MG_MIN ( fabs(a[3]), m );
472  m = MG_MIN ( fabs(a[5]), m );
473 
474  if ( m >= MMG5_EPSD ) {
475  /* Normalization */
476  m = 1.0/m;
477  }
478  else {
479  /* Unable to normalized */
480  m = 1.;
481  }
482 
483  for(i=0;i<6;i++){
484  as[i] = a[i]*m;
485  }
486 
487  det = as[0]*(as[3]*as[5]-as[4]*as[4]) - as[1]*(as[1]*as[5]-as[2]*as[4]) \
488  + as[2]*(as[1]*as[4]-as[2]*as[3]);
489 
490  if(fabs(det) < MMG5_EPSD)
491  return 0;
492 
493  det = 1.0/det;
494 
495  ia[0] = (as[3]*as[5]-as[4]*as[4]);
496  ia[1] = - (as[1]*as[5]-as[2]*as[4]);
497  ia[2] = (as[1]*as[4]-as[2]*as[3]);
498  ia[3] = (as[0]*as[5]-as[2]*as[2]);
499  ia[4] = -(as[0]*as[4]-as[2]*as[1]);
500  ia[5] = (as[0]*as[3]-as[1]*as[1]);
501 
502  r[0] = ia[0]*b[0] + ia[1]*b[1] + ia[2]*b[2];
503  r[1] = ia[1]*b[0] + ia[3]*b[1] + ia[4]*b[2];
504  r[2] = ia[2]*b[0] + ia[4]*b[1] + ia[5]*b[2];
505 
506  r[0] *= (det*m);
507  r[1] *= (det*m);
508  r[2] *= (det*m);
509 
510  return 1;
511 }

959  int (*delElt)(MMG5_pMesh,int) ) {
960  MMG5_pTria pt;

977  * not be adjacent to k anymore */
