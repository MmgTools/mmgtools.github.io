90 static inline
91 int MMGS_packMesh(MMG5_pMesh mesh,MMG5_pSol sol,MMG5_pSol met) {
92  MMG5_pTria pt,ptnew;
93  MMG5_pPoint ppt,pptnew;
94  int np,nc,nr, k,nt,nbl,imet,imetnew,i,na,jel;
95  int iadr,iadrnew,iadrv,*adjav,*adja,*adjanew,voy;
96  int8_t i1,i2;
97 
98  /* Remove non wanted subdomains if needed */
99  MMGS_keep_only1Subdomain ( mesh, mesh->info.nsd );
100 
101  /* compact vertices */
102  np = nc = nr = 0;
103  for (k=1; k<=mesh->np; k++) {
104  ppt = &mesh->point[k];
105  if ( !MG_VOK(ppt) ) continue;
106  ppt->tmp = ++np;
107  if ( ppt->tag & MG_CRN ) nc++;
108  ppt->ref = abs(ppt->ref);
109  }
110 
111  /* compact triangles */
112  nt = 0;
113  na = 0;
114  nbl = 1;
115  for (k=1; k<=mesh->nt; k++) {
116  pt = &mesh->tria[k];
117  if ( !MG_EOK(pt) ) continue;
118 
119  pt->v[0] = mesh->point[pt->v[0]].tmp;
120  pt->v[1] = mesh->point[pt->v[1]].tmp;
121  pt->v[2] = mesh->point[pt->v[2]].tmp;
122  nt++;
123  if ( k!=nbl ) {
124  ptnew = &mesh->tria[nbl];
125  memcpy(ptnew,pt,sizeof(MMG5_Tria));
126 
127  iadr = 3*(k-1) + 1;
128  adja = &mesh->adja[iadr];
129  iadrnew = 3*(nbl-1) + 1;
130  adjanew = &mesh->adja[iadrnew];
131  for(i=0 ; i<3 ; i++) {
132  adjanew[i] = adja[i];
133  if(!adja[i]) continue;
134  iadrv = 3*(adja[i]/3-1) +1;
135  adjav = &mesh->adja[iadrv];
136  voy = i;
137  adjav[adja[i]%3] = 3*nbl + voy;
138  }
139  }
140  nbl++;
141 
142  /* Count the edges */
143  for(i=0 ; i<3 ; i++) {
144  if ( !MG_EDG(pt->tag[i]) ) continue;
145 
146  adja = &mesh->adja[3*(k-1)+1];
147  jel = adja[i] / 3;
148 
149  assert ( jel != k );
150  if ( jel ) {
151  /* If we have an adjacent, treat the edge either from the tria with
152  * higher ref or, if both tria have the same references, from the tria
153  * with higher index */
154  if ( mesh->tria[jel].ref < mesh->tria[k].ref ) {
155  continue;
156  }
157  else if ( (mesh->tria[jel].ref == mesh->tria[k].ref) && (k < jel) ) {
158  continue;
159  }
160  }
161  ++na;
162  }
163 
164  }
165  mesh->nt = nt;
166 
167  /* compact solutions (metric, level-set, displacement...) */
168  if ( met && met->m ) {
169  nbl = 1;
170  for (k=1; k<=mesh->np; k++) {
171  ppt = &mesh->point[k];
172  if ( !MG_VOK(ppt) ) continue;
173  imet = k * met->size;
174  imetnew = nbl * met->size;
175 
176  for (i=0; i<met->size; i++)
177  met->m[imetnew + i] = met->m[imet + i];
178  ++nbl;
179  }
180  }
181 
182  if ( sol && sol->m ) {
183  nbl = 1;
184  for (k=1; k<=mesh->np; k++) {
185  ppt = &mesh->point[k];
186  if ( !MG_VOK(ppt) ) continue;
187  imet = k * sol->size;
188  imetnew = nbl * sol->size;
189 
190  for (i=0; i<sol->size; i++)
191  sol->m[imetnew + i] = sol->m[imet + i];
192  ++nbl;
193  }
194  }
195 
196  /* compact vertices */
197  np = 0;
198  nbl = 1;
199  for (k=1; k<=mesh->np; k++) {
200  ppt = &mesh->point[k];
201  if ( !MG_VOK(ppt) ) continue;
202  np++;
203  if ( k!=nbl ) {
204  pptnew = &mesh->point[nbl];
205  memmove(pptnew,ppt,sizeof(MMG5_Point));
206  memset(ppt,0,sizeof(MMG5_Point));
207  ppt->tag = MG_NUL;
208  }
209  nbl++;
210  }
211  mesh->np = np;
212 
213  if ( met && met->m )
214  met->np = np;
215 
216  if ( sol && sol->m )
217  sol->np = np;
218 
219  /* memory alloc */
220  mesh->na = 0;
221  if ( mesh->edge ) {
222  MMG5_DEL_MEM(mesh,mesh->edge);
223  MMG5_SAFE_FREE(mesh->edge);
224  }
225 
226  if ( na ) {
227  MMG5_ADD_MEM(mesh,(na+1)*sizeof(MMG5_Edge),"final edges",
228  na = 0;
229  printf(" ## Warning: uncomplete mesh\n")
230  );
231  }
232 
233  if ( na ) {
234  MMG5_SAFE_CALLOC(mesh->edge,na+1,MMG5_Edge,return 0);
235  for (k=1; k<=mesh->nt; k++) {
236  pt = &mesh->tria[k];
237  if ( MG_EOK(pt) ) {
238  for (i=0; i<3; i++) {
239  if ( !MG_EDG(pt->tag[i]) ) continue;
240 
241  adja = &mesh->adja[3*(k-1)+1];
242  jel = adja[i] / 3;
243  if ( jel ) {
244  /* If we have an adjacent, treat the edge either from the tria with
245  * higher ref or, if both tria have the same references, from the tria
246  * with higher index */
247  if ( mesh->tria[jel].ref < mesh->tria[k].ref ) {
248  continue;
249  }
250  else if ( (mesh->tria[jel].ref == mesh->tria[k].ref) && (k < jel) ) {
251  continue;
252  }
253  }
254 
255  i1 = MMG5_inxt2[i];
256  i2 = MMG5_inxt2[i1];
257  mesh->na++;
258  mesh->edge[mesh->na].a = mesh->point[pt->v[i1]].tmp;
259  mesh->edge[mesh->na].b = mesh->point[pt->v[i2]].tmp;
260  mesh->edge[mesh->na].ref = pt->edg[i];
261  mesh->edge[mesh->na].tag |= pt->tag[i];
262  if ( pt->tag[i] & MG_GEO ) nr++;
263  }
264  }
265  }
266  }
267 
268  for(k=1 ; k<=mesh->np ; k++)
269  mesh->point[k].tmp = 0;
270 
271  mesh->npnil = mesh->np + 1;
272  for(k=mesh->npnil; k<mesh->npmax-1; k++)
273  mesh->point[k].tmp = k+1;
274 
275  mesh->nenil = mesh->nt + 1;
276  for(k=mesh->nenil; k<mesh->ntmax-1; k++)
277  mesh->tria[k].v[2] = k+1;
278 
279  /* to could save the mesh, the adjacency have to be correct */
280  if ( mesh->info.ddebug && (!MMG5_chkmsh(mesh,1,1) ) ) {
281  fprintf(stderr,"\n ## Warning: %s: invalid mesh.\n",__func__);
282  return 0;
283  }
284 
285  if ( mesh->info.imprim > 0 ) {
286  fprintf(stdout," NUMBER OF VERTICES %8d CORNERS %8d\n",mesh->np,nc);
287  fprintf(stdout," NUMBER OF TRIANGLES %8d\n",mesh->nt);
288 
289  if ( mesh->na )
290  fprintf(stdout," NUMBER OF EDGES %8d RIDGES %8d\n",mesh->na,nr);
291  }
292  return 1;
293 }
