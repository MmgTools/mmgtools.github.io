55 static void MMG5_InvMat_set(MMG5_pMesh mesh,MMG5_pInvMat pim,int k) {
56  MMG5_pMat pm;
57  int key;
58 
59  /* Get material */
60  pm = &mesh->info.mat[k];
61 
63  key = MMG5_InvMat_key(pim,pm->ref);
64  pim->lookup[key] = 4*(k+1)+pm->dospl;
65 
73  if( pm->dospl ) {
74  key = MMG5_InvMat_key(pim,pm->rin);
75  pim->lookup[key] = 4*(k+1)+MG_MINUS;
76  key = MMG5_InvMat_key(pim,pm->rex);
77  pim->lookup[key] = 4*(k+1)+MG_PLUS;
78  }

255 int MMG5_MultiMat_init(MMG5_pMesh mesh) {
256  MMG5_pMat pm;
257  MMG5_pInvMat pim;
258  int k;
259  int refmax,refmin;
260 
261  /* Nothing to do if no multi-material option */
262  if( !mesh->info.nmat ) return 1;
263 
264  /* Error if all the materials have not been set */
265  if( mesh->info.nmati < mesh->info.nmat ) {
266  fprintf(stderr,"\n ## Error: %s: Only %d materials out of %d have been set.\n",
267  __func__,mesh->info.nmati,mesh->info.nmat);
268  return 0;
269  }
270 
271  /* Get pointer to the structure */
272  pim = &mesh->info.invmat;
273 
274  /* Initialize the max and min reference */
275  refmax = 0;
276  refmin = INT_MAX;
277 
278  /* Look for the max/min reference */
279  for( k = 0; k < mesh->info.nmat; k++ ) {
280  pm = &mesh->info.mat[k];
281  /* Update max and min val for original ref */
282  if( pm->ref > refmax ) refmax = pm->ref;
283  if( pm->ref < refmin ) refmin = pm->ref;
284  if( !pm->dospl ) continue;
285  /* Update max and min val with interior ref */
286  if( pm->rin > refmax ) refmax = pm->rin;
287  if( pm->rin < refmin ) refmin = pm->rin;
288  /* Update max and min val with exterior ref */
289  if( pm->rex > refmax ) refmax = pm->rex;
290  if( pm->rex < refmin ) refmin = pm->rex;
291  }
292 
293  /* Get span of the lookup table */
294  pim->offset = refmin;
295  pim->size = refmax - refmin + 1;
296  assert( pim->size > 0 );
297 
298  /* Allocate lookup table */
299  MMG5_ADD_MEM(mesh,pim->size*sizeof(int),"materials lookup table",return 0);
300  MMG5_SAFE_CALLOC(pim->lookup,pim->size,int,return 0);
301 
302  /* Fill lookup table */
303  for( k = 0; k < mesh->info.nmat; k++ ) {
304  MMG5_InvMat_set(mesh,pim,k);
305  }
306 
307 // MMG5_InvMat_print(mesh,pim);
308  return 1;
309 }

392

394  else return 0;

396  } else {
397  /* Single material, check references directly */
398  if( ( ref0 == MG_MINUS && ref1 == MG_PLUS ) ||

400  else return 0;
