464 int MMGS_Get_numberOfNonBdyEdges(MMG5_pMesh mesh, int* nb_edges) {
465  MMG5_pTria pt,pt1;
466  MMG5_pEdge ped;
467  int *adja,k,i,j,i1,i2,iel;
468 
469  *nb_edges = 0;
470  if ( mesh->tria ) {
471  /* Create the triangle adjacency if needed */
472  if ( !mesh->adja ) {
473  if ( !MMGS_hashTria(mesh) ) {
474  fprintf(stderr,"\n ## Error: %s: unable to create "
475  "adjacency table.\n",__func__);
476  return 0;
477  }
478  }
479 
480  /* Count the number of non boundary edges */
481  for ( k=1; k<=mesh->nt; k++ ) {
482  pt = &mesh->tria[k];
483  if ( !MG_EOK(pt) ) continue;
484 
485  adja = &mesh->adja[3*(k-1)+1];
486 
487  for ( i=0; i<3; i++ ) {
488  /* Do not treat boundary edges */
489  if ( MG_EDG(pt->tag[i]) ) continue;
490 
491  iel = adja[i] / 3;
492  assert ( iel != k );
493 
494  pt1 = &mesh->tria[iel];
495 
496  if ( (!iel) || (pt->ref != pt1->ref) ) {
497  /* Do not treat boundary edges */
498  continue;
499  }
500  if ( k < iel ) {
501  /* Treat edge from the triangle with lowest index */
502  ++(*nb_edges);
503  }
504  }
505  }
506 
507  /* Append the non boundary edges to the boundary edges array */
508  if ( mesh->na ) {
509  MMG5_ADD_MEM(mesh,(*nb_edges)*sizeof(MMG5_Edge),"non boundary edges",
510  printf(" Exit program.\n");
511  return 0);
512  MMG5_SAFE_RECALLOC(mesh->edge,(mesh->na+1),(mesh->na+(*nb_edges)+1),
513  MMG5_Edge,"non bdy edges arrray",return 0);
514  }
515  else {
516  MMG5_ADD_MEM(mesh,((*nb_edges)+1)*sizeof(MMG5_Edge),"non boundary edges",
517  printf(" Exit program.\n");
518  return 0);
519  MMG5_SAFE_RECALLOC(mesh->edge,0,(*nb_edges)+1,
520  MMG5_Edge,"non bdy edges arrray",return 0);
521  }
522 
523  j = mesh->na+1;
524  for ( k=1; k<=mesh->nt; k++ ) {
525  pt = &mesh->tria[k];
526  if ( !MG_EOK(pt) ) continue;
527 
528  adja = &mesh->adja[3*(k-1)+1];
529 
530  for ( i=0; i<3; i++ ) {
531  /* Do not treat boundary edges */
532  if ( MG_EDG(pt->tag[i]) ) continue;
533 
534  i1 = MMG5_inxt2[i];
535  i2 = MMG5_iprv2[i];
536  iel = adja[i] / 3;
537  assert ( iel != k );
538 
539  pt1 = &mesh->tria[iel];
540 
541  if ( (!iel) || (pt->ref != pt1->ref) ) {
542  /* Do not treat boundary edges */
543  continue;
544  }
545  if ( k < iel ) {
546  /* Treat edge from the triangle with lowest index */
547  ped = &mesh->edge[j++];
548  assert ( ped );
549  ped->a = pt->v[i1];
550  ped->b = pt->v[i2];
551  ped->ref = pt->edg[i];
552  }
553  }
554  }
555  }
556  return 1;
557 }

615 int MMGS_Get_adjaVerticesFast(MMG5_pMesh mesh, int ip,int start, int lispoi[MMGS_LMAX])
616 {
617  MMG5_pTria pt;
618  int k,prevk,nbpoi,iploc,i,i1,i2,*adja;
619 
620  pt = &mesh->tria[start];
621 
622  for ( iploc=0; iploc<3; ++iploc ) {
623  if ( pt->v[iploc] == ip ) break;
624  }
625 
626  assert(iploc!=3);
627 
628  k = start;
629  i = iploc;
630  nbpoi = 0;
631  do {
632  if ( nbpoi == MMGS_LMAX ) {
633  fprintf(stderr,"\n ## Warning: %s: unable to compute adjacent"
634  " vertices of the vertex %d:\nthe ball of point contain too many"
635  " elements.\n",__func__,ip);
636  return 0;
637  }
638  i1 = MMG5_inxt2[i];
639  lispoi[nbpoi] = mesh->tria[k].v[i1];
640  ++nbpoi;
641

643  prevk = k;
644  k = adja[i1] / 3;

649

662

689 int MMGS_doSol(MMG5_pMesh mesh,MMG5_pSol met) {
690  MMG5_pTria ptt;
691  MMG5_pPoint p1,p2;
692  double ux,uy,uz,dd;
693  int i,k,iadr,ipa,ipb,type;
694  int *mark;
695 
696  MMG5_SAFE_CALLOC(mark,mesh->np+1,int,return 0);
697 
698  /* Memory alloc */
699  if ( met->size==1 ) type=1;
700  else if ( met->size==6 ) type = 3;
701  else {
702  fprintf(stderr,"\n ## Error: %s: unexpected size of metric: %d.\n",
703  __func__,met->size);
704  return 0;
705  }
706 
707  if ( !MMGS_Set_solSize(mesh,met,MMG5_Vertex,mesh->np,type) )
708  return 0;
709 
710  /* edges */
711  dd = 0.;
712  for (k=1; k<=mesh->nt; k++) {
713  ptt = &mesh->tria[k];
714  if ( !MG_EOK(ptt) ) continue;
715 
716  if ( met->size == 1 ) {
717  for (i=0; i<3; i++) {
718  ipa = ptt->v[i];
719  ipb = ptt->v[MMG5_inxt2[i]];
720  p1 = &mesh->point[ipa];
721  p2 = &mesh->point[ipb];
722 
723  ux = p1->c[0] - p2->c[0];
724  uy = p1->c[1] - p2->c[1];
725  uz = p1->c[2] - p2->c[2];
726  dd = sqrt(ux*ux + uy*uy + uz*uz);
727 
728  met->m[ipa] += dd;
729  mark[ipa]++;
730  met->m[ipb] += dd;
731  mark[ipb]++;
732  }
733  }
734  else if ( met->size == 6 ) {
735  for (i=0; i<3; i++) {
736  ipa = ptt->v[i];
737  ipb = ptt->v[MMG5_inxt2[i]];
738  p1 = &mesh->point[ipa];
739  p2 = &mesh->point[ipb];
740 
741  ux = p1->c[0] - p2->c[0];
742  uy = p1->c[1] - p2->c[1];
743  uz = p1->c[2] - p2->c[2];
744  dd = sqrt(ux*ux + uy*uy + uz*uz);
745 
746  iadr = 6*ipa;
747  met->m[iadr] += dd;
748  mark[ipa]++;
749 
750  iadr = 6*ipb;
751  met->m[iadr] += dd;
752  mark[ipb]++;
753  }
754  }
755  else {
756  MMG5_SAFE_FREE(mark);
757  return 0;
758  }
759  }
760 
761  /* if hmax is not specified, compute it from the metric */
762  if ( mesh->info.hmax < 0. ) {
763  if ( met->size == 1 ) {
764  dd = 0.;
765  for (k=1; k<=mesh->np; k++) {
766  if ( !mark[k] ) continue;
767  dd = MG_MAX(dd,met->m[k]);
768  }
769  assert ( dd );
770  }
771  else if ( met->size == 6 ) {
772  dd = FLT_MAX;
773  for (k=1; k<=mesh->np; k++) {
774  if ( !mark[k] ) continue;
775  iadr = 6*k;
776  dd = MG_MIN(dd,met->m[iadr]);
777  }
778  assert ( dd < FLT_MAX );
779  dd = 1./sqrt(dd);
780  }
781  mesh->info.hmax = 10.*dd;
782  }
783 
784  /* vertex size */
785  if ( met->size == 1 ) {
786  for (k=1; k<=mesh->np; k++) {
787  if ( !mark[k] ) {
788  met->m[k] = mesh->info.hmax;
789  continue;
790  }
791  else
792  met->m[k] = met->m[k] / (double)mark[k];
793  }
794  }
795  else if ( met->size == 6 ) {
796  for (k=1; k<=mesh->np; k++) {
797  iadr = 6*k;
798  if ( !mark[k] ) {
799  met->m[iadr] = 1./(mesh->info.hmax*mesh->info.hmax);
800  met->m[iadr+3] = met->m[iadr];
801  met->m[iadr+5] = met->m[iadr];
802  continue;
803  }
804  else {
805  met->m[iadr] = (double)mark[k]*(double)mark[k]/(met->m[iadr]*met->m[iadr]);
806  met->m[iadr+3] = met->m[iadr];
807  met->m[iadr+5] = met->m[iadr];
808  }
809  }
810  }
811 
812  MMG5_SAFE_FREE(mark);
813  return 1;
814 }
