112  hausdloc = 0;
113  if ( mesh->info.npar ) {

116  iel = list[k]/3;
117  pt = &mesh->tria[iel];
118  assert ( MG_EOK(pt) );
119

121  for (l=0; l<mesh->info.npar; l++) {

123  if ( ppa->elt == MMG5_Triangle && ppa->ref == pt->ref ) {
124  if ( !isloc ) {
125  hmin = ppa->hmin;
126  hmax = ppa->hmax;
127  isloc = 1;
128  }
129  else {
130  hmin = MG_MAX ( hmin, ppa->hmin );
131  hmax = MG_MIN ( hmax, ppa->hmax );
132  }
133  break;
134  }
135  }

137  for ( i=0; i<3; i++ ) {
138  if ( !MG_EDG(pt->tag[i]) ) continue;
139

142  ppa = &mesh->info.par[l];

144  if ( !hausdloc ) {
145  hausd = ppa->hausd;
146  hausdloc = 1;
147  }
148  else {

150  }
151  if ( !isloc ) {
152  hmax = ppa->hmax;
153  hmin = ppa->hmin;
154  }
155  else {
156  hmin = MG_MAX ( hmin, ppa->hmin );
157  hmax = MG_MIN ( hmax, ppa->hmax );
158  }
159  break;
160  }
161  }
162  }
163  }
164  }

168  ppa = &mesh->info.par[l];
169  if ( ppa->elt == MMG5_Vertex && ppa->ref == p0->ref ) {
170  if ( !isloc ) {
171  hmin = ppa->hmin;
172  hmax = ppa->hmax;
173  isloc = 1;
174  }
175  else {
176  hmin = MG_MAX ( hmin, ppa->hmin );
177  hmax = MG_MIN ( hmax, ppa->hmax );

179  break;
180  }
181  }
182  }

185  if ( !mmgWarn2 ) {

187  fprintf(stderr,"\n ## Warning: %s: Non compatible local parameters:\n"
188  " hmin (%.15lg) > hmax (%.15lg).\nhmax ignored.",__func__,hmin,hmax);

200 
201  i0 = list[l] % 3;

203  i2 = MMG5_iprv2[i0];

209  }

211  if ( ip2 == 0 ) {
212  ip2 = pt->v[i2];

216  if ( !mmgWarn0 ) {

222  }
223  }
224  }
225 
226  if ( MG_EDG(pt->tag[i2]) ) {

240  " intersection of 3 edges. abort.\n",__func__);

249  if ( ip1 == 0 || ip2 == 0 ) {

253  "at the intersection of 2 edges. abort.\n",__func__);

259 
260  /* Curvature of both boundary edges meeting at ip */
261  for (j=0; j<2; j++) {
262  iel = it[j] / 3;
263  pt = &mesh->tria[iel];
264  i0 = it[j] % 3;
265  i1 = MMG5_inxt2[i0];
266  i2 = MMG5_iprv2[i0];
267  ip1 = pt->v[i1];
268  ip2 = pt->v[i2];

271  p2 = &mesh->point[ip2];
272 
273  ux = p2->c[0] - p1->c[0];
274  uy = p2->c[1] - p1->c[1];
275  ll = ux*ux + uy*uy;
276  if ( ll < MMG5_EPSD ) continue;
277  li = 1.0 / sqrt(ll);
278 
279  /* Tangent vector at p1 */
280  if ( (MG_CRN & p1->tag) || p1->tag & MG_NOM ) {
281  t1[0] = li*ux;
282  t1[1] = li*uy;
283  }
284  else {
285  t1[0] = p1->n[1];
286  t1[1] = -p1->n[0];
287  }
288 
289  /* Tangent vector at p2 */
290  if ( (MG_CRN & p2->tag) || p2->tag & MG_NOM ) {
291  t2[0] = li*ux;
292  t2[1] = li*uy;
293  }
294  else {
295  t2[0] = p2->n[1];
296  t2[1] = -p2->n[0];
297  }
298 
299  /* Calculation of the two Bezier coefficients along the boundary curve */
300  ps1 = ux*t1[0] + uy*t1[1];
301  b1[0] = p1->c[0] + MMG5_ATHIRD*ps1*t1[0];
302  b1[1] = p1->c[1] + MMG5_ATHIRD*ps1*t1[1];
303 
304  ps2 = ux*t2[0] + uy*t2[1];
305  b2[0] = p2->c[0] - MMG5_ATHIRD*ps2*t2[0];
306  b2[1] = p2->c[1] - MMG5_ATHIRD*ps2*t2[1];
307 
308  ps1 *= ps1;
309  ps2 *= ps2;
310 
311  if ( ps1 < MMG5_EPSD || ps2 < MMG5_EPSD ) continue;
312 
313  /* \gamma^{\prime\prime}(0); \gamma^\prime(0) = ps*t1 by construction */
314  gpp1[0] = 6.0*(p1->c[0] - 2.0*b1[0] + b2[0]);
315  gpp1[1] = 6.0*(p1->c[1] - 2.0*b1[1] + b2[1]);
316 
317  /* Vector product gpp1 ^ t1 */
318  pv = gpp1[0]*t1[1] - gpp1[1]*t1[0];
319  M1 = fabs(pv)/ps1;
320 
321  /* \gamma^{\prime\prime}(1); \gamma^\prime(1) = -ps*t2 by construction */
322  gpp2[0] = 6.0*(p2->c[0] - 2.0*b2[0] + b1[0]);
323  gpp2[1] = 6.0*(p2->c[1] - 2.0*b2[1] + b1[1]);
324 
325  /* Vector product gpp2 ^ t2 */
326  pv = gpp2[0]*t2[1] - gpp2[1]*t2[0];
327  M2 = fabs(pv)/ps2;
328 
329  M1 = MG_MAX(M1,M2);
330  if ( M1 < MMG5_EPSD) continue;
331  else {
332  ltmp = 8.0*hausd / M1;
333  lm = MG_MAX(sqhmin,MG_MIN(ltmp,lm));
334  }
335  }
336 
337  /* Expression of the metric tensor diag(lm,hmax) (in the (tau, n) basis) in the canonical basis */
338  n = &p0->n[0];
339  sqhmax = 1.0 / sqhmax;
340  lm = 1.0 / lm;
341 
342  m[0] = lm*n[1]*n[1] + sqhmax*n[0]*n[0];
343  m[1] = n[0]*n[1]*(sqhmax-lm);
344  m[2] = lm*n[0]*n[0] + sqhmax*n[1]*n[1];
345 
346  p0->flag = 2;
347 
348  return 1;
349 }

497 void MMG2D_gradEigenv(double dm[2],double dn[2],double difsiz,int8_t dir,int8_t *ier) {
498  double hm,hn;
499 
500  /* Gradation of sizes = 1/sqrt(eigenv of the tensors) in the first direction */
501  hm = 1.0 / sqrt(dm[dir]);
502  hn = 1.0 / sqrt(dn[dir]);
503 
504  if ( hn > hm + difsiz + MMG5_EPSOK ) {
505  hn = hm+difsiz;
506  dn[dir] = 1.0 / (hn*hn);
507  (*ier) = (*ier) | 2;
508  }
509  else if ( hm > hn + difsiz + MMG5_EPSOK ) {
510  hm = hn+difsiz;
511  dm[dir] = 1.0 / (hm*hm);
512  (*ier) = (*ier) | 1;
513  }
514 }

531 int MMG2D_updatemet_ani(double *m,double *n,double dm[2],double dn[2],
532  double vp[2][2],int8_t ier ) {
533  double det,ip[4];
534 
535  det = vp[0][0]*vp[1][1] - vp[0][1]*vp[1][0];
536  if ( fabs(det) < MMG5_EPS ) return 0;
537  det = 1.0 / det;
538 
539  ip[0] = vp[1][1]*det;
540  ip[1] = -vp[1][0]*det;
541  ip[2] = -vp[0][1]*det;
542  ip[3] = vp[0][0]*det;
543 
544  if ( ier & 1 ) {
545  m[0] = dm[0]*ip[0]*ip[0] + dm[1]*ip[2]*ip[2];
546  m[1] = dm[0]*ip[0]*ip[1] + dm[1]*ip[2]*ip[3];
547  m[2] = dm[0]*ip[1]*ip[1] + dm[1]*ip[3]*ip[3];
548  }
549  if ( ier & 2 ) {
550  n[0] = dn[0]*ip[0]*ip[0] + dn[1]*ip[2]*ip[2];
551  n[1] = dn[0]*ip[0]*ip[1] + dn[1]*ip[2]*ip[3];
552  n[2] = dn[0]*ip[1]*ip[1] + dn[1]*ip[3]*ip[3];
553  }
554  return 1;
555 }

574 int MMG2D_grad2met_ani(MMG5_pMesh mesh,MMG5_pSol met,MMG5_pTria pt,int np1,int np2) {
575  MMG5_pPoint p1,p2;
576  double dm[2],dn[2];
577  double vp[2][2],*m,*n,ll,difsiz;
578  int8_t ier;
579 
580  ier = 0;
581 
582  p1 = &mesh->point[np1];
583  p2 = &mesh->point[np2];
584 
585  /* Maximum allowed difference between the prescribed sizes in p1 and p2 */
586  ll = (p2->c[0]-p1->c[0])*(p2->c[0]-p1->c[0])
587  + (p2->c[1]-p1->c[1])*(p2->c[1]-p1->c[1]);
588  ll = sqrt(ll);
589 
590  difsiz = ll*mesh->info.hgrad;
591 
592  m = &met->m[met->size*np1];
593  n = &met->m[met->size*np2];
594 
595  /* Simultaneous reduction of m1 and m2 */
596  if ( !MMG5_simred(mesh,m,n,dm,dn,vp) ) {
597  return 0;
598  }
599 
600  /* Gradation of sizes = 1/sqrt(eigenv of the tensors) in the first direction */
601  MMG2D_gradEigenv(dm,dn,difsiz,0,&ier);
602 
603  /* Gradation of sizes = 1/sqrt(eigenv of the tensors) in the second direction */
604  MMG2D_gradEigenv(dm,dn,difsiz,1,&ier);
605 
606  if ( !ier ) {
607  return 0;
608  }
609 
610  /* Update of the metrics = tP^-1 diag(d0,d1)P^-1, P = (vp[0], vp[1]) stored in
611  * columns */
612  if ( !MMG2D_updatemet_ani(m,n,dm,dn,vp,ier ) ) {
613  return 0;
614  }
615 
616  return ier;
