39 
41 ! if the header file is in the "include" directory

47  mmg5_data_ptr_t :: mmgmesh

52  WRITE(*,*) " -- TEST MMGSLIB"

54  argc = command_argument_count();
55  CALL get_command_argument(0, exec_name)
56 
57  IF ( argc /=3 ) THEN
58  print*," Usage: ",trim(exec_name)," meshfile lsfile meshout"
59  CALL exit(1);

61 
62  ! Name and path of the mesh file
63  CALL get_command_argument(1, inname)
64  CALL get_command_argument(2, lsname)
65  CALL get_command_argument(3, outname)
66 
68  ! 1) Initialisation of mesh and sol structures
69  ! args of InitMesh:
70  ! MMG5_ARG_start: we start to give the args of a variadic func
71  ! MMG5_ARG_ppMesh: next arg will be a pointer over a MMG5_pMesh
72  ! mmgMesh: your MMG5_pMesh (that store your mesh)
73  ! MMG5_ARG_ppLs: next arg will be a pointer over a MMG5_pSol storing a level-set
74  ! mmgLs: pointer toward your MMG5_pSol (that store your level-set)
75  ! MMG5_ARG_ppMet: next arg will be a pointer over a MMG5_pSol that will
76  ! store the input metric
77  ! mmgMet: pointer toward your MMG5_pSol (that will store the input metric)
78  mmgmesh = 0
79  mmgls = 0
80  mmgmet = 0
81 
82  CALL mmgs_init_mesh(mmg5_arg_start, &
83  mmg5_arg_ppmesh,mmgmesh,mmg5_arg_ppls,mmgls, &
84  mmg5_arg_ppmet,mmgmet, &
85  mmg5_arg_end)
86 
87  !!------------------- Level set discretization option ---------------------
88  ! Ask for level set discretization: note that it is important to do this step
89  ! here because in iso mode, some filters are applied at mesh loading
90  CALL mmgs_set_iparameter(mmgmesh,mmgls,mmgs_iparam_iso, 1,ier)
91  IF ( ier == 0 ) CALL exit(101)
92 
96 
98  CALL mmgs_loadmesh(mmgmesh,trim(adjustl(inname)),&
99  len(trim(adjustl(inname))),ier)
100  IF ( ier == 0 ) CALL exit(102)
101 
103  ! Two solutions: just use the MMGS_loadSol function that will read a .sol(b)
104  ! file formatted or manually set your level-set using the MMGS_Set* functions
105 
107  CALL mmgs_loadsol(mmgmesh,mmgls,trim(adjustl(lsname)),&
108  len(trim(adjustl(lsname))),ier)
109  IF ( ier /= 1 ) THEN
110  CALL exit(104)
111  ENDIF
112 
113  ! ------------------- Give the Metric to Mmg ------------------------------
114  ! Manually for example
115  ! a) give info for the metric: the metric is applied on vertex
116  ! entities, number of vertices np is recoverd using get_meshSize and the sol
117  ! is tensorial
118  CALL mmgs_get_meshsize(mmgmesh,np,%val(0),%val(0),ier)
119  IF ( ier /= 1 ) THEN
120  CALL exit(204)
121  ENDIF
122 
123 
124  CALL mmgs_set_solsize(mmgmesh,mmgmet,mmg5_vertex,np,mmg5_tensor,ier)
125  IF ( ier /= 1 ) THEN
126  CALL exit(304)
127  ENDIF
128 
129  ! b) give metric values and positions
130  DO k=1,np
131  ! the Metric is constant over the mesh and follow the canonical
132  ! directions: it is given by the tensor (10000,0,100)
133  CALL mmgs_set_tensorsol(mmgmet,10000.d0,0.d0,0.d0,100.d0,0.d0,1.d0,k,ier)
134  IF ( ier /= 1 ) THEN
135  CALL exit(404)
136  ENDIF
137  ENDDO
138 
140  CALL mmgs_chk_meshdata(mmgmesh,mmgls,ier)
141  IF ( ier /= 1 ) CALL exit(105)
142 
144  ! isovalue discretization followed by an adaptation step over the input
145  ! Metric mmgMet. The mmgMet structure is updated so at the end it contains
146  ! the output metric of Mmg.
147  CALL mmgs_mmgsls(mmgmesh,mmgls,mmgmet,ier)
148 
149  IF ( ier == mmg5_strongfailure ) THEN
150  print*,"BAD ENDING OF MMGSLIB: UNABLE TO SAVE MESH"
151  stop 2
152  ELSE IF ( ier == mmg5_lowfailure ) THEN
153  print*,"BAD ENDING OF MMGSLIB"
154  ENDIF
155 
161 
163  CALL mmgs_savemesh(mmgmesh,trim(adjustl(outname)),&
164  len(trim(adjustl(outname))),ier)
165  IF ( ier /= 1 ) THEN
166  CALL exit(106)
167  ENDIF
168 
170  CALL mmgs_savesol(mmgmesh,mmgmet,trim(adjustl(outname)),&
171  len(trim(adjustl(outname))),ier)
172  IF ( ier /= 1 ) THEN
173  CALL exit(107)
174  ENDIF
175 
177  CALL mmgs_free_all(mmg5_arg_start, &
178  mmg5_arg_ppmesh,mmgmesh,mmg5_arg_ppls,mmgls, &
179  mmg5_arg_ppmet,mmgmet, &
180  mmg5_arg_end)
