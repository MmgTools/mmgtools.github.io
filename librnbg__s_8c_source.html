50 void MMG5_swapTri(MMG5_pTria trias, int* perm, int ind1, int ind2) {
51  MMG5_Tria pttmp;
52  int tmp;
53 
54  /* 2-- swap the triangles */
55  memcpy(&pttmp ,&trias[ind2],sizeof(MMG5_Tria));
56  memcpy(&trias[ind2],&trias[ind1],sizeof(MMG5_Tria));
57  memcpy(&trias[ind1],&pttmp ,sizeof(MMG5_Tria));
58 
59  /* 3-- swap the permutation table */
60  tmp = perm[ind2];
61  perm[ind2] = perm[ind1];
62  perm[ind1] = tmp;
63 }

104  }

109  return 1;

116 
117  if (!memset(vertTab, ~0, sizeof(SCOTCH_Num)*(vertNbr + 2))) {
118  perror(" ## Memory problem: memset");
119  MMG5_DEL_MEM(mesh,vertOldTab);
120  MMG5_DEL_MEM(mesh,vertTab);
121  return 1;
122  }
123 
124  edgeNbr = 1;
125  /* Euler-Poincare formulae edgeSiz = 20*mesh->np~4*mesh->nt;
126  (2*(12*mesh->np (triangles)-2*mesh->np (boundary triangles))) */
127  edgeSiz = vertNbr*4;
128 
129  MMG5_ADD_MEM(mesh,edgeSiz*sizeof(SCOTCH_Num),"edgeTab",
130  MMG5_DEL_MEM(mesh,vertOldTab);
131  MMG5_DEL_MEM(mesh,vertTab);
132  return 1);

134

136  /* Computing the adjacency list for each vertex */
137  for(triaIdx = 1 ; triaIdx < mesh->nt + 1 ; triaIdx++) {
138 
139  /* Testing if the tria exists */
140  if (!mesh->tria[triaIdx].v[0]) continue;
141 
142  iadr = 3*(triaIdx-1) + 1;
143  adja = &mesh->adja[iadr];
144  for (i=0; i<3; i++) {
145  ballTriIdx = adja[i] / 3;
146 
147  if (!ballTriIdx) continue;
148 
149  /* Testing if one neighbour of triaIdx has already been added */
150  if (vertTab[vertOldTab[triaIdx]] < 0)
151  vertTab[vertOldTab[triaIdx]] = edgeNbr;
152 
153  /* Testing if edgeTab memory is enough */
154  if (edgeNbr >= edgeSiz) {
155  int oldsize = edgeSiz;
156  MMG5_ADD_MEM(mesh,0.2*sizeof(SCOTCH_Num),"edgeTab",
157  MMG5_DEL_MEM(mesh,vertOldTab);
158  MMG5_DEL_MEM(mesh,vertTab);
159  return 1);
160  edgeSiz *= 1.2;
161  MMG5_SAFE_REALLOC(edgeTab,oldsize,edgeSiz,SCOTCH_Num,"scotch table",return 1);
162  }
163 
164  edgeTab[edgeNbr++] = vertOldTab[ballTriIdx];
165  }
166  }
167  vertTab[vertNbr+1] = edgeNbr;
168  edgeNbr--;
169  /*check if some tria are alone*/
170  for(triaIdx = 1 ; triaIdx < mesh->nt + 1 ; triaIdx++) {
171 
172  /* Testing if the tria exists */
173  if (!mesh->tria[triaIdx].v[0]) continue;
174  if (vertTab[vertOldTab[triaIdx]] < 0) {
175  if(vertOldTab[triaIdx] == vertNbr) {
176  fprintf(stderr," ## Warning: %s: graph error, no renumbering.\n",
177  __func__);
178  MMG5_DEL_MEM(mesh,edgeTab);
179  MMG5_DEL_MEM(mesh,vertTab);
180  return 1;
181  }
182  if(vertTab[vertOldTab[triaIdx] + 1] > 0)
183  vertTab[vertOldTab[triaIdx]] = vertTab[vertOldTab[triaIdx] + 1];
184  else {
185  if(vertOldTab[triaIdx]+1 == vertNbr) {
186  fprintf(stderr," ## Warning: %s: graph error, no renumbering.\n",
187  __func__);
188  MMG5_DEL_MEM(mesh,edgeTab);
189  MMG5_DEL_MEM(mesh,vertTab);
190  return 1;
191  }
192  i = 1;
193  do {
194  i++;
195  } while((vertTab[vertOldTab[triaIdx] + i] < 0) && ((vertOldTab[triaIdx] + i) < vertNbr));
196  if(vertOldTab[triaIdx] + i == vertNbr) {
197  fprintf(stderr," ## Warning: %s: graph error, no renumbering.\n",
198  __func__);
199  MMG5_DEL_MEM(mesh,edgeTab);
200  MMG5_DEL_MEM(mesh,vertTab);
201  return 1;
202  }
203  vertTab[vertOldTab[triaIdx]] = vertTab[vertOldTab[triaIdx] + i];
204  }
205  }
206  }
207 
208  /* free adjacents to gain memory space */
209  MMG5_DEL_MEM(mesh,mesh->adja);
210 
211  /* Building the graph by calling Scotch functions */
212  SCOTCH_graphInit(&graf) ;
213  CHECK_SCOTCH(SCOTCH_graphBuild(&graf, (SCOTCH_Num) 1, vertNbr, vertTab+1,
214  NULL, NULL, NULL, edgeNbr, edgeTab+1, NULL),
215  "scotch_graphbuild", 0) ;
216 
217 #ifndef NDEBUG

219  if ( mesh->info.imprim > 6 || mesh->info.ddebug )

223 #endif

305  for(j = 0 ; j < 3 ; j++) {

307  }
308  }
309

311  if ( permNodGlob ) {
312  for ( k=1; k<=mesh->np; ++k ) {
313  permNodGlob[k] = permNodTab[permNodGlob[k]];
314  }
315  }
316

319  while ( permNodTab[j] != j && permNodTab[j] )

321  }
322  MMG5_DEL_MEM(mesh,permNodTab);
323 
324  mesh->nt = ntreal;
325  mesh->np = npreal;
326

331 
332  if ( mesh->nt == mesh->ntmax )

339  mesh->point[k].tmp = k+1;

348  }

350  mesh->tria[mesh->ntmax ].v[2] = 0;
