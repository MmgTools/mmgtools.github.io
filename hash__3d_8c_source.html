98 
99  mins = MG_MIN(ia,MG_MIN(ib,ic));
100  maxs = MG_MAX(ia,MG_MAX(ib,ic));
101 
102  /* compute key */
103  sum = ia + ib + ic;
104  key = (MMG5_KA*mins + MMG5_KB*maxs) % hash->siz;
105  ph = &hash->item[key];
106 
107  if ( ph->a ) {
108  if ( ph->a == mins && ph->b == maxs && ph->s == sum )
109  return ph->k;
110  else {
111  while ( ph->nxt && ph->nxt < hash->max ) {
112  ph = &hash->item[ph->nxt];
113  if ( ph->a == mins && ph->b == maxs && ph->s == sum ) return ph->k;
114  }
115  }
116  ph->nxt = hash->nxt;
117  ph = &hash->item[hash->nxt];
118  ph->a = mins;
119  ph->b = maxs;
120  ph->s = sum;
121  ph->k = k;
122  hash->nxt = ph->nxt;
123  ph->nxt = 0;
124 
125  if ( hash->nxt >= hash->max ) {
126  MMG5_TAB_RECALLOC(mesh,hash->item,hash->max,MMG5_GAP,MMG5_hedge,"face",return 0;);
127  for (j=hash->nxt; j<hash->max; j++) hash->item[j].nxt = j+1;
128  }
129  return -1;
130  }
131 
132  /* insert new face */
133  ph->a = mins;
134  ph->b = maxs;
135  ph->s = sum;
136  ph->k = k;
137  ph->nxt = 0;
138 
139  return -1;

215  hcode[k] = -inival;

231  key = key % hsize + 1;
232  iadr++;
233  link[iadr] = hcode[key];
234  hcode[key] = -iadr;
235  }
236  }

238  /* set adjacency */
239  if ( mesh->info.ddebug ) fprintf(stdout," h- stage 2: adjacencies\n");
240  for (l=iadr; l>0; l--) {
241  if ( link[l] >= 0 ) continue;
242 
243  /* current element */
244  k = (l-1) / 4 + 1;
245  i = (l-1) % 4;
246  i1 = MMG5_idir[i][0];
247  i2 = MMG5_idir[i][1];
248  i3 = MMG5_idir[i][2];

251  maxs = MG_MAX(pt->v[i1],MG_MAX(pt->v[i2],pt->v[i3]));
252  sum = pt->v[i1] + pt->v[i2] + pt->v[i3];
253

256  pp = 0;
257  link[l] = 0;
258  while ( ll != inival ) {
259  kk = (ll-1) / 4 + 1;
260  ii = (ll-1) % 4;
261  i1 = MMG5_idir[ii][0];
262  i2 = MMG5_idir[ii][1];
263  i3 = MMG5_idir[ii][2];
264  pt1 = &mesh->tetra[kk];
265  sum1 = pt1->v[i1] + pt1->v[i2] + pt1->v[i3];
266  if ( sum1 == sum ) {
267  mins1 = MG_MIN(pt1->v[i1],MG_MIN(pt1->v[i2],pt1->v[i3]));
268  maxs1 = MG_MAX(pt1->v[i1],MG_MAX(pt1->v[i2],pt1->v[i3]));
269 
270  /* adjacent found */
271  if ( mins1 == mins && maxs1 == maxs ) {
272  if ( pp != 0 ) link[pp] = link[ll];
273  link[l] = 4*kk + ii;
274  link[ll] = 4*k + i;
275  break;
276  }
277  }
278  pp = ll;
279  ll = -link[ll];
280  }
281  }
282  MMG5_SAFE_FREE(hcode);
283  return 1;

517 int MMG5_setEdgeNmTag(MMG5_pMesh mesh, MMG5_Hash *hash) {
518  MMG5_pTetra pt;
519  MMG5_pxTetra pxt;
520  MMG5_pTria ptt;
521  MMG5_hedge *ph;
522  int adj,pradj,piv,ilist,list[MMG3D_LMAX+2];
523  int k,l,i1,i2,na,nb,ia,it1,it2, nr;
524  int ipa,ipb,nbdy,start;
525  unsigned int key;
526  int8_t iface,hasadja,i;
527  static int8_t mmgWarn0=0,mmgWarn1=0;
528 
529  nr = 0;
530 
531  /* First: seek edges at the interface of two distinct domains and mark it as
532  * required */
533  for (k=1; k<=mesh->nt; k++) {
534  ptt = &mesh->tria[k];
535 
536  if ( !MG_EOK(ptt) ) continue;
537 
538  for (l=0; l<3; l++) {
539 
540  /* Skip parallel edges */
541  if ( (ptt->tag[l] & MG_PARBDY) || (ptt->tag[l] & MG_BDY) ) continue;
542 
543  if ( ptt->tag[l] & MG_NOM ) {
544  i1 = MMG5_inxt2[l];
545  i2 = MMG5_iprv2[l];
546 
547  /* compute key */
548  na = MG_MIN(ptt->v[i1],ptt->v[i2]);
549  nb = MG_MAX(ptt->v[i1],ptt->v[i2]);
550  key = (MMG5_KA*na + MMG5_KB*nb) % hash->siz;
551  ph = &hash->item[key];
552 
553  assert(ph->a);
554  while ( ph->a ) {
555  if ( ph->a == na && ph->b == nb ) break;
556  assert(ph->nxt);
557  ph = &hash->item[ph->nxt];
558  }
559  /* Set edge tag and point tags to MG_REQ if the non-manifold edge shared
560  * separated domains */
561  if ( ph->s > 3 ) {
562  start = ptt->cc/4;
563  assert(start);
564  pt = &mesh->tetra[start];
565 
566 
567  for (ia=0; ia<6; ++ia) {
568  ipa = MMG5_iare[ia][0];
569  ipb = MMG5_iare[ia][1];
570  if ( (pt->v[ipa] == na && pt->v[ipb] == nb) ||
571  (pt->v[ipa] == nb && pt->v[ipb] == na)) break;
572  }
573  assert(ia<6);
574 
575 
576  /* Travel throug the shell of the edge until reaching a tetra without adjacent
577  * or until reaching the starting tetra */
578  iface = ptt->cc%4;
579  MMG3D_coquilFaceFirstLoop(mesh,start,na,nb,iface,ia,list,&ilist,&it1,&it2,
580  &piv,&adj,&hasadja,&nbdy,1);
581 
582  /* At this point, the first travel, in one direction, of the shell is
583  complete. Now, analyze why the travel ended. */
584  if ( adj == start ) {
585  if ( !it2 ) {
586  if ( !mmgWarn0 ) {
587  mmgWarn0 = 1;
588  fprintf(stderr,"\n ## Warning: %s: at least 1 wrong boundary tag:"
589  " Only 0 or 1 boundary triangles founded in the shell of the edge\n",
590  __func__);
591  }
592  }
593  if ( nbdy < 2 )
594  MMG5_coquilFaceErrorMessage(mesh, it1/4, it2/4);
595  }
596  else {
597  /* A boundary has been detected : slightly different configuration */
598  if ( hasadja ) {
599 
600  /* Start back everything from this tetra adj */
601  MMG3D_coquilFaceSecondLoopInit(mesh,piv,&iface,&i,list,&ilist,&it1,
602  &pradj,&adj);
603 
604  nbdy = 1;
605  while ( adj ) {
606  pradj = adj;
607 
608  if ( MMG5_openCoquilTravel(mesh,na,nb,&adj,&piv,&iface,&i)<0 ) {
609  return 0;
610  }
611 
612  /* overflow */
613  if ( ++ilist > MMG3D_LMAX-2 ) {
614  if ( !mmgWarn1 ) {
615  mmgWarn1 = 1;
616  fprintf(stderr,"\n ## Warning: %s: problem in surface remesh"
617  " process. At least 1 shell of edge (%d-%d) contains"
618  " too many elts.\n",__func__,MMG3D_indPt(mesh,na),
619  MMG3D_indPt(mesh,nb));
620  fprintf(stderr,"\n ## Try to modify the hausdorff"
621  " number, or/and the maximum mesh.\n");
622  }
623  return 0;
624  }
625 
626  pt = &mesh->tetra[pradj];
627  if ( pt->xt ) {
628  pxt = &mesh->xtetra[pt->xt];
629  if ( pxt->ftag[iface] & MG_BDY ) ++nbdy;
630  }
631  }
632 
633  assert(!adj);
634  it2 = 4*pradj + iface;
635 
636  if ( (!it1 || !it2) || (it1 == it2) ) {
637  MMG5_coquilFaceErrorMessage(mesh, it1/4, it2/4);
638  return 0;
639  }
640  }
641  }
642 
643  /* If ph->s do not match the number of encountred boundaries we have
644  separated domains. */
645  if ( nbdy != ph->s ) {
646  if ( !(ptt->tag[l] & MG_REQ) ) {
647  ptt->tag[l] |= MG_REQ;
648  ptt->tag[l] &= ~MG_NOSURF;
649  ++nr;
650  }
651  mesh->point[ptt->v[MMG5_inxt2[l]]].tag |= MG_REQ;
652  mesh->point[ptt->v[MMG5_iprv2[l]]].tag |= MG_REQ;
653  mesh->point[ptt->v[MMG5_inxt2[l]]].tag &= ~MG_NOSURF;
654  mesh->point[ptt->v[MMG5_iprv2[l]]].tag &= ~MG_NOSURF;
655  }
656 
657  /* Work done for this edge: reset ph->s/ */
658  ph->s = 0;
659  }
660  }
