50 int MMG5_mmgHashTria(MMG5_pMesh mesh, int *adjt, MMG5_Hash *hash, int chkISO) {
51  MMG5_pTria pt,pt1,pt2;
52  MMG5_hedge *ph;
53  int *adja,k,kk,jel,lel,hmax,dup,nmf,ia,ib;
54  int8_t i,i1,i2,j,l;
55  unsigned int key;
56 
57  /* adjust hash table params */
58  hmax =(int)(3.71*mesh->np);
59  hash->siz = mesh->np;
60  hash->max = hmax + 1;
61  hash->nxt = hash->siz;
62  MMG5_ADD_MEM(mesh,(hash->max+1)*sizeof(MMG5_hedge),"hash table",return 0);
63  MMG5_SAFE_CALLOC(hash->item,hash->max+1,MMG5_hedge,return 0);
64 
65  for (k=hash->siz; k<hash->max; k++)
66  hash->item[k].nxt = k+1;
67 
68  if ( mesh->info.ddebug ) fprintf(stdout," h- stage 1: init\n");
69 
70  /* hash triangles */
71  mesh->base = 1;
72  dup = nmf = 0;
73  for (k=1; k<=mesh->nt; k++) {
74  pt = &mesh->tria[k];
75  if ( !MG_EOK(pt) ) continue;
76 
77  pt->flag = 0;
78  pt->base = mesh->base;
79  adja = &adjt[3*(k-1)+1];
80  for (i=0; i<3; i++) {
81 
82  /* Skip parallel edges */
83  if( (pt->tag[i] & MG_PARBDY) && !(pt->tag[i] & MG_PARBDYBDY) ) continue;
84 
85  i1 = MMG5_inxt2[i];
86  i2 = MMG5_iprv2[i];
87 
88  /* compute key */
89  ia = MG_MIN(pt->v[i1],pt->v[i2]);
90  ib = MG_MAX(pt->v[i1],pt->v[i2]);
91  key = (MMG5_KA*ia + MMG5_KB*ib) % hash->siz;
92  ph = &hash->item[key];
93 
94  /* store edge */
95  if ( ph->a == 0 ) {
96  ph->a = ia;
97  ph->b = ib;
98  ph->k = 3*k + i;
99  ph->nxt = 0;
100  ++ph->s;
101  continue;
102  }
103  /* update info about adjacent */
104 #ifndef NDEBUG
105  int8_t ok = 0;
106 #endif
107  while ( ph->a ) {
108  if ( ph->a == ia && ph->b == ib ) {
109  jel = ph->k / 3;
110  j = ph->k % 3;
111  pt1 = &mesh->tria[jel];
112  /* discard duplicate face */
113  if ( pt1->v[j] == pt->v[i] ) {
114  pt1->v[0] = 0;
115  dup++;
116  }
117  /* update adjacent */
118  else if ( !adjt[3*(jel-1)+1+j] ) {
119  adja[i] = 3*jel + j;
120  adjt[3*(jel-1)+1+j] = 3*k + i;
121  ++ph->s;
122  }
123  /* non-manifold case */
124  else if ( adja[i] != 3*jel+j ) {
125  lel = adjt[3*(jel-1)+1+j]/3;
126  l = adjt[3*(jel-1)+1+j]%3;
127  pt2 = &mesh->tria[lel];
128  if ( chkISO && ( (pt->ref == MG_ISO) || (pt->ref < 0)) ) {
129  adjt[3*(lel-1)+1+l] = 0;
130  adja[i] = 3*jel+j;
131  adjt[3*(jel-1)+1+j] = 3*k + i;
132  }
133  pt->tag[i] |= MG_GEO + MG_NOM;
134  pt1->tag[j] |= MG_GEO + MG_NOM;
135  pt2->tag[l] |= MG_GEO + MG_NOM;
136  nmf++;
137  ++ph->s;
138  }
139 #ifndef NDEBUG
140  ok = 1;
141 #endif
142  break;
143  }
144  else if ( !ph->nxt ) {
145  ph->nxt = hash->nxt;
146  ph = &hash->item[ph->nxt];
147  assert(ph);
148 
149  if ( hash->nxt >= hash->max-1 ) {
150  if ( mesh->info.ddebug ) {
151  fprintf(stderr,"\n ## Warning: %s: memory alloc problem (edge):"
152  " %d\n",__func__,hash->max);
153  }
154  MMG5_TAB_RECALLOC(mesh,hash->item,hash->max,MMG5_GAP,MMG5_hedge,
155  "MMG5_edge",
156  MMG5_DEL_MEM(mesh,hash->item);
157  return 0);
158 
159  ph = &hash->item[hash->nxt];
160 
161  for (kk=ph->nxt; kk<hash->max; kk++)
162  hash->item[kk].nxt = kk+1;
163  }
164 
165  hash->nxt = ph->nxt;
166  ph->a = ia;
167  ph->b = ib;
168  ph->k = 3*k + i;
169  ph->nxt = 0;
170  ++ph->s;
171 #ifndef NDEBUG
172  ok = 1;
173 #endif
174  break;
175  }
176  else
177  ph = &hash->item[ph->nxt];
178  }
179  assert(ok);
180  }
181  }
182 
183  /* Now loop on "only" parallel edges in order to add a MG_BDY tag on those
184  * that are found in the hash table and their adjacents (if manifold; in the
185  * non-manifold case the MG_NOM tag suffices).
186  *
187  * Rationale:
188  * - put MG_BDY tags on edges that are contact edges between a true
189  * boundary (!MG_PARBDY || (MG_PARBDY && MG_PARBDYBDY) and a parallel one
190  * (MG_PARBDY && !MG_PARBDYBDY);
191  * - add also a MG_PARBDYBDY tag on those edges (as it cannot be inherited
192  * from any triangle there) and their extremities.
193  */
194  for (k=1; k<=mesh->nt; k++) {
195  pt = &mesh->tria[k];
196  if ( !MG_EOK(pt) ) continue;
197  for (i=0; i<3; i++) {
198  if( !(pt->tag[i] & MG_PARBDY) || (pt->tag[i] & MG_PARBDYBDY) ) continue;
199  i1 = MMG5_inxt2[i];
200  i2 = MMG5_iprv2[i];
201 
202  /* compute key */
203  ia = MG_MIN(pt->v[i1],pt->v[i2]);
204  ib = MG_MAX(pt->v[i1],pt->v[i2]);
205  key = (MMG5_KA*ia + MMG5_KB*ib) % hash->siz;
206  ph = &hash->item[key];
207 
208  if ( ph->a == 0 ) continue;
209  while( ph->a ) {
210  if ( ph->a == ia && ph->b == ib ) {
211  jel = ph->k / 3;
212  j = ph->k % 3;
213  pt1 = &mesh->tria[jel];
214  pt1->tag[j] |= MG_BDY + MG_PARBDYBDY;
215  mesh->point[ia].tag |= MG_PARBDYBDY;
216  mesh->point[ib].tag |= MG_PARBDYBDY;
217  /* update adjacent */
218  lel = adjt[3*(jel-1)+1+j]/3;
219  l = adjt[3*(jel-1)+1+j]%3;
220  if( lel ) {
221  pt2 = &mesh->tria[lel];
222  pt2->tag[l] |= MG_BDY + MG_PARBDYBDY;
223  mesh->point[ia].tag |= MG_PARBDYBDY;
224  mesh->point[ib].tag |= MG_PARBDYBDY;
225  }
226  break;
227  } else if ( !ph->nxt ) {
228  break;
229  } else {
230  ph = &hash->item[ph->nxt];
231  }
232  }
233  }
234  }
235 
236  /* set tag */
237  for (k=1; k<=mesh->nt; k++) {
238  pt = &mesh->tria[k];
239  for (i=0; i<3; i++) {
240  if ( pt->tag[i] & MG_NOM ) {
241  mesh->point[pt->v[MMG5_inxt2[i]]].tag |= MG_NOM;
242  mesh->point[pt->v[MMG5_iprv2[i]]].tag |= MG_NOM;
243  }
244  }
245  }
246 
247  if ( abs(mesh->info.imprim) > 5 && dup+nmf > 0 ) {
248  fprintf(stdout," ## "); fflush(stdout);
249  if ( nmf > 0 ) fprintf(stdout,"[non-manifold model] ");
250  if ( dup > 0 ) fprintf(stdout," %d duplicate removed",dup);
251  fprintf(stdout,"\n");
252  }
253  if ( mesh->info.ddebug ) fprintf(stdout," h- completed.\n");
254 
255  return 1;
256 }

269 int MMG5_hashEdge(MMG5_pMesh mesh,MMG5_Hash *hash, int a,int b,int k) {
270  MMG5_hedge *ph;
271  int key,ia,ib,j;
272 
273  ia = MG_MIN(a,b);
274  ib = MG_MAX(a,b);
275  key = (MMG5_KA*ia + MMG5_KB*ib) % hash->siz;
276  ph = &hash->item[key];
277 
278  if ( ph->a == ia && ph->b == ib )
279  return 1;
280  else if ( ph->a ) {
281  while ( ph->nxt && ph->nxt < hash->max ) {
282  ph = &hash->item[ph->nxt];
283  if ( ph->a == ia && ph->b == ib ) return 1;
284  }
285  ph->nxt = hash->nxt;
286  ph = &hash->item[hash->nxt];
287 
288  if ( hash->nxt >= hash->max-1 ) {
289  if ( mesh->info.ddebug )
290  fprintf(stderr,"\n ## Warning: %s: memory alloc problem (edge):"
291  " %d\n",__func__,hash->max);
292 
293  MMG5_TAB_RECALLOC(mesh,hash->item,hash->max,MMG5_GAP,MMG5_hedge,
294  "MMG5_edge",return 0);
295  /* ph pointer may be false after realloc */
296  ph = &hash->item[hash->nxt];
297 
298  for (j=ph->nxt; j<hash->max; j++) hash->item[j].nxt = j+1;
299  }
300  hash->nxt = ph->nxt;
301  }
302 
303  /* insert new edge */
304  ph->a = ia;
305  ph->b = ib;
306  ph->k = k;
307  ph->nxt = 0;
308 
309  return 1;
310 }
