47 static inline
48 double MMG5_surf(MMG5_pMesh mesh,double m[3][6],MMG5_pTria ptt) {
49  MMG5_Bezier b;
50  double surf,dens,J[3][2],mJ[3][2],tJmJ[2][2];
51  int8_t i,nullDens;
52  static int8_t mmgErr=0;
53 
54  surf = 0.0;
55 
56  if ( !MMG5_bezierCP(mesh,ptt,&b,1) ) return 0.0;
57 
58  /* Compute density integrand of volume at the 3 vertices of T */
59  nullDens = 0;
60  for (i=0; i<3; i++) {
61  if ( i == 0 ) {
62  J[0][0] = 3.0*( b.b[7][0] - b.b[0][0] ) ; J[0][1] = 3.0*( b.b[6][0] - b.b[0][0] );
63  J[1][0] = 3.0*( b.b[7][1] - b.b[0][1] ) ; J[1][1] = 3.0*( b.b[6][1] - b.b[0][1] );
64  J[2][0] = 3.0*( b.b[7][2] - b.b[0][2] ) ; J[2][1] = 3.0*( b.b[6][2] - b.b[0][2] );
65  }
66  else if ( i == 1 ) {
67  J[0][0] = 3.0*( b.b[1][0] - b.b[8][0] ) ; J[0][1] = 3.0*( b.b[3][0] - b.b[8][0] );
68  J[1][0] = 3.0*( b.b[1][1] - b.b[8][1] ) ; J[1][1] = 3.0*( b.b[3][1] - b.b[8][1] );
69  J[2][0] = 3.0*( b.b[1][2] - b.b[8][2] ) ; J[2][1] = 3.0*( b.b[3][2] - b.b[8][2] );
70  }
71  else {
72  J[0][0] = 3.0*( b.b[4][0] - b.b[5][0] ) ; J[0][1] = 3.0*( b.b[2][0] - b.b[5][0] );
73  J[1][0] = 3.0*( b.b[4][1] - b.b[5][1] ) ; J[1][1] = 3.0*( b.b[2][1] - b.b[5][1] );
74  J[2][0] = 3.0*( b.b[4][2] - b.b[5][2] ) ; J[2][1] = 3.0*( b.b[2][2] - b.b[5][2] );
75  }
76 
77  mJ[0][0] = m[i][0]*J[0][0] + m[i][1]*J[1][0] + m[i][2]*J[2][0];
78  mJ[1][0] = m[i][1]*J[0][0] + m[i][3]*J[1][0] + m[i][4]*J[2][0];
79  mJ[2][0] = m[i][2]*J[0][0] + m[i][4]*J[1][0] + m[i][5]*J[2][0];
80 
81  mJ[0][1] = m[i][0]*J[0][1] + m[i][1]*J[1][1] + m[i][2]*J[2][1];
82  mJ[1][1] = m[i][1]*J[0][1] + m[i][3]*J[1][1] + m[i][4]*J[2][1];
83  mJ[2][1] = m[i][2]*J[0][1] + m[i][4]*J[1][1] + m[i][5]*J[2][1];
84 
85  /* dens = sqrt(tJacsigma * M * Jacsigma )*/
86  tJmJ[0][0] = J[0][0]*mJ[0][0] + J[1][0]*mJ[1][0] + J[2][0]*mJ[2][0];
87  tJmJ[0][1] = J[0][0]*mJ[0][1] + J[1][0]*mJ[1][1] + J[2][0]*mJ[2][1];
88  tJmJ[1][0] = J[0][1]*mJ[0][0] + J[1][1]*mJ[1][0] + J[2][1]*mJ[2][0];
89  tJmJ[1][1] = J[0][1]*mJ[0][1] + J[1][1]*mJ[1][1] + J[2][1]*mJ[2][1];
90 
91  dens = tJmJ[0][0]*tJmJ[1][1] - tJmJ[1][0]*tJmJ[0][1];
92  if ( dens <= MMG5_EPSD2 ) {
93 #ifndef NDEBUG
94  if ( !mmgErr ) {
95  fprintf(stderr,"\n ## Warning: %s: at least 1 negative or null density.\n",
96  __func__);
97  mmgErr = 1;
98  }
99 #endif
100  ++nullDens;
101  continue;
102  }
103  surf += sqrt(fabs(dens));
104  }
105 
106  if ( nullDens==3 ) return 0;
107 
108  surf *= MMG5_ATHIRD;
109  return surf;
110 }

169 double MMG5_surftri33_ani(MMG5_pMesh mesh,MMG5_pTria ptt,
170  double ma[6], double mb[6], double mc[6]) {
171  double mm[6];
172  double *a,*b,*c,abx,aby,abz,acx,acy,acz,dens[3],surf;
173  int i,ia,ib,ic;
174 
175  ia = ptt->v[0];
176  ib = ptt->v[1];
177  ic = ptt->v[2];
178 
179  a = &mesh->point[ia].c[0];
180  b = &mesh->point[ib].c[0];
181  c = &mesh->point[ic].c[0];
182 
183  abx = b[0] - a[0];
184  aby = b[1] - a[1];
185  abz = b[2] - a[2];
186  acx = c[0] - a[0];
187  acy = c[1] - a[1];
188  acz = c[2] - a[2];
189 
190  /* Compute the mean of the metrics over the triangle */
191  for (i=0; i<6; i++)
192  mm[i] = MMG5_ATHIRD * (ma[i] + mb[i]+ mc[i]);
193 
194  /* Compute sqrt(det(t^JmJ)) (= int_T sqrt(t^JmJ) for a non-curve element) */
195  dens[0] = (abx*abx*mm[0]+abx*aby*mm[1]+abx*abz*mm[2])
196  + (aby*abx*mm[1]+aby*aby*mm[3]+aby*abz*mm[4])
197  + (abz*abx*mm[2]+abz*aby*mm[4]+abz*abz*mm[5]);
198 
199  dens[1] = (abx*acx*mm[0]+abx*acy*mm[1]+abx*acz*mm[2])
200  + (aby*acx*mm[1]+aby*acy*mm[3]+aby*acz*mm[4])
201  + (abz*acx*mm[2]+abz*acy*mm[4]+abz*acz*mm[5]);
202 
203  dens[2] = (acx*acx*mm[0]+acx*acy*mm[1]+acx*acz*mm[2])
204  + (acy*acx*mm[1]+acy*acy*mm[3]+acy*acz*mm[4])
205  + (acz*acx*mm[2]+acz*acy*mm[4]+acz*acz*mm[5]);
206 
207  surf = dens[0]*dens[2]-dens[1]*dens[1];
208 
209  if ( surf < MMG5_EPSD ) return 0.0;
210 
211  surf = sqrt(surf);
212 
213  return surf;
214 }

446  m[0] = isqhmax;
447  m[1] = 0;
448  m[2] = 0;
449  m[3] = isqhmax;
450  m[4] = 0;
451  m[5] = isqhmax;
452  return 1;
453  }
454 
455  /* solve now (a b c) = tAA^{-1} * tAb */
456  if ( !MMG5_sys33sym(tAA,tAb,c) ) {
457  if ( !mmgWarn0 ) {
458  fprintf(stderr,"\n ## Warning: %s: unable to solve the system on at"
459  " least 1 point.\n",__func__);
460  mmgWarn0 = 1;

464  intm[0] = 2.0*c[0];

466  intm[2] = 2.0*c[1];

469  are the two pr. directions of curvature, and the two curvatures can be inferred from lambdas*/
470  MMG5_eigensym(intm,kappa,vp);

483  intm[0] = kappa[0]*vp[0][0]*vp[0][0] + kappa[1]*vp[1][0]*vp[1][0];
484  intm[1] = kappa[0]*vp[0][0]*vp[0][1] + kappa[1]*vp[1][0]*vp[1][1];
485  intm[2] = kappa[0]*vp[0][1]*vp[0][1] + kappa[1]*vp[1][1]*vp[1][1];
486

488  desired metric, except it is expressed in the rotated bc, that is intm = R
489  * metric in bc * ^t R, so metric in bc = ^tR*intm*R */
490 
491  /* b0, b1 and b2 are the lines of matrix intm*R */
492  // intm = intm[0] intm[1] 0
493  // intm[1] intm[2] 0
494  // 0 0 isqhmax
495

497  b0[1] = intm[0]*r[0][1] + intm[1]*r[1][1];
498  b0[2] = intm[0]*r[0][2] + intm[1]*r[1][2];

501  b1[2] = intm[1]*r[0][2] + intm[2]*r[1][2];
502  b2[0] = isqhmax*r[2][0];
503  b2[1] = isqhmax*r[2][1];
504  b2[2] = isqhmax*r[2][2];
505 
506  m[0] = r[0][0] * b0[0] + r[1][0] * b1[0] + r[2][0] * b2[0];
507  m[1] = r[0][0] * b0[1] + r[1][0] * b1[1] + r[2][0] * b2[1];

513  m[5] = r[0][2] * b0[2] + r[1][2] * b1[2] + r[2][2] * b2[2];
514 
515  return 1;

536 int MMG5_solveDefmetrefSys( MMG5_pMesh mesh, MMG5_pPoint p0, int ipref[2],
537  double r[3][3], double c[3],
538  double tAA[6], double tAb[3], double *m,
539  double isqhmin, double isqhmax, double hausd)
540 {
541  MMG5_pPoint p1;
542  double intm[3], kappa[2], vp[2][2], b0[3], b1[3], b2[3], kappacur;
543  double gammasec[3],tau[2], ux, uy, uz, ps1, l, ll, *t, *t1;
544  int i;
545  static int8_t mmgWarn=0;
546 
547  memset(intm,0x0,3*sizeof(double));
548 
549  /* case planar surface : tAb = 0 => no curvature */
550  /* isotropic metric with hmax size*/
551  if((tAb[0]*tAb[0] + tAb[1]*tAb[1] + tAb[2]*tAb[2]) < MMG5_EPSD) {
552  m[0] = isqhmax;
553  m[1] = 0;
554  m[2] = 0;
555  m[3] = isqhmax;
556  m[4] = 0;
557  m[5] = isqhmax;
558  return 1;
559  }
560 
561  /* solve now (a b c) = tAA^{-1} * tAb */
562  if ( !MMG5_sys33sym(tAA,tAb,c) ) {
563  if ( !mmgWarn ) {
564  fprintf(stderr,"\n ## Warning: %s: unable to solve the system on at"
565  " least 1 point.\n", __func__);
566  mmgWarn = 1;
567  }
568  return 0;
569  }
570  intm[0] = 2.0*c[0];
571  intm[1] = c[2];
572  intm[2] = 2.0*c[1];
573 
574  /* At this point, intm stands for the integral matrix of Taubin's approach :
575  vp[0] and vp[1] are the two pr. directions of curvature, and the two
576  curvatures can be inferred from lambdas*/
577  MMG5_eigensym(intm,kappa,vp);
578 
579  /* Truncation of eigenvalues */
580  kappa[0] = 2.0/9.0 * fabs(kappa[0])/mesh->info.hausd;
581  kappa[0] = MG_MIN(kappa[0],isqhmin);
582  kappa[0] = MG_MAX(kappa[0],isqhmax);
583 
584  kappa[1] = 2.0/9.0 * fabs(kappa[1])/mesh->info.hausd;
585  kappa[1] = MG_MIN(kappa[1],isqhmin);
586  kappa[1] = MG_MAX(kappa[1],isqhmax);
587 
588  /* Send back the metric to the canonical basis of tangent plane :
589  diag(lambda) = {^t}vp * M * vp, M = vp * diag(lambda) * {^t}vp */
590  intm[0] = kappa[0]*vp[0][0]*vp[0][0] + kappa[1]*vp[1][0]*vp[1][0];
591  intm[1] = kappa[0]*vp[0][0]*vp[0][1] + kappa[1]*vp[1][0]*vp[1][1];
592  intm[2] = kappa[0]*vp[0][1]*vp[0][1] + kappa[1]*vp[1][1]*vp[1][1];
593 
594  /* Now express metric with respect to the approx of the underlying ref
595  * curve */
596  t = &p0->n[0];
597  kappacur = 0.0;
598 
599  for (i=0; i<2; i++) {
600  p1 = &mesh->point[ipref[i]];
601  ux = p1->c[0] - p0->c[0];
602  uy = p1->c[1] - p0->c[1];
603  uz = p1->c[2] - p0->c[2];
604 
605  ps1 = ux*t[0] + uy*t[1] + uz*t[2];
606  c[0] = MMG5_ATHIRD*ps1*t[0];
607  c[1] = MMG5_ATHIRD*ps1*t[1];
608  c[2] = MMG5_ATHIRD*ps1*t[2];
609 
610  b0[0] = r[0][0]*c[0] + r[0][1]*c[1] + r[0][2]*c[2];
611  b0[1] = r[1][0]*c[0] + r[1][1]*c[1] + r[1][2]*c[2];
612  b0[2] = r[2][0]*c[0] + r[2][1]*c[1] + r[2][2]*c[2];
613 
614  if ( (MG_CRN & p1->tag) || (MG_NOM & p1->tag) ) {
615  c[0] = p1->c[0] - MMG5_ATHIRD*ux;
616  c[1] = p1->c[1] - MMG5_ATHIRD*uy;
617  c[2] = p1->c[2] - MMG5_ATHIRD*uz;
618  }
619  else {
620  assert(MG_REF & p1->tag);
621  t1 = &(p1->n[0]);
622  ps1 = -(ux*t1[0] + uy*t1[1] + uz*t1[2]);
623  c[0] = p1->c[0] + MMG5_ATHIRD*ps1*t1[0];
624  c[1] = p1->c[1] + MMG5_ATHIRD*ps1*t1[1];
625  c[2] = p1->c[2] + MMG5_ATHIRD*ps1*t1[2];
626  }
627  c[0] -= p0->c[0];
628  c[1] -= p0->c[1];
629  c[2] -= p0->c[2];
630 
631  b1[0] = r[0][0]*c[0] + r[0][1]*c[1] + r[0][2]*c[2];
632  b1[1] = r[1][0]*c[0] + r[1][1]*c[1] + r[1][2]*c[2];
633  b1[2] = r[2][0]*c[0] + r[2][1]*c[1] + r[2][2]*c[2];
634 
635  /* Everything is expressed in the rotated frame */
636  tau[0] = 3.0*b0[0];
637  tau[1] = 3.0*b0[1];
638  ll = tau[0]*tau[0] + tau[1]*tau[1];
639  if ( ll < MMG5_EPSD ) {
640  kappacur = isqhmax;
641  continue;
642  }
643  l = 1.0 / sqrt(ll);
644  tau[0] *= l;
645  tau[1] *= l;
646 
647  gammasec[0] = -12.0*b0[0] + 6.0*b1[0];
648  gammasec[1] = -12.0*b0[1] + 6.0*b1[1];
649  gammasec[2] = -12.0*b0[2] + 6.0*b1[2];
650 
651  ps1 = tau[0]*gammasec[0] + tau[1]*gammasec[1];
652  c[0] = gammasec[0] - ps1*tau[0];
653  c[1] = gammasec[1] - ps1*tau[1];
654  c[2] = gammasec[2];
655 
656  // p.s. with normal at p0
657  kappacur = MG_MAX(kappacur,MG_MAX(0.0,1.0/ll*fabs(c[2])));
658  }
659 
660  /* Rotation of tangent vector : tau is reused */
661  c[0] = r[0][0]*t[0] + r[0][1]*t[1] + r[0][2]*t[2];
662  c[1] = r[1][0]*t[0] + r[1][1]*t[1] + r[1][2]*t[2];
663  c[2] = r[2][0]*t[0] + r[2][1]*t[1] + r[2][2]*t[2];
664  memcpy(tau,&c[0],2*sizeof(double));
665 
666  /* Truncation of curvature */
667  kappacur = 1.0/8.0 * kappacur/mesh->info.hausd;
668  kappacur = MG_MIN(kappacur,isqhmin);
669  kappacur = MG_MAX(kappacur,isqhmax);
670 
671  /* The associated matrix in basis (rt, orth rt) */
672  c[0] = kappacur*tau[0]*tau[0] + isqhmax*tau[1]*tau[1];
673  c[1] = (kappacur - isqhmax)*tau[0]*tau[1];
674  c[2] = kappacur*tau[1]*tau[1] + isqhmax*tau[0]*tau[0];
675 
676  /* Reuse b0 for commodity */
677  MMG5_intmetsavedir(mesh,c,intm,b0);
678  memcpy(intm,b0,3*sizeof(double));
679 
680  /* At this point, intm (with 0 replaced by isqhmax in the z direction) is the
681  desired metric, except it is expressed in the rotated bc, that is intm = R
682  * metric in bc * ^t R, so metric in bc = ^tR*intm*R */
683  // intm = intm[0] intm[1] 0
684  // intm[1] intm[2] 0
685  // 0 0 isqhmax
686 
687  /* b0 and b1 serve now for nothing : let them be the lines of matrix intm*R */
688  b0[0] = intm[0]*r[0][0] + intm[1]*r[1][0];
689  b0[1] = intm[0]*r[0][1] + intm[1]*r[1][1];
690  b0[2] = intm[0]*r[0][2] + intm[1]*r[1][2];
691  b1[0] = intm[1]*r[0][0] + intm[2]*r[1][0];
692  b1[1] = intm[1]*r[0][1] + intm[2]*r[1][1];
693  b1[2] = intm[1]*r[0][2] + intm[2]*r[1][2];
694  b2[0] = isqhmax*r[2][0];
695  b2[1] = isqhmax*r[2][1];
696  b2[2] = isqhmax*r[2][2];
697 
698  m[0] = r[0][0] * b0[0] + r[1][0] * b1[0] + r[2][0] * b2[0];
699  m[1] = r[0][0] * b0[1] + r[1][0] * b1[1] + r[2][0] * b2[1];
700  m[2] = r[0][0] * b0[2] + r[1][0] * b1[2] + r[2][0] * b2[2];
701 
702  m[3] = r[0][1] * b0[1] + r[1][1] * b1[1] + r[2][1] * b2[1];
703  m[4] = r[0][1] * b0[2] + r[1][1] * b1[2] + r[2][1] * b2[2];
704 
705  m[5] = r[0][2] * b0[2] + r[1][2] * b1[2] + r[2][2] * b2[2];
706 
707  return 1;
708 }
