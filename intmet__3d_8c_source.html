51 int MMG5_intmet_ani(MMG5_pMesh mesh,MMG5_pSol met,int k,int8_t i,int ip,
52  double s) {
53  MMG5_pTetra pt;
54  MMG5_pxTetra pxt;
55  MMG5_pPoint ppt;
56  MMG5_pxPoint pxp;
57  double *m;
58  int ip1,ip2;
59 
60  pt = &mesh->tetra[k];
61  m = &met->m[6*ip];
62  ip1 = pt->v[MMG5_iare[i][0]];
63  ip2 = pt->v[MMG5_iare[i][1]];
64 
65  if ( pt->xt ) {
66  pxt = &mesh->xtetra[pt->xt];
67  if ( pxt->tag[i] & MG_GEO && !(pxt->tag[i] & MG_NOM) ) {
68  ppt = &mesh->point[ip];
69  assert(ppt->xp);
70  pxp = &mesh->xpoint[ppt->xp];
71  return MMG5_intridmet(mesh,met,ip1,ip2,s,pxp->n1,m);
72  }
73  else if ( pxt->tag[i] & MG_BDY ) {
74  return MMG5_intregmet(mesh,met,k,i,s,m);
75  }
76  else {
77  /* The edge is an internal edge. */
78  return MMG5_intvolmet(mesh,met,k,i,s,m);
79  }
80  }
81  else {
82  /* The edge is an internal edge. */
83  return MMG5_intvolmet(mesh,met,k,i,s,m);
84  }
85  return 0;
86 }

162 int MMG5_intregmet(MMG5_pMesh mesh,MMG5_pSol met,int k,int8_t i,double s,
163  double mr[6]) {
164  MMG5_pTetra pt;
165  MMG5_pxTetra pxt;
166  MMG5_Tria ptt;
167  int ifa0, ifa1, iloc;
168 
169  pt = &mesh->tetra[k];
170  pxt = &mesh->xtetra[pt->xt];
171  ifa0 = MMG5_ifar[i][0];
172  ifa1 = MMG5_ifar[i][1];
173 
174  if ( pxt->ftag[ifa0] & MG_BDY ) {
175  MMG5_tet2tri( mesh,k,ifa0,&ptt);
176  iloc = MMG5_iarfinv[ifa0][i];
177  assert(iloc >= 0);
178  return MMG5_interpreg_ani(mesh,met,&ptt,iloc,s,mr);
179  }
180  else if ( pxt->ftag[ifa1] & MG_BDY ) {
181  MMG5_tet2tri( mesh,k,ifa1,&ptt);
182  iloc = MMG5_iarfinv[ifa1][i];
183  assert(iloc >= 0);
184  return MMG5_interpreg_ani(mesh,met,&ptt,iloc,s,mr);
185  }
186 
187  /* if we pass here, then i is a boundary edge but the tet has no bdy
188  * face. Don't do anything, the edge will be split via a boundary tetra. */
189  return -1;
190 }

203 static inline int
204 MMG5_intregvolmet(double *ma,double *mb,double *mp,double t) {
205  double dma[6],dmb[6],mai[6],mbi[6],mi[6];
206  int i;
207  static int8_t mmgWarn=0;
208 
209  for (i=0; i<6; i++) {
210  dma[i] = ma[i];
211  dmb[i] = mb[i];
212  }
213  if ( !MMG5_invmat(dma,mai) || !MMG5_invmat(dmb,mbi) ) {
214  if ( !mmgWarn ) {
215  mmgWarn = 1;
216  fprintf(stderr,"\n ## Warning: %s: at least 1 invalid metric.\n",__func__);
217  }
218  return 0;
219  }
220  for (i=0; i<6; i++)
221  mi[i] = (1.0-t)*mai[i] + t*mbi[i];
222 
223  if ( !MMG5_invmat(mi,mai) ) {
224  if ( !mmgWarn ) {
225  mmgWarn = 1;
226  fprintf(stderr,"\n ## Warning: %s: at least 1 invalid metric.\n",__func__);
227  }
228  return 0;
229  }
230 
231  for (i=0; i<6; i++) mp[i] = mai[i];
232  return 1;
233 }

249 int MMG5_intvolmet(MMG5_pMesh mesh,MMG5_pSol met,int k,int8_t i,double s,
250  double mr[6]) {
251  MMG5_pTetra pt;
252  MMG5_pPoint pp1,pp2;
253  double m1[6],m2[6];
254  int ip1,ip2,l,ier;
255 
256  pt = &mesh->tetra[k];
257 
258  ip1 = pt->v[MMG5_iare[i][0]];
259  ip2 = pt->v[MMG5_iare[i][1]];
260 
261  pp1 = &mesh->point[ip1];
262  pp2 = &mesh->point[ip2];
263 
264  // build metric at ma and mb points
265  if ( !(MG_SIN(pp1->tag) || (MG_NOM & pp1->tag)) && (pp1->tag & MG_GEO) ) {
266  if (!MMG5_moymet(mesh,met,pt,m1)) return 0;
267  } else {
268  for ( l=0; l<6; ++l )
269  m1[l] = met->m[6*ip1+l];
270  }
271  if ( !(MG_SIN(pp2->tag)|| (MG_NOM & pp2->tag)) && (pp2->tag & MG_GEO) ) {
272  if (!MMG5_moymet(mesh,met,pt,m2)) return 0;
273  } else {
274  for ( l=0; l<6; ++l )
275  m2[l] = met->m[6*ip2+l];
276  }
277 
278  ier = MMG5_intregvolmet(m1,m2,mr,s);
279  if ( mesh->info.ddebug && ( (!ier) || (fabs(mr[5]) < 1e-6) ) ) {
280  fprintf(stderr," ## Error: %s:\n",__func__);
281  fprintf(stderr," pp1 : %d %d \n",
282  MG_SIN(pp1->tag) || (MG_NOM & pp1->tag),pp1->tag & MG_GEO);
283  fprintf(stderr," m1 %e %e %e %e %e %e\n",
284  m1[0],m1[1],m1[2],m1[3],m1[4],m1[5]);
285  fprintf(stderr," pp2 : %d %d \n",
286  MG_SIN(pp2->tag) || (MG_NOM & pp2->tag),pp2->tag & MG_GEO);
287  fprintf(stderr," m2 %e %e %e %e %e %e\n",
288  m2[0],m2[1],m2[2],m2[3],m2[4],m2[5]);
289  fprintf(stderr," mr %e %e %e %e %e %e\n",
290  mr[0],mr[1],mr[2],mr[3],mr[4],mr[5]);
291  return 0;
292  }
293 
294  return 1;
295 }

336 int MMG5_interp4barintern(MMG5_pSol met,int ip,double cb[4],double dm0[6],
337  double dm1[6],double dm2[6],double dm3[6]) {
338  double m0i[6],m1i[6],m2i[6],m3i[6],mi[6];
339  int i;
340  static int8_t mmgWarn=0;
341 
342  if ( !MMG5_invmat(dm0,m0i) || !MMG5_invmat(dm1,m1i) ||
343  !MMG5_invmat(dm2,m2i) || !MMG5_invmat(dm3,m3i) ) {
344  if ( !mmgWarn ) {
345  mmgWarn = 1;
346  fprintf(stderr,"\n ## Warning: %s: at least 1 invalid metric.\n",__func__);
347  }
348  return 0;
349  }
350  for (i=0; i<6; i++)
351  mi[i] = cb[0]*m0i[i] + cb[1]*m1i[i] + cb[2]*m2i[i] + cb[3]*m3i[i];
352 
353  if ( !MMG5_invmat(mi,m0i) ) {
354  if ( !mmgWarn ) {
355  mmgWarn = 1;
356  fprintf(stderr,"\n ## Warning: %s: at least 1 invalid metric.\n",__func__);
357  }
358  return 0;
359  }
360 
361  for (i=0; i<6; i++) met->m[met->size*ip+i] = m0i[i];
362 
363  return 1;
364 }
