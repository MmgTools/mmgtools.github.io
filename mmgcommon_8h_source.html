208static inline
209void * mycalloc(size_t c, size_t s) {
210 char *ptr;
211 ptr = (char *)calloc(c*s+sizeof(size_t),1);
212 if (ptr == NULL)
213 return NULL;
214 else {
215 *((size_t*)ptr)=c*s;
216 ptr+=sizeof(size_t);
217 return (void*)ptr;

234static inline
235void * myrealloc(void * ptr_in, size_t s, size_t oldsize) {
236 char *ptr;
237 char *ptr_in_c = (char*)ptr_in;
238
239 if ( !ptr_in ) {
240 assert ( !oldsize );
241 return mymalloc( s );
242 }
243
244 ptr_in_c -= sizeof(size_t);
245 if (oldsize != *((size_t*)ptr_in_c)) {
246 fprintf(stderr, "myrealloc: Error: freed memory mismatch\n");
247 assert(0);
248 }
249 ptr = (char *)realloc(ptr_in_c, s+sizeof(size_t));
250 if (ptr == NULL)
251 return NULL;
252 else {
253 *((size_t*)ptr)=s;
254 ptr+=sizeof(size_t);
255 return (void*)ptr;
256 }
257}

259static inline
260size_t myfree(void *ptr) {
261 size_t s;
262 char * ptr_c = (char*)ptr;
263
264 if ( !ptr ) return 0;
265
266 ptr_c = ptr_c-sizeof(size_t);
267 s = *((size_t*)ptr_c);
268 free(ptr_c);
269
270 return s;
271}

322 \

334#define MMG5_SAFE_RECALLOC(ptr,prevSize,newSize,type,message,law) do \

336 type* tmp; \

338 \
339 tmp = (type *)myrealloc((ptr),size_to_allocate,(prevSize)*sizeof(type)); \

357#define MMG5_TAB_RECALLOC(mesh,ptr,initSize,wantedGap,type,message,law) do \
358 { \
359 int gap; \
360 \
361 assert ( mesh->memCur < mesh->memMax ); \
362 \
363 gap = (int)(wantedGap * initSize); \
364 if ( !gap ) gap = 1; \
365 \
366 if ( mesh->memMax < mesh->memCur + gap*sizeof(type) ) { \
367 gap = (int)((mesh->memMax-mesh->memCur)/sizeof(type)); \
368 if(gap<1) { \
369 fprintf(stderr," ## Error:"); \
370 fprintf(stderr," unable to allocate %s.\n",message); \
371 fprintf(stderr," ## Check the mesh size or "); \
372 fprintf(stderr,"increase maximal authorized memory with the -m option.\n"); \
373 law; \
374 } \
375 } \
376 \
377 MMG5_ADD_MEM(mesh,gap*sizeof(type),message,law); \
378 MMG5_SAFE_RECALLOC((ptr),initSize+1,initSize+gap+1,type,message,law); \
379 initSize = initSize+gap; \
380 }while(0);
