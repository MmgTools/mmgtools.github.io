54 MMGS_ismaniball(MMG5_pMesh mesh, MMG5_pSol sol, int start, int8_t istart) {
55  MMG5_pTria pt;
56  double v1, v2;
57  int *adja,k,ip1,ip2,end1;
58  int8_t i,i1,smsgn;
59  static int8_t mmgWarn=0;
60 
61  k = start;
62  i = MMG5_inxt2[istart];
63 
64  /* First loop: stop if an external boundary, or a change in signs (or a 0) is
65  met recall that MG_SMGSGN(a,b) = 1 provided a*b >0 */
66  do{
67  adja = &mesh->adja[3*(k-1)+1];
68  k = adja[i] / 3;
69  i1 = adja[i] % 3;
70  i = MMG5_iprv2[i1];
71 
72  if ( k==0 ) break;
73 
74  pt = &mesh->tria[k];
75  ip1 = pt->v[i1];
76  ip2 = pt->v[i];
77 
78  v1 = sol->m[ip1]-mesh->info.ls;
79  v2 = sol->m[ip2]-mesh->info.ls;
80 
81  smsgn = MG_SMSGN(v1,v2) ? 1 : 0;
82  }
83  while ( smsgn && k!=start );
84 
85  assert(k!=start); //unexpected case
86 
87  end1 = k;
88  k = start;
89  i = MMG5_iprv2[istart];
90 
91  /* Second loop: same travel in the opposite sense */
92  do{
93  adja = &mesh->adja[3*(k-1)+1];
94  k = adja[i] / 3;
95  i1 = adja[i] % 3;
96  i = MMG5_inxt2[i1];
97 
98  if ( k==0 ) break;
99 
100  pt = &mesh->tria[k];
101  ip1 = pt->v[i1];
102  ip2 = pt->v[i];
103 
104  v1 = sol->m[ip1]-mesh->info.ls;
105  v2 = sol->m[ip2]-mesh->info.ls;
106 
107  smsgn = MG_SMSGN(v1,v2) ? 1 : 0;
108  }
109  while ( smsgn && k!=start );
110 
111  assert(k!=start); //unexpected case
112 
113  /* If first stop was due to an external boundary, the second one must too
114  (end1 = k = 0);
115  else, the final triangle for the first travel must be that
116  of the second one */
117  if ( k != end1 ) {
118  if ( !mmgWarn ) {
119  mmgWarn = 1;
120  fprintf(stderr,"\n ## Warning: %s: triangle %d, point %d; unsnap \n",
121  __func__,start,mesh->tria[start].v[istart]);
122  }
123  return 0;
124  }
125  return 1;
126 }

139 int MMGS_snpval_ls(MMG5_pMesh mesh,MMG5_pSol sol) {
140  MMG5_pTria pt;
141  MMG5_pPoint p0;
142  double *tmp,v1,v2;
143  int k,nc,ns,ip,ip1,ip2;
144  int8_t i;
145 
146  MMG5_ADD_MEM(mesh,(mesh->npmax+1)*sizeof(double),"temporary table",
147  fprintf(stderr," Exit program.\n");
148  return 0);
149  MMG5_SAFE_CALLOC(tmp,mesh->npmax+1,double,return 0);
150 
151  /* create tetra adjacency */
152  if ( !MMGS_hashTria(mesh) ) {
153  fprintf(stderr,"\n ## Error: %s: hashing problem (1). Exit program.\n",
154  __func__);
155  return 0;
156  }
157 
158  /* Reset point flags */
159  for (k=1; k<=mesh->np; k++)
160  mesh->point[k].flag = 0;
161 
162  /* Snap values of sol that are close to 0 to 0 exactly */
163  ns = nc = 0;
164  for (k=1; k<=mesh->np; k++) {
165  p0 = &mesh->point[k];
166  if ( !MG_VOK(p0) ) continue;
167  if ( fabs(sol->m[k]-mesh->info.ls) < MMG5_EPS ) {
168  if ( mesh->info.ddebug )
169  fprintf(stderr," ## Warning: %s: snapping value %d; "
170  "previous value : %E\n",__func__,k,fabs(sol->m[k]));
171 
172  tmp[k] = ( fabs(sol->m[k]-mesh->info.ls) < MMG5_EPSD ) ?
173  (mesh->info.ls-100.0*MMG5_EPS) : sol->m[k];
174  p0->flag = 1;
175  sol->m[k] = mesh->info.ls;
176  ns++;
177  }
178  }
179 
180  /* Check that the snapping process has not led to a nonmanifold situation */
181  for (k=1; k<=mesh->nt; k++) {
182  pt = &mesh->tria[k];
183  if ( !MG_EOK(pt) ) continue;
184  for (i=0; i<3; i++) {
185  ip = pt->v[i];
186  ip1 = pt->v[MMG5_inxt2[i]];
187  ip2 = pt->v[MMG5_iprv2[i]];
188 
189  p0 = &mesh->point[ip];
190  v1 = sol->m[ip1]-mesh->info.ls;
191  v2 = sol->m[ip2]-mesh->info.ls;
192  if ( p0->flag && !(MG_SMSGN(v1,v2)) ) {
193  if ( !MMGS_ismaniball(mesh,sol,k,i) ) {
194  sol->m[ip] = tmp[ip];
195  nc++;
196  }
197  p0->flag = 0;
198  tmp[ip] = mesh->info.ls;
199  }
200  }
201  }
202  if ( (abs(mesh->info.imprim) > 5 || mesh->info.ddebug) && ns+nc > 0 )
203  fprintf(stdout," %8d points snapped, %d corrected\n",ns,nc);
204 
205  /* memory free */
206  MMG5_DEL_MEM(mesh,mesh->adja);
207  MMG5_DEL_MEM(mesh,tmp);
208 
209  return 1;
210 }

235 #endif

281 
282  assert(k!=start); //unexpected case
283 
284  return !k;
285  }
286

288  i = MMG5_inxt2[i];

292

295

298  i = MMG5_inxt2[i];
299  }

301 
302  /* At least 3 boundary segments meeting at p */
303  if ( k != start )
304  return 0;
305 
306  return 1;

317 int MMGS_chkmanimesh(MMG5_pMesh mesh) {
318  MMG5_pTria pt;
319  int *adja,k,cnt,iel;
320  int8_t i,i1;
321  static int8_t mmgWarn0 = 0;
322 
323 
324  /* First check: check whether one triangle in the mesh has 3 boundary faces */
325  for (k=1; k<=mesh->nt; k++) {
326  pt = &mesh->tria[k];
327  if ( !MG_EOK(pt) ) continue;
328 
329  cnt = 0;
330  adja = &mesh->adja[3*(k-1)+1];
331  for (i=0; i<3; i++) {
332  iel = adja[i] / 3;
333 
334  if (!iel ) {
335  cnt++;
336  continue;
337  }
338  else {
339  if ( pt->edg[i] == MG_ISO ) cnt++;
340  }
341  }
342  if( cnt == 3 ) {
343  if ( !mmgWarn0 ) {
344  mmgWarn0 = 1;
345  fprintf(stderr,"\n ## Warning: %s: at least 1 triangle with 3 boundary"
346  " edges.\n",__func__);
347  }
348  return 0;
349  }
350  }
351 
352  /* Second check: check whether the configuration is manifold in the ball of
353  each point; each vertex on the implicit boundary is caught in such a way
354  that i1 inxt[i1] is one edge of the implicit boundary */
355  for (k=1; k<=mesh->nt; k++) {
356  pt = &mesh->tria[k];
357  if ( !MG_EOK(pt) ) continue;
358 
359  for (i=0; i<3; i++) {
360  adja = &mesh->adja[3*(k-1)+1];
361  iel = adja[i] / 3;
362 
363  if ( (!iel) || (pt->edg[i] != MG_ISO) ) continue;
364 
365  i1 = MMG5_inxt2[i];
366  if ( !MMGS_chkmaniball(mesh,k,i1) )
367  return 0;
368  }
369  }
370 
371  if ( mesh->info.imprim > 0 || mesh->info.ddebug )
372  fprintf(stdout," *** Manifold implicit surface.\n");
373 
374  return 1;

392  int vx[3],nb,k,ip0,ip1,np,ns,nt,ier;
393  int8_t ia;
394  /* reset point flags and h */
395  for (k=1; k<=mesh->np; k++)
396  mesh->point[k].flag = 0;
397 
398  /* compute the number nb of intersection points on edges */
399  nb = 0;
400  for (k=1; k<=mesh->nt; k++) {
401  pt = &mesh->tria[k];
402  for (ia=0; ia<3; ia++) {
403  ip0 = pt->v[MMG5_inxt2[ia]];
404  ip1 = pt->v[MMG5_iprv2[ia]];
405  p0 = &mesh->point[ip0];
406  p1 = &mesh->point[ip1];
407  if ( p0->flag && p1->flag ) continue;
408  v0 = sol->m[ip0]-mesh->info.ls;
409  v1 = sol->m[ip1]-mesh->info.ls;
410  if ( fabs(v0) > MMG5_EPSD2 && fabs(v1) > MMG5_EPSD2 && v0*v1 < 0.0 ) {
411  if ( !p0->flag ) {
412  p0->flag = ++nb;
413  }
414  if ( !p1->flag ) {
415  p1->flag = ++nb;
416  }
417  }
418  }
419  }
420  if ( ! nb ) return 1;
421 
422  /* Create intersection points at 0 isovalue and set flags to trias */
423  if ( !MMG5_hashNew(mesh,&hash,nb,3*nb) ) return 0;
424  for (k=1; k<=mesh->nt; k++) {
425  pt = &mesh->tria[k];
426  if ( !MG_EOK(pt) ) continue;
427 
428  for (ia=0; ia<3; ia++) {
429  ip0 = pt->v[MMG5_inxt2[ia]];
430  ip1 = pt->v[MMG5_iprv2[ia]];
431  np = MMG5_hashGet(&hash,ip0,ip1);
432  if ( np ) continue;
433 
434  p0 = &mesh->point[ip0];
435  p1 = &mesh->point[ip1];
436  v0 = sol->m[ip0]-mesh->info.ls;
437  v1 = sol->m[ip1]-mesh->info.ls;
438  if ( fabs(v0) < MMG5_EPSD2 || fabs(v1) < MMG5_EPSD2 ) continue;
439  else if ( MG_SMSGN(v0,v1) ) continue;
440  else if ( !p0->flag || !p1->flag ) continue;
441 
442  s = v0 / (v0-v1);
443 
444  s = MG_MAX(MG_MIN(s,1.0-MMG5_EPS),MMG5_EPS);
445  c[0] = p0->c[0] + s*(p1->c[0]-p0->c[0]);
446  c[1] = p0->c[1] + s*(p1->c[1]-p0->c[1]);
447  c[2] = p0->c[2] + s*(p1->c[2]-p0->c[2]);
448 
449  np = MMGS_newPt(mesh,c,NULL);
450  if ( !np ) {
451  MMGS_POINT_REALLOC(mesh,sol,np,0.2,
452  fprintf(stderr,"\n ## Error: %s: unable to"
453  " allocate a new point\n",__func__);
454  MMG5_INCREASE_MEM_MESSAGE();
455  return 0
456  ,c,NULL);
457  }
458  sol->m[np] = mesh->info.ls;
459 
460  /* If user provide a metric, interpolate it at the new point */
461  if ( met && met->m ) {
462  if ( met->size > 1 ) {
463  ier = MMGS_intmet33_ani(mesh,met,k,ia,np,s);
464  }
465  else {
466  ier = intmet_iso(mesh,met,k,ia,np,s);
467  }
468  if ( ier <= 0 ) {
469  // Unable to compute the metric
470  fprintf(stderr,"\n ## Error: %s: unable to"
471  " interpolate the metric during the level-set"
472  " discretization\n",__func__);
473  return 0;
474  }
475  }
476 
477  MMG5_hashEdge(mesh,&hash,ip0,ip1,np);
478  }
479  }
480 
481  /* Proceed to splitting, according to flags to tris */
482  nt = mesh->nt;
483  ns = 0;
484  ier = 1;
485  for (k=1; k<=nt; k++) {
486  pt = &mesh->tria[k];
487  if ( !MG_EOK(pt) ) continue;
488  pt->flag = 0;
489  memset(vx,0,3*sizeof(int));
490  for (ia=0; ia<3; ia++) {
491  vx[ia] = MMG5_hashGet(&hash,pt->v[MMG5_inxt2[ia]],pt->v[MMG5_iprv2[ia]]);
492  if ( vx[ia] ) {
493  MG_SET(pt->flag,ia);
494  }
495  }
496  switch (pt->flag) {
497  case 1: /* 1 edge split */
498  ier = MMGS_split1(mesh,met,k,0,vx);
499  ns++;
500  break;
501 
502  case 2: /* 1 edge split */
503  ier = MMGS_split1(mesh,met,k,1,vx);
504  ns++;
505  break;
506 
507  case 4: /* 1 edge split */
508  ier = MMGS_split1(mesh,met,k,2,vx);
509  ns++;
510  break;
511 
512  case 3: case 5: case 6: /* 2 edges split */
513  ier = MMGS_split2(mesh,met,k,vx);
514  ns++;
515  break;
516 
517  case 7: /* 3 edges splitted */
518  ier =MMGS_split3(mesh,met,k,vx);
519  ns++;
520  break;
521 
522  default :
523  assert(pt->flag == 0);
524  break;
525  }
526  if ( !ier ) return 0;
527  }
528  if ( (mesh->info.ddebug || abs(mesh->info.imprim) > 5) && ns > 0 )
529  fprintf(stdout," %7d splitted\n",ns);
530 
531  /* reset point flags */
532  for (k=1; k<=mesh->np; k++)
533  mesh->point[k].flag = 0;
534 
535  MMG5_DEL_MEM(mesh,hash.item);
536  return ns;
537 }
