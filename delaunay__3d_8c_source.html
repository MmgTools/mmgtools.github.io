54 /* hash mesh edge v[0],v[1] (face i of iel) */
55 int MMG5_hashEdgeDelone(MMG5_pMesh mesh,MMG5_Hash *hash,int iel,int i,int *v) {
56  int *adja,iadr,jel,j,key,mins,maxs;
57  MMG5_hedge *ha;
58 
59  /* compute key */
60  if ( v[0] < v[1] ) {
61  mins = v[0];
62  maxs = v[1];
63  }
64  else {
65  mins = v[1];
66  maxs = v[0];
67  }
68  key = MMG3D_KTA*mins + MMG3D_KTB*maxs;
69  key = key % hash->siz;
70  ha = &hash->item[key];
71 
72  if ( ha->a ) {
73  /* identical face */
74  if ( ha->a == mins && ha->b == maxs ) {
75  iadr = (iel-1)*4 + 1;
76  adja = &mesh->adja[iadr];
77  adja[i] = ha->k;
78 
79  jel = ha->k >> 2;
80  j = ha->k % 4;
81  iadr = (jel-1)*4 + 1;
82  adja = &mesh->adja[iadr];
83  adja[j] = iel*4 + i;
84  return 1;
85  }
86  else {
87  while ( ha->nxt && ha->nxt < hash->max ) {
88  ha = &hash->item[ha->nxt];
89  if ( ha->a == mins && ha->b == maxs ) {
90  iadr = (iel-1)*4 + 1;
91  adja = &mesh->adja[iadr];
92  adja[i] = ha->k;
93 
94  jel = ha->k >> 2;
95  j = ha->k % 4;
96  iadr = (jel-1)*4 + 1;
97  adja = &mesh->adja[iadr];
98  adja[j] = iel*4 + i;
99  return 1;
100  }
101  }
102  }
103  ha->nxt = hash->nxt;
104  ha = &hash->item[hash->nxt];
105  ha->a = mins;
106  ha->b = maxs;
107  ha->k = iel*4 + i;
108  hash->nxt = ha->nxt;
109  ha->nxt = 0;
110 
111  if ( hash->nxt >= hash->max ) {
112  MMG5_TAB_RECALLOC(mesh,hash->item,hash->max,MMG5_GAP,MMG5_hedge,"face",
113  return 0;);
114  for (j=hash->nxt; j<hash->max; j++) hash->item[j].nxt = j+1;
115  }
116  return 1;
117  }
118 
119  /* insert */
120  ha->a = mins;
121  ha->b = maxs;
122  ha->k = iel*4 + i;
123  ha->nxt = 0;
124 
125  return 1;
126 }

351 static int MMG5_correction_ani(MMG5_pMesh mesh,MMG5_pSol met,int ip,int* list,
352  int ilist,int nedep,double volmin) {
353  MMG5_pPoint ppt,p1,p2,p3;
354  MMG5_pTetra pt;
355  double dd,det,nn,eps,eps2,ux,uy,uz,vx,vy,vz,v1,v2,v3;
356  double *ma,*mb,*mc,*md,mm[6],h1,h2,h3;
357  int *adja,i,j,ipil,iel,lon,iadr,adj,ib,ic,id,base,ncor;
358  int vois[4];
359 
360  ppt = &mesh->point[ip];
361  if ( ppt->tag & MG_NUL ) return ilist;
362  base = mesh->base;
363  lon = ilist;
364  eps = MMG3D_EPSCON;
365  eps2 = eps*eps;
366 
367  /* average metric */
368  memset(mm,0,6*sizeof(double));
369  ma = &met->m[6*ip];
370 
371  do {
372  ipil = lon-1;
373  ncor = 0;
374 
375  while ( ipil >= 0 ) {
376  iel = list[ipil];
377  iadr = (iel-1)*4 + 1;
378  adja = &mesh->adja[iadr];
379  vois[0] = adja[0] >> 2;
380  vois[1] = adja[1] >> 2;
381  vois[2] = adja[2] >> 2;
382  vois[3] = adja[3] >> 2;
383  pt = &mesh->tetra[iel];
384 
385  // MMG_cas=0; // uncomment to debug
386  for (i=0; i<4; i++) {
387  adj = vois[i];
388  // MMG_cas = 0;// uncomment to debug
389  if ( adj && mesh->tetra[adj].flag == base) continue;
390 
391  ib = pt->v[ MMG5_idir[i][0] ];
392  ic = pt->v[ MMG5_idir[i][1] ];
393  id = pt->v[ MMG5_idir[i][2] ];
394 
395  p1 = &mesh->point[ib];
396  p2 = &mesh->point[ic];
397  p3 = &mesh->point[id];
398 
399  ux = p2->c[0] - p1->c[0];
400  uy = p2->c[1] - p1->c[1];
401  uz = p2->c[2] - p1->c[2];
402 
403  vx = p3->c[0] - p1->c[0];
404  vy = p3->c[1] - p1->c[1];
405  vz = p3->c[2] - p1->c[2];
406 
407  /* volume PABC */
408  v1 = uz*vy - uy*vz;
409  v2 = ux*vz - uz*vx;
410  v3 = uy*vx - ux*vy;
411  dd = v1*(ppt->c[0]-p1->c[0]) + v2*(ppt->c[1]-p1->c[1]) \
412  + v3*(ppt->c[2]-p1->c[2]);
413  // MMG_cas=1; // uncomment to debug
414 
415  /*test sur le volume avec un eps local*/
416  h1 = ux*ux + uy*uy + uz*uz;
417  h2 = vx*vx + vy*vy + vz*vz;
418  h3 = (p2->c[0] - p3->c[0])*(p2->c[0] - p3->c[0]) + (p2->c[1] - p3->c[1])*(p2->c[1] - p3->c[1])
419  + (p2->c[2] - p3->c[2])*(p2->c[2] - p3->c[2]);
420  if ( dd < volmin*sqrt(h1*h2*h3) ) break;
421 
422  /* average metric */
423  mb = &met->m[6*ib];
424  mc = &met->m[6*ic];
425  md = &met->m[6*id];
426  for (j=0; j<6; j++)
427  mm[j] = 0.25 * (ma[j]+mb[j]+mc[j]+md[j]);
428 
429  det = mm[0] * ( mm[3]*mm[5] - mm[4]*mm[4]) \
430  - mm[1] * ( mm[1]*mm[5] - mm[2]*mm[4]) \
431  + mm[2] * ( mm[1]*mm[4] - mm[2]*mm[3]);
432  if ( det < MMG5_EPSOK ) break;
433 
434  /* point close to face */
435  // MMG_cas=2; // uncomment to debug
436  nn = mm[0]*v1*v1 + mm[3]*v2*v2 + mm[5]*v3*v3 \
437  + 2.0*(mm[1]*v1*v2 + mm[2]*v1*v3 + mm[4]*v2*v3);
438 
439  if ( det*dd*dd < nn * eps2 ) break;
440  // MMG_cas=0; // uncomment to debug
441  }
442  if ( i < 4 || pt->tag & MG_REQ ) {
443  if ( ipil <= nedep ) {
444  return 0;
445  }
446  /* remove iel from list */
447  pt->flag = base-1;
448  list[ipil] = list[--lon];
449  ncor = 1;
450  break;
451  }
452  else
453  ipil--;
454  }
455  }
456  while ( ncor > 0 && lon >= nedep );
457 
458  return lon;
459 }

694  }

699  list[ilist++] = adj;
700  }
