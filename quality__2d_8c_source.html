59 double MMG2D_caltri_iso(MMG5_pMesh mesh,MMG5_pSol met,MMG5_pTria pt) {
60  double abx,aby,acx,acy,bcx,bcy;
61  double *a,*b,*c,h1,h2,h3,area,hm;
62 
63  a = mesh->point[pt->v[0]].c;
64  b = mesh->point[pt->v[1]].c;
65  c = mesh->point[pt->v[2]].c;
66 
67  abx = b[0] - a[0];
68  aby = b[1] - a[1];
69  acx = c[0] - a[0];
70  acy = c[1] - a[1];
71  bcx = c[0] - b[0];
72  bcy = c[1] - b[1];
73 
74  /* orientation */
75  area = abx*acy - aby*acx;
76  if ( area <= 0.0 ) return 0.0;
77 
78  /* edge lengths */
79  h1 = abx*abx + aby*aby;
80  h2 = acx*acx + acy*acy;
81  h3 = bcx*bcx + bcy*bcy;
82 
83  hm = h1 + h2 + h3;
84 
85  if ( hm > 0. ) {
86  return area / hm;
87  }
88  else {
89  return 0.0;
90  }
91 }

94  return 0 in the case that the triangle has inverted orientation */
95 double MMG2D_caltri_ani(MMG5_pMesh mesh,MMG5_pSol met,MMG5_pTria pt) {
96  double abx,aby,acx,acy,bcx,bcy;
97  double *a,*b,*c,*ma,*mb,*mc;
98  double area,aream,hm,m[6],h1,h2,h3;
99  int ipa,ipb,ipc,i;
100 
101  ipa = pt->v[0];
102  ipb = pt->v[1];
103  ipc = pt->v[2];
104 
105  a = mesh->point[ipa].c;
106  b = mesh->point[ipb].c;
107  c = mesh->point[ipc].c;
108 
109  ma = &met->m[3*ipa];
110  mb = &met->m[3*ipb];
111  mc = &met->m[3*ipc];
112 
113  /* Isotropic area of pt */
114  abx = b[0] - a[0];
115  aby = b[1] - a[1];
116  acx = c[0] - a[0];
117  acy = c[1] - a[1];
118  bcx = c[0] - b[0];
119  bcy = c[1] - b[1];
120  area = abx*acy - aby*acx;
121  if ( area <= 0.0 ) return 0.0;
122 
123  for (i=0; i<3; i++) m[i] = (ma[i]+mb[i]+mc[i]) / 3.0;
124 
125  /* Anisotropic edge lengths */
126  h1 = m[0]*abx*abx + m[2]*aby*aby + 2.0*m[1]*abx*aby;
127  h1 = h1 > 0.0 ? sqrt(h1) : 0.0;
128  h2 = m[0]*acx*acx + m[2]*acy*acy + 2.0*m[1]*acx*acy;
129  h2 = h2 > 0.0 ? sqrt(h2) : 0.0;
130  h3 = m[0]*bcx*bcx + m[2]*bcy*bcy + 2.0*m[1]*bcx*bcy;
131  h3 = h3 > 0.0 ? sqrt(h3) : 0.0;
132 
133  hm = h1*h1 + h2*h2 + h3*h3;
134 
135  /* Anisotropic volume of pt */
136  aream = sqrt(m[0]*m[2]-m[1]*m[1])*area;
137 
138  /* Quality measure = (Vol_M(T) / (l(ab)^2+l(ac)^2+l(bc)^2)) */
139  if ( hm > 0. ) {
140  return aream/hm;
141  }
142  else {
143  return (0.0);
144  }
145 }
