126  int k,iel,idp,ifac,isloc,init_s;

132  p0 = &mesh->point[idp];
133 
134  /* local parameters at vertex: useless for now because new points are created
135  * without reference (inside the domain) */

137  isqhmin = mesh->info.hmin;
138  isqhmax = mesh->info.hmax;
139  isloc = 0;
140 
141  if ( mesh->adja[4*(kel-1)+iface+1] ) return 0;
142  ilist = MMG5_boulesurfvolp(mesh,kel,ip,iface,
143  listv,&ilistv,lists,&ilists,(p0->tag & MG_NOM));
144 
145  if ( ilist!=1 ) {
146  if ( !mmgWarn ) {
147  fprintf(stderr,"\n ## Warning: %s: at least 1 metric not computed:"
148  " unable to compute the ball of point\n",__func__);
149  mmgWarn = 1;
150  }
151  return 0;
152  }
153 
154 
155  /* travel across the ball of ip to find the minimal local params imposed on
156  * tetras */
157  if ( mesh->info.parTyp & MG_Tetra ) {
158  i = 0;
159  do

161  if ( isloc ) break;
162 
163  par = &mesh->info.par[i];

171  isqhmax = par->hmax;
172  isloc = 1;
173  break;
174  }
175  }
176  } while ( ++i<mesh->info.npar );
177 
178  for ( ; i<mesh->info.npar; ++i ) {
179  par = &mesh->info.par[i];
180  if ( par->elt != MMG5_Tetrahedron ) continue;
181 
182  for ( k=0; k<ilistv; ++k ) {
183  pt = &mesh->tetra[listv[k]/4];
184  if ( par->ref == pt->ref ) {
185  hausd_v = MG_MIN(hausd_v,par->hausd);
186  isqhmin = MG_MAX(isqhmin,par->hmin);
187  isqhmax = MG_MIN(isqhmax,par->hmax);
188  break;
189  }
190  }
191  }
192  }
193 
194 
195  maxkappa = 0.0;
196  init_s = 0;
197 
198  for (k=0; k<ilists; k++) {
199  iel = lists[k] / 4;
200  ifac = lists[k] % 4;
201  pt = &mesh->tetra[iel];
202  assert(pt->xt && (mesh->xtetra[pt->xt].ftag[ifac] & MG_BDY) );
203  pxt = &mesh->xtetra[pt->xt];
204 
205  for ( i = 0; i < 3; i++ ) {
206  if ( pt->v[MMG5_idir[ifac][i]] == idp ) break;
207  }
208  assert(i<3);
209 
210  i0 = MMG5_idir[ifac][i];
211  i1 = MMG5_idir[ifac][MMG5_inxt2[i]];
212  i = MMG5_iprv2[i];
213  i2 = MMG5_idir[ifac][i];
214 
215  /* Computation of the two control points associated to edge p0p1 with
216  * p1=mesh->point[pt->v[i1]]: p0 is singular */
217  MMG5_norpts(mesh,pt->v[i0],pt->v[i1],pt->v[i2],n);

220  MG_EDG(pxt->tag[MMG5_iarf[ifac][i]]),n);

223  tau[0] = 3.0*(b0[0] - p0->c[0]);
224  tau[1] = 3.0*(b0[1] - p0->c[1]);
225  tau[2] = 3.0*(b0[2] - p0->c[2]);

230  gammasec[1] = 6.0*p0->c[1] - 12.0*b0[1] + 6.0*b1[1];
231  gammasec[2] = 6.0*p0->c[2] - 12.0*b0[2] + 6.0*b1[2];
232  if ( ntau2 < MMG5_EPSD ) continue;
233  ntau2 = 1.0 / ntau2;
234 
235  /* derivative via the normal parametrization */
236  ps1 = gammasec[0]*tau[0] + gammasec[1]*tau[1] + gammasec[2]*tau[2];

239  c[2] = gammasec[2] - ps1*tau[2]*ntau2;
240

242 
243  /* local parameters for triangle */
244  hausd = hausd_v;
245 
246  if ( mesh->info.parTyp & MG_Tria ) {
247  if ( !isloc ) {
248  for ( i=0; i<mesh->info.npar; ++i ) {
249  par = &mesh->info.par[i];
250 
251  if ( (par->elt != MMG5_Triangle) || (pxt->ref[ifac] != par->ref ) )
252  continue;
253 
254  hausd = par->hausd;
255  if ( !init_s ) {
256  isqhmin = par->hmin;
257  isqhmax = par->hmax;
258  init_s = 1;
259  }
260  else {
261  isqhmin = MG_MAX(par->hmin,isqhmin);
262  isqhmax = MG_MIN(par->hmax,isqhmax);
263  }
264  break;
265  }
266  }
267  else {
268  for ( i=0 ; i<mesh->info.npar; ++i) {
269  par = &mesh->info.par[i];
270 
271  if ( (par->elt != MMG5_Triangle) || (pxt->ref[ifac] != par->ref ) )
272  continue;
273 
274  hausd = MG_MIN(par->hausd,hausd);
275  isqhmin = MG_MAX(par->hmin,isqhmin);
276  isqhmax = MG_MIN(par->hmax,isqhmax);
277  break;
278  }
279  }
280  }
281  maxkappa = MG_MAX(kappa/hausd,maxkappa);
282  }
283 
284  isqhmin = 1.0 / (isqhmin*isqhmin);
285  isqhmax = 1.0 / (isqhmax*isqhmax);
286 
287  alpha = 1.0 / 8.0 * maxkappa;
288  alpha = MG_MIN(alpha,isqhmin);
289  alpha = MG_MAX(alpha,isqhmax);
290 
291  m = &met->m[6*idp];
292  memset(m,0,6*sizeof(double));
293  m[0] = m[3] = m[5] = alpha;
294 
295  return 1;
296 }

336  pxt = &mesh->xtetra[pt->xt];
337

340  isqhmax = mesh->info.hmax;
341 
342  if ( mesh->info.parTyp ) {
343  isloc = 0;
344 
345  i = 0;
346  do

348  if ( isloc ) break;
349 
350  par = &mesh->info.par[i];
351  if ( ( (par->elt != MMG5_Triangle) || (pxt->ref[iface] != par->ref ) ) &&
352  ( (par->elt != MMG5_Tetrahedron) || (pt->ref != par->ref ) ) )

356  isqhmax = par->hmax;
357  isloc = 1;
358  } while ( ++i<mesh->info.npar );
359 
360  for ( ; i<mesh->info.npar; ++i) {
361  par = &mesh->info.par[i];
362 
363  if ( ( (par->elt != MMG5_Triangle) || (pxt->ref[iface] != par->ref ) ) &&
364  ( (par->elt != MMG5_Tetrahedron) || (pt->ref != par->ref ) ) )
365  continue;
366 
367  isqhmin = MG_MAX(isqhmin,par->hmin);
368  isqhmax = MG_MIN(isqhmax,par->hmax);
369  }
370  }
371 
372  isqhmin = 1.0 / (isqhmin*isqhmin);
373  isqhmax = 1.0 / (isqhmax*isqhmax);
374 
375  n1 = &mesh->xpoint[p0->xp].n1[0];
376  n2 = &mesh->xpoint[p0->xp].n2[0];
377  t = p0->n;
378 
379  m = &met->m[6*idp];
380  memset(m,0,6*sizeof(double));
381  m[0] = isqhmax;
382  m[1] = isqhmax;
383  m[2] = isqhmax;
384  m[3] = isqhmax;
385  m[4] = isqhmax;

388  ier = MMG5_bouletrid(mesh,kel,iface,ip,&ilist1,list1,&ilist2,list2,
389  &iprid[0],&iprid[1] );

392  fprintf(stderr,"\n ## Warning: %s: at least 1 metric not computed:"
393  " unable to compute the ball of point\n",__func__);
394  mmgWarn = 1;
395  }
396  }
397

399  m[0] = MG_MAX(m[0],MMG5_ridSizeInTangentDir(mesh,p0,idp,iprid,isqhmin,isqhmax));
400 
401  /* Characteristic sizes in directions u1 and u2 */
402  for (i=0; i<2; i++) {
403  if ( i==0 ) {
404  n = n1;
405  ilist = ilist1;
406  list = &list1[0];

414

416  i1 = 0;
417  ifac = -1; // Remove uninitialized warning
418  for (k=0; k<ilist; k++) {
419  iel = list[k] / 4;
420  ifac = list[k] % 4;
421  pt = &mesh->tetra[iel];
422  for ( i0=0; i0!=3; ++i0 ) {

424  }

428 
429  ux = p1->c[0] - p0->c[0];
430  uy = p1->c[1] - p0->c[1];
431  uz = p1->c[2] - p0->c[2];
432 
433  lispoi[3*k+1] = r[0][0]*ux + r[0][1]*uy + r[0][2]*uz;
434  lispoi[3*k+2] = r[1][0]*ux + r[1][1]*uy + r[1][2]*uz;
435  lispoi[3*k+3] = r[2][0]*ux + r[2][1]*uy + r[2][2]*uz;
436  }
437 
438  /* last point : the half-ball is open : ilist tria, and ilist +1 points ;
439  lists are enumerated in direct order */
440  i2 = MMG5_inxt2[i1];
441  p2 = &mesh->point[pt->v[MMG5_idir[ifac][i2]]];
442 
443  ux = p2->c[0] - p0->c[0];
444  uy = p2->c[1] - p0->c[1];
445  uz = p2->c[2] - p0->c[2];
446 
447  lispoi[3*ilist+1] = r[0][0]*ux + r[0][1]*uy + r[0][2]*uz;
448  lispoi[3*ilist+2] = r[1][0]*ux + r[1][1]*uy + r[1][2]*uz;
449  lispoi[3*ilist+3] = r[2][0]*ux + r[2][1]*uy + r[2][2]*uz;
450 
451  /* At this point, lispoi contains all the points of the half-ball of p0, rotated
452  so that t_{p_0}S = [z = 0] */
453 
454  /* Rotated tangent vector (trot[2] = 0), and third direction */
455  trot[0] = r[0][0]*t[0] + r[0][1]*t[1] + r[0][2]*t[2];
456  trot[1] = r[1][0]*t[0] + r[1][1]*t[1] + r[1][2]*t[2];
457 
458  u[0] = -trot[1];
459  u[1] = trot[0];
460 
461  /* Travel half-ball at p0 and stop at first triangle containing u */
462  for (k=0; k<ilist; k++) {
463  detg = lispoi[3*k+1]*u[1] - lispoi[3*k+2]*u[0];
464  detd = u[0]*lispoi[3*(k+1)+2] - u[1]*lispoi[3*(k+1)+1];
465  if ( detg > 0.0 && detd > 0.0 ) break;
466  }
467 
468  /* If triangle not found, try with -u */
469  if ( k == ilist ) {
470  u[0] *= -1.0;
471  u[1] *= -1.0;
472 
473  for (k=0; k<ilist; k++) {
474  detg = lispoi[3*k+1]*u[1] - lispoi[3*k+2]*u[0];
475  detd = u[0]*lispoi[3*(k+1)+2] - u[1]*lispoi[3*(k+1)+1];
476  if ( detg > 0.0 && detd > 0.0 ) break;
477  }
478  }
479  if ( k == ilist ) continue;
480 
481  iel = list[k] / 4;
482  ifac = list[k] % 4;
483  pt = &mesh->tetra[iel];
484  for ( i0=0; i0!=3; ++i0 ) {
485  if ( pt->v[MMG5_idir[ifac][i0]]==idp ) break;
486  }
487  assert(i0!=3);
488 
489  MMG5_tet2tri(mesh,iel,ifac,&ptt);
490  assert(pt->xt);
491  pxt = &mesh->xtetra[pt->xt];
492  if ( !MMG5_bezierCP(mesh,&ptt,&b,MG_GET(pxt->ori,i)) ) continue;
493 
494  /* Barycentric coordinates of vector u in tria iel */
495  detg = lispoi[3*k+1]*u[1] - lispoi[3*k+2]*u[0];
496  detd = u[0]*lispoi[3*(k+1)+2] - u[1]*lispoi[3*(k+1)+1];
497  det = detg + detd;
498  if ( det < MMG5_EPSD ) continue;
499 
500  det = 1.0 / det;
501  bcu[0] = 0.0;
502  bcu[1] = u[0]*lispoi[3*(k+1)+2] - u[1]*lispoi[3*(k+1)+1];
503  bcu[1] *= det;
504  assert(bcu[1] <= 1.0);
505  bcu[2] = 1.0 - bcu[1];
506 
507  /* Computation of tangent vector and second derivative of curve t \mapsto
508  b(tbcu) (not in rotated frame) */
509  m[i+1] = MG_MAX(m[i+1],
510  MMG5_ridSizeInNormalDir(mesh,i0,bcu,&b,isqhmin,isqhmax));
511  }
512 
513  return 1;
514 }

1400  return -1;

1404 
1405  if ( (!( MG_SIN(p2->tag) || (p2->tag & MG_NOM) )) && p2->tag & MG_GEO ) {

1407  if( !MMG5_buildridmet(mesh,met,np2,ux,uy,uz,m2,rbasis2) )
1408  return -1;
1409  }

1416  t[1] = uy/l;
1417  t[2] = uz/l;

1420  ps1 = m1[0]*t[0]*t[0] + 2.0*m1[1]*t[0]*t[1] + m1[3]*t[1]*t[1]
1421  + 2.0*m1[2]*t[0]*t[2] + 2.0*m1[4]*t[1]*t[2] + m1[5]*t[2]*t[2];

1433  if( ps1 >= alpha -MMG5_EPS )
1434  return -1;

1633  " simultaneous reduction.\n",__func__);

1635  return 0;
1636  }
1637 
1638  if ( order == 3 ) {
1639  /* First case : matrices m and n are homothetic: n = lambda0*m */
1640  if ( (fabs(m[1]) < MMG5_EPS && fabs(m[2]) < MMG5_EPS

1663  dn[2] = lambda[0]*dm[2];
