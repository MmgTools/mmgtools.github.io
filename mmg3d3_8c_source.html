44 double MMG5_estavglen(MMG5_pMesh mesh) {
45  MMG5_pTetra pt;
46  MMG5_pPoint p1,p2;
47  int k,na;
48  double len,lent,dna;
49  int8_t i,i1,i2;
50 
51  na = 0;
52  lent = 0.0;
53 
54  for (k=1; k<=mesh->ne; k++) {
55  pt = &mesh->tetra[k];
56  for (i=0; i<6; i++) {
57  i1 = MMG5_iare[i][0];
58  i2 = MMG5_iare[i][1];
59 
60  p1 = &mesh->point[pt->v[i1]];
61  p2 = &mesh->point[pt->v[i2]];
62 
63  len = (p2->c[0] - p1->c[0])*(p2->c[0] - p1->c[0])
64  + (p2->c[1] - p1->c[1])*(p2->c[1] - p1->c[1])
65  + (p2->c[2] - p1->c[2])*(p2->c[2] - p1->c[2]);
66 
67  lent += sqrt(len);
68  na++;
69  }
70  }
71 
72  dna = (double)na;
73  dna = 1.0 / dna;
74  lent *= dna;
75 
76  return lent;
77 }

103  MMG5_pTetra pt;
104  MMG5_pxTetra pxt;
105  MMG5_pPoint p0,p1;
106  double len,lmax,o[3],hma2;
107  double *m1,*m2,*mp;
108  int k,ip,ip1,ip2,list[MMG3D_LMAX+2],ilist,ns,ier,iadr,src;
109  int8_t imax,i,i1,i2;
110  static int8_t mmgWarn0 = 0;
111 
112  *warn=0;
113  ns = 0;
114  hma2 = mesh->info.hmax*mesh->info.hmax;
115 
116  for (k=1; k<=mesh->ne; k++) {
117  pt = &mesh->tetra[k];
118  pxt = pt->xt ? &mesh->xtetra[pt->xt] : NULL;
119 
120  if ( !MG_EOK(pt) || (pt->tag & MG_REQ) ) continue;
121  if ( pt->mark != itdeg ) continue;
122 
123  /* find longest, internal edge */
124  imax = -1;
125  lmax = 0.0;
126  for (i=0; i<6; i++) {
127  i1 = MMG5_iare[i][0];
128  i2 = MMG5_iare[i][1];
129  ip1 = pt->v[i1];
130  ip2 = pt->v[i2];
131  p0 = &mesh->point[ip1];
132  p1 = &mesh->point[ip2];
133 
134  /* Skip the non-internal edges */
135  if ( pxt && (pxt->tag[i] & MG_BDY) ) continue;
136  /* Skip boundary edges - WARNING: This also skips edges connecting two
137  * boundary points. */
138  if( (p0->tag & MG_BDY) && (p1->tag & MG_BDY) ) continue;
139 
140  len = (p1->c[0]-p0->c[0])*(p1->c[0]-p0->c[0])
141  + (p1->c[1]-p0->c[1])*(p1->c[1]-p0->c[1])
142  + (p1->c[2]-p0->c[2])*(p1->c[2]-p0->c[2]);
143 
144  if ( len > lmax ) {
145  lmax = len;
146  imax = i;
147  }
148  }
149  if ( imax==-1 ) {
150  if ( !mmgWarn0 ){
151  mmgWarn0 = 1;
152  fprintf(stderr,"\n ## Warning: %s: all edges of tetra %d are required"
153  " or of length null.\n",__func__,k);
154  }
155  continue;
156  }
157 
158  if ( lmax < hma2 ) continue;
159 
160  /* proceed edges according to lengths */
161  i1 = MMG5_iare[imax][0];
162  i2 = MMG5_iare[imax][1];
163  ip1 = pt->v[i1];
164  ip2 = pt->v[i2];
165  p0 = &mesh->point[ip1];
166  p1 = &mesh->point[ip2];
167 
168  /* Deal only with internal faces */
169 #ifndef NDEBUG
170  if ( pxt ) {
171  assert( !(pxt->tag[imax] & MG_BDY) ); }
172 #endif
173  ilist = MMG5_coquil(mesh,k,imax,list);
174 
175  if ( !ilist ) continue;
176  else if ( ilist<0 ) return -1;
177 
178  o[0] = 0.5*(p0->c[0] + p1->c[0]);
179  o[1] = 0.5*(p0->c[1] + p1->c[1]);
180  o[2] = 0.5*(p0->c[2] + p1->c[2]);
181 
182 #ifdef USE_POINTMAP
183  src = p0->src;
184 #else
185  src = 1;
186 #endif
187  ip = MMG3D_newPt(mesh,o,MG_NOTAG,src);
188 
189  if ( !ip ) {
190  /* reallocation of point table */
191  int oldnpmax = mesh->npmax;
192  MMG3D_POINT_REALLOC(mesh,met,ip,mesh->gap,*warn=1;break,o,MG_NOTAG,src);
193  if( disp->m ) {
194  MMG5_ADD_MEM(mesh,(disp->size*(mesh->npmax-disp->npmax))*sizeof(double),
195  "larger displacement",
196  MMG5_SAFE_RECALLOC(mesh->point,mesh->npmax+1,oldnpmax+1,MMG5_Point,,);
197  mesh->memCur -= (mesh->npmax - oldnpmax)*sizeof(MMG5_Point);
198  mesh->npmax = oldnpmax;
199  mesh->np = mesh->npmax-1;
200  mesh->npnil = 0;
201  break);
202  MMG5_SAFE_REALLOC(disp->m,disp->size*(disp->npmax+1),
203  disp->size*(mesh->npmax+1),
204  double,"larger displacement",
205  MMG5_SAFE_RECALLOC(mesh->point,mesh->npmax+1,oldnpmax+1,MMG5_Point,,);
206  mesh->memCur -= (mesh->npmax - oldnpmax)*sizeof(MMG5_Point);
207  mesh->npmax = oldnpmax;
208  mesh->np = mesh->npmax-1;
209  mesh->npnil = 0;
210  break);
211  }
212  disp->npmax = mesh->npmax;
213  }
214 
215  /* Interpolation of metric, if any */
216  if ( met->m ) {
217  if ( !MMG5_intmet(mesh,met,k,imax,ip,0.5) ) {
218  MMG3D_delPt(mesh,ip);
219  return -1;
220  }
221  }
222 
223  /* Interpolation of displacement */
224  if ( disp->m ) {
225  iadr = disp->size*ip1;
226  m1 = &disp->m[iadr];
227  iadr = disp->size*ip2;
228  m2 = &disp->m[iadr];
229  iadr = disp->size*ip;
230  mp = &disp->m[iadr];
231 
232  if ( !MMG5_intdispvol(m1,m2,mp,0.5) ) {
233  MMG3D_delPt(mesh,ip);
234  return -1;
235  }
236  }
237 
238  /* Il y a un check sur la taille des arÃªtes ici aussi ! */
239  ier = MMG5_split1b(mesh,met,list,ilist,ip,1,1,0);
240  if ( ier < 0 ) {
241  fprintf(stderr,"\n ## Error: %s: unable to split.\n",__func__);
242  return -1;
243  }
244  else if ( !ier ) {
245  MMG3D_delPt(mesh,ip);
246  }
247  else {
248  ns++;
249  }
250  }
251 
252  return ns;
253 }

382 
383  nc = nnm = 0;

426  }
427  }

429  }

431  p1->flag = base;
432  if ( isnm ) nnm++;
433  nc++;
434  break;
435  }
436  }
437  }

500 short MMG5_dikmov(MMG5_pMesh mesh,MMG5_pSol disp,short *lastt) {
501  int it,maxit;
502  short t,tmin,tmax;
503  int8_t ier;
504 
505  maxit = 200;
506  it = 0;
507 
508  tmin = 0;
509  tmax = MMG3D_SHORTMAX;
510 
511  *lastt = 0;
512 
513  /* If full displacement can be achieved */
514  if ( !MMG5_chkmovmesh(mesh,disp,tmax,NULL) )
515  return tmax;
516 
517  /* Else, find the largest displacement by dichotomy */
518  while( tmin != tmax && it < maxit ) {
519  t = (tmin+tmax)/2;
520 
521  /* Case that tmax = tmin +1 : check move with tmax */
522  if ( t == tmin ) {
523  ier = MMG5_chkmovmesh(mesh,disp,tmax,NULL);
524  if ( !ier ) {
525  return tmax;
526  }
527  else {
528  if ( tmin==0 ) {
529  *lastt = tmax;
530  }
531  return tmin;
532  }
533  }
534 
535  /* General case: check move with t */
536  ier = MMG5_chkmovmesh(mesh,disp,t,NULL);
537  if ( !ier ) {
538  tmin = t;
539  }
540  else
541  tmax = t;
542 
543  it++;
544  }
545 
546  if ( tmin==0 ) {
547  *lastt=t;
548  }
549 
550  return tmin;

554 int MMG5_dispmesh(MMG5_pMesh mesh,MMG5_pSol disp,short t,int itdeg) {
555  MMG5_pTetra pt;
556  MMG5_pPoint ppt;
557  double *v,tau,ctau,c[4][3],ocal,ncal;
558  int k,np;
559  int8_t i,j;
560 
561  tau = (double)t /MMG3D_SHORTMAX;
562  ctau = 1.0 - tau;
563 
564  /* Identify elements which are very distorted in the process */
565  for (k=1; k<=mesh->ne; k++) {
566  pt = &mesh->tetra[k];
567  if ( !MG_EOK(pt) ) continue;
568 
569  for (i=0; i<4; i++) {
570  np = pt->v[i];
571  ppt = &mesh->point[np];
572  for (j=0; j<3; j++)
573  c[i][j] = ppt->c[j];
574  }
575 
576  ocal = MMG5_caltet_iso_4pt(c[0],c[1],c[2],c[3]);
577 
578  for (i=0; i<4; i++) {
579  np = pt->v[i];
580  v = &disp->m[3*np];
581  for (j=0; j<3; j++)
582  c[i][j] += tau*v[j];
583  }
584 
585  ncal = MMG5_caltet_iso_4pt(c[0],c[1],c[2],c[3]);
586 
587  if ( ncal < MMG5_DEGTOL*ocal )
588  pt->mark = itdeg;
589 
590  }
591 
592  /* Perform physical displacement */
593  for (k=1; k<=mesh->np; k++) {
594  ppt = &mesh->point[k];
595 
596  if ( !MG_VOK(ppt) ) continue;
597  v = &disp->m[3*k];
598 
599  for (i=0; i<3; i++) {
600  ppt->c[i] = ppt->c[i] + tau*v[i];
601  v[i] *= ctau;
602  }
603  }
604 
605  return 1;
606 }

651 int MMG5_mmg3d3(MMG5_pMesh mesh,MMG5_pSol disp,MMG5_pSol met,int **invalidTets) {
652  double avlen,tau;
653  int itdc,itmn,maxitmn,maxitdc,nspl,ns,nm,nc,iit,k,warn;
654  int nns,nnm,nnc,nnspl,nnns,nnnm,nnnc,nnnspl,ninvalidTets;
655  short t,lastt;
656  int8_t ier;
657 
658  tau = 0.0;
659  maxitmn = 10;
660  maxitdc = 100;
661  t = 0;
662  ninvalidTets = 0;
663  lastt = 0;
664 
665  //++mesh->info.fem;
666 
667  if ( abs(mesh->info.imprim) > 4 || mesh->info.ddebug )
668  fprintf(stdout," ** LAGRANGIAN MOTION\n");
669 
670  /* Field mark stores information about whether a tetra has been greatly deformed during current step */
671  for (k=1; k<=mesh->ne; k++)
672  mesh->tetra[k].mark = 0;
673 
674  /* Estimates of the minimum and maximum edge lengths in the mesh */
675  avlen = MMG5_estavglen(mesh);
676  mesh->info.hmax = MMG3D_LLONG*avlen;
677  mesh->info.hmin = MMG3D_LOPTS*avlen;
678 
679  //printf("Average length: %f ; proceed with hmin = %f, hmax = %f\n",avlen,mesh->info.hmin,mesh->info.hmax);
680 
681  for (itmn=0; itmn<maxitmn; itmn++) {
682  nnnspl = nnnc = nnns = nnnm = 0;
683 
684 #ifdef USE_ELAS
685  /* Extension of the velocity field */
686  if ( !MMG5_velextLS(mesh,disp) ) {
687  fprintf(stderr,"\n ## Problem in func. MMG5_packLS. Exit program.\n");
688  return 0;
689  }
690 #else
691  fprintf(stderr,"\n ## Error: %s: you need to compile with the USE_ELAS"
692  " CMake's flag set to ON to use the rigidbody movement.\n",__func__);
693  return 0;
694 #endif
695 
696  //MMG5_saveDisp(mesh,disp);
697 
698  /* Dichotomy loop */
699  for (itdc=0; itdc<maxitdc; itdc++) {
700  nnspl = nnc = nns = nnm = 0;
701 
702  t = MMG5_dikmov(mesh,disp,&lastt);
703  if ( t == 0 ) {
704  if ( abs(mesh->info.imprim) > 4 || mesh->info.ddebug )
705  fprintf(stderr,"\n *** Stop: impossible to proceed further\n");
706  break;
707  }
708 
709  ier = MMG5_dispmesh(mesh,disp,t,itdc);
710  if ( !ier ) {
711  fprintf(stderr,"\n ** Impossible motion\n");
712  return 0;
713  }
714 
715  tau = tau + ((double)t /MMG3D_SHORTMAX)*(1.0-tau);
716  if ( (abs(mesh->info.imprim) > 3 ) || mesh->info.ddebug )
717  printf(" ---> Realized displacement: %f\n",tau);
718 
719  /* Local remeshing depending on the option */
720  if ( mesh->info.lag > 0 ) {
721  for ( iit=0; iit<5; iit++) {
722 
723  nspl = nc = ns = nm = 0;
724 
725  if ( mesh->info.lag > 1 ) {
726  /* Split of points */
727  nspl = MMG5_spllag(mesh,disp,met,itdc,&warn);
728  if ( nspl < 0 ) {
