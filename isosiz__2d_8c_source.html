75 int MMG2D_set_metricAtPointsOnReqEdges ( MMG5_pMesh mesh,MMG5_pSol met, int8_t ismet ) {
76  MMG5_pTria pt;
77  int k,i,iadj;
78 
79  /* Reset the tria flag */
80  for ( k=1; k<=mesh->nt; k++ ) {
81  mesh->tria[k].flag = 0;
82  }
83 
84  /* Reset the input metric at required edges extremities */
85  if ( !MMG5_reset_metricAtReqEdges_surf (mesh, met, ismet ) ) {
86  return 0;
87  }
88 
89  /* Process the required edges and add the edge length to the metric of the
90  * edge extremities */
91  for ( k=1; k<=mesh->nt; k++ ) {
92  pt = &mesh->tria[k];
93  if ( !MG_EOK(pt) ) continue;
94 
95  /* Mark the tria as proceeded */
96  pt->flag = 1;
97  for ( i=0; i<3; i++ ) {
98  if ( (pt->tag[i] & MG_REQ) || (pt->tag[i] & MG_NOSURF) ||
99  (pt->tag[i] & MG_PARBDY) ) {
100  /* Check if the edge has been proceeded by the neighbour triangle */
101  iadj = mesh->adja[3*(k-1)+i+1];
102  if ( iadj && mesh->tria[iadj/3].flag ) continue;
103 
104  if ( !MMG2D_sum_reqEdgeLengthsAtPoint(mesh,met,pt,i) ) {
105  return 0;
106  }
107  }
108  }
109  }
110 
111  /* Travel the points and compute the metric of the points belonging to
112  * required edges as the mean of the required edges length */
113  if ( !MMG5_compute_meanMetricAtMarkedPoints ( mesh,met ) ) {
114  return 0;
115  }
116 
117 return 1;
118 }

130 int MMG2D_defsiz_iso(MMG5_pMesh mesh,MMG5_pSol met) {
131  MMG5_pTria pt;
132  MMG5_pPoint p0,p1,p2;
133  MMG5_pPar ppa;
134  double t1[2],t2[2],b1[2],b2[2],gpp1[2],gpp2[2],pv,M1,M2;
135  double ps1,ps2,ux,uy,ll,li,lm,hmax,hausd,hmin,lhmax,lhausd;
136  int k,l,ip,ip1,ip2;
137  int8_t ismet;
138  uint8_t i,i1,i2;
139 
140  if ( !MMG5_defsiz_startingMessage (mesh,met,__func__) ) {
141  return 0;
142  }
143 
144  /* Reset the s and flag field : s is used to count the number of req edges to
145  * which a req point belongs and flag is used to differenciate the point not
146  * yet treated, the points with a required metric and the classical points */
147  for (k=1; k<=mesh->np; k++) {
148  p0 = &mesh->point[k];
149  p0->flag = 0;
150  p0->s = 0;
151  }
152 
154  hmax = mesh->info.hmax;
155  hausd = mesh->info.hausd;
156  hmin = mesh->info.hmin;
157 
158  /* Allocate the structure */
159  if ( !met->np ) {
160  ismet = 0;
161 
162  /* Allocate and store the header informations for each solution */
163  if ( !MMG2D_Set_solSize(mesh,met,MMG5_Vertex,mesh->np,MMG5_Scalar) ) {
164  return 0;
165  }
166  }
167  else {
168  ismet = 1;
169  assert ( met->m );
170  }
171 
175  if ( !mesh->info.nosizreq ) {
176  if ( !MMG2D_set_metricAtPointsOnReqEdges ( mesh,met,ismet ) ) {
177  return 0;
178  }
179  }
180 
182  if ( !ismet ) {
183  /* Initialize metric with a constant size */
184  for ( k=1; k<=mesh->np; k++ ) {
185  if ( mesh->point[k].flag ) {
186  continue;
187  }
188  met->m[k] = hmax;
189  mesh->point[k].flag = 1;
190  }
191  }
192 
194  for ( k=1; k<=mesh->nt; k++ ) {
195  pt = &mesh->tria[k];
196  if ( !MG_EOK(pt) ) continue;
197 
198  for ( i=0; i<3; i++ ) {
199 
200  if ( !MG_EDG(pt->tag[i]) ) continue;
201 
202  i1 = MMG5_inxt2[i];
203  i2 = MMG5_iprv2[i];
204  ip1 = pt->v[i1];
205  ip2 = pt->v[i2];
206 
207  p1 = &mesh->point[ip1];
208  p2 = &mesh->point[ip2];
209 
210  if ( p1->flag>1 && p2->flag>1 ) continue;
211 
212  lhmax = hmax;
213  lhausd = hausd;
214 
215  /* Retrieve local parameters associated to edge i */
216  if ( mesh->info.npar ) {
217  for (l=0; l<mesh->info.npar; l++) {
218  ppa = &mesh->info.par[l];
219  if ( ppa->elt == MMG5_Edg && ppa->ref == pt->edg[i] ) {
220  lhmax = ppa->hmax;
221  lhausd = ppa->hausd;
222  break;
223  }
224  }
225  }
226 
227 
228  ux = p2->c[0] - p1->c[0];
229  uy = p2->c[1] - p1->c[1];
230  ll = ux*ux + uy*uy;
231  if ( ll < MMG5_EPSD ) continue;
232  li = 1.0 / sqrt(ll);
233 
234  /* Recovery of the two tangent vectors associated to points p1,p2; they
235  * need not be oriented in the same fashion */
236  if ( MG_CRN & p1->tag || (p1->tag & MG_NOM) ) {
237  t1[0] = li*ux;
238  t1[1] = li*uy;
239  }
240  else {
241  t1[0] = -p1->n[1];
242  t1[1] = p1->n[0];
243  }
244 
245  if ( MG_CRN & p2->tag || (p2->tag & MG_NOM) ) {
246  li = 1.0 / sqrt(ll);
247  t2[0] = li*ux;
248  t2[1] = li*uy;
249  }
250  else {
251  t2[0] = -p2->n[1];
252  t2[1] = p2->n[0];
253  }
254 
255  /* Calculation of the two Bezier coefficients along the curve */
256  ps1 = ux*t1[0] + uy*t1[1];
257  b1[0] = p1->c[0] + MMG5_ATHIRD*ps1*t1[0];
258  b1[1] = p1->c[1] + MMG5_ATHIRD*ps1*t1[1];
259 
260  ps2 = ux*t2[0]+uy*t2[1];
261  b2[0] = p2->c[0] - MMG5_ATHIRD*ps2*t2[0];
262  b2[1] = p2->c[1] - MMG5_ATHIRD*ps2*t2[1];
263 
264  ps1 *= ps1;
265  ps2 *= ps2;
266 
267  if ( ps1 < MMG5_EPSD || ps2 < MMG5_EPSD ) continue;
268 
269  /* \gamma^{\prime\prime}(0); \gamma^\prime(0) = ps*t1 by construction */
270  gpp1[0] = 6.0*(p1->c[0] - 2.0*b1[0] + b2[0]);
271  gpp1[1] = 6.0*(p1->c[1] - 2.0*b1[1] + b2[1]);
272 
273  /* Vector product gpp1 ^ t1 */
274  pv = gpp1[0]*t1[1] - gpp1[1]*t1[0];
275  M1 = fabs(pv)/ps1;
276 
277  /* \gamma^{\prime\prime}(1); \gamma^\prime(1) = -ps*t2 by construction */
278  gpp2[0] = 6.0*(p2->c[0] - 2.0*b2[0] + b1[0]);
279  gpp2[1] = 6.0*(p2->c[1] - 2.0*b2[1] + b1[1]);
280 
281  /* Vector product gpp2 ^ t2 */
282  pv = gpp2[0]*t2[1] - gpp2[1]*t2[0];
283  M2 = fabs(pv)/ps2;
284 
285  M1 = MG_MAX(M1,M2);
286  if ( M1 < MMG5_EPSD )
287  lm = lhmax;
288  else {
289  lm = 8.0*lhausd / M1;
290  lm = MG_MIN(lhmax,sqrt(lm));
291  }
292 
293  if ( p1->flag < 3 ) {
294  met->m[ip1] = MG_MAX(hmin,MG_MIN(met->m[ip1],lm));
295  }
296  if ( p2->flag < 3 ) {
297  met->m[ip2] = MG_MAX(hmin,MG_MIN(met->m[ip2],lm));
298  }
299  }
300  }
301 
303  /* Without local parameters information, only the boundary edges impose a
304  * minimum size feature */
305  if ( mesh->info.npar ) {
306  /* Minimum size feature imposed by triangles */
307  for (k=1; k<=mesh->nt; k++) {
308  pt = &mesh->tria[k];
309  if ( !MG_EOK(pt) ) continue;
310 
311  /* Retrieve local parameters associated to triangle k */
312  for (l=0; l<mesh->info.npar; l++) {
313  ppa = &mesh->info.par[l];
314  if ( ppa->elt == MMG5_Triangle && ppa->ref == pt->ref ) {
315  for (i=0; i<3; i++) {
316  ip = pt->v[i];
317  if ( mesh->point[ip].flag < 3 ) {
318  met->m[ip] = MG_MAX(hmin,MG_MIN(met->m[ip],ppa->hmax));
319  }
320  }
321  break;
322  }
323  }
324  }
325  /* Minimum size feature imposed by vertices */
326  for (k=1; k<=mesh->np; k++) {
327  p0 = &mesh->point[k];
328  if ( (!MG_VOK(p0)) || p0->flag == 3 ) continue;
329 
330  /* Retrieve local parameters associated to vertex k */
331  for (l=0; l<mesh->info.npar; l++) {
332  ppa = &mesh->info.par[l];
333  if ( ppa->elt == MMG5_Vertex && ppa->ref == p0->ref ) {
334  met->m[k] = MG_MAX(hmin,MG_MIN(met->m[k],ppa->hmax));
335  break;
336  }
337  }
338  }
339  }
340 
341  return 1;
342 }
