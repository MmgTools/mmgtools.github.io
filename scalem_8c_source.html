45 int MMG5_boundingBox(MMG5_pMesh mesh) {
46  MMG5_pPoint ppt;
47  int k,i;
48  double dd;
49 
50  /* compute bounding box */
51  for (i=0; i<mesh->dim; i++) {
52  mesh->info.min[i] = DBL_MAX;
53  mesh->info.max[i] = -DBL_MAX;
54  }
55  for (k=1; k<=mesh->np; k++) {
56  ppt = &mesh->point[k];
57  if ( !MG_VOK(ppt) ) continue;
58  for (i=0; i<mesh->dim; i++) {
59  if ( ppt->c[i] > mesh->info.max[i] ) mesh->info.max[i] = ppt->c[i];
60  if ( ppt->c[i] < mesh->info.min[i] ) mesh->info.min[i] = ppt->c[i];
61  }
62  ppt->tmp = 0;
63  }
64  mesh->info.delta = 0.0;
65  for (i=0; i<mesh->dim; i++) {
66  dd = mesh->info.max[i] - mesh->info.min[i];
67  if ( dd > mesh->info.delta ) mesh->info.delta = dd;
68  }
69  if ( mesh->info.delta < MMG5_EPSD ) {
70  fprintf(stderr,"\n ## Error: %s: unable to scale mesh:"
71  " Check that your mesh contains non-zero points and "
72  "valid elements.\n",__func__);
73  return 0;
74  }
75 
76  return 1;
77 }

122 int MMG5_scale_scalarMetric(MMG5_pMesh mesh, MMG5_pSol met, double dd,
123  int8_t sethmin, int8_t sethmax) {
124  int k;
125  static int8_t mmgWarn0 = 0;
126 
127  for (k=1; k<=mesh->np; k++) {
128  if( !MG_VOK( &mesh->point[k] ) ) continue;
129  /* Check the metric */
130  if ( met->m[k] <= 0 ) {
131  if ( !mmgWarn0 ) {
132  mmgWarn0 = 1;
133  fprintf(stderr,"\n ## Error: %s: at least 1 wrong metric.\n",
134  __func__);
135  return 0;
136  }
137  }
138  /* normalization */
139  met->m[k] *= dd;
140  }
141 
142  /* compute hmin and hmax parameters if not provided by the user */
143  if ( !sethmin ) {
144  mesh->info.hmin = FLT_MAX;
145  for (k=1; k<=mesh->np; k++) {
146  mesh->info.hmin = MG_MIN(mesh->info.hmin,met->m[k]);
147  }
148  }
149  if ( !sethmax ) {
150  mesh->info.hmax = 0.;
151  for (k=1; k<=mesh->np; k++) {
152  mesh->info.hmax = MG_MAX(mesh->info.hmax,met->m[k]);
153  }
154  }
155 
156  /* Check the compatibility between the user settings and the automatically
157  * computed values */
158  MMG5_check_hminhmax(mesh,sethmin,sethmax);
159 
160  /* Truncature */
161  for (k=1; k<=mesh->np; k++) {
162  met->m[k]=MG_MAX(mesh->info.hmin,met->m[k]);
163  met->m[k]=MG_MIN(mesh->info.hmax,met->m[k]);
164  }
165  return 1;
166 }

182 int MMG5_scale_meshAndSol(MMG5_pMesh mesh,MMG5_pSol met,MMG5_pSol sol,double *dd,
183  int8_t* sethmin,int8_t *sethmax ) {
184  MMG5_pPoint ppt;
185  MMG5_pPar par;
186  int k,i;
187  int8_t hsizOrOptim;
188 
189  /* sol is a level-set or a displacement so it cannot be an aniso metric */
190  if ( sol ) { assert ( sol->type == MMG5_Scalar || sol->type == MMG5_Vector ); }
191 
192  /* met is a metric so it cannot be a vector */
193  if ( met ) { assert ( met->type != MMG5_Vector ); }
194 
195  /* if we are not in iso or lagrangian mode, check that sol isn't allocated */
196  if ( (!mesh->info.iso) && mesh->info.lag < 0 ) {
197  assert ( !(sol && sol->m) );
198  }
199 
200  /* compute bounding box */
201  if ( ! MMG5_boundingBox(mesh) ) return 0;
202 
203  /* normalize coordinates */
204  *dd = 1.0 / mesh->info.delta;
205  for (k=1; k<=mesh->np; k++) {
206  ppt = &mesh->point[k];
207  if ( !MG_VOK(ppt) ) continue;
208  for (i=0 ; i<mesh->dim ; i++)
209  ppt->c[i] = (*dd) * (ppt->c[i] - mesh->info.min[i]);
210  }
211 
212  mesh->info.hausd *= (*dd);
213  mesh->info.ls *= (*dd);
214  mesh->info.hsiz *= (*dd);
215 
216  /* normalize local parameters */
217  for (k=0; k<mesh->info.npar; k++) {
218  par = &mesh->info.par[k];
219  par->hmin *= (*dd);
220  par->hmax *= (*dd);
221  par->hausd *= (*dd);
222  }
223 
224  /* Check if hmin/hmax have been provided by the user and scale it if yes */
225  *sethmin = 0;
226  *sethmax = 0;
227 
228  if ( mesh->info.hmin > 0. ) {
229  mesh->info.hmin *= (*dd);
230  *sethmin = 1;
231  }
232  if ( mesh->info.hmax > 0. ) {
233  mesh->info.hmax *= (*dd);
234  *sethmax = 1;
235  }
236 
237  hsizOrOptim = ( mesh->info.hsiz > 0. || mesh->info.optim )? 1 : 0;
238 
239  /* if we are not in optim or hsiz mode and if we don't have a metric, compute
240  * default hmin/hmax */
241  if ( (!hsizOrOptim) && (!(met && met->np)) ) {
242  /* Set default values to hmin/hmax from the bounding box if not provided by
243  * the user */
244  if ( !MMG5_Set_defaultTruncatureSizes(mesh,*sethmin,*sethmax) ) {
245  fprintf(stderr,"\n ## Error: %s: Exit program.\n",__func__);
246  return 0;
247  }
248  *sethmin = 1;
249  *sethmax = 1;
250  }
251 
252  if ( sol && sol->np ) {
253  for ( k=sol->size; k<sol->size*(mesh->np+1); k++ ) {
254  sol->m[k] *= (*dd);
255  }
256  }
257 
258  return 1;
259 
260 }
273 int MMG5_scaleMesh(MMG5_pMesh mesh,MMG5_pSol met,MMG5_pSol sol) {
274  double dd,d1;
275  int k,i;
276  double *m;
277  double lambda[3],v[3][3],isqhmin,isqhmax;
278  int8_t sethmin,sethmax;
279 
280  if ( !MMG5_scale_meshAndSol(mesh,met,sol,&dd,&sethmin,&sethmax) ) {
281  return 0;
282  }
283 
284  if ( (!met) || (met && !met->np) || (!met->m) ) {
285  return 1;
286  }
287 
288  switch ( met->size ) {
289  case 1:
290  if ( !MMG5_scale_scalarMetric ( mesh, met, dd, sethmin, sethmax ) ) {
291  return 0;
292  }
293  break;
294 
295  case 6:
296  d1 = 1.0 / (dd*dd);
297  /* Normalization */
298  for (k=1; k<=mesh->np; k++) {
299 
300  if( !MG_VOK( &mesh->point[k] ) ) continue;
301 
302  for ( i=0; i<met->size; ++i ) {
303  met->m[6*k+i] *= d1;
304  }
305  }
306 
307  /* compute hmin and hmax parameters if not provided by the user and check
308  * the input metric */
309  if ( !sethmin ) {
310  mesh->info.hmin = FLT_MAX;
311  }
312 
313  if ( !sethmax ) {
314  mesh->info.hmax = 0.;
315  }
316 
317  for (k=1; k<=mesh->np; k++) {
318  if( !MG_VOK( &mesh->point[k] ) ) continue;
319 
320  m = &met->m[6*k];
321 
322  /* Check the input metric */
323  if ( !MMG5_eigenv(1,m,lambda,v) ) {
324  fprintf(stderr,"\n ## Error: %s: unable to diagonalize at least"
325  " 1 metric (point %d).\n",__func__,k);
326  return 0;
327  }
328  for (i=0; i<3; i++) {
329  if(lambda[i]<=0) {
330  fprintf(stderr,"\n ## Error: %s: at least 1 wrong metric "
331  "(point %d -> eigenvalues : %e %e %e).\n"
332  " metric tensor: %e %e %e %e %e %e.\n",
333  __func__,k,lambda[0],lambda[1],lambda[2],
334  m[0],m[1],m[2],m[3],m[4],m[5]);
335  return 0;
336  }
337  if ( !sethmin )
338  mesh->info.hmin = MG_MIN(mesh->info.hmin,1./sqrt(lambda[i]));
339  if ( !sethmax )
340  mesh->info.hmax = MG_MAX(mesh->info.hmax,1./sqrt(lambda[i]));
341  }
342  }
343 
344  /* Check the compatibility between the user settings and the automatically
345  * computed values */
346  MMG5_check_hminhmax(mesh,sethmin,sethmax);
347 
348  /* Truncature */
349  isqhmin = 1.0 / (mesh->info.hmin*mesh->info.hmin);
350  isqhmax = 1.0 / (mesh->info.hmax*mesh->info.hmax);
351  for (k=1; k<=mesh->np; k++) {
352 
353  if( !MG_VOK( &mesh->point[k] ) ) continue;
354 
355  m = &met->m[6*k];
356 
357  if ( !MMG5_eigenv(1,m,lambda,v) ) {
358  fprintf(stderr,"\n ## Error: %s: unable to diagonalize at least"
359  " 1 metric (point %d).\n",__func__,k);
360  return 0;
361  }
362  for (i=0; i<3; i++) {
363  if(lambda[i]<=0) {
364  fprintf(stderr,"\n ## Error: %s: at least 1 wrong metric "
365  "(point %d -> eigenvalues : %e %e %e).\n"
366  " metric tensor: %e %e %e %e %e %e.\n",
367  __func__,k,lambda[0],lambda[1],lambda[2],
368  m[0],m[1],m[2],m[3],m[4],m[5]);
369  return 0;
370  }
371  lambda[i]=MG_MIN(isqhmin,lambda[i]);
372  lambda[i]=MG_MAX(isqhmax,lambda[i]);
373  }
374  m[0] = v[0][0]*v[0][0]*lambda[0] + v[1][0]*v[1][0]*lambda[1] + v[2][0]*v[2][0]*lambda[2];
375  m[1] = v[0][0]*v[0][1]*lambda[0] + v[1][0]*v[1][1]*lambda[1] + v[2][0]*v[2][1]*lambda[2];
376  m[2] = v[0][0]*v[0][2]*lambda[0] + v[1][0]*v[1][2]*lambda[1] + v[2][0]*v[2][2]*lambda[2];
377  m[3] = v[0][1]*v[0][1]*lambda[0] + v[1][1]*v[1][1]*lambda[1] + v[2][1]*v[2][1]*lambda[2];
378  m[4] = v[0][1]*v[0][2]*lambda[0] + v[1][1]*v[1][2]*lambda[1] + v[2][1]*v[2][2]*lambda[2];
379  m[5] = v[0][2]*v[0][2]*lambda[0] + v[1][2]*v[1][2]*lambda[1] + v[2][2]*v[2][2]*lambda[2];
380  }
381 
382  break;
383  default:
384  fprintf(stderr,"\n ## Error: %s: unexpected metric size (%d)\n",__func__,met->size);
385  break;
386  }
387 
388  return 1;
389 }

401 int MMG5_unscaleMesh(MMG5_pMesh mesh,MMG5_pSol met,MMG5_pSol sol) {
402  MMG5_pPoint ppt;
403  double dd;
404  int k,i,iadr;
405  MMG5_pPar par;
406 
407  /* de-normalize coordinates */
408  dd = mesh->info.delta;
409  for (k=1; k<=mesh->np; k++) {
410  ppt = &mesh->point[k];
411  if ( !MG_VOK(ppt) ) continue;
412  for ( i=0; i<mesh->dim; ++i ) {
413  ppt->c[i] = ppt->c[i] * dd + mesh->info.min[i];
414  }
415  }
416 
417  /* unscale paramter values */
418  if ( !mesh->info.sethmin ) {
419  mesh->info.hmin = MMG5_NONSET_HMIN;
420  }
421  else {
422  mesh->info.hmin *= dd;
423  }
424 
425  if ( !mesh->info.sethmax ) {
426  mesh->info.hmax = MMG5_NONSET_HMAX;
427  }
428  else {
429  mesh->info.hmax *= dd;
430  }
431  mesh->info.hausd *= dd;
432  mesh->info.ls *= dd;
433  mesh->info.hsiz *= dd;
434 
435  /* normalize local parameters */
436  for (k=0; k<mesh->info.npar; k++) {
437  par = &mesh->info.par[k];
438  par->hmin *= dd;
439  par->hmax *= dd;
440  par->hausd *= dd;
441  }
442 
443  /* de-normalize level-set or displacement */
444  if ( sol && sol->np && sol->m ) {
445  assert ( sol->size != MMG5_Tensor );
446  for (k=1; k<=mesh->np; k++) {
447  ppt = &mesh->point[k];
448  if ( !MG_VOK(ppt) ) continue;
449  iadr = k*sol->size;
450  for (i=0; i<sol->size; i++) sol->m[iadr+i] *= dd;
451  }
452  }
453 
454  /* reset the scaling data to ensure that if we try to unscale again, we will
455  * do nothing */
456  mesh->info.delta = 1.;
457  mesh->info.min[0]= 0.;
458  mesh->info.min[1]= 0.;
459  mesh->info.min[2]= 0.;
460 
461  /* de-normalize metric */
462  if ( !(met && met->np && met->m) ) return 1;
463 
464  /* unscale sizes */
465  switch (met->type) {
466  case 1:
467  for (k=1; k<=mesh->np; k++) {
468  ppt = &mesh->point[k];
469  if ( !MG_VOK(ppt) ) continue;
470  met->m[k] *= dd;
471  }
472  break;
473  case 3:
474  dd = 1.0 / (dd*dd);
475  for (k=1; k<=mesh->np; k++) {
476  ppt = &mesh->point[k];
477  if ( !MG_VOK(ppt) ) continue;
478  for (i=0; i<met->size; i++) met->m[met->size*k+i] *= dd;
479  }
480  break;
481  default:
482  fprintf(stderr,"\n ## Error: %s: unexpected metric size (%d)\n",__func__,met->size);
483  break;
484  }
485 
486  return 1;
487 }
