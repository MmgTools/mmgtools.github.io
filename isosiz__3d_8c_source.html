60 inline double MMG5_lenedgCoor_iso(double *ca,double *cb,double *ma,double *mb) {
61  double h1,h2,l,r,len;
62 
63  h1 = *ma;
64  h2 = *mb;
65  l = (cb[0]-ca[0])*(cb[0]-ca[0]) + (cb[1]-ca[1])*(cb[1]-ca[1]) \
66  + (cb[2]-ca[2])*(cb[2]-ca[2]);
67  l = sqrt(l);
68  r = h2 / h1 - 1.0;
69  len = fabs(r) < MMG5_EPS ? l / h1 : l / (h2-h1) * log1p(r);
70 
71  return len;
72 }
