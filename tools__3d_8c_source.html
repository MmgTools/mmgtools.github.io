166  p0 = &mesh->point[ip0]; /* Ref point, from which step is counted */
167  p1 = &mesh->point[ip1];
168  if ( !(MG_GEO & p0->tag) || !(MG_GEO & p1->tag) ) {
169  return 0;
170  }
171

173  uy = p1->c[1] - p0->c[1];
174  uz = p1->c[2] - p0->c[2];
175  ll = ux*ux + uy*uy + uz*uz;
176  if ( ll < MMG5_EPSD2 ) {
177  return 0;
178  }
179  il = 1.0 / sqrt(ll);
180 
181  if ( MG_SIN(p0->tag) ) {
182  t0[0] = ux * il;
183  t0[1] = uy * il;
184  t0[2] = uz * il;
185  }
186  else {
187  memcpy(t0,&(p0->n[0]),3*sizeof(double));
188  ps = t0[0]*ux + t0[1]*uy + t0[2]*uz;
189  if ( ps < 0.0 ) {
190  t0[0] *= -1.0;
191  t0[1] *= -1.0;
192  t0[2] *= -1.0;
193  }
194  }
195  if ( MG_SIN(p1->tag) ) {
196  t1[0] = -ux * il;
197  t1[1] = -uy * il;
198  t1[2] = -uz * il;
199  }
200  else {
201  memcpy(t1,&(p1->n[0]),3*sizeof(double));
202  ps = - ( t1[0]*ux + t1[1]*uy + t1[2]*uz );
203  if ( ps < 0.0 ) {
204  t1[0] *= -1.0;
205  t1[1] *= -1.0;
206  t1[2] *= -1.0;
207  }
208  }
209  alpha = MMG5_BezierGeod(p0->c,p1->c,t0,t1);
210 
211  b0[0] = p0->c[0] + alpha * t0[0];
212  b0[1] = p0->c[1] + alpha * t0[1];
213  b0[2] = p0->c[2] + alpha * t0[2];
214 
215  b1[0] = p1->c[0] + alpha * t1[0];
216  b1[1] = p1->c[1] + alpha * t1[1];
217  b1[2] = p1->c[2] + alpha * t1[2];
218 
219  o[0] = (1.0-s)*(1.0-s)*(1.0-s)*p0->c[0] + 3.0*s*(1.0-s)*(1.0-s)*b0[0] + \
220  3.0*s*s*(1.0-s)*b1[0] + s*s*s*p1->c[0];
221 
222  o[1] = (1.0-s)*(1.0-s)*(1.0-s)*p0->c[1] + 3.0*s*(1.0-s)*(1.0-s)*b0[1] + \
223  3.0*s*s*(1.0-s)*b1[1] + s*s*s*p1->c[1];
224 
225  o[2] = (1.0-s)*(1.0-s)*(1.0-s)*p0->c[2] + 3.0*s*(1.0-s)*(1.0-s)*b0[2] + \
226  3.0*s*s*(1.0-s)*b1[2] + s*s*s*p1->c[2];
227 
228  if ( MG_SIN(p0->tag) && MG_SIN(p1->tag) ) {
229  memcpy(to,t0,3*sizeof(double));
230  return 1;
231  }
232  else if ( MG_SIN(p0->tag) ) {
233  memcpy(n11,&(mesh->xpoint[p1->xp].n1[0]),3*sizeof(double));
234  memcpy(n12,&(mesh->xpoint[p1->xp].n2[0]),3*sizeof(double));
235  memcpy(n01,&(mesh->xpoint[p1->xp].n1[0]),3*sizeof(double));
236  memcpy(n02,&(mesh->xpoint[p1->xp].n2[0]),3*sizeof(double));
237  }
238  else if ( MG_SIN(p1->tag) ) {
239  memcpy(n01,&(mesh->xpoint[p0->xp].n1[0]),3*sizeof(double));
240  memcpy(n02,&(mesh->xpoint[p0->xp].n2[0]),3*sizeof(double));
241  memcpy(n11,&(mesh->xpoint[p0->xp].n1[0]),3*sizeof(double));
242  memcpy(n12,&(mesh->xpoint[p0->xp].n2[0]),3*sizeof(double));
243  }
244  else {
245  memcpy(n01,&(mesh->xpoint[p0->xp].n1[0]),3*sizeof(double));
246  memcpy(n02,&(mesh->xpoint[p0->xp].n2[0]),3*sizeof(double));
247  memcpy(n11,&(mesh->xpoint[p1->xp].n1[0]),3*sizeof(double));
248  memcpy(n12,&(mesh->xpoint[p1->xp].n2[0]),3*sizeof(double));
249 
250  /* Switch normals of p1 for pairing */
251  ps = n01[0] * n11[0] + n01[1] * n11[1] + n01[2] * n11[2];
252  ps2 = n01[0] * n12[0] + n01[1] * n12[1] + n01[2] * n12[2];
253  if ( ps2 > ps ) {
254  memcpy(ntemp,n11,3*sizeof(double));
255  memcpy(n11,n12,3*sizeof(double));
256  memcpy(n12,ntemp,3*sizeof(double));
257  }
258  }
259 
260  /* Normal n1 interpolation */
261  ps = ux*(n01[0] + n11[0]) + uy*(n01[1] + n11[1]) + uz*(n01[2] + n11[2]);
262  ps = 2.0*ps / ll;
263 
264  bn[0] = n01[0] + n11[0] -ps*ux;
265  bn[1] = n01[1] + n11[1] -ps*uy;
266  bn[2] = n01[2] + n11[2] -ps*uz;
267

269  if ( dd > MMG5_EPSD2 ) {
270  dd = 1.0 / sqrt(dd);
271  bn[0] *= dd;
272  bn[1] *= dd;
273  bn[2] *= dd;
274  }
275  no1[0] = (1.0-s)*(1.0-s)*n01[0] + 2.0*s*(1.0-s)*bn[0] + s*s*n11[0];
276  no1[1] = (1.0-s)*(1.0-s)*n01[1] + 2.0*s*(1.0-s)*bn[1] + s*s*n11[1];
277  no1[2] = (1.0-s)*(1.0-s)*n01[2] + 2.0*s*(1.0-s)*bn[2] + s*s*n11[2];
278  dd = no1[0]*no1[0] + no1[1]*no1[1] + no1[2]*no1[2];
279  if ( dd > MMG5_EPSD2 ) {
280  dd = 1.0 / sqrt(dd);
281  no1[0] *= dd;
282  no1[1] *= dd;
283  no1[2] *= dd;
284  }
285 
286  /* Normal n2 interpolation */
287  ps = ux*(n02[0] + n12[0]) + uy*(n02[1] + n12[1]) + uz*(n02[2] + n12[2]);
288  ps = 2.0*ps/ll;
289 
290  bn[0] = n02[0] + n12[0] -ps*ux;
291  bn[1] = n02[1] + n12[1] -ps*uy;
292  bn[2] = n02[2] + n12[2] -ps*uz;
293 
294  dd = bn[0]*bn[0] + bn[1]*bn[1] + bn[2]*bn[2];
295  if ( dd > MMG5_EPSD2 ) {
296  dd = 1.0 / sqrt(dd);
297  bn[0] *= dd;
298  bn[1] *= dd;
299  bn[2] *= dd;
300  }
301  no2[0] = (1.0-s)*(1.0-s)*n02[0] + 2.0*s*(1.0-s)*bn[0] + s*s*n12[0];
302  no2[1] = (1.0-s)*(1.0-s)*n02[1] + 2.0*s*(1.0-s)*bn[1] + s*s*n12[1];
303  no2[2] = (1.0-s)*(1.0-s)*n02[2] + 2.0*s*(1.0-s)*bn[2] + s*s*n12[2];
304  dd = no2[0]*no2[0] + no2[1]*no2[1] + no2[2]*no2[2];
305  if ( dd > MMG5_EPSD2 ) {
306  dd = 1.0 / sqrt(dd);
307  no2[0] *= dd;
308  no2[1] *= dd;
309  no2[2] *= dd;
310 
311  to[0] = no1[1]*no2[2] - no1[2]*no2[1];
312  to[1] = no1[2]*no2[0] - no1[0]*no2[2];
313  to[2] = no1[0]*no2[1] - no1[1]*no2[0];
314  }
315  else {
316  /* Open boundary: tangent interpolation (possibly flip (back) t1) */
317  ps = t0[0]*t1[0] + t0[1]*t1[1] + t0[2]*t1[2];
318  if ( ps < 0.0 ) {
319  t1[0] *= -1.0;
320  t1[1] *= -1.0;
321  t1[2] *= -1.0;
322  }
323  to[0] = (1.0-s)*t0[0] + s*t1[0];
324  to[1] = (1.0-s)*t0[1] + s*t1[1];
325  to[2] = (1.0-s)*t0[2] + s*t1[2];
326 
327  /* Projection of the tangent in the tangent plane defined by no */
328  ps = to[0]*no1[0] + to[1]*no1[1] + to[2]*no1[2];
329  to[0] = to[0] -ps*no1[0];
330  to[1] = to[1] -ps*no1[1];
331  to[2] = to[2] -ps*no1[2];
332  }
333 
334  dd = to[0]*to[0] + to[1]*to[1] + to[2]*to[2];
335  if ( dd > MMG5_EPSD2 ) {
336  dd = 1.0/sqrt(dd);
337  to[0] *= dd;
338  to[1] *= dd;
339  to[2] *= dd;
340  }
341 
342  return 1;
343 }
