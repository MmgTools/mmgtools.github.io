47 int lissmet_ani(MMG5_pMesh mesh,MMG5_pSol sol) {
48  MMG5_Hash edgeTable;
49  MMG5_hedge *pht;
50  MMG5_pTria pt;
51  MMG5_pPoint p1,p2;
52  double logh,logs,*ma,*mb,ux,uy,d1,d2,dd,rap,dh;
53  double tail,coef,ma1[3],mb1[3],m[3],dd1,dd2;
54  double SQRT3DIV2=0.8660254037844386;
55  int i,nc,k,itour,maxtou,ncor,a,b,iadr;
56  int8_t ier;
57  static int8_t mmgWarn = 0;
58 
59  if ( abs(mesh->info.imprim) > 5 || mesh->info.ddebug ) {
60  fprintf(stdout," ** Grading mesh\n");
61  }
62 
63  MMG5_mark_pointsOnReqEdge_fromTria ( mesh );
64 
65  logh = mesh->info.hgrad;
66  logs = 0.001 + logh;
67  maxtou = 100;
68  ncor = 0;
69  itour = 0;
70 
71  /* alloc hashtable */
72  if ( !MMG5_hashNew(mesh,&edgeTable,mesh->ntmax,3*mesh->ntmax) ) {
73  fprintf(stderr,"\n ## Error: %s: unable to allocate hash table.\n",__func__);
74  return 0;
75  }
76 
77  /* build edge table */
78  for(k=1 ; k<=mesh->nt ; k++) {
79  pt = &mesh->tria[k];
80  if ( !MG_EOK(pt) ) {
81  continue;
82  }
83  for(i=0 ; i<3 ; i++) {
84  a = pt->v[MMG2D_iare[i][0]];
85  b = pt->v[MMG2D_iare[i][1]];
86 
87  /* Skip edges with a required vertex */
88  if ( mesh->point[a].s || mesh->point[b].s ) {
89  continue;
90  }
91  ier = MMG5_hashEdge(mesh,&edgeTable,a,b,k);
92  if ( !ier ) {
93  if ( !mmgWarn ) {
94  mmgWarn = 1;
95  fprintf(stderr,"\n ## Warning: %s: unable to hash at least one edge"
96  " (tria %d, edge %d).\n",__func__,MMG2D_indElt(mesh,k),i);
97  }
98  }
99  }
100  }
101 
102  /* reset color */
103  for (k=1; k<=mesh->np; k++)
104  (&mesh->point[k])->tagdel = mesh->base+1;
105 
106  /* analyze mesh edges via hash table */
107  do {
108  ++mesh->base;
109  nc = 0;
110  for (k=0; k<edgeTable.siz; k++) {
111  pht = &edgeTable.item[k];
112  /* analyze linked list */
113  while ( pht ) {
114  if ( !pht->a ) break;
115  a = pht->a;
116  b = pht->b;
117  p1 = &mesh->point[a];
118  p2 = &mesh->point[b];
119  iadr = a*sol->size;
120  ma = &sol->m[iadr];
121  iadr = b*sol->size;
122  mb = &sol->m[iadr];
123 
124  if ( (p1->tagdel < mesh->base) && (p2->tagdel < mesh->base) ) {
125  pht = pht->nxt ? &edgeTable.item[pht->nxt] : 0;
126  continue;
127  }
128 
129  /* compute edge lengths */
130  ux = p2->c[0] - p1->c[0];
131  uy = p2->c[1] - p1->c[1];
132 
133  d1 = ma[0]*ux*ux + ma[2]*uy*uy + 2.0*ma[1]*ux*uy;
134  assert(d1 >=0);
135  if ( d1 < 0.0 ) d1 = 0.0;
136  dd1 = M_MAX(MMG2D_EPSD,sqrt(d1));
137 
138  d2 = mb[0]*ux*ux + mb[2]*uy*uy+ 2.0*mb[1]*ux*uy;
139  assert(d2 >=0);
140  if ( d2 < 0.0 ) d2 = 0.0;
141  dd2 = M_MAX(MMG2D_EPSD,sqrt(d2));
142 
143  /* swap vertices */
144  if ( dd1 > dd2 ) {
145  p1 = &mesh->point[b];
146  p2 = &mesh->point[a];
147  mb = ma;
148  iadr = b*sol->size;
149  ma = &sol->m[iadr];
150  dd = dd1;
151  dd1 = dd2;
152  dd2 = dd;
153  }
154  rap = dd2 / dd1;
155  dh = rap - 1.0;
156  if ( fabs(dh) > MMG2D_EPSD ) {
157  // Edge length in the metric
158  tail = (dd1+dd2+4*sqrt(0.5*(d1+d2))) / 6.0;
159  coef = log(rap) / tail;
160  p1->tagdel = mesh->base+1;
161  p2->tagdel = mesh->base+1;
162 
163  /* update sizes */
164  if ( coef > logs ) {
165  coef = exp(tail*logh);
166  p1->tagdel = mesh->base;
167  p2->tagdel = mesh->base;
168 
169  /* metric intersection */
170  coef = 1.0 / (coef*coef);
171  for (i=0; i<3; i++) {
172  ma1[i] = coef * ma[i];
173  mb1[i] = coef * mb[i];
174  }
175 
176  if ( MMG5_intersecmet22(mesh,ma,mb1,m) ) {
177  for (i=0; i<3; i++) ma[i] = m[i];
178  }
179  else {
180  for (i=0; i<3; i++) ma[i] = SQRT3DIV2 * (ma[i]+mb1[i]);
181  }
182  if ( MMG5_intersecmet22(mesh,ma1,mb,m) ) {
183  for (i=0; i<3; i++) mb[i] = m[i];
184  }
185  else {
186  for (i=0; i<3; i++) mb[i] = SQRT3DIV2 * (mb[i]+ma1[i]);
187  }
188  nc++;
189  }
190  }
191  /* next edge */
192  pht = pht->nxt ? &edgeTable.item[pht->nxt] : 0;
193  }
194  }
195  ncor += nc;
196  } while ( nc && ++itour < maxtou );
197  MMG5_SAFE_FREE(edgeTable.item);
198 
199  if ( abs(mesh->info.imprim) > 3 && ncor ) {
200  fprintf(stdout," gradation: %7d updated, %d iter.\n",ncor,itour);
201  }
202 
203  return 1;
204 }
