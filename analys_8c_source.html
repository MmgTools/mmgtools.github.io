72  /* Mmgs */
73  adja = mesh->adja;
74  }

79  lm1 = 0.4;
80  lm2 = 0.399;
81  while ( it++ < nit ) {
82  /* step 1: laplacian */
83  for (k=1; k<=mesh->np; k++) {
84  ppt = &mesh->point[k];
85  if ( !MG_VOK(ppt) ) continue;
86  if ( ppt->tag & MG_CRN || ppt->tag & MG_NOM || MG_EDG(ppt->tag) ) continue;
87

89  if ( !iel ) continue; // Mmg3d
90 
91  pt = &mesh->tria[iel];
92  i = 0;
93  if ( pt->v[1] == k ) i = 1;
94  else if ( pt->v[2] == k ) i = 2;
95 
96  ilist = MMG5_boulep(mesh,iel,i,adja,list);
97 
98  /* average normal */
99  nx = ny = nz = 0.0;
100  for (i=1; i<=ilist; i++) {
101  p0 = &mesh->point[list[i]];
102  if ( p0->tag & MG_CRN || p0->tag & MG_NOM || MG_EDG(p0->tag) ) continue;
103 
104  if ( p0->xp ) {
105  /* Mmg3d */
106  pxp = &mesh->xpoint[p0->xp];
107  nptr = pxp->n1;
108  }
109  else {
110  /* Mmgs */
111  nptr = p0->n;
112  }
113 
114  nx += nptr[0];
115  ny += nptr[1];
116  nz += nptr[2];
117  }
118 
119  dd = nx*nx + ny*ny + nz*nz;
120  if ( dd > MMG5_EPSD2 ) {
121  dd = 1.0 / sqrt(dd);
122  nx *= dd;
123  ny *= dd;
124  nz *= dd;
125  }
126 
127  /* Laplacian */
128  if ( ppt->xp ) {
129  /* Mmg3d */
130  pxp = &mesh->xpoint[ppt->xp];
131  nptr = pxp->n1;
132  }
133  else {
134  /* Mmgs */
135  nptr = ppt->n;
136  }
137

139  tabl[iad+0] = nptr[0] + lm1 * (nx - nptr[0]);
140  tabl[iad+1] = nptr[1] + lm1 * (ny - nptr[1]);
141  tabl[iad+2] = nptr[2] + lm1 * (nz - nptr[2]);
142  }
143

149

159  else if ( pt->v[2] == k ) i = 2;
160 
161  ilist = MMG5_boulep(mesh,iel,i,adja,list);
162 
163  /* average normal */
164  nx = ny = nz = 0.0;
165  for (i=1; i<=ilist; i++) {
166  iad = 3*(list[i]-1) + 1;
167  nx += tabl[iad+0];
168  ny += tabl[iad+1];
169  nz += tabl[iad+2];
170  }
171  dd = nx*nx + ny*ny + nz*nz;
172  if ( dd > MMG5_EPSD2 ) {
173  dd = 1.0 / sqrt(dd);
174  nx *= dd;
175  ny *= dd;
176  nz *= dd;
177  }
178 
179  /* antiLaplacian */
180  iad = 3*(k-1)+1;
181  n[0] = tabl[iad+0] - lm2 * (nx - tabl[iad+0]);
182  n[1] = tabl[iad+1] - lm2 * (ny - tabl[iad+1]);
183  n[2] = tabl[iad+2] - lm2 * (nz - tabl[iad+2]);
184  nn++;
185 
186  if ( ppt->xp ) {
187  /* Mmg3d */
188  pxp = &mesh->xpoint[ppt->xp];
189  nptr = pxp->n1;
190  }
191  else {
192  /* Mmgs */
193  nptr = ppt->n;
194  }
195  res += (nptr[0]-n[0])*(nptr[0]*n[0]) + (nptr[1]-n[1])*(nptr[1]*n[1]) + (nptr[2]-n[2])*(nptr[2]*n[2]);
196 
197  nptr[0] = n[0];
198  nptr[1] = n[1];
199  nptr[2] = n[2];
200 
201  }
202 
203  if ( it == 1 ) res0 = res;
204  if ( res0 > MMG5_EPSD ) res = res / res0;
205  if ( mesh->info.imprim < -1 || mesh->info.ddebug ) {
206  fprintf(stdout," iter %5d res %.3E\r",it,res);
207  fflush(stdout);
208  }
209  if ( it > 1 && res < MMG5_EPS ) break;
210  }
211 
212  /* reset the ppt->s tag */
213  for (k=1; k<=mesh->np; ++k) {
214  mesh->point[k].s = 0;
215  }
216 
217  if ( mesh->info.imprim < -1 || mesh->info.ddebug ) fprintf(stdout,"\n");
218 
219  if ( abs(mesh->info.imprim) > 4 )
220  fprintf(stdout," %d normals regularized: %.3e\n",nn,res);
221 
222  MMG5_SAFE_FREE(tabl);
223  return 1;
