37 int MMG2D_movintpt_ani(MMG5_pMesh mesh,MMG5_pSol met,int ilist,int *list,int8_t improve) {
38  MMG5_pTria pt,pt0;
39  MMG5_pPoint ppt0,p0,p1,p2;
40  double calold,calnew,area,det,alpha,ps,ps1,ps2,step,sqdetm1,sqdetm2;
41  double gr[2],grp[2],*m0,*m1,*m2;

43  int8_t i,i1,i2;
44  static int8_t mmgWarn0=0;
45 
46  pt0 = &mesh->tria[0];
47  ppt0 = &mesh->point[0];
48  gr[0] = gr[1] = 0.0;
49  calold = calnew = DBL_MAX;
50  step = 0.1;
51 
52  /* Step 1: Calculation of the gradient of the variance function; store the quality of the previous
53  configuration */
54  p0 = p1 = p2 = NULL;
55  m0 = NULL;
56  assert ( ilist );
57  for (k=0; k<ilist; k++) {
58  iel = list[k] / 3;
59  pt = &mesh->tria[iel];
60 
61  /* Quality of pt */
62  calold = MG_MIN(MMG2D_caltri(mesh,met,pt),calold);
63 
64  i = list[k] % 3;
65  i1 = MMG5_inxt2[i];
66  i2 = MMG5_iprv2[i];
67 
68  ip0 = pt->v[i];
69  ip1 = pt->v[i1];
70  ip2 = pt->v[i2];
71 
72  p0 = &mesh->point[ip0];
73  p1 = &mesh->point[ip1];
74  p2 = &mesh->point[ip2];
75 
76  area = (p1->c[0]-p0->c[0])*(p2->c[1]-p0->c[1]) - (p1->c[1]-p0->c[1])*(p2->c[0]-p0->c[0]);
77  area = 0.5*fabs(area);
78 
79  m0 = &met->m[3*ip0];
80  m1 = &met->m[3*ip1];
81  m2 = &met->m[3*ip2];
82

88  }
89

91  assert(m0 && (m0+1) && (m0+2));
92  det = m0[0]*m0[2]-m0[1]*m0[1];
93 
94  if ( det < MMG5_EPSD ) return 0;
95  det = 1.0 / det;
96

98  grp[1] = det*(-m0[1]*gr[0]+m0[0]*gr[1]);
99 
100  /* Step 2: Identification of the triangle such that gr is comprised in the associated angular sector */
101  ps1 = ps2 = 0.;
102  for (k=0; k<ilist; k++) {
103  iel = list[k] / 3;
104  pt = &mesh->tria[iel];
105 
106  i = list[k] % 3;
107  i1 = MMG5_inxt2[i];
108  i2 = MMG5_iprv2[i];
109

111  ip1 = pt->v[i1];
112  ip2 = pt->v[i2];
113

115  p1 = &mesh->point[ip1];
116  p2 = &mesh->point[ip2];
117 
118  ps1 = (p1->c[0]-p0->c[0])*grp[1] - (p1->c[1]-p0->c[1])*grp[0];
119  ps2 = grp[0]*(p2->c[1]-p0->c[1]) - grp[1]*(p2->c[0]-p0->c[0]);

122  }
123 
124  if ( k == ilist ) {
125  if ( !mmgWarn0 ) {
126  mmgWarn0=1;

130  return 0;
131  }
132 
133  /* coordinates of the proposed position for relocation p = p0 + alpha*set*grp, so that
134  the new point is inside the triangle */
135  det = (p1->c[0]-p0->c[0])*(p2->c[1]-p0->c[1]) - (p1->c[1]-p0->c[1])*(p2->c[0]-p0->c[0]);
136  ps = ps1+ps2;
137  if ( ps < MMG5_EPSD ) return 0;
138  alpha = det / ps;
139 
140  ppt0->c[0] = p0->c[0] + alpha*step*grp[0];
141  ppt0->c[1] = p0->c[1] + alpha*step*grp[1];
142 
143  /* Step 3: Vertex relocation and checks */
144  i = 0;
145  pt = NULL;
146  for (k=0; k<ilist; k++) {
147  iel = list[k] / 3;
148  i = list[k] % 3;
149  pt = &mesh->tria[iel];
150  memcpy(pt0,pt,sizeof(MMG5_Tria));
151  pt0->v[i] = 0;
152 
153  calnew = MG_MIN(MMG2D_caltri(mesh,met,pt0),calnew);
154  }
155 
156  if (calold < MMG2D_NULKAL && calnew <= calold) return 0;
157  else if (calnew < MMG2D_NULKAL) return 0;
158  else if ( improve && calnew < 1.02 * calold ) return 0;
159  else if ( calnew < 0.3 * calold ) return 0;
160 
161  /* Update of the coordinates of the point */
162  p0 = &mesh->point[pt->v[i]];
163  p0->c[0] = ppt0->c[0];
164  p0->c[1] = ppt0->c[1];
165 
166  return 1;
167 }
