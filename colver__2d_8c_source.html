269 /* Perform effective collapse of edge i in tria k, i1->i2 */
270 int MMG2D_colver(MMG5_pMesh mesh,int ilist,int *list) {
271  MMG5_pTria pt,pt1,pt2;
272  int iel,jel,ip1,ip2,k,kel,*adja;
273  uint8_t i,j,jj,i1,i2,open;
274 
275  iel = list[0] / 3;
276  i1 = list[0] % 3;
277  i = MMG5_iprv2[i1];
278  i2 = MMG5_inxt2[i1];
279  pt = &mesh->tria[iel];
280  ip1 = pt->v[i1];
281  ip2 = pt->v[i2];
282 
283  /* Check for open ball */
284  adja = &mesh->adja[3*(iel-1)+1];
285  open = adja[i] == 0;
286 
287  /* Update of the vertex ip1 -> ip2 */
288  mesh->point[ip2].tag |= mesh->point[ip1].tag;
289 
290  for (k=1; k<ilist-1+open; k++) {
291  jel = list[k] / 3;
292  jj = list[k] % 3;
293  pt1 = &mesh->tria[jel];
294  pt1->v[jj] = ip2;
295  pt1->base = mesh->base; // What is the use of updating the base field?
296  }
297 
298  /* Update adjacent with the first element (which necessarily exists) */
299  jel = list[1] / 3;
300  jj = list[1] % 3;
301  j = MMG5_iprv2[jj];
302  pt1 = &mesh->tria[jel];
303 
304  pt1->tag[j] |= pt->tag[i1];
305  pt1->edg[j] = MG_MAX(pt1->edg[j],pt->edg[i1]);
306 
307  if ( adja[i1] ) {
308  kel = adja[i1] / 3;
309  k = adja[i1] % 3;
310  mesh->adja[3*(kel-1)+1+k] = 3*jel+j;
311  mesh->adja[3*(jel-1)+1+j] = 3*kel+k;
312  pt2 = &mesh->tria[kel];
313  pt2->tag[k] |= pt1->tag[j];
314  pt2->edg[k] = MG_MAX(pt1->edg[j],pt2->edg[k]);
315  }
316  else
317  mesh->adja[3*(jel-1)+1+j] = 0;
318 
319  /* When the boundary is not open, update the adjacency relation with the second element */
320  if ( !open ) {
321  iel = list[ilist-1] / 3;
322  i1 = list[ilist-1] % 3;
323  pt = &mesh->tria[iel];
324 
325  jel = list[ilist-2] / 3;
326  jj = list[ilist-2] % 3;
327  j = MMG5_inxt2[jj];
328  pt1 = &mesh->tria[jel];
329  pt1->tag[j] |= pt->tag[i1];
330  pt1->edg[j] = MG_MAX(pt->edg[i1],pt1->edg[j]);
331  adja = &mesh->adja[3*(iel-1)+1];
332 
333  if ( adja[i1] ) {
334  kel = adja[i1] / 3;
335  k = adja[i1] % 3;
336  mesh->adja[3*(kel-1)+1+k] = 3*jel + j;
337  mesh->adja[3*(jel-1)+1+j] = 3*kel + k;
338  pt2 = &mesh->tria[kel];
339  pt2->tag[k] |= pt1->tag[j];
340  pt2->edg[k] = MG_MAX(pt1->edg[j],pt2->edg[k]);
341  }
342  else
343  mesh->adja[3*(jel-1)+1+j] = 0;
344  }
345 
346  /* Delete removed point and elements */
347  MMG2D_delPt(mesh,ip1);
348  MMG2D_delElt(mesh,list[0] / 3);
349  if ( !open ) MMG2D_delElt(mesh,list[ilist-1] / 3);
350 
351  return 1;
352 }

363  i = list[0] % 3;
364  i1 = MMG5_inxt2[i];
365  pt = &mesh->tria[iel];
366  ip = pt->v[i];

368  j = list[1] % 3;
369  j1 = MMG5_inxt2[j];
370  j2 = MMG5_iprv2[j];
371  pt1 = &mesh->tria[jel];
372 
373  kel = list[2] / 3;
374  k = list[2] % 3;
375  pt2 = &mesh->tria[kel];
376 
377  /* update info */
378  pt1->v[j] = pt->v[i1];
379  pt1->tag[j1] |= pt2->tag[k];
380  pt1->edg[j1] = MG_MAX(pt1->edg[j1],pt2->edg[k]);
381  pt1->tag[j2] |= pt->tag[i];
382  pt1->edg[j2] = MG_MAX(pt1->edg[j2],pt->edg[i]);
383  pt1->base = mesh->base;
384 
385  /* Update adjacency relations */
386  adja = &mesh->adja[3*(jel-1)+1];
387  adja[j1] = mesh->adja[3*(kel-1)+1+k];
388  adja[j2] = mesh->adja[3*(iel-1)+1+i];
389 
390  mel = adja[j2] / 3;
391  if ( mel ) {
392  m = adja[j2] % 3;
393  pt = &mesh->tria[mel];
394  pt->tag[m] = pt1->tag[j2];
395  pt->edg[m] = pt1->edg[j2];
396  mesh->adja[3*(mel-1)+1+m] = 3*jel + j2;
397  }
398 
399  mel = adja[j1] / 3;
400  if ( mel ) {
401  m = adja[j1] % 3;
402  pt = &mesh->tria[mel];
403  pt->tag[m] = pt1->tag[j1];
404  pt->edg[m] = pt1->edg[j1];
405  mesh->adja[3*(mel-1)+1+m] = 3*jel + j1;
406  }
407 
408  /* remove vertex + elements */
409  MMG2D_delPt(mesh,ip);
410  MMG2D_delElt(mesh,iel);
411  MMG2D_delElt(mesh,kel);
412 
413  return 1;
414 }
